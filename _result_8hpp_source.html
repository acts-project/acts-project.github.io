279 template <typename C>
280 auto and_then(C&& callable) const&
281 requires std::invocable<C, const T&>
282 {
283 using R = decltype(std::declval<C>()(std::declval<T>()));
284
285 static_assert(std::same_as<typename R::ErrorType, ErrorType>,
286 "bind must take a callable with the same error type");

288 if (ok()) {
289 return callable(value());
290 } else {
291 return R::failure(error());

293 }

327 void checkValueAccess() const {
328 if (m_var.index() != 0) {
329 if constexpr (std::is_same_v<E, std::error_code>) {
330 std::stringstream ss;
331 const auto& e = std::get<E>(m_var);
332 ss << "Value called on error value: " << e.category().name() << ": "
333 << e.message() << " [" << e.value() << "]";
334 throw std::runtime_error(ss.str());
335 } else {
336 throw std::runtime_error("Value called on error value");
