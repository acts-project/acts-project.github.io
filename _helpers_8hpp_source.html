114template <template <std::size_t> class Callable, std::size_t N,
115 std::size_t NMAX, typename... Args>
116auto template_switch(std::size_t v, Args&&... args) {
117 if (v == N) {
118 return Callable<N>::invoke(std::forward<Args>(args)...);
119 }
120 if (v == 0) {
121 std::cerr << "template_switch<Fn, " << N << ", " << NMAX << ">(v=" << v
122 << ") is not valid (v == 0 and N != 0)" << std::endl;
123 std::abort();
124 }
125 if constexpr (N < NMAX) {
126 return template_switch<Callable, N + 1, NMAX>(v,
127 std::forward<Args>(args)...);
128 }
129 std::cerr << "template_switch<Fn, " << N << ", " << NMAX << ">(v=" << v
130 << ") is not valid (v > NMAX)" << std::endl;
131 std::abort();
132}

142template <std::size_t N, std::size_t NMAX, typename Lambda, typename... Args>
143auto template_switch_lambda(std::size_t v, Lambda&& func, Args&&... args) {
144 if (v == N) {
145 return func(std::integral_constant<std::size_t, N>{},
146 std::forward<Args>(args)...);
147 }
148 if (v == 0) {
149 std::cerr << "template_switch<Fn, " << N << ", " << NMAX << ">(v=" << v
150 << ") is not valid (v == 0 and N != 0)" << std::endl;
151 std::abort();
152 }
153 if constexpr (N < NMAX) {
154 return template_switch_lambda<N + 1, NMAX>(v, func,
155 std::forward<Args>(args)...);
156 }
157 std::cerr << "template_switch<Fn, " << N << ", " << NMAX << ">(v=" << v
158 << ") is not valid (v > NMAX)" << std::endl;
159 std::abort();
160}

167template <typename T, typename U>
168T clampValue(U value) {
169 if (std::numeric_limits<U>::has_infinity && std::isinf(value)) {
170 if (!std::numeric_limits<T>::has_infinity) {
171 throw std::logic_error(
172 "Cannot convert infinite value to type without infinity support");
173 }
174 return (value > 0) ? std::numeric_limits<T>::infinity()
175 : -std::numeric_limits<T>::infinity();
176 }
177 if (std::numeric_limits<U>::has_quiet_NaN && std::isnan(value)) {
178 if (!std::numeric_limits<T>::has_quiet_NaN) {
179 throw std::logic_error(
180 "Cannot convert NaN value to type without NaN support");
181 }
182 return std::numeric_limits<T>::quiet_NaN();
183 }
184 return static_cast<T>(
185 std::clamp(value, static_cast<U>(std::numeric_limits<T>::lowest()),
186 static_cast<U>(std::numeric_limits<T>::max())));
187}

196template <typename T>
197std::tuple<typename T::value_type, double> range_medium(const T& tseries) {
198 auto [minIt, maxIt] = std::ranges::minmax_element(tseries);
199 typename T::value_type range = (*maxIt - *minIt);
200 double medium = static_cast<double>((*maxIt + *minIt) * 0.5);
201 return {range, medium};
202}

270inline auto getMinMaxAndBinCount(std::vector<double>& xPos) {
271 // sort the values for unique()
272 std::ranges::sort(xPos);
273
274 // get the number of bins over unique values
275 auto it = std::unique(xPos.begin(), xPos.end());
276 const std::size_t xBinCount = std::distance(xPos.begin(), it);
277
278 // get the minimum and maximum
279 auto [xMin, xMax] = std::ranges::minmax(xPos);
280
281 // calculate maxima (add one last bin, because bin value always corresponds to
282 // left boundary)
283 const double stepX = (xMax - xMin) / static_cast<double>(xBinCount - 1);
284 xMax += stepX;
285
286 // Return all values as a tuple
287 return std::make_tuple(xMin, xMax, xBinCount);
288}
