42 template <typename traj_t>
43 Result<void> operator()(const GeometryContext& gctx, traj_t& trajectory,
44 std::size_t entryIndex,
45 const Logger& logger = getDummyLogger()) const {
46 (void)gctx;
47
48 using TrackStateProxy = typename traj_t::TrackStateProxy;
49
50 GetParameters filtered;
51 GetCovariance filteredCovariance;
52 GetParameters smoothed;
53 GetParameters predicted;
54 GetCovariance predictedCovariance;
55 GetCovariance smoothedCovariance;
56 GetCovariance jacobian;
57
58 filtered.connect([](const void*, void* ts) {
59 return static_cast<TrackStateProxy*>(ts)->filtered();
60 });
61 filteredCovariance.connect([](const void*, void* ts) {
62 return static_cast<TrackStateProxy*>(ts)->filteredCovariance();
63 });
64
65 smoothed.connect([](const void*, void* ts) {
66 return static_cast<TrackStateProxy*>(ts)->smoothed();
67 });
68 smoothedCovariance.connect([](const void*, void* ts) {
69 return static_cast<TrackStateProxy*>(ts)->smoothedCovariance();
70 });
71
72 predicted.connect([](const void*, void* ts) {
73 return static_cast<TrackStateProxy*>(ts)->predicted();
74 });
75 predictedCovariance.connect([](const void*, void* ts) {
76 return static_cast<TrackStateProxy*>(ts)->predictedCovariance();
77 });
78
79 jacobian.connect([](const void*, void* ts) {
80 return static_cast<TrackStateProxy*>(ts)->jacobian();
81 });
82
83 ACTS_VERBOSE("Invoked GainMatrixSmoother on entry index: " << entryIndex);
84
85 // For the last state: smoothed is filtered - also: switch to next
86 ACTS_VERBOSE("Getting previous track state");
87 auto prev_ts = trajectory.getTrackState(entryIndex);
88
89 prev_ts.shareFrom(TrackStatePropMask::Filtered,
90 TrackStatePropMask::Smoothed);
91
92 // make sure there is more than one track state
93 if (!prev_ts.hasPrevious()) {
94 ACTS_VERBOSE("Only one track state given, smoothing terminates early");
95 return Result<void>::success();
96 }
97
98 ACTS_VERBOSE("Start smoothing from previous track state at index: "
99 << prev_ts.previous());
100
101 // default-constructed error represents success, i.e. an invalid error code
102 std::error_code error;
103 trajectory.applyBackwards(prev_ts.previous(), [&, this](auto ts) {
104 // should have filtered and predicted, this should also include the
105 // covariances.
106 assert(ts.hasFiltered());
107 assert(ts.hasPredicted());
108
109 // previous trackstate should have smoothed and predicted
110 assert(prev_ts.hasSmoothed());
111 assert(prev_ts.hasPredicted());
112 assert(prev_ts.hasJacobian());
113
114 ACTS_VERBOSE("Calculate smoothing matrix:");
115 ACTS_VERBOSE("Filtered covariance:\n" << ts.filteredCovariance());
116 ACTS_VERBOSE("Jacobian:\n" << prev_ts.jacobian());
117
118 // ensure the track state has a smoothed component
119 ts.addComponents(TrackStatePropMask::Smoothed);
120
121 if (auto res = calculate(&ts, &prev_ts, filtered, filteredCovariance,
122 smoothed, predicted, predictedCovariance,
123 smoothedCovariance, jacobian, logger);
124 !res.ok()) {
125 error = res.error();
126 return false;
127 }
128
129 prev_ts = ts;
130 return true; // continue execution
131 });
132
133 return error ? Result<void>::failure(error) : Result<void>::success();
134 }
135

138 Acts::Delegate<TrackStateTraits<MultiTrajectoryTraits::MeasurementSizeMax,
139 false>::Parameters(void*)>;
141 using GetCovariance =
142 Acts::Delegate<TrackStateTraits<MultiTrajectoryTraits::MeasurementSizeMax,
143 false>::Covariance(void*)>;

164 const GetCovariance& smoothedCovariance,
