19template <typename T>
20concept MutableProxyType = requires(T t, HashedString key) {
21 requires !T::ReadOnly;
22
23 {
24 t.template component<int>(key)
25 } -> std::same_as<std::conditional_t<T::ReadOnly, const int&, int&>>;
26};
27
28template <typename T>
29concept ConstProxyType = requires(T t, HashedString key) {
30 requires T::ReadOnly;
31 { t.template component<int>(key) } -> std::same_as<const int&>;
32};
33
34template <typename T>
35concept ProxyType = (MutableProxyType<T> || ConstProxyType<T>) && requires {
36 typename T::ConstProxyType;
37
38 requires ConstProxyType<typename T::ConstProxyType>;
39};
40
41} // namespace Acts::detail

67template <typename T, bool ReadOnly>
68struct ProxyAccessorBase {
70 HashedString key;
71
74 explicit constexpr ProxyAccessorBase(HashedString _key) : key{_key} {}
75
78 explicit constexpr ProxyAccessorBase(const std::string& _key)
79 : key{hashStringDynamic(_key)} {}
80
85 template <detail::MutableProxyType proxy_t>
86 T& operator()(proxy_t proxy) const
87 requires(!ReadOnly)
88 {
89 static_assert(!proxy_t::ReadOnly,
90 "Cannot get mutable ref for const track proxy");
91 return proxy.template component<T>(key);
92 }
93
98 template <detail::ProxyType proxy_t>
99 const T& operator()(proxy_t proxy) const
100 requires(ReadOnly)
101 {
102 if constexpr (proxy_t::ReadOnly) {
103 return proxy.template component<T>(key);
104
105 } else {
106 using const_proxy_t = typename proxy_t::ConstProxyType;
107 const_proxy_t cproxy{proxy};
108 return cproxy.template component<T>(key);
109 }
110 }
111
116 template <detail::ProxyType proxy_t>
117 bool hasColumn(proxy_t proxy) const {
118 return proxy.container().hasColumn(key);
119 }
120};
