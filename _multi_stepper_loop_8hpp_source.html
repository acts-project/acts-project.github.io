45struct MaxMomentumComponent {
46 template <typename component_range_t>
47 auto operator()(const component_range_t& cmps) const {
48 return std::max_element(cmps.begin(), cmps.end(),
49 [&](const auto& a, const auto& b) {
50 return std::abs(a.state.pars[eFreeQOverP]) >
51 std::abs(b.state.pars[eFreeQOverP]);
52 });
53 }
54};
55
56struct MaxWeightComponent {
57 template <typename component_range_t>
58 auto operator()(const component_range_t& cmps) {
59 return std::max_element(
60 cmps.begin(), cmps.end(),
61 [&](const auto& a, const auto& b) { return a.weight < b.weight; });
62 }
63};

65template <typename component_chooser_t>
66struct SingleComponentReducer {
67 template <typename stepper_state_t>
68 static Vector3 position(const stepper_state_t& s) {
69 return component_chooser_t{}(s.components)
70 ->state.pars.template segment<3>(eFreePos0);
71 }
72
73 template <typename stepper_state_t>
74 static Vector3 direction(const stepper_state_t& s) {
75 return component_chooser_t{}(s.components)
76 ->state.pars.template segment<3>(eFreeDir0);
77 }
78
79 template <typename stepper_state_t>
80 static double qOverP(const stepper_state_t& s) {
81 const auto cmp = component_chooser_t{}(s.components);
82 return cmp->state.pars[eFreeQOverP];
83 }
84
85 template <typename stepper_state_t>
86 static double absoluteMomentum(const stepper_state_t& s) {
87 const auto cmp = component_chooser_t{}(s.components);
88 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]);
89 }
90
91 template <typename stepper_state_t>
92 static Vector3 momentum(const stepper_state_t& s) {
93 const auto cmp = component_chooser_t{}(s.components);
94 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]) *
95 cmp->state.pars.template segment<3>(eFreeDir0);
96 }
97
98 template <typename stepper_state_t>
99 static double charge(const stepper_state_t& s) {
100 const auto cmp = component_chooser_t{}(s.components);
101 return s.particleHypothesis.extractCharge(cmp->state.pars[eFreeQOverP]);
102 }
103
104 template <typename stepper_state_t>
105 static double time(const stepper_state_t& s) {
106 return component_chooser_t{}(s.components)->state.pars[eFreeTime];
107 }
108
109 template <typename stepper_state_t>
110 static FreeVector pars(const stepper_state_t& s) {
111 return component_chooser_t{}(s.components)->state.pars;
112 }
113
114 template <typename stepper_state_t>
115 static FreeVector cov(const stepper_state_t& s) {
116 return component_chooser_t{}(s.components)->state.cov;
117 }

271 void initialize(State& state,
272 const MultiComponentBoundTrackParameters& par) const {
273 if (par.components().empty()) {
274 throw std::invalid_argument(
275 "Cannot construct MultiEigenStepperLoop::State with empty "
276 "multi-component parameters");
277 }
278
279 state.particleHypothesis = par.particleHypothesis();
280
281 const auto surface = par.referenceSurface().getSharedPtr();
282
283 for (auto i = 0ul; i < par.components().size(); ++i) {
284 const auto& [weight, singlePars] = par[i];
285 auto& cmp =
286 state.components.emplace_back(SingleStepper::makeState(state.options),
287 weight, IntersectionStatus::onSurface);
288 SingleStepper::initialize(cmp.state, singlePars);
289 }
290
291 if (std::get<2>(par.components().front())) {
292 state.covTransport = true;
293 }
294 }

406 void reweightComponents(State& state) const {
407 double sumOfWeights = 0.0;
408 for (const auto& cmp : state.components) {
409 sumOfWeights += cmp.weight;
410 }
411 for (auto& cmp : state.components) {
412 cmp.weight /= sumOfWeights;
413 }

530 IntersectionStatus updateSurfaceStatus(
531 State& state, const Surface& surface, std::uint8_t index,
532 Direction navDir, const BoundaryTolerance& boundaryTolerance,
533 double surfaceTolerance, ConstrainedStep::Type stype,
534 const Logger& logger = getDummyLogger()) const {
535 using Status = IntersectionStatus;
536
537 std::array<int, 3> counts = {0, 0, 0};
538
539 for (auto& component : state.components) {
540 component.status = detail::updateSingleSurfaceStatus<SingleStepper>(
541 *this, component.state, surface, index, navDir, boundaryTolerance,
542 surfaceTolerance, stype, logger);
543 ++counts[static_cast<std::size_t>(component.status)];
544 }
545
546 // If at least one component is on a surface, we can remove all missed
547 // components before the step. If not, we must keep them for the case that
548 // all components miss and we need to retarget
549 if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
550 removeMissedComponents(state);
551 reweightComponents(state);
552 }
553
554 ACTS_VERBOSE("Component status wrt "
555 << surface.geometryId() << " at {"
556 << surface.center(state.options.geoContext).transpose()
557 << "}:\t" << [&]() {
558 std::stringstream ss;
559 for (auto& component : state.components) {
560 ss << component.status << "\t";
561 }
562 return ss.str();
563 }());
564
565 // Switch on stepCounter if one or more components reached a surface, but
566 // some are still in progress of reaching the surface
567 if (!state.stepCounterAfterFirstComponentOnSurface &&
568 counts[static_cast<std::size_t>(Status::onSurface)] > 0 &&
569 counts[static_cast<std::size_t>(Status::reachable)] > 0) {
570 state.stepCounterAfterFirstComponentOnSurface = 0;
571 ACTS_VERBOSE("started stepCounterAfterFirstComponentOnSurface");
572 }
573
574 // If there are no components onSurface, but the counter is switched on
575 // (e.g., if the navigator changes the target surface), we need to switch it
576 // off again
577 if (state.stepCounterAfterFirstComponentOnSurface &&
578 counts[static_cast<std::size_t>(Status::onSurface)] == 0) {
579 state.stepCounterAfterFirstComponentOnSurface.reset();
580 ACTS_VERBOSE("switch off stepCounterAfterFirstComponentOnSurface");
581 }
582
583 // This is a 'any_of' criterium. As long as any of the components has a
584 // certain state, this determines the total state (in the order of a
585 // somewhat importance)
586 if (counts[static_cast<std::size_t>(Status::reachable)] > 0) {
587 return Status::reachable;
588 } else if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
589 state.stepCounterAfterFirstComponentOnSurface.reset();
590 return Status::onSurface;
591 } else {
592 return Status::unreachable;
593 }
594 }

607 template <typename object_intersection_t>
608 void updateStepSize(State& state, const object_intersection_t& oIntersection,
609 Direction direction, ConstrainedStep::Type stype) const {
610 const Surface& surface = *oIntersection.object();
611
612 for (auto& component : state.components) {
613 auto intersection = surface.intersect(
614 component.state.options.geoContext,
615 SingleStepper::position(component.state),
616 direction * SingleStepper::direction(component.state),
617 BoundaryTolerance::None())[oIntersection.index()];
618
619 SingleStepper::updateStepSize(component.state, intersection, direction,
620 stype);
621 }
622 }
