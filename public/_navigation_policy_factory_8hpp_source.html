50class NavigationPolicyFactory {
51 private:
53 using factory_type = std::function<std::unique_ptr<INavigationPolicy>(
54 const GeometryContext&, const TrackingVolume&, const Logger&)>;
55
58 explicit NavigationPolicyFactory(std::vector<factory_type>&& factories)
59 : m_factories(std::move(factories)) {}

61 public:

72 template <NavigationPolicyConcept P, typename... Args>
73 requires(std::is_constructible_v<P, const GeometryContext&,
74 const TrackingVolume&, const Logger&,
75 Args...> &&
76 (std::is_copy_constructible_v<Args> && ...))
77 constexpr NavigationPolicyFactory add(Args&&... args) && {
78 auto factory = [=](const GeometryContext& gctx,
79 const TrackingVolume& volume, const Logger& logger) {
80 return std::make_unique<P>(gctx, volume, logger, args...);
81 };
82
83 m_factories.push_back(std::move(factory));
84 return std::move(*this);
85 }
86
95 template <typename Fn, typename... Args>
96 requires(detail::NavigationPolicyIsolatedFactoryConcept<Fn, Args...>)
97 constexpr NavigationPolicyFactory add(Fn&& fn, Args&&... args) && {
98 auto factory = [=](const GeometryContext& gctx,
99 const TrackingVolume& volume, const Logger& logger) {
100 using policy_type = decltype(fn(gctx, volume, logger, args...));
101 return std::make_unique<policy_type>(fn(gctx, volume, logger, args...));
102 };
103
104 m_factories.push_back(std::move(factory));
105 return std::move(*this);
106 }
107
110 std::unique_ptr<NavigationPolicyFactory> asUniquePtr() && {
111 return std::make_unique<NavigationPolicyFactory>(std::move(*this));
112 }
113
120 std::unique_ptr<MultiNavigationPolicy> operator()(
121 const GeometryContext& gctx, const TrackingVolume& volume,
122 const Logger& logger) const {
123 if (m_factories.empty()) {
124 throw std::runtime_error(
125 "No factories registered in the navigation policy factory");
126 }
127
128 std::vector<std::unique_ptr<INavigationPolicy>> policies;
129 policies.reserve(m_factories.size());
130 for (auto& factory : m_factories) {
131 policies.push_back(factory(gctx, volume, logger));
132 }
133
134 return std::make_unique<MultiNavigationPolicy>(std::move(policies));
135 }

142 std::unique_ptr<INavigationPolicy> build(const GeometryContext& gctx,
143 const TrackingVolume& volume,
144 const Logger& logger) const {
145 return operator()(gctx, volume, logger);
146 }
147
148 private:
150 std::vector<factory_type> m_factories;
151};
