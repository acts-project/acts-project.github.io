101 template <typename propagator_state_t, typename stepper_t,
102 typename navigator_t>
103 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
104 const navigator_t& navigator, result_type& result,
105 const Logger& logger) const {
106 auto currentVolume = navigator.currentVolume(state.navigation);
107
108 // The current volume has been assigned by the navigator
109 if (currentVolume && selector(*currentVolume)) {
110 // Create for recording
111 VolumeHit volume_hit;
112 volume_hit.volume = currentVolume;
113 volume_hit.position = stepper.position(state.stepping);
114 volume_hit.direction = stepper.direction(state.stepping);
115 bool save = true;
116 // Check if the Volume ws already encountered
117 for (auto const& res : result.collected) {
118 if (res.volume == volume_hit.volume) {
119 save = false;
120 break;
121 }
122 }
123 // Save if in the result if it does not already exist
124 if (save) {
125 result.collected.push_back(volume_hit);
126 // Screen output
127 ACTS_VERBOSE("Collect volume " << currentVolume->geometryId());
128 }
129 }
130
131 return {};
132 }
133};
