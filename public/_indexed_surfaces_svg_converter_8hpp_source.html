59template <typename surface_container, typename index_grid>
60ProtoIndexedSurfaceGrid convertImpl(const Acts::GeometryContext& gctx,
61 const surface_container& surfaces,
62 const index_grid& indexGrid,
63 const Options& cOptions) {
64 // The return surfaces
65 ProtoSurfaces pSurfaces;
66 pSurfaces.reserve(surfaces.size());
67
68 // Make a local copy of the grid Options, in case we have to estimate
69 // an additional bound
70 GridConverter::Options gridOptions = cOptions.gridOptions;
71 Acts::Extent constrain;
72
73 // Estimate the radial extension
74 // - for 1D phi
75 // - for 2D z-phi or phi-z
76 bool estimateR = (index_grid::grid_type::DIM == 1 &&
77 indexGrid.casts[0u] == Acts::AxisDirection::AxisPhi) ||
78 (index_grid::grid_type::DIM == 2 &&
79 (indexGrid.casts[0u] == Acts::AxisDirection::AxisPhi ||
80 indexGrid.casts[1u] == Acts::AxisDirection::AxisPhi));
81
82 for (auto [is, s] : Acts::enumerate(surfaces)) {
83 // Create the surface converter options
84 SurfaceConverter::Options sOptions;
85 // Try to see if you have predefined surface styles
86 auto sfIter = cOptions.surfaceStyles.find(s->geometryId());
87 if (sfIter != cOptions.surfaceStyles.end()) {
88 sOptions.style = *sfIter;
89 }
90 auto pSurface = SurfaceConverter::convert(gctx, *s, sOptions);
91 // Run the r estimation
92 if (estimateR) {
93 auto sExtent = s->polyhedronRepresentation(gctx, 4u).extent();
94 if constexpr (index_grid::grid_type::DIM == 2u) {
95 pSurface._radii[0u] =
96 static_cast<float>(sExtent.medium(Acts::AxisDirection::AxisR));
97 }
98 constrain.extend(sExtent, {Acts::AxisDirection::AxisR});
99 }
100 // Add center info
101 std::string centerInfo = " - center = (";
102 const Acts::Vector3& center = s->center(gctx);
103 centerInfo +=
104 std::to_string(Acts::VectorHelpers::cast(center, indexGrid.casts[0u]));
105 if (indexGrid.casts.size() > 1u) {
106 centerInfo += ", ";
107 centerInfo += std::to_string(
108 Acts::VectorHelpers::cast(center, indexGrid.casts[1u]));
109 centerInfo += ")";
110 }
111 pSurface._aux_info["center"] = {centerInfo};
112 // Add the center info
113 pSurfaces.push_back(pSurface);
114 }
115
116 // Adjust the grid options
117 if constexpr (index_grid::grid_type::DIM == 1u) {
118 if (indexGrid.casts[0u] == Acts::AxisDirection::AxisPhi) {
119 auto estRangeR = constrain.range(Acts::AxisDirection::AxisR);
120 std::array<double, 2u> rRange = {estRangeR.min(), estRangeR.max()};
121 gridOptions.optionalBound = {rRange, Acts::AxisDirection::AxisR};
122 }
123 }
124
125 // Create the grid
126 ProtoGrid pGrid =
127 GridConverter::convert(indexGrid.grid, indexGrid.casts, gridOptions);
128
129 auto axes = indexGrid.grid.axes();
130
131 // Specify the highlight indices
132 ProtoAssociations highlightIndices;
133
134 // 1D connections
135 if constexpr (index_grid::grid_type::DIM == 1u) {
136 const auto& binEdges = axes[0u]->getBinEdges();
137 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
138 typename index_grid::grid_type::index_t lbin;
139 lbin[0u] = ib0;
140 highlightIndices.push_back(indexGrid.grid.atLocalBins(lbin));
141 // Register the bin naming
142 std::string binInfo =
143 std::string("- bin : [") + std::to_string(ib0) + std::string("]");
144 double binCenter = 0.5 * (binEdges[ib0] + binEdges[ib0 - 1u]);
145 binInfo += "\n - center : (" + std::to_string(binCenter) + ")";
146 pGrid._bin_ids.push_back(binInfo);
147 }
148 }
149 // 2D connections
150 if constexpr (index_grid::grid_type::DIM == 2u) {
151 const auto& binEdges0 = axes[0u]->getBinEdges();
152 const auto& binEdges1 = axes[1u]->getBinEdges();
153 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
154 for (unsigned int ib1 = 1u; ib1 <= axes[1u]->getNBins(); ++ib1) {
155 typename index_grid::grid_type::index_t lbin;
156 lbin[0u] = ib0;
157 lbin[1u] = ib1;
158 highlightIndices.push_back(indexGrid.grid.atLocalBins(lbin));
159 // Register the bin naming
160 std::string binInfo = std::string("- bin : [") + std::to_string(ib0) +
161 std::string(", ") + std::to_string(ib1) +
162 std::string("]");
163 double binCenter0 = 0.5 * (binEdges0[ib0] + binEdges0[ib0 - 1u]);
164 double binCenter1 = 0.5 * (binEdges1[ib1] + binEdges1[ib1 - 1u]);
165 binInfo += "\n - center : (" + std::to_string(binCenter0) + ", " +
166 std::to_string(binCenter1) + ")";
167 pGrid._bin_ids.push_back(binInfo);
168 if (estimateR) {
169 pGrid._reference_r =
170 static_cast<float>(constrain.medium(Acts::AxisDirection::AxisR));
171 }
172 }
173 }
174 }
175 return {pSurfaces, pGrid, highlightIndices};
176}

189static inline actsvg::svg::object xy(const ProtoIndexedSurfaceGrid& pIndexGrid,
190 const std::string& identification) {
191 auto [pSurfaces, pGrid, pIndices] = pIndexGrid;
192
193 // Create the master object
194 actsvg::svg::object xyIndexedGrid;
195 xyIndexedGrid._id = identification;
196 xyIndexedGrid._tag = "g";
197
198 // The surfaces
199 std::vector<actsvg::svg::object> sObs;
200 for (const auto& s : pSurfaces) {
201 if (pGrid._type == ProtoGrid::e_z_phi) {
202 sObs.push_back(zrphi(s, s._name));
203 } else {
204 sObs.push_back(xy(s, s._name));
205 }
206 }
207
208 // The grid as provided
209 auto gOb =
210 actsvg::display::grid(identification + std::string("_grid"), pGrid);
211
212 // The connectors
213 actsvg::connectors::connect_action(gOb._sub_objects, sObs, pIndices);
214
215 // Add them all
216 xyIndexedGrid.add_objects(sObs);
217
218 auto xmax = xyIndexedGrid._x_range[1u];
219 // The association info boxes
220 for (auto [ig, gTile] : Acts::enumerate(gOb._sub_objects)) {
221 // Target surface text
222 std::vector<std::string> binText;
223 binText.push_back("Source:");
224 binText.push_back(pGrid._bin_ids[ig]);
225 binText.push_back("Target:");
226 for (const auto [is, sis] : Acts::enumerate(pIndices[ig])) {
227 const auto& ps = pSurfaces[sis];
228 std::string oInfo = std::string("- object: ") + std::to_string(sis);
229 if (ps._aux_info.contains("center")) {
230 for (const auto& ci : ps._aux_info.at("center")) {
231 oInfo += ci;
232 }
233 }
234 binText.push_back(oInfo);
235 }
236 // Make the connected text
237 std::string cTextId =
238 identification + std::string("_ct_") + std::to_string(ig);
239 auto cText = actsvg::draw::connected_text(
240 cTextId, {static_cast<actsvg::scalar>(1.1 * xmax), 0}, binText,
241 actsvg::style::font{}, actsvg::style::transform{}, gTile);
242 xyIndexedGrid.add_object(cText);
243 }
244 xyIndexedGrid.add_object(gOb);
245 // Return the grid
246 return xyIndexedGrid;
247}

261 return xy(pIndexGrid, identification);
