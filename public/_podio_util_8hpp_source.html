58template <typename T>
59decltype(auto) getDataMutable(T&& object) {
60 if constexpr (podio::version::build_version.major >= 1) {
61 return std::forward<T>(object).getData();
62 } else {
63 return std::forward<T>(object).data();
64 }
65}
