78 explicit KDTree(vector_t &&d) : m_elems(d) {
79 // To start out, we need to check whether we need to construct a leaf node
80 // or an internal node. We create a leaf only if we have at most as many
81 // elements as the number of elements that can fit into a leaf node.
82 // Hopefully most invocations of this constructor will have more than a few
83 // elements!
84 //
85 // One interesting thing to note is that all of the nodes in the k-d tree
86 // have a range in the element vector of the outermost node. They simply
87 // make in-place changes to this array, and they hold no memory of their
88 // own.
89 m_root = std::make_unique<KDTreeNode>(m_elems.begin(), m_elems.end(),
90 m_elems.size() > LeafSize
91 ? KDTreeNode::NodeType::Internal
92 : KDTreeNode::NodeType::Leaf,
93 0UL);
94 }

273 static Scalar nextRepresentable(Scalar v) {
274 // I'm not super happy with this bit of code, but since 1D ranges are
275 // semi-open, we can't simply incorporate values by setting the maximum to
276 // them. Instead, what we need to do is get the next representable value.
277 // For integer values, this means adding one. For floating point types, we
278 // rely on the nextafter method to get the smallest possible value that is
279 // larger than the one we requested.
280 if constexpr (std::is_integral_v<Scalar>) {
281 return v + 1;
282 } else if constexpr (std::is_floating_point_v<Scalar>) {
283 return std::nextafter(v, std::numeric_limits<Scalar>::max());
284 }
285 }
286
287 static range_t boundingBox(iterator_t b, iterator_t e) {
288 // Firstly, we find the minimum and maximum value in each dimension to
289 // construct a bounding box around this node's values.
290 std::array<Scalar, Dims> min_v{}, max_v{};
291
292 for (std::size_t i = 0; i < Dims; ++i) {
293 min_v[i] = std::numeric_limits<Scalar>::max();
294 max_v[i] = std::numeric_limits<Scalar>::lowest();
295 }
296
297 for (iterator_t i = b; i != e; ++i) {
298 for (std::size_t j = 0; j < Dims; ++j) {
299 min_v[j] = std::min(min_v[j], i->first[j]);
300 max_v[j] = std::max(max_v[j], i->first[j]);
301 }
302 }
303
304 // Then, we construct a k-dimensional range from the given minima and
305 // maxima, which again is just a bounding box.
306 range_t r;
307
308 for (std::size_t j = 0; j < Dims; ++j) {
309 r[j] = Range1D<Scalar>{min_v[j], nextRepresentable(max_v[j])};
310 }
311
312 return r;
313 }

321 class KDTreeNode {
322 public:
324 enum class NodeType { Internal, Leaf };
325
332 KDTreeNode(iterator_t _b, iterator_t _e, NodeType _t, std::size_t _d)
333 : m_type(_t),
334 m_begin_it(_b),
335 m_end_it(_e),
336 m_range(boundingBox(m_begin_it, m_end_it)) {
337 if (m_type == NodeType::Internal) {
338 // This constant determines the maximum number of elements where we
339 // still
340 // calculate the exact median of the values for the purposes of
341 // splitting. In general, the closer the pivot value is to the true
342 // median, the more balanced the tree will be. However, calculating the
343 // median exactly is an O(n log n) operation, while approximating it is
344 // an O(1) time.
345 constexpr std::size_t max_exact_median = 128;
346
347 iterator_t pivot;
348
349 // Next, we need to determine the pivot point of this node, that is to
350 // say the point in the selected pivot dimension along which point we
351 // will split the range. To do this, we check how large the set of
352 // elements is. If it is sufficiently small, we use the median.
353 // Otherwise we use the mean.
354 if (size() > max_exact_median) {
355 // In this case, we have a lot of elements, and sorting the range to
356 // find the true median might be too expensive. Therefore, we will
357 // just use the middle value between the minimum and maximum. This is
358 // not nearly as accurate as using the median, but it's a nice cheat.
359 Scalar mid = static_cast<Scalar>(0.5) *
360 (m_range[_d].max() + m_range[_d].min());
361
362 pivot = std::partition(m_begin_it, m_end_it, [=](const pair_t &i) {
363 return i.first[_d] < mid;
364 });
365 } else {
366 // If the number of elements is fairly small, we will just calculate
367 // the median exactly. We do this by finding the values in the
368 // dimension, sorting it, and then taking the middle one.
369 std::sort(m_begin_it, m_end_it,
370 [_d](const typename iterator_t::value_type &a,
371 const typename iterator_t::value_type &b) {
372 return a.first[_d] < b.first[_d];
373 });
374
375 pivot = m_begin_it + (std::distance(m_begin_it, m_end_it) / 2);
376 }
377
378 // This should never really happen, but in very select cases where there
379 // are a lot of equal values in the range, the pivot can end up all the
380 // way at the end of the array and we end up in an infinite loop. We
381 // check for pivot points which would not split the range, and fix them
382 // if they occur.
383 if (pivot == m_begin_it || pivot == std::prev(m_end_it)) {
384 pivot = std::next(m_begin_it, LeafSize);
385 }
386
387 // Calculate the number of elements on the left-hand side, as well as
388 // the right-hand side. We do this by calculating the difference from
389 // the begin and end of the array to the pivot point.
390 std::size_t lhs_size = std::distance(m_begin_it, pivot);
391 std::size_t rhs_size = std::distance(pivot, m_end_it);
392
393 // Next, we check whether the left-hand node should be another internal
394 // node or a leaf node, and we construct the node recursively.
395 m_lhs = std::make_unique<KDTreeNode>(
396 m_begin_it, pivot,
397 lhs_size > LeafSize ? NodeType::Internal : NodeType::Leaf,
398 (_d + 1) % Dims);
399
400 // Same on the right hand side.
401 m_rhs = std::make_unique<KDTreeNode>(
402 pivot, m_end_it,
403 rhs_size > LeafSize ? NodeType::Internal : NodeType::Leaf,
404 (_d + 1) % Dims);
405 }
406 }

417 template <typename Callable>
418 void rangeSearchMapDiscard(const range_t &r, Callable &&f) const {
419 // Determine whether the range completely covers the bounding box of
420 // this leaf node. If it is, we can copy all values without having to
421 // check for them being inside the range again.
422 bool contained = r >= m_range;
423
424 if (m_type == NodeType::Internal) {
425 // Firstly, we can check if the range completely contains the bounding
426 // box of this node. If that is the case, we know for certain that any
427 // value contained below this node should end up in the output, and we
428 // can stop recursively looking for them.
429 if (contained) {
430 // We can also pre-allocate space for the number of elements, since we
431 // are inserting all of them anyway.
432 for (iterator_t i = m_begin_it; i != m_end_it; ++i) {
433 f(i->first, i->second);
434 }
435
436 return;
437 }
438
439 assert(m_lhs && m_rhs && "Did not find lhs and rhs");
440
441 // If we have a left-hand node (which we should!), then we check if
442 // there is any overlap between the target range and the bounding box of
443 // the left-hand node. If there is, we recursively search in that node.
444 if (m_lhs->range() && r) {
445 m_lhs->rangeSearchMapDiscard(r, std::forward<Callable>(f));
446 }
447
448 // Then, we perform exactly the same procedure for the right hand side.
449 if (m_rhs->range() && r) {
450 m_rhs->rangeSearchMapDiscard(r, std::forward<Callable>(f));
451 }
452 } else {
453 // Iterate over all the elements in this leaf node. This should be a
454 // relatively small number (the LeafSize template parameter).
455 for (iterator_t i = m_begin_it; i != m_end_it; ++i) {
456 // We need to check whether the element is actually inside the range.
457 // In case this node's bounding box is fully contained within the
458 // range, we don't actually need to check this.
459 if (contained || r.contains(i->first)) {
460 f(i->first, i->second);
461 }
462 }
463 }
464 }
465
473 std::size_t size() const { return std::distance(m_begin_it, m_end_it); }
