35template <std::ranges::sized_range index_range_t>
36inline static bool checkSubspaceIndices(const index_range_t& indexRange,
37 std::size_t fullSize,
38 std::size_t subspaceSize) {
39 if (subspaceSize > fullSize) {
40 return false;
41 }
42 if (static_cast<std::size_t>(indexRange.size()) != subspaceSize) {
43 return false;
44 }
45 for (auto it = indexRange.begin(); it != indexRange.end();) {
46 auto index = *it;
47 if (index >= fullSize) {
48 return false;
49 }
50 ++it;
51 if (std::find(it, indexRange.end(), index) != indexRange.end()) {
52 return false;
53 }
54 }
55 return true;
56}

65template <std::size_t FullSize>
66inline static SerializedSubspaceIndices serializeSubspaceIndices(
67 const SubspaceIndices<FullSize>& indices)
68 requires(FullSize <= 8)
69{
70 SerializedSubspaceIndices result = 0;
71 for (std::size_t i = 0; i < FullSize; ++i) {
72 result |= static_cast<SerializedSubspaceIndices>(indices[i] & 0xFF)
73 << (i * 8);
74 }
75 return result;
76}

85template <std::size_t FullSize>
86inline static SubspaceIndices<FullSize> deserializeSubspaceIndices(
87 SerializedSubspaceIndices serialized)
88 requires(FullSize <= 8)
89{
90 SubspaceIndices<FullSize> result;
91 for (std::size_t i = 0; i < FullSize; ++i) {
92 result[i] = static_cast<std::uint8_t>((serialized >> (i * 8)) & 0xFF);
93 }
94 return result;
95}

104template <typename Derived, std::size_t FullSize>
105class SubspaceHelperBase {
106 public:
107 static constexpr std::size_t kFullSize = FullSize;
108
109 using FullSquareMatrix = ActsSquareMatrix<kFullSize>;
110
111 using size_type = std::size_t;
112
113 bool empty() const { return self().empty(); }
114 std::size_t size() const { return self().size(); }
115
116 auto operator[](std::size_t i) const { return self()[i]; }
117
118 auto begin() const { return self().begin(); }
119 auto end() const { return self().end(); }
120
121 bool contains(std::uint8_t index) const {
122 return std::find(begin(), end(), index) != end();
123 }
124 std::size_t indexOf(std::uint8_t index) const {
125 auto it = std::find(begin(), end(), index);
126 return it != end() ? std::distance(begin(), it) : kFullSize;
127 }
128
129 template <typename EigenDerived>
130 ActsVector<kFullSize> expandVector(
131 const Eigen::DenseBase<EigenDerived>& vector) const {
132 ActsVector<kFullSize> result = ActsVector<kFullSize>::Zero();
133 for (auto [i, index] : enumerate(*this)) {
134 result(index) = vector(i);
135 }
136 return result;
137 }
138
139 template <typename EigenDerived>
140 FullSquareMatrix expandMatrix(
141 const Eigen::DenseBase<EigenDerived>& matrix) const {
142 FullSquareMatrix result = FullSquareMatrix::Zero();
143 for (auto [i, indexI] : enumerate(*this)) {
144 for (auto [j, indexJ] : enumerate(*this)) {
145 result(indexI, indexJ) = matrix(i, j);
146 }
147 }
148 return result;
149 }
150
151 FullSquareMatrix fullProjector() const {
152 FullSquareMatrix result = FullSquareMatrix::Zero();
153 for (auto [i, index] : enumerate(*this)) {
154 result(i, index) = 1;
155 }
156 return result;
157 }
158
159 FullSquareMatrix fullExpander() const {
160 FullSquareMatrix result = FullSquareMatrix::Zero();
161 for (auto [i, index] : enumerate(*this)) {
162 result(index, i) = 1;
163 }
164 return result;
165 }
166
167 ProjectorBitset projectorBitset() const
168 requires(kFullSize <= 8)
169 {
170 return matrixToBitset(fullProjector()).to_ullong();
171 }
172
173 private:
174 const Derived& self() const { return static_cast<const Derived&>(*this); }
175};

183template <std::size_t FullSize, typename index_t = std::uint8_t>
184class VariableSubspaceHelper
185 : public detail::SubspaceHelperBase<
186 VariableSubspaceHelper<FullSize, index_t>, FullSize> {
187 public:
189 static constexpr std::size_t kFullSize = FullSize;
190
192 using IndexType = index_t;
194 using Container = boost::container::static_vector<IndexType, FullSize>;
195
200 template <std::ranges::sized_range other_index_range_t>
201 explicit VariableSubspaceHelper(const other_index_range_t& indices) {
202 assert(checkSubspaceIndices(indices, kFullSize, indices.size()) &&
203 "Invalid indices");
204 m_indices.resize(indices.size());
205 std::transform(indices.begin(), indices.end(), m_indices.begin(),
206 [](auto index) { return static_cast<IndexType>(index); });
207 }
208
211 bool empty() const { return m_indices.empty(); }
214 std::size_t size() const { return m_indices.size(); }
217 const Container& indices() const { return m_indices; }
218
222 IndexType operator[](std::size_t i) const { return m_indices[i]; }
223
226 auto begin() const { return m_indices.begin(); }
229 auto end() const { return m_indices.end(); }
230
231 private:
232 Container m_indices;
233};

334 template <std::size_t K, typename Derived>
335 ApplyLeftResult<K> applyLeftOf(
336 const Eigen::DenseBase<Derived>& matrix) const {
337 assert(matrix.rows() == kFullSize && "Invalid matrix size");
338 ApplyLeftResult<K> result = ApplyLeftResult<K>::Zero();
339 for (auto [i, indexI] : enumerate(*this)) {
340 for (std::size_t j = 0; j < K; ++j) {
341 result(i, j) = matrix(indexI, j);
342 }
343 }
344 return result;
345 }

352 template <std::size_t N, typename Derived>
353 ApplyRightResult<N> applyRightOf(
354 const Eigen::DenseBase<Derived>& matrix) const {
355 assert(matrix.cols() == kSubspaceSize && "Invalid matrix size");
356 ApplyRightResult<N> result = ApplyRightResult<N>::Zero();
357 for (std::size_t i = 0; i < N; ++i) {
358 for (auto [j, indexJ] : enumerate(*this)) {
359 result(i, j) = matrix(i, indexJ);
360 }
361 }
362 return result;

370 Vector projectVector(const Eigen::DenseBase<Derived>& fullVector) const {
371 assert(fullVector.size() == kFullSize && "Invalid full vector size");
372 Vector result = Vector::Zero();
373 for (auto [i, index] : enumerate(*this)) {
374 result(i) = fullVector(index);
375 }
376 return result;

384 SquareMatrix projectMatrix(
385 const Eigen::DenseBase<Derived>& fullMatrix) const {
386 assert(fullMatrix.rows() == kFullSize && fullMatrix.cols() == kFullSize &&
387 "Invalid full matrix size");
388 SquareMatrix result = SquareMatrix::Zero();
389 for (auto [i, indexI] : enumerate(*this)) {
390 for (auto [j, indexJ] : enumerate(*this)) {
391 result(i, j) = fullMatrix(indexI, indexJ);

422template <std::size_t kFullSize, typename Derived>
423SubspaceIndices<kFullSize> projectorToSubspaceIndices(
424 const Eigen::DenseBase<Derived>& projector) {
425 auto rows = static_cast<std::size_t>(projector.rows());
426 auto cols = static_cast<std::size_t>(projector.cols());
427 assert(cols == kFullSize && rows <= kFullSize && "Invalid projector size");
428 SubspaceIndices<kFullSize> result;
429 result.fill(kFullSize);
430 for (std::size_t i = 0; i < rows; ++i) {
431 for (std::size_t j = 0; j < cols; ++j) {
432 assert((projector(i, j) == 0 || projector(i, j) == 1) &&
433 "Invalid projector value");
434 if (projector(i, j) == 1) {
435 result[i] = j;
436 }
437 }
438 }
439 return result;
440}
