68 do { \

169inline std::string_view levelName(Level level) {
170 switch (level) {
171 case Level::VERBOSE:
172 return "VERBOSE";
173 case Level::DEBUG:
174 return "DEBUG";
175 case Level::INFO:
176 return "INFO";
177 case Level::WARNING:
178 return "WARNING";
179 case Level::ERROR:
180 return "ERROR";
181 case Level::FATAL:
182 return "FATAL";
183 case Level::MAX:
184 return "MAX";
185 default:
186 throw std::invalid_argument{"Unknown level"};
187 }
188}

490class TimedOutputDecorator final : public OutputDecorator {
491 public:
496 explicit TimedOutputDecorator(std::unique_ptr<OutputPrintPolicy> wrappee,
497 const std::string& format = "%X")
498 : OutputDecorator(std::move(wrappee)), m_format(format) {}
499
507 void flush(const Level& lvl, const std::string& input) override {
508 std::ostringstream os;
509 os << std::left << std::setw(12) << now() << input;
510 OutputDecorator::flush(lvl, os.str());
511 }
512
516 std::unique_ptr<OutputPrintPolicy> clone(
517 const std::string& name) const override {
518 return std::make_unique<TimedOutputDecorator>(m_wrappee->clone(name),
519 m_format);
520 }
521
522 private:
526 std::string now() const {
527 char buffer[20];
528 time_t t{};
529 std::time(&t);
530 struct tm tbuf{};
531 std::strftime(buffer, sizeof(buffer), m_format.c_str(),
532 localtime_r(&t, &tbuf));
533 return buffer;
534 }
535
537 std::string m_format;

622class DefaultPrintPolicy final : public OutputPrintPolicy {
623 public:
629 explicit DefaultPrintPolicy(std::ostream* out = &std::cout) : m_out(out) {}
630
635 void flush(const Level& lvl, const std::string& input) final {
636 // Mutex to serialize access to std::cout
637 static std::mutex s_stdoutMutex;
638 std::unique_lock lock{s_stdoutMutex,
639 std::defer_lock}; // prep empty, we might not need it
640
641 if (m_out == &std::cout) {
642 lock.lock(); // lock only if we are printing to std::cout
643 }
644
645 (*m_out) << input << std::endl;
646 if (lvl >= getFailureThreshold()) {
647 throw ThresholdFailure(
648 "Previous debug message exceeds the "
649 "ACTS_LOG_FAILURE_THRESHOLD=" +
650 std::string{levelName(getFailureThreshold())} +
651 " configuration, bailing out. See "
652 "https://acts.readthedocs.io/en/latest/core/misc/"
653 "logging.html#logging-thresholds");
654 }
655 }
656
663 const std::string& name() const override {
664 throw std::runtime_error{
665 "Default print policy doesn't have a name. Is there no named output in "
666 "the decorator chain?"};
667 };
668
671 std::unique_ptr<OutputPrintPolicy> clone(
672 const std::string& /*name*/) const override {
673 return std::make_unique<DefaultPrintPolicy>(m_out);
674 };
675
676 private:
678 std::ostream* m_out;
679};

691class Logger {
692 public:
697 Logger(std::unique_ptr<Logging::OutputPrintPolicy> pPrint,
698 std::unique_ptr<Logging::OutputFilterPolicy> pFilter)
699 : m_printPolicy(std::move(pPrint)), m_filterPolicy(std::move(pFilter)) {}
700
706 bool doPrint(const Logging::Level& lvl) const {
707 return m_filterPolicy->doPrint(lvl);

709
714 void log(const Logging::Level& lvl, const std::string& input) const {
715 if (doPrint(lvl)) {

734 Logging::Level level() const { return m_filterPolicy->level(); }

744 std::unique_ptr<Logger> clone(
745 const std::optional<std::string>& _name = std::nullopt,
746 const std::optional<Logging::Level>& _level = std::nullopt) const {
747 return std::make_unique<Logger>(

751
