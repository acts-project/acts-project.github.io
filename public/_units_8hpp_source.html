103// Shortcuts for commonly used area and volume units. This intentionally
104// contains not all possible combinations to avoid cluttering the namespace.
105// Missing area or volume units can always be defined on the fly using the
106// existing length units e.g. 1fm³ -> 1fm * 1fm * 1fm
107// Area, native unit mm²
109constexpr double mm2 = mm * mm;
111constexpr double cm2 = cm * cm;
113constexpr double m2 = m * m;
114// Volume, native unit mm³
116constexpr double mm3 = mm * mm * mm;
118constexpr double cm3 = cm * cm * cm;
120constexpr double m3 = m * m * m;
121// Time, native unit mm = [speed-of-light * time] = mm/s * s
123constexpr double s = 299792458000.0; // = 299792458.0 * (m / 1.0) * 1.0;
