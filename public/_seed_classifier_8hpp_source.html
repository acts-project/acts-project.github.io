49 std::vector<std::size_t> seedSelection(
50 std::vector<std::vector<std::size_t>>& clusters,
51 std::vector<std::vector<float>>& outputTensor,
52 float minSeedScore = 0.1) const {
53 std::vector<std::size_t> goodSeeds;
54 // Loop over all the cluster and only keep the seed with the highest score
55 // in each cluster
56 for (const auto& cluster : clusters) {
57 std::size_t bestseedID = 0;
58 float bestSeedScore = 0;
59 for (const auto& seed : cluster) {
60 if (outputTensor[seed][0] > bestSeedScore) {
61 bestSeedScore = outputTensor[seed][0];
62 bestseedID = seed;
63 }
64 }
65 if (bestSeedScore >= minSeedScore) {
66 goodSeeds.push_back(bestseedID);
67 }
68 }
69 return goodSeeds;
70 }
71
78 std::vector<std::size_t> solveAmbiguity(
79 std::vector<std::vector<std::size_t>>& clusters,
80 NetworkBatchInput& networkInput, float minSeedScore = 0.1) const {
81 std::vector<std::vector<float>> outputTensor = inferScores(networkInput);
82 std::vector<std::size_t> goodSeeds =
83 seedSelection(clusters, outputTensor, minSeedScore);
84 return goodSeeds;
85 }
