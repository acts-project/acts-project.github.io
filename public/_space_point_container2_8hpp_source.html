43enum class SpacePointColumns : std::uint32_t {
44 None = 0,
45
46 SourceLinks = 1 << 0,
47 X = 1 << 1,
48 Y = 1 << 2,
49 Z = 1 << 3,
50 R = 1 << 4,
51 Phi = 1 << 5,
52 Time = 1 << 6,
53 VarianceZ = 1 << 7,
54 VarianceR = 1 << 8,
55 TopStripVector = 1 << 9,
56 BottomStripVector = 1 << 10,
57 StripCenterDistance = 1 << 11,
58 TopStripCenter = 1 << 12,
59 CopyFromIndex = 1 << 13,
60
61 // packed columns for performance reasons
62 XY = 1 << 14,
63 ZR = 1 << 15,
64 XYZ = 1 << 16,
65 XYZR = 1 << 17,
66 VarianceZR = 1 << 18,
67
69 Strip =
70 TopStripVector | BottomStripVector | StripCenterDistance | TopStripCenter,
71};
72
73ACTS_DEFINE_ENUM_BITWISE_OPERATORS(SpacePointColumns);
