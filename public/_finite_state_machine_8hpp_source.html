99template <typename Derived, typename... States>
100class FiniteStateMachine {
101 public:
104 struct Terminated {
106 constexpr static std::string_view name = "Terminated";
107 };
108
111 using StateVariant = std::variant<Terminated, States...>;
112
113 protected:
115 using fsm_base = FiniteStateMachine<Derived, States...>;
116
118 using event_return = std::optional<StateVariant>;
119
120 public:
123 FiniteStateMachine()
124 : m_state(typename std::tuple_element_t<0, std::tuple<States...>>{}) {}
125
128 explicit FiniteStateMachine(StateVariant state) : m_state(std::move(state)) {}
129
132 const StateVariant& getState() const noexcept { return m_state; }
133
134 public:
141 template <typename State, typename... Args>
142 void setState(State state, Args&&... args) {
143 Derived& child = static_cast<Derived&>(*this);
144
145 // call on exit function
146 std::visit([&](auto& s) { child.on_exit(s, std::forward<Args>(args)...); },
147 m_state);
148
149 m_state = std::move(state);
150
151 // call on enter function, the type is known from the template argument.
152 child.on_enter(std::get<State>(m_state), std::forward<Args>(args)...);
153 }
154
158 template <typename S>
159 bool is(const S& /*state*/) const noexcept {
160 return is<S>();
161 }
162
167 template <typename S>
168 bool is() const noexcept {
169 if (std::get_if<S>(&m_state)) {
170 return true;
171 }
172 return false;
173 }
174
177 bool terminated() const noexcept { return is<Terminated>(); }
178
179 protected:
188 template <typename Event, typename... Args>
189 event_return process_event(Event&& event, Args&&... args) {
190 Derived& child = static_cast<Derived&>(*this);
191
192 child.on_process(event);
193
194 auto new_state = std::visit(
195 [&](auto& s) -> std::optional<StateVariant> {
196 auto s2 = child.on_event(s, std::forward<Event>(event),
197 std::forward<Args>(args)...);
198
199 if (s2) {
200 std::visit([&](auto& s2_) { child.on_process(s, event, s2_); },
201 *s2);
202 } else {
203 child.on_process(s, event);
204 }
205 return s2;
206 },
207 m_state);
208 return new_state;
209 }
210
211 public:
218 template <typename Event, typename... Args>
219 void dispatch(Event&& event, Args&&... args) {
220 auto new_state = process_event(std::forward<Event>(event), args...);
221 if (new_state) {
222 std::visit(
223 [&](auto& s) { setState(std::move(s), std::forward<Args>(args)...); },
224 *new_state);
225 }
226 }
227
228 private:
229 StateVariant m_state;
230};
