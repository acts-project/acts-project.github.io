25inline nlohmann::json encodeIdentifier(const GeometryIdentifier& geoId,
26 bool compact = false) {
27 nlohmann::json encoded;
28 if (compact) {
29 // store the raw value directly
30 encoded = geoId.value();
31 return encoded;
32 }
33
34 // only store non-zero identifiers
35 if (geoId.volume() != 0u) {
36 encoded["volume"] = geoId.volume();
37 }
38 if (geoId.boundary() != 0u) {
39 encoded["boundary"] = geoId.boundary();
40 }
41 if (geoId.layer() != 0u) {
42 encoded["layer"] = geoId.layer();
43 }
44 if (geoId.approach() != 0u) {
45 encoded["approach"] = geoId.approach();
46 }
47 if (geoId.sensitive() != 0u) {
48 encoded["sensitive"] = geoId.sensitive();
49 }
50 if (geoId.extra() != 0u) {
51 encoded["extra"] = geoId.extra();
52 }
53 return encoded;
54}

59inline GeometryIdentifier decodeIdentifier(const nlohmann::json& encoded) {
60 return GeometryIdentifier()
61 .withVolume(encoded.value("volume", GeometryIdentifier::Value{0u}))
62 .withBoundary(encoded.value("boundary", GeometryIdentifier::Value{0u}))
63 .withLayer(encoded.value("layer", GeometryIdentifier::Value{0u}))
64 .withApproach(encoded.value("approach", GeometryIdentifier::Value{0u}))
65 .withSensitive(encoded.value("sensitive", GeometryIdentifier::Value{0u}))
66 .withExtra(encoded.value("extra", GeometryIdentifier::Value{0u}));
67}
