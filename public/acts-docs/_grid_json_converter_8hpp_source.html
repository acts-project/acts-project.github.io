143template <typename grid_type>
144nlohmann::json toJson(const grid_type& grid) {
145 nlohmann::json jGrid;
146
147 auto axes = grid.axes();
148 nlohmann::json jAxes;
149 for (unsigned int ia = 0u; ia < grid_type::DIM; ++ia) {
150 auto jAxis = AxisJsonConverter::toJson(*axes[ia]);
151 jAxes.push_back(jAxis);
152 }
153 jGrid["axes"] = jAxes;
154
155 nlohmann::json jData;
156 // 1D connections
157 if constexpr (grid_type::DIM == 1u) {
158 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
159 typename grid_type::index_t lbin;
160 lbin[0u] = ib0;
161 jData.push_back(std::tie(lbin, grid.atLocalBins(lbin)));
162 }
163 }
164 // 2D connections
165 if constexpr (grid_type::DIM == 2u) {
166 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
167 for (unsigned int ib1 = 1u; ib1 <= axes[1u]->getNBins(); ++ib1) {
168 typename grid_type::index_t lbin;
169 lbin[0u] = ib0;
170 lbin[1u] = ib1;
171 jData.push_back(std::tie(lbin, grid.atLocalBins(lbin)));
172 }
173 }
174 }
175 jGrid["data"] = jData;
176
177 return jGrid;
178}

190template <typename grid_type>
191nlohmann::json toJsonDetray(const grid_type& grid, bool swapAxis = false) {
192 nlohmann::json jGrid;
193 // Get the grid axes & potentially swap them
194 auto axes = grid.axes();
195 if (swapAxis && grid_type::DIM == 2u) {
196 std::swap(axes[0u], axes[1u]);
197 }
198
199 nlohmann::json jAxes;
200
201 // Fill the axes in the order they are
202 for (unsigned int ia = 0u; ia < grid_type::DIM; ++ia) {
203 auto jAxis = AxisJsonConverter::toJsonDetray(*axes[ia]);
204 jAxis["label"] = ia;
205 jAxes.push_back(jAxis);
206 }
207 jGrid["axes"] = jAxes;
208
209 nlohmann::json jData;
210 // 1D connections
211 if constexpr (grid_type::DIM == 1u) {
212 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
213 // Lookup bin
214 typename grid_type::index_t lbin;
215 lbin[0u] = ib0;
216 nlohmann::json jBin;
217 jBin["content"] = grid.atLocalBins(lbin);
218 // Corrected bin for detray
219 lbin[0u] = ib0 - 1u;
220 jBin["loc_index"] = lbin;
221 jData.push_back(jBin);
222 }
223 }
224
225 // 2D connections
226 if constexpr (grid_type::DIM == 2u) {
227 for (unsigned int ib0 = 1u; ib0 <= axes[0u]->getNBins(); ++ib0) {
228 for (unsigned int ib1 = 1u; ib1 <= axes[1u]->getNBins(); ++ib1) {
229 typename grid_type::index_t lbin;
230 // Lookup bin - respect swap (if it happened) for the lookup
231 lbin[0u] = swapAxis ? ib1 : ib0;
232 lbin[1u] = swapAxis ? ib0 : ib1;
233 nlohmann::json jBin;
234 jBin["content"] = grid.atLocalBins(lbin);
235 // Corrected bin for detray
236 lbin[0u] = ib0 - 1u;
237 lbin[1u] = ib1 - 1u;
238 jBin["loc_index"] = lbin;
239 jData.push_back(jBin);
240 }
241 }
242 }
243 jGrid["bins"] = jData;
244
245 return jGrid;
246}

259template <typename axis_generator_type,
260 typename value_type = std::vector<std::size_t>>
261auto fromJson(const nlohmann::json& jGrid,
262 const axis_generator_type& aGenerator) {
263 // Generate the grid
264 using GridType = typename axis_generator_type::template grid_type<value_type>;
265 GridType grid(aGenerator());
266 nlohmann::json jData = jGrid["data"];
267 // Index filling
268 if constexpr (GridType::DIM == 1u) {
269 for (const auto& jd : jData) {
270 std::array<std::size_t, 1u> lbin = jd[0u];
271 if (!jd[1u].is_null()) {
272 grid.atLocalBins(lbin) = jd[1u].get<value_type>();
273 }
274 }
275 }
276 if constexpr (GridType::DIM == 2u) {
277 for (const auto& jd : jData) {
278 std::array<std::size_t, 2u> lbin = jd[0u];
279 if (!jd[1u].is_null()) {
280 grid.atLocalBins(lbin) = jd[1u].get<value_type>();
281 }
282 }
283 }
284 return grid;
285}
