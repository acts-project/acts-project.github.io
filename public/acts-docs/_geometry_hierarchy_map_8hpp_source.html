150 // NOTE this class assumes that it knows the ordering of the levels within
151 // the geometry id. if the geometry id changes, this code has to be
152 // adapted too. the asserts ensure that such a change is caught.
153 static_assert(GeometryIdentifier().withVolume(1).value() <
154 GeometryIdentifier().withVolume(1).withBoundary(1).value(),
155 "Incompatible GeometryIdentifier hierarchy");
156 static_assert(GeometryIdentifier().withBoundary(1).value() <
157 GeometryIdentifier().withBoundary(1).withLayer(1).value(),
158 "Incompatible GeometryIdentifier hierarchy");
159 static_assert(GeometryIdentifier().withLayer(1).value() <

161 "Incompatible GeometryIdentifier hierarchy");

163 GeometryIdentifier().withApproach(1).value() <
164 GeometryIdentifier().withApproach(1).withSensitive(1).value(),
165 "Incompatible GeometryIdentifier hierarchy");

167 using Identifier = GeometryIdentifier::Value;
168
169 // encoded ids for all elements for faster lookup.
170 std::vector<Identifier> m_ids;
171 // validity bit masks for the ids: which parts to use for comparison
172 std::vector<Identifier> m_masks;
173 std::vector<Value> m_values;
174
176 static constexpr Identifier makeLeadingLevelsMask(GeometryIdentifier id) {
177 // construct id from encoded value with all bits set
178 const auto allSet = GeometryIdentifier(~GeometryIdentifier::Value{0u});
179 // manually iterate over identifier levels starting from the lowest
180 if (id.sensitive() != 0u) {
181 // all levels are valid; keep all bits set.
182 return allSet.withExtra(0u).value();
183 }
184 if (id.approach() != 0u) {
185 return allSet.withExtra(0u).withSensitive(0u).value();
186 }
187 if (id.layer() != 0u) {
188 return allSet.withExtra(0u).withSensitive(0u).withApproach(0u).value();
189 }
190 if (id.boundary() != 0u) {
191 return allSet.withExtra(0u)
192 .withSensitive(0u)
193 .withApproach(0u)
194 .withLayer(0u)
195 .value();
196 }
197 if (id.volume() != 0u) {
198 return allSet.withExtra(0u)
199 .withSensitive(0u)
200 .withApproach(0u)
201 .withLayer(0u)
202 .withBoundary(0u)
203 .value();
204 }
205 // no valid levels; all bits are zero.
206 return Identifier{0u};
207 }
208
210 static constexpr Identifier makeHighestLevelMask() {
211 return makeLeadingLevelsMask(GeometryIdentifier(0u).withVolume(1u));
212 }
213
215 static constexpr bool equalWithinMask(Identifier lhs, Identifier rhs,
216 Identifier mask) {
217 return (lhs & mask) == (rhs & mask);
218 }
219
221 static void sortAndCheckDuplicates(std::vector<InputElement>& elements);
222
227 void fill(const std::vector<InputElement>& elements);

245template <typename value_t>
246inline void GeometryHierarchyMap<value_t>::sortAndCheckDuplicates(
247 std::vector<InputElement>& elements) {
248 // ensure elements are sorted by identifier
249 std::ranges::sort(elements, [=](const auto& lhs, const auto& rhs) {
250 return lhs.first < rhs.first;
251 });
252
253 // Check that all elements have unique identifier
254 auto dup = std::ranges::adjacent_find(
255 elements,
256 [](const auto& lhs, const auto& rhs) { return lhs.first == rhs.first; });
257
258 if (dup != elements.end()) {
259 throw std::invalid_argument("Input elements contain duplicates");
260 }
261}

263template <typename value_t>
264inline void GeometryHierarchyMap<value_t>::fill(
265 const std::vector<InputElement>& elements) {
266 m_ids.clear();
267 m_masks.clear();
268 m_values.clear();
269
270 m_ids.reserve(elements.size());
271 m_masks.reserve(elements.size());
272 m_values.reserve(elements.size());
273
274 for (const auto& element : elements) {
275 m_ids.push_back(element.first.value());
276 m_masks.push_back(
277 makeLeadingLevelsMask(GeometryIdentifier(element.first.value())));
278 m_values.push_back(std::move(element.second));
279 }
280}

282template <typename value_t>
283inline auto GeometryHierarchyMap<value_t>::find(
284 const GeometryIdentifier& id) const -> Iterator {
285 assert((m_ids.size() == m_values.size()) &&
286 "Inconsistent container state: #ids != # values");
287 assert((m_masks.size() == m_values.size()) &&
288 "Inconsistent container state: #masks != #values");
289
290 // we can not search for the element directly since the relevant one
291 // might be stored at a higher level. ids for higher levels would always
292 // be sorted before the requested id. searching for the first element
293 // after the requested ensures that we include the full hierarchy.
294 const auto it = std::upper_bound(m_ids.begin(), m_ids.end(), id.value());
295 auto i = std::distance(m_ids.begin(), it);
296
297 // now go up the hierarchy to find the first matching element.
298 // example: the container stores four identifiers
299 //
300 // 2|x|x (volume-only)
301 // 2|2|1 (volume, layer, and sensitive)
302 // 2|3|x (volume and layer)
303 // 2|3|4 (volume, layer, and sensitive)
304 //
305 // where | marks level boundaries. searching for either 2|3|4, 2|3|7, or
306 // 2|4|x would first point to 2|3|4 and thus needs to go up the hierarchy.
307 while (0 < i) {
308 // index always starts after item of interest due to upper bound search
309 --i;
310
311 // if the input id does not even match at the highest hierarchy level
312 // with the current comparison id, then have reached the end of this
313 // hierarchy. having a special check for the highest level avoids an
314 // unbounded search window all the way to the beginning of the container for
315 // the global default entry.
316 if (!equalWithinMask(id.value(), m_ids[i], makeHighestLevelMask())) {
317 // check if a global default entry exists
318 if (m_ids.front() == Identifier{0u}) {
319 return begin();
320 } else {
321 return end();
322 }
323 }
324
325 // since the search is going backwards in the sorted container, it
326 // progresses from more specific to less specific elements. the first
327 // match is automatically the appropriate one.
328 if (equalWithinMask(id.value(), m_ids[i], m_masks[i])) {
329 return std::next(begin(), i);
330 }
331 }
332
333 // all options are exhausted and no matching element was found.
334 return end();
335}
