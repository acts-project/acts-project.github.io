30template <typename T>
31concept ConstTrackContainerBackend =
32 requires(const T& cv, HashedString key, TrackIndexType itrack) {
33 { cv.size_impl() } -> std::same_as<std::size_t>;
34
35 { cv.component_impl(key, itrack) } -> std::same_as<std::any>;
36
37 { cv.parameters(itrack) } -> std::same_as<detail::ConstParameters>;
38
39 { cv.covariance(itrack) } -> std::same_as<detail::ConstCovariance>;
40
41 { cv.hasColumn_impl(key) } -> std::same_as<bool>;
42
43 { cv.referenceSurface_impl(itrack) } -> std::same_as<const Surface*>;
44
45 {
46 cv.particleHypothesis_impl(itrack)
47 } -> std::same_as<ParticleHypothesis>;
48
49 { cv.dynamicKeys_impl() };
50 requires detail::RangeLike<decltype(cv.dynamicKeys_impl())>;
51 };

53template <typename T>
54concept MutableTrackContainerBackend =
55 ConstTrackContainerBackend<T> &&
56 requires(T v, HashedString key, TrackIndexType itrack, std::string col,
57 const T& other, std::shared_ptr<const Surface> sharedSurface) {
58 { v.parameters(itrack) } -> std::same_as<detail::Parameters>;
59
60 { v.covariance(itrack) } -> std::same_as<detail::Covariance>;
61
62 { v.addTrack_impl() } -> std::same_as<TrackIndexType>;
63
64 { v.removeTrack_impl(itrack) };
65
66 // As far as I know there's no good way to assert that there's a
67 // generic template function
68 { v.template addColumn_impl<std::uint32_t>(col) };
69 { v.template addColumn_impl<std::uint64_t>(col) };
70 { v.template addColumn_impl<std::int32_t>(col) };
71 { v.template addColumn_impl<std::int64_t>(col) };
72 { v.template addColumn_impl<float>(col) };
73 { v.template addColumn_impl<double>(col) };
74
75 { v.copyDynamicFrom_impl(itrack, key, std::declval<const std::any&>()) };
76
77 { v.ensureDynamicColumns_impl(other) };
78
79 { v.reserve(itrack) };
80
81 { v.setReferenceSurface_impl(itrack, sharedSurface) };
82
83 {
84 v.setParticleHypothesis_impl(
85 itrack, std::declval<const Acts::ParticleHypothesis&>())
86 };
87
88 { v.clear() };
89 };
