70 template <int i, typename stepper_t>
71 bool k(const typename stepper_t::State& state, const stepper_t& stepper,
72 const IVolumeMaterial* volumeMaterial, Vector3& knew,
73 const Vector3& bField, std::array<double, 4>& kQoP,
74 const double h = 0., const Vector3& kprev = Vector3::Zero())
75 requires(i >= 0 && i <= 3)
76 {
77 if (volumeMaterial == nullptr) {
78 return defaultExtension.template k<i>(state, stepper, volumeMaterial,
79 knew, bField, kQoP, h, kprev);
80 }
81
82 double q = stepper.charge(state);
83 const auto& particleHypothesis = stepper.particleHypothesis(state);
84 float mass = particleHypothesis.mass();
85
86 // i = 0 is used for setup and evaluation of k
87 if constexpr (i == 0) {
88 // Set up for energy loss
89 Vector3 position = stepper.position(state);
90 material = volumeMaterial->material(position.template cast<double>());
91 initialMomentum = stepper.absoluteMomentum(state);
92 currentMomentum = initialMomentum;
93 qop[0] = stepper.qOverP(state);
94 initializeEnergyLoss(state, stepper);
95 // Evaluate k
96 knew = qop[0] * stepper.direction(state).cross(bField);
97 // Evaluate k for the time propagation
98 Lambdappi[0] = -qop[0] * qop[0] * qop[0] * g * energy[0] / (q * q);
99 //~ tKi[0] = std::hypot(1, mass / initialMomentum);
100 tKi[0] = fastHypot(1, mass * qop[0]);
101 kQoP[0] = Lambdappi[0];
102 } else {
103 // Update parameters and check for momentum condition
104 updateEnergyLoss(mass, h, state, stepper, i);
105 if (currentMomentum < state.options.dense.momentumCutOff) {
106 return false;
107 }
108 // Evaluate k
109 knew = qop[i] * (stepper.direction(state) + h * kprev).cross(bField);
110 // Evaluate k_i for the time propagation
111 auto qopNew = qop[0] + h * Lambdappi[i - 1];
112 Lambdappi[i] = -qopNew * qopNew * qopNew * g * energy[i] / (q * q);
113 tKi[i] = fastHypot(1, mass * qopNew);
114 kQoP[i] = Lambdappi[i];
115 }
116 return true;
117 }

132 template <typename stepper_t>
133 bool finalize(typename stepper_t::State& state, const stepper_t& stepper,
134 const IVolumeMaterial* volumeMaterial, const double h) const {
135 if (volumeMaterial == nullptr) {
136 return defaultExtension.finalize(state, stepper, volumeMaterial, h);
137 }
138
139 const auto& particleHypothesis = stepper.particleHypothesis(state);
140 float mass = particleHypothesis.mass();
141
142 // Evaluate the new momentum
143 auto newMomentum =
144 stepper.absoluteMomentum(state) +
145 (h / 6.) * (dPds[0] + 2. * (dPds[1] + dPds[2]) + dPds[3]);
146
147 // Break propagation if momentum becomes below cut-off
148 if (newMomentum < state.options.dense.momentumCutOff) {
149 return false;
150 }
151
152 // Add derivative dlambda/ds = Lambda''
153 state.derivative(7) = -fastHypot(mass, newMomentum) * g /
154 (newMomentum * newMomentum * newMomentum);
155
156 // Update momentum
157 state.pars[eFreeQOverP] = stepper.charge(state) / newMomentum;
158 // Add derivative dt/ds = 1/(beta * c) = sqrt(m^2 * p^{-2} + c^{-2})
159 state.derivative(3) = fastHypot(1, mass / newMomentum);
160 // Update time
161 state.pars[eFreeTime] +=
162 (h / 6.) * (tKi[0] + 2. * (tKi[1] + tKi[2]) + tKi[3]);
163
164 return true;
165 }

207 template <typename stepper_t>
208 bool transportMatrix(typename stepper_t::State& state,
209 const stepper_t& stepper, const double h,
210 FreeMatrix& D) const {
230
231 auto& sd = state.stepData;
232 auto dir = stepper.direction(state);
233 const auto& particleHypothesis = stepper.particleHypothesis(state);
234 float mass = particleHypothesis.mass();
235
236 D = FreeMatrix::Identity();
237 const double half_h = h * 0.5;
238
239 // This sets the reference to the sub matrices
240 // dFdx is already initialised as (3x3) zero
241 auto dFdT = D.block<3, 3>(0, 4);
242 auto dFdL = D.block<3, 1>(0, 7);
243 // dGdx is already initialised as (3x3) identity
244 auto dGdT = D.block<3, 3>(4, 4);
245 auto dGdL = D.block<3, 1>(4, 7);
246
247 SquareMatrix3 dk1dT = SquareMatrix3::Zero();
248 SquareMatrix3 dk2dT = SquareMatrix3::Identity();
249 SquareMatrix3 dk3dT = SquareMatrix3::Identity();
250 SquareMatrix3 dk4dT = SquareMatrix3::Identity();
251
252 Vector3 dk1dL = Vector3::Zero();
253 Vector3 dk2dL = Vector3::Zero();
254 Vector3 dk3dL = Vector3::Zero();
255 Vector3 dk4dL = Vector3::Zero();
256
258 std::array<double, 4> jdL{};
259
260 // Evaluation of the rightmost column without the last term.
261 jdL[0] = dLdl[0];
262 dk1dL = dir.cross(sd.B_first);
263
264 jdL[1] = dLdl[1] * (1. + half_h * jdL[0]);
265 dk2dL = (1. + half_h * jdL[0]) * (dir + half_h * sd.k1).cross(sd.B_middle) +
266 qop[1] * half_h * dk1dL.cross(sd.B_middle);
267
268 jdL[2] = dLdl[2] * (1. + half_h * jdL[1]);
269 dk3dL = (1. + half_h * jdL[1]) * (dir + half_h * sd.k2).cross(sd.B_middle) +
270 qop[2] * half_h * dk2dL.cross(sd.B_middle);
271
272 jdL[3] = dLdl[3] * (1. + h * jdL[2]);
273 dk4dL = (1. + h * jdL[2]) * (dir + h * sd.k3).cross(sd.B_last) +
274 qop[3] * h * dk3dL.cross(sd.B_last);
275
276 dk1dT(0, 1) = sd.B_first.z();
277 dk1dT(0, 2) = -sd.B_first.y();
278 dk1dT(1, 0) = -sd.B_first.z();
279 dk1dT(1, 2) = sd.B_first.x();
280 dk1dT(2, 0) = sd.B_first.y();
281 dk1dT(2, 1) = -sd.B_first.x();
282 dk1dT *= qop[0];
283
284 dk2dT += half_h * dk1dT;
285 dk2dT = qop[1] * VectorHelpers::cross(dk2dT, sd.B_middle);
286
287 dk3dT += half_h * dk2dT;
288 dk3dT = qop[2] * VectorHelpers::cross(dk3dT, sd.B_middle);
289
290 dk4dT += h * dk3dT;
291 dk4dT = qop[3] * VectorHelpers::cross(dk4dT, sd.B_last);
292
293 dFdT.setIdentity();
294 dFdT += h / 6. * (dk1dT + dk2dT + dk3dT);
295 dFdT *= h;
296
297 dFdL = h * h / 6. * (dk1dL + dk2dL + dk3dL);
298
299 dGdT += h / 6. * (dk1dT + 2. * (dk2dT + dk3dT) + dk4dT);
300
301 dGdL = h / 6. * (dk1dL + 2. * (dk2dL + dk3dL) + dk4dL);
302
303 // Evaluation of the dLambda''/dlambda term
304 D(7, 7) += (h / 6.) * (jdL[0] + 2. * (jdL[1] + jdL[2]) + jdL[3]);
305
306 // The following comment lines refer to the application of the time being
307 // treated as a position. Since t and qop are treated independently for now,
308 // this just serves as entry point for building their relation
309 //~ double dtpp1dl = -mass * mass * qop[0] * qop[0] *
310 //~ (3. * g + qop[0] * dgdqop(energy[0], .mass,
311 //~ absPdg, meanEnergyLoss));
312
313 double dtp1dl = qop[0] * mass * mass / fastHypot(1, qop[0] * mass);
314 double qopNew = qop[0] + half_h * Lambdappi[0];
315
316 //~ double dtpp2dl = -mass * mass * qopNew *
317 //~ qopNew *
318 //~ (3. * g * (1. + half_h * jdL[0]) +
319 //~ qopNew * dgdqop(energy[1], mass, absPdgCode, meanEnergyLoss));
320
321 double dtp2dl = qopNew * mass * mass / fastHypot(1, qopNew * mass);
322 qopNew = qop[0] + half_h * Lambdappi[1];
323
324 //~ double dtpp3dl = -mass * mass * qopNew *
325 //~ qopNew *
326 //~ (3. * g * (1. + half_h * jdL[1]) +
327 //~ qopNew * dgdqop(energy[2], mass, absPdg, meanEnergyLoss));
328
329 double dtp3dl = qopNew * mass * mass / fastHypot(1, qopNew * mass);
330 qopNew = qop[0] + half_h * Lambdappi[2];
331 double dtp4dl = qopNew * mass * mass / fastHypot(1, qopNew * mass);
332
333 //~ D(3, 7) = h * mass * mass * qop[0] /
334 //~ std::hypot(1., mass * qop[0])
335 //~ + h * h / 6. * (dtpp1dl + dtpp2dl + dtpp3dl);
336
337 D(3, 7) = (h / 6.) * (dtp1dl + 2. * (dtp2dl + dtp3dl) + dtp4dl);
338 return true;
339 }

348 template <typename stepper_t>
349 void initializeEnergyLoss(const typename stepper_t::State& state,
350 const stepper_t& stepper) {
351 const auto& particleHypothesis = stepper.particleHypothesis(state);
352 float mass = particleHypothesis.mass();
353 PdgParticle absPdg = particleHypothesis.absolutePdg();
354 float absQ = particleHypothesis.absoluteCharge();
355
356 energy[0] = fastHypot(initialMomentum, mass);
357 // use unit length as thickness to compute the energy loss per unit length
358 MaterialSlab slab(material, 1);
359 // Use the same energy loss throughout the step.
360 if (state.options.dense.meanEnergyLoss) {
361 g = -computeEnergyLossMean(slab, absPdg, mass, static_cast<float>(qop[0]),
362 absQ);
363 } else {
364 // TODO using the unit path length is not quite right since the most
365 // probably energy loss is not independent from the path length.
366 g = -computeEnergyLossMode(slab, absPdg, mass, static_cast<float>(qop[0]),
367 absQ);
368 }
369 // Change of the momentum per path length
370 // dPds = dPdE * dEds
371 dPds[0] = g * energy[0] / initialMomentum;
372 if (state.covTransport) {
373 // Calculate the change of the energy loss per path length and
374 // inverse momentum
375 if (state.options.dense.includeGradient) {
376 if (state.options.dense.meanEnergyLoss) {
377 dgdqopValue = deriveEnergyLossMeanQOverP(
378 slab, absPdg, mass, static_cast<float>(qop[0]), absQ);
379 } else {
380 // TODO path length dependence; see above
381 dgdqopValue = deriveEnergyLossModeQOverP(
382 slab, absPdg, mass, static_cast<float>(qop[0]), absQ);
383 }
384 }
385 // Calculate term for later error propagation
386 dLdl[0] = (-qop[0] * qop[0] * g * energy[0] *
387 (3. - (initialMomentum * initialMomentum) /
388 (energy[0] * energy[0])) -
389 qop[0] * qop[0] * qop[0] * energy[0] * dgdqopValue);
390 }
391 }

403 template <typename stepper_t>
404 void updateEnergyLoss(const double mass, const double h,
405 const typename stepper_t::State& state,
406 const stepper_t& stepper, const int i) {
407 // Update parameters related to a changed momentum
408 currentMomentum = initialMomentum + h * dPds[i - 1];
409 energy[i] = fastHypot(currentMomentum, mass);
410 dPds[i] = g * energy[i] / currentMomentum;
411 qop[i] = stepper.charge(state) / currentMomentum;
412 // Calculate term for later error propagation
413 if (state.covTransport) {
414 dLdl[i] = (-qop[i] * qop[i] * g * energy[i] *
415 (3. - (currentMomentum * currentMomentum) /
416 (energy[i] * energy[i])) -
417 qop[i] * qop[i] * qop[i] * energy[i] * dgdqopValue);
418 }
419 }
