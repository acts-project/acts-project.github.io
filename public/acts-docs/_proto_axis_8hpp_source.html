128template <typename payload_t>
129std::unique_ptr<IGrid> makeGrid(const ProtoAxis& a) {
130 if (a.isAutorange()) {
131 throw std::invalid_argument(
132 "ProtoAxis::makeGrid: Auto-range of the proto axis is not (yet) "
133 "resolved, call setRange() first.");
134 }
135
136 return a.getAxis().visit(
137 [&]<typename AxisTypeA>(const AxisTypeA& axis) -> std::unique_ptr<IGrid> {
138 using GridType = Grid<payload_t, AxisTypeA>;
139 return std::make_unique<GridType>(axis);
140 });
141}
142
151template <typename payload_t>
152std::unique_ptr<IGrid> makeGrid(const ProtoAxis& a, const ProtoAxis& b) {
153 if (a.isAutorange() || b.isAutorange()) {
154 throw std::invalid_argument(
155 "ProtoAxis::makeGrid: Auto-range of the proto axis is not (yet) "
156 "resolved, call setRange() first.");
157 }
158
159 return a.getAxis().visit([&]<typename AxisTypeA>(const AxisTypeA& axisA)
160 -> std::unique_ptr<IGrid> {
161 return b.getAxis().visit([&]<typename AxisTypeB>(const AxisTypeB& axisB)
162 -> std::unique_ptr<IGrid> {
163 using GridType = Grid<payload_t, AxisTypeA, AxisTypeB>;
164 return std::make_unique<GridType>(axisA, axisB);
165 });
166 });
167}
