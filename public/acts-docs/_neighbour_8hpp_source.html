48template <typename grid_t>
49Neighbour<grid_t>::Neighbour(const grid_t& grid, std::size_t idx,
50 const float lowerBound)
51 : index(idx) {
53 const auto& collection = grid.at(idx);
56 if (collection.size() == 0) {
57 itr = collection.begin();
58 return;
59 }
60
63 if (collection.front()->radius() > lowerBound) {
64 itr = collection.begin();
65 }
70 else if (collection.back()->radius() < lowerBound) {
71 itr = collection.end();
72 }
75 else {
76 // custom binary search which was observed to be faster than
77 // `std::lower_bound` see https://github.com/acts-project/acts/pull/3095
78 std::size_t start = 0ul;
79 std::size_t stop = collection.size() - 1;
80 while (start <= stop) {
81 std::size_t mid = (start + stop) / 2;
82 if (collection[mid]->radius() == lowerBound) {
83 itr = collection.begin() + mid;
84 return;
85 } else if (collection[mid]->radius() > lowerBound) {
86 if (mid > 0 && collection[mid - 1]->radius() < lowerBound) {
87 itr = collection.begin() + mid;
88 return;
89 }
90 stop = mid - 1;
91 } else {
92 if (mid + 1 < collection.size() &&
93 collection[mid + 1]->radius() > lowerBound) {
94 itr = collection.begin() + mid + 1;
95 return;
96 }
97 start = mid + 1;
98 }
99 } // while loop
100 }
