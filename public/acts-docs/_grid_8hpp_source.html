523 template <class Point>
524 T interpolate(const Point& point) const
525 requires(Concepts::interpolatable<T, Point, std::array<double, DIM>,
526 std::array<double, DIM>>)
527 {
528 // there are 2^DIM corner points used during the interpolation
529 constexpr std::size_t nCorners = 1 << DIM;
530
531 // construct vector of pairs of adjacent bin centers and values
532 std::array<value_type, nCorners> neighbors{};
533
534 // get local indices for current bin
535 // value of bin is interpreted as being the field value at its lower left
536 // corner
537 const auto& llIndices = localBinsFromPosition(point);
538
539 // get global indices for all surrounding corner points
540 const auto& closestIndices = rawClosestPointsIndices(llIndices);
541
542 // get values on grid points
543 std::size_t i = 0;
544 for (std::size_t index : closestIndices) {
545 neighbors.at(i++) = at(index);
546 }
547
548 return Acts::interpolate(point, lowerLeftBinEdge(llIndices),
549 upperRightBinEdge(llIndices), neighbors);
550 }

618 std::size_t size(bool fullCounter = true) const {
619 index_t nBinsArray = numLocalBins();
620 std::size_t current_size = 1;
621 // add under-and overflow bins for each axis and multiply all bins
622 if (fullCounter) {
623 for (const auto& value : nBinsArray) {
624 current_size *= value + 2;
625 }
626 }
627 // ignore under-and overflow bins for each axis and multiply all bins
628 else {
629 for (const auto& value : nBinsArray) {
630 current_size *= value;
631 }
632 }
633 return current_size;
634 }
