45template <typename Scalar>
46Scalar etaFromTheta(Scalar theta) {
47 if (theta <= EtaThetaConversionTraits<Scalar>::minTheta) {
48 return std::numeric_limits<Scalar>::infinity();
49 }
50 if (theta >= EtaThetaConversionTraits<Scalar>::maxTheta) {
51 return -std::numeric_limits<Scalar>::infinity();
52 }
53
54 return -std::log(std::tan(theta / 2));
55}

65template <typename Scalar>
66Scalar thetaFromEta(Scalar eta) {
67 if (eta <= EtaThetaConversionTraits<Scalar>::minEta) {
68 return std::numbers::pi_v<Scalar>;
69 }
70 if (eta >= EtaThetaConversionTraits<Scalar>::maxEta) {
71 return 0;
72 }
73
74 return 2 * std::atan(std::exp(-eta));
75}
