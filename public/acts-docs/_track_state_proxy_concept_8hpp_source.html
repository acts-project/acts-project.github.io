207template <typename T>
208concept MutableTrackStateProxyConcept =
209 TrackStateProxyConcept<T> &&
210 requires(T v, HashedString key, TrackStatePropMask mask,
211 TrackIndexType index, std::shared_ptr<const Surface> surface,
212 Eigen::Matrix<double, 3, 6> projector,
213 ProjectorBitset projectorBitset, SourceLink sl,
214 std::size_t measdim) {
215 { v.unset(mask) };
216
217 // Cannot verify for all types, so just check int
218 {
219 v.template component<int, hashString("blubb")>()
220 } -> std::same_as<int&>;
221
222 { v.template component<int>(key) } -> std::same_as<int&>;
223
224 { v.predicted() } -> std::same_as<detail::Parameters>;
225 { v.predictedCovariance() } -> std::same_as<detail::Covariance>;
226
227 { v.filtered() } -> std::same_as<detail::Parameters>;
228 { v.filteredCovariance() } -> std::same_as<detail::Covariance>;
229
230 { v.smoothed() } -> std::same_as<detail::Parameters>;
231 { v.smoothedCovariance() } -> std::same_as<detail::Covariance>;

233 { v.jacobian() } -> std::same_as<detail::Covariance>;
234
235 requires requires(BoundSubspaceIndices m) {
236 v.setProjectorSubspaceIndices(m);
237 };
238
239 { v.setUncalibratedSourceLink(std::move(sl)) };
240
241 { v.template calibrated<2>() } -> std::same_as<detail::Measurement>;
242 {
243 v.template calibratedCovariance<2>()
244 } -> std::same_as<detail::MeasurementCovariance>;
245
246 { v.effectiveCalibrated() } -> std::same_as<detail::DynamicMeasurement>;
247 {
248 v.effectiveCalibratedCovariance()
249 } -> std::same_as<detail::DynamicMeasurementCovariance>;
250
251 { v.allocateCalibrated(measdim) };
252
253 { v.allocateCalibrated(ActsVector<1>{}, ActsSquareMatrix<1>{}) };
254 // Assuming intermediate values are also allowed
255 {
256 v.allocateCalibrated(ActsVector<eBoundSize>{},
257 ActsSquareMatrix<eBoundSize>{})
258 };
259
260 { v.chi2() } -> std::same_as<float&>;
261
262 { v.pathLength() } -> std::same_as<double&>;

264 { v.typeFlags() } -> std::same_as<TrackStateType>;
