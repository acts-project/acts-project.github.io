28template <typename T, std::size_t... BitsPerLevel>
29class MultiIndex {
30 public:
31 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
32 "The underlying storage type must be an unsigned integer");
33 static_assert(0 < sizeof...(BitsPerLevel),
34 "At least one level must be defined");
35 static_assert((sizeof(T) * CHAR_BIT) == (... + BitsPerLevel),
36 "The sum of bits per level must match the underlying storage");
37
39 using Value = T;
41 static constexpr std::size_t kNumLevels = sizeof...(BitsPerLevel);
42
45 static constexpr MultiIndex Zeros() { return MultiIndex(0u); }
55 template <typename... Us>
56 static constexpr MultiIndex Encode(Us&&... us) {
57 static_assert(sizeof...(Us) <= kNumLevels,
58 "Can only encode as many levels as in the MultiIndex");
59
60 MultiIndex index(0u);
61 std::size_t lvl = 0;
62 for (Value val : std::array<Value, sizeof...(Us)>{us...}) {
63 index.set(lvl++, val);
64 }
65 return index;
66 }
67
70 explicit constexpr MultiIndex(Value encoded) : m_value(encoded) {}
72 MultiIndex() = default;
74 MultiIndex(const MultiIndex&) = default;
76 MultiIndex(MultiIndex&) = default;
79 MultiIndex& operator=(const MultiIndex&) = default;
82 MultiIndex& operator=(MultiIndex&&) noexcept = default;

87 m_value = encoded;
88 return *this;
89 }

93 constexpr Value value() const { return m_value; }
97 constexpr Value level(std::size_t lvl) const {
98 assert((lvl < kNumLevels) && "Index level outside allowed range");
99 return (m_value >> shift(lvl)) & mask(lvl);
100 }
101
105 constexpr MultiIndex& set(std::size_t lvl, Value val) {
106 assert((lvl < kNumLevels) && "Index level outside allowed range");
107 if (val > maxValue(lvl)) {
108 throw std::out_of_range(
109 "Value " + std::to_string(val) + " for index level " +
110 std::to_string(lvl) +
111 " exceeds allowed range (max=" + std::to_string(maxValue(lvl)) + ")");
112 }
113 // mask of valid bits at the encoded positions for the index level
114 Value shiftedMask = (mask(lvl) << shift(lvl));
115 // value of the index level shifted to its encoded position
116 Value shiftedValue = (val << shift(lvl));
117 // combine existing values with the value for the index level
118 m_value = (m_value & ~shiftedMask) | (shiftedValue & shiftedMask);
119 return *this;
120 }
121
125 constexpr std::size_t maxValue(std::size_t lvl) const {
126 assert((lvl < kNumLevels) && "Index level outside allowed range");
127 return (1 << s_bits.at(lvl)) - 1;
128 }
129
133 constexpr MultiIndex makeNextSibling(std::size_t lvl) const {
134 assert((lvl < kNumLevels) && "Index level outside allowed range");
135 // remove lower levels by shifting the upper levels to the left edge
136 Value upper = (m_value >> shift(lvl));
137 // increase to create sibling and shift back to zero lower levels again
138 return MultiIndex{(upper + 1u) << shift(lvl)};
139 }
140
143 constexpr MultiIndex makeLastDescendant(std::size_t lvl) const {
144 assert((lvl < kNumLevels) && "Index level outside allowed range");
145 // mask everything below the selected level
146 Value maskLower = (Value{1u} << shift(lvl)) - 1u;
147 // replace the masked lower levels w/ ones
148 return MultiIndex{(m_value & ~maskLower) | maskLower};
149 }
150
154 static constexpr std::size_t bits(std::size_t lvl) {
155 assert((lvl < kNumLevels) && "Index level outside allowed range");
156 return s_bits[lvl];
157 }
158
159 private:
160 // per-level mask and right-most bit position for shifting
161 static constexpr std::array<std::size_t, kNumLevels> s_bits{BitsPerLevel...};
162 static constexpr std::size_t shift(std::size_t lvl) {
163 std::size_t s = 0u;
164 // sum up all bits below the requested level
165 for (std::size_t i = (lvl + 1); i < s_bits.size(); ++i) {
166 s += s_bits[i];
167 }
168 return s;
169 }
170 static constexpr Value mask(std::size_t lvl) {
171 return (Value{1u} << s_bits[lvl]) - 1u;
172 }
173
174 Value m_value;
175
176 friend constexpr bool operator<(MultiIndex lhs, MultiIndex rhs) {
177 return lhs.m_value < rhs.m_value;
178 }
179
180 friend constexpr bool operator==(MultiIndex lhs, MultiIndex rhs) {
181 return lhs.m_value == rhs.m_value;
182 }
183
184 friend inline std::ostream& operator<<(std::ostream& os, MultiIndex idx) {
185 // one level is always defined
186 os << idx.level(0u);
187 for (std::size_t lvl = 1; lvl < kNumLevels; ++lvl) {
188 os << '|' << idx.level(lvl);
189 }
190 return os;
191 }
192};
