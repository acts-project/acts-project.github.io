54template <typename external_space_point_t>
55concept SatisfyCandidateConcept = requires(external_space_point_t spacePoint) {
56 { spacePoint.x() } -> std::convertible_to<float>;
57 { spacePoint.y() } -> std::convertible_to<float>;
58 { spacePoint.z() } -> std::convertible_to<float>;
59};
