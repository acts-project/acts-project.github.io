331inline std::ostream& operator<<(std::ostream& os,
332 const TrackSelector::Config& cuts) {
333 // for printing cuts set up with `within`
334 auto printMinMax = [&](const char* name, const auto& min, const auto& max) {
335 os << " - " << min << " <= " << name << " < " << max << "\n";
336 };
337 // for printing cuts set up with `checkMin`
338 auto printMin = [&](const char* name, const auto& min) {
339 os << " - " << min << " <= " << name << "\n";
340 };
341 // for printing cuts set up with `checkMax`
342 auto printMax = [&](const char* name, const auto& max) {
343 os << " - " << name << " <= " << max << "\n";
344 };
345
346 printMinMax("loc0", cuts.loc0Min, cuts.loc0Max);
347 printMinMax("loc1", cuts.loc1Min, cuts.loc1Max);
348 printMinMax("time", cuts.timeMin, cuts.timeMax);
349 printMinMax("phi", cuts.phiMin, cuts.phiMax);
350 printMinMax("eta", cuts.etaMin, cuts.etaMax);
351 printMinMax("absEta", cuts.absEtaMin, cuts.absEtaMax);
352 printMinMax("pt", cuts.ptMin, cuts.ptMax);
353 printMax("nHoles", cuts.maxHoles);
354 printMax("nOutliers", cuts.maxOutliers);
355 printMax("nHoles + nOutliers", cuts.maxHolesAndOutliers);
356 printMax("nSharedHits", cuts.maxSharedHits);
357 printMax("chi2", cuts.maxChi2);
358 printMin("nMeasurements", cuts.minMeasurements);
359 return os;

362inline TrackSelector::EtaBinnedConfig& TrackSelector::EtaBinnedConfig::addCuts(
363 double etaMax, const std::function<void(Config&)>& callback) {
364 if (etaMax <= absEtaEdges.back()) {
365 throw std::invalid_argument{
366 "Abs Eta bin edges must be in increasing order"};
367 }
368
369 if (etaMax < 0.0) {
370 throw std::invalid_argument{"Abs Eta bin edges must be positive"};
371 }
372
373 absEtaEdges.push_back(etaMax);
374 cutSets.emplace_back();
375 if (callback) {
376 callback(cutSets.back());
377 }
378 return *this;
379}

398inline std::size_t TrackSelector::EtaBinnedConfig::binIndexNoCheck(
399 double eta) const {
400 auto binIt =
401 std::upper_bound(absEtaEdges.begin(), absEtaEdges.end(), std::abs(eta));
402 std::size_t index = std::distance(absEtaEdges.begin(), binIt);
403 if (index == 0) {
404 index = absEtaEdges.size() + 1; // positive value to check for underflow
405 }
406 return index - 1;

439template <TrackProxyConcept track_proxy_t>
440bool TrackSelector::isValidTrack(const track_proxy_t& track) const {
441 auto checkMin = [](auto x, auto min) { return min <= x; };
442 auto checkMax = [](auto x, auto max) { return x <= max; };
443 auto within = [](double x, double min, double max) {
444 return (min <= x) && (x < max);
445 };
446
447 const auto theta = track.theta();
448
449 constexpr double kUnset = -std::numeric_limits<double>::infinity();
450
451 double _eta = kUnset;
452 double _absEta = kUnset;
453
454 auto absEta = [&]() {
455 if (_absEta == kUnset) {
456 _eta = AngleHelpers::etaFromTheta(theta);
457 _absEta = std::abs(_eta);
458 }
459 return _absEta;
460 };
461
462 const Config* cutsPtr{nullptr};
463 if (!m_isUnbinned) {
464 // return false if |eta| is outside its range, or nan.
465 if (!(absEta() >= m_cfg.absEtaEdges.front() &&
466 _absEta < m_cfg.absEtaEdges.back())) {
467 return false;
468 }
469 cutsPtr = &m_cfg.getCuts(_eta);
470 } else {
471 cutsPtr = &m_cfg.cutSets.front();
472 }
473
474 const Config& cuts = *cutsPtr;
475
476 auto parameterCuts = [&]() {
477 return within(track.transverseMomentum(), cuts.ptMin, cuts.ptMax) &&
478 (!m_isUnbinned ||
479 (within(absEta(), cuts.absEtaMin, cuts.absEtaMax) &&
480 within(_eta, cuts.etaMin, cuts.etaMax))) &&
481 within(track.phi(), cuts.phiMin, cuts.phiMax) &&
482 within(track.loc0(), cuts.loc0Min, cuts.loc0Max) &&
483 within(track.loc1(), cuts.loc1Min, cuts.loc1Max) &&
484 within(track.time(), cuts.timeMin, cuts.timeMax);
485 };
486
487 auto trackCuts = [&]() {
488 return checkMin(track.nMeasurements(), cuts.minMeasurements) &&
489 checkMax(track.nHoles(), cuts.maxHoles) &&
490 checkMax(track.nOutliers(), cuts.maxOutliers) &&
491 checkMax(track.nHoles() + track.nOutliers(),
492 cuts.maxHolesAndOutliers) &&
493 checkMax(track.nSharedHits(), cuts.maxSharedHits) &&
494 checkMax(track.chi2(), cuts.maxChi2) &&
495 cuts.measurementCounter.isValidTrack(track);
496 };
497
498 if (cuts.requireReferenceSurface) {
499 return track.hasReferenceSurface() && parameterCuts() && trackCuts();
500 } else {
501 return trackCuts();
502 }
503}

505inline TrackSelector::TrackSelector(
506 const TrackSelector::EtaBinnedConfig& config)
507 : m_cfg(config) {
508 if (m_cfg.cutSets.size() != m_cfg.nEtaBins()) {
509 throw std::invalid_argument{
510 "TrackSelector cut / eta bin configuration is inconsistent"};
511 }
512
513 if (m_cfg.nEtaBins() == 1) {
514 static const std::vector<double> infVec = {0, inf};
515 m_isUnbinned =
516 m_cfg.absEtaEdges == infVec; // single bin, no eta edges given
517 }
518
519 if (!m_isUnbinned) {
520 for (const auto& cuts : m_cfg.cutSets) {
521 if (cuts.etaMin != -inf || cuts.etaMax != inf || cuts.absEtaMin != 0.0 ||
522 cuts.absEtaMax != inf) {
523 throw std::invalid_argument{
524 "Explicit eta cuts are only valid for single eta bin"};
525 }
526 }
527 }
528}

533template <TrackProxyConcept track_proxy_t>
534bool TrackSelector::MeasurementCounter::isValidTrack(
535 const track_proxy_t& track) const {
536 // No hit cuts, accept everything
537 if (counters.empty()) {
538 return true;
539 }
540
541 boost::container::small_vector<unsigned int, 4> counterValues;
542 counterValues.resize(counters.size(), 0);
543
544 for (const auto& ts : track.trackStatesReversed()) {
545 if (!ts.typeFlags().test(Acts::TrackStateFlag::MeasurementFlag)) {
546 continue;
547 }
548
549 const auto geoId = ts.referenceSurface().geometryId();
550
551 for (std::size_t i = 0; i < counters.size(); i++) {
552 const auto& [counterMap, threshold] = counters[i];
553 if (const auto it = counterMap.find(geoId); it != counterMap.end()) {
554 counterValues[i]++;
555 }
556 }
557 }
558
559 for (std::size_t i = 0; i < counters.size(); i++) {
560 const auto& [counterMap, threshold] = counters[i];
561 const unsigned int value = counterValues[i];
562 if (value < threshold) {
563 return false;
564 }
565 }
566
567 return true;
568}
