50template <typename queries_type, typename expansion_type, std::size_t kDIM>
51void expand(queries_type& gridQueries,
52 const expansion_type& referenceExpansion) {
53 queries_type copiedQueries = gridQueries;
54 // Sort them for smaller bigger
55 std::ranges::sort(copiedQueries, [](const auto& a, const auto& b) {
56 return a[kDIM] < b[kDIM];
57 });
58 // Get a mid point
59 auto midPoint =
60 0.5 * (copiedQueries.front()[kDIM] + copiedQueries.back()[kDIM]);
61 // Loop and correct the first coordinate
62 for (auto& pq : gridQueries) {
63 if (pq[kDIM] < midPoint) {
64 pq[kDIM] -= referenceExpansion[kDIM];
65 } else {
66 pq[kDIM] += referenceExpansion[kDIM];
67 }
68 }
69}

75template <typename grid_type>
76void applyReferenceExpansion(
77 std::vector<typename grid_type::point_t>& gridQueries,
78 const std::vector<double>& referenceExpansion) {
79 if (referenceExpansion.empty()) {
80 return;
81 }
82 if (referenceExpansion.size() != grid_type::DIM) {
83 throw std::runtime_error(
84 "IndexedSurfaceGridFiller: wrong dimension of reference expansion "
85 "given.");
86 }
87 expand<decltype(gridQueries), decltype(referenceExpansion), 0u>(
88 gridQueries, referenceExpansion);
89 if constexpr (grid_type::DIM >= 2u) {
90 expand<decltype(gridQueries), decltype(referenceExpansion), 1u>(
91 gridQueries, referenceExpansion);
92 }
93 if constexpr (grid_type::DIM == 3u) {
94 expand<decltype(gridQueries), decltype(referenceExpansion), 2u>(
95 gridQueries, referenceExpansion);
96 }
97}

111template <typename grid_type>
112std::set<typename grid_type::index_t> localIndices(
113 const grid_type& grid,
114 const std::vector<typename grid_type::point_t>& queries,
115 const std::vector<std::size_t>& expansion = {}) {
116 // Return indices
117 std::set<typename grid_type::index_t> lIndices;
118
119 if (queries.empty()) {
120 throw std::runtime_error("IndexedSurfaceGridFiller: no query point given.");
121 }
122
123 if (!expansion.empty() && expansion.size() != grid_type::DIM) {
124 throw std::runtime_error(
125 "IndexedSurfaceGridFiller: wrong dimension of bin expansion given.");
126 }
127
129 std::array<Acts::AxisBoundaryType, grid_type::DIM> axisTypes{};
130 std::array<std::size_t, grid_type::DIM> axisBins{};
131 // Fill the axis types
132 for (auto [ia, a] : enumerate(grid.axes())) {
133 axisTypes[ia] = a->getBoundaryType();
134 axisBins[ia] = a->getNBins();
135 }
136
137 // Initialize the bin ranges
138 std::array<std::array<std::size_t, 2u>, grid_type::DIM> binRanges = {};
139 for (auto& br : binRanges) {
140 br[0u] = std::numeric_limits<std::size_t>::max();
141 br[1u] = 0u;
142 }
143 // Bin range bounding box - estimated from the query points
144 for (const auto& q : queries) {
145 auto qbin = grid.localBinsFromPosition(q);
146 for (std::size_t ib = 0; ib < grid_type::DIM; ++ib) {
147 auto iqb = qbin[ib];
148 binRanges[ib][0u] = std::min(iqb, binRanges[ib][0u]);
149 binRanges[ib][1u] = std::max(iqb, binRanges[ib][1u]);
150 }
151 }
152 // Fill the bins - 1D case
153 if constexpr (grid_type::DIM == 1u) {
154 // Take the expansion if available & generate the local bin sequence
155 std::size_t expand = expansion.empty() ? 0u : expansion[0u];
156 auto localBins0 =
157 binSequence(binRanges[0u], expand, axisBins[0u], axisTypes[0u]);
158 for (auto l0 : localBins0) {
159 typename grid_type::index_t b;
160 b[0u] = l0;
161 lIndices.insert(b);
162 }
163 }
164 // Fill the bins - 2D case
165 if constexpr (grid_type::DIM == 2u) {
166 // Take the expansion if available & generate the local bin sequence
167 std::size_t expand = expansion.empty() ? 0u : expansion[0u];
168 auto localBins0 =
169 binSequence(binRanges[0u], expand, axisBins[0u], axisTypes[0u]);
170 expand = expansion.empty() ? 0u : expansion[1u];
171 auto localBins1 =
172 binSequence(binRanges[1u], expand, axisBins[1u], axisTypes[1u]);
173 for (auto l0 : localBins0) {
174 for (auto l1 : localBins1) {
175 typename grid_type::index_t b;
176 b[0u] = l0;
177 b[1u] = l1;
178 lIndices.insert(b);
179 }
180 }
181 }
182 return lIndices;
183}

192template <typename local_bin>
193std::string outputIndices(const std::set<local_bin>& lbins) {
194 std::string rString;
195 for (auto [ilb, lb] : Acts::enumerate(lbins)) {
196 if (ilb == 0) {
197 rString = "bins: [";
198 } else {
199 rString += ", [";
200 }
201 for (auto [ib, b] : Acts::enumerate(lb)) {
202 if (ib != 0u) {
203 rString += ", ";
204 }
205 rString += std::to_string(b);
206 }
207 rString += "]";
208 }
209 return rString;
210}

282 template <typename index_grid, typename indexed_objects,
283 typename reference_generator>
284 void fill(
285 const GeometryContext& gctx, index_grid& iGrid,
286 const indexed_objects& iObjects, const reference_generator& rGenerator,
287 const typename index_grid::grid_type::value_type& aToAll = {}) const {
288 // Loop over the surfaces to be filled
289 for (auto [io, o] : enumerate(iObjects)) {
290 // Exclude indices that should be handled differently
291 if (rangeContainsValue(aToAll, io)) {
292 continue;
293 }
294 // Get the reference positions
295 auto refs = rGenerator.references(gctx, *o);
296 std::vector<typename index_grid::grid_type::point_t> gridQueries;
297 gridQueries.reserve(refs.size());
298 for (const auto& ref : refs) {
299 // Cast the transform according to the grid binning
300 gridQueries.push_back(
301 GridAccessHelpers::castPosition<decltype(iGrid.grid)>(
302 iGrid.transform * ref, iGrid.casts));
303 }
304 ACTS_DEBUG(gridQueries.size() << " reference points generated.");
305 // These are now in the grid frame, can be expanded
306 if (!referenceExpansion.empty()) {
307 ACTS_DEBUG("Applying reference expansion.");
308 applyReferenceExpansion<decltype(iGrid.grid)>(gridQueries,
309 referenceExpansion);
310 }
311 // Now generate the local indices
312 auto lIndices = localIndices<decltype(iGrid.grid)>(
313 iGrid.grid, gridQueries, binExpansion);
314 ACTS_DEBUG(lIndices.size() << " indices assigned.");
315 if (oLogger->level() <= Logging::VERBOSE) {
316 ACTS_VERBOSE("- list of indices: " << outputIndices(lIndices));
317 }
318 // Now fill the surface indices
319 for (const auto& li : lIndices) {
320 auto& bContent = iGrid.grid.atLocalBins(li);
321 if (!rangeContainsValue(bContent, io)) {
322 bContent.push_back(io);
323 }
324 }
325 }
326
327 // Assign the indices into all
328 if (!aToAll.empty()) {
329 assignToAll(iGrid, aToAll);
330 }
331 }

338 template <typename index_grid, typename indices>
339 void assignToAll(index_grid& iGrid, const indices& idcs) const {
340 for (std::size_t gi = 0; gi < iGrid.grid.size(true); ++gi) {
341 auto& bContent = iGrid.grid.at(gi);
342 for (const auto& io : idcs) {
343 if (!rangeContainsValue(bContent, io)) {
344 bContent.push_back(io);
345 }
346 }
347 }
348 }
