25struct ImpactParametersAndSigma {
26 // Impact parameters ...
27 double d0 = 0.;
28 double z0 = 0.;
29 // ... and their standard deviations wrt a vertex, e.g.:
30 // sigmaD0 = sqrt(Var(X) + Var(Y) + Var(d0)),
31 // where X and Y are the x- and y-coordinate of the vertex
32 double sigmaD0 = 0.;
33 double sigmaZ0 = 0.;
34 // Absolute difference in time between the vertex and the track at the 2D PCA
35 // ...
36 std::optional<double> deltaT = std::nullopt;
37 // ... and standard deviation wrt a vertex
38 std::optional<double> sigmaDeltaT = std::nullopt;
39};

171 template <int nDim>

173 getDistanceAndMomentum(const GeometryContext& gctx,
174 const BoundTrackParameters& trkParams,
175 const ActsVector<nDim>& vtxPos, State& state) const {
176 static_assert(nDim == 3 || nDim == 4,
177 "Only 3D and 4D vertex positions allowed");
178 auto res =
179 getDistanceAndMomentum(gctx, trkParams, {vtxPos.data(), nDim}, state);
180 if (!res.ok()) {
181 return res.error();
182 }
183 auto& [distance, momentum] = *res;
184 return std::pair{distance.template head<nDim>(), momentum};
