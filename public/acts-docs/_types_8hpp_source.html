61template <std::size_t Size, bool ReadOnlyMaps = true>
62struct FixedSizeTypes {
63 constexpr static auto Flags = Eigen::ColMajor | Eigen::AutoAlign;
64
65 // single items
66 using Coefficients = Eigen::Matrix<double, Size, 1, Flags>;
67 using Covariance = Eigen::Matrix<double, Size, Size, Flags>;
68 using CoefficientsMap = Eigen::Map<const_if_t<ReadOnlyMaps, Coefficients>>;
69 using CovarianceMap = Eigen::Map<const_if_t<ReadOnlyMaps, Covariance>>;
70
71 using DynamicCoefficients = Eigen::Matrix<double, Eigen::Dynamic, 1, Flags>;
72 using DynamicCovariance =
73 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, Flags>;
74 using DynamicCoefficientsMap =
75 Eigen::Map<const_if_t<ReadOnlyMaps, DynamicCoefficients>>;
76 using DynamicCovarianceMap =
77 Eigen::Map<const_if_t<ReadOnlyMaps, DynamicCovariance>>;
78};
