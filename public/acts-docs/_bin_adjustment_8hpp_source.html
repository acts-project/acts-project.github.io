31static inline BinUtility adjustBinUtility(const BinUtility& bu,
32 const RadialBounds& rBounds,
33 const Transform3& transform) {
34 // Default constructor
35 BinUtility uBinUtil(transform);
36
37 // The parameters from the cylinder bounds
38 double minR = rBounds.get(RadialBounds::eMinR);
39 double maxR = rBounds.get(RadialBounds::eMaxR);
40 double minPhi = rBounds.get(RadialBounds::eAveragePhi) -
41 rBounds.get(RadialBounds::eHalfPhiSector);
42 double maxPhi = rBounds.get(RadialBounds::eAveragePhi) +
43 rBounds.get(RadialBounds::eHalfPhiSector);
44 // Retrieve the binning data
45 const std::vector<BinningData>& bData = bu.binningData();
46 // Loop over the binning data and adjust the dimensions
47 for (auto& bd : bData) {
48 // The binning value
49 AxisDirection bval = bd.binvalue;
50 // Throw exceptions is stuff doesn't make sense:
51 // - not the right binning value
52 // - not equidistant
53 if (bd.type == arbitrary) {
54 throw std::invalid_argument("Arbitrary binning can not be adjusted.");
55 } else if (bval != AxisDirection::AxisR && bval != AxisDirection::AxisPhi) {
56 throw std::invalid_argument("Disc binning must be: phi, r");
57 }
58 float min = 0., max = 0.;
59 // Perform the value adjustment
60 if (bval == AxisDirection::AxisPhi) {
61 min = minPhi;
62 max = maxPhi;
63 } else {
64 min = minR;
65 max = maxR;
66 }
67 // Create the updated BinningData
68 BinningData uBinData(bd.option, bval, bd.bins(), min, max);
69 uBinUtil += BinUtility(uBinData);
70 }
71 return uBinUtil;
72}

81static inline BinUtility adjustBinUtility(const BinUtility& bu,
82 const CylinderBounds& cBounds,
83 const Transform3& transform) {
84 // Default constructor
85 BinUtility uBinUtil(transform);
86
87 // The parameters from the cylinder bounds
88 double cR = cBounds.get(CylinderBounds::eR);
89 double cHz = cBounds.get(CylinderBounds::eHalfLengthZ);
90 double avgPhi = cBounds.get(CylinderBounds::eAveragePhi);
91 double halfPhi = cBounds.get(CylinderBounds::eHalfPhiSector);
92 double minPhi = avgPhi - halfPhi;
93 double maxPhi = avgPhi + halfPhi;
94
95 // Retrieve the binning data
96 const std::vector<BinningData>& bData = bu.binningData();
97 // Loop over the binning data and adjust the dimensions
98 for (auto& bd : bData) {
99 // The binning value
100 AxisDirection bval = bd.binvalue;
101 // Throw exceptions if stuff doesn't make sense:
102 // - not the right binning value
103 // - not equidistant
104 if (bd.type == arbitrary) {
105 throw std::invalid_argument("Arbitrary binning can not be adjusted.");
106 } else if (bval != AxisDirection::AxisRPhi &&
107 bval != AxisDirection::AxisPhi && bval != AxisDirection::AxisZ) {
108 throw std::invalid_argument("Cylinder binning must be: rphi, phi, z");
109 }
110 float min = 0., max = 0.;
111 // Perform the value adjustment
112 if (bval == AxisDirection::AxisPhi) {
113 min = minPhi;
114 max = maxPhi;
115 } else if (bval == AxisDirection::AxisRPhi) {
116 min = cR * minPhi;
117 max = cR * maxPhi;
118 } else {
119 min = -cHz;
120 max = cHz;
121 }
122 // Create the updated BinningData
123 BinningData uBinData(bd.option, bval, bd.bins(), min, max);
124 uBinUtil += BinUtility(uBinData);
125 }
126 return uBinUtil;
127}

186static inline BinUtility adjustBinUtility(const BinUtility& bu,
187 const TrapezoidBounds& pBounds,
188 const Transform3& transform) {
189 // Default constructor
190 BinUtility uBinUtil(transform);
191
192 // The parameters from the cylinder bounds
193
194 double halfX = std::max(pBounds.get(Acts::TrapezoidBounds::eHalfLengthXnegY),
195 pBounds.get(Acts::TrapezoidBounds::eHalfLengthXposY));
196 double halfY = pBounds.get(Acts::TrapezoidBounds::eHalfLengthY);
197
198 // Retrieve the binning data
199 const std::vector<BinningData>& bData = bu.binningData();
200 // Loop over the binning data and adjust the dimensions
201 for (auto& bd : bData) {
202 // The binning value
203 AxisDirection bval = bd.binvalue;
204 // Throw exceptions if stuff doesn't make sense:
205 // - not the right binning value
206 // - not equidistant
207 if (bd.type == arbitrary) {
208 throw std::invalid_argument("Arbitrary binning can not be adjusted.");
209 } else if (bval != AxisDirection::AxisX && bval != AxisDirection::AxisY) {
210 throw std::invalid_argument("Rectangle binning must be: x, y. ");
211 }
212 float min = 0., max = 0.;
213 // Perform the value adjustment
214 if (bval == AxisDirection::AxisX) {
215 min = -1 * halfX;
216 max = halfX;
217 } else {
218 min = -1 * halfY;
219 max = halfY;
220 }
221 // Create the updated BinningData
222 BinningData uBinData(bd.option, bval, bd.bins(), min, max);
223 uBinUtil += BinUtility(uBinData);
224 }
225
226 return uBinUtil;
227}

236static inline BinUtility adjustBinUtility(const BinUtility& bu,
237 const Surface& surface,
238 const GeometryContext& gctx) {
239 if (auto b = dynamic_cast<const CylinderBounds*>(&(surface.bounds()));
240 b != nullptr) {
241 return adjustBinUtility(bu, *b, surface.transform(gctx));
242 }
243 if (auto b = dynamic_cast<const RadialBounds*>(&(surface.bounds()));
244 b != nullptr) {
245 return adjustBinUtility(bu, *b, surface.transform(gctx));
246 }
247 if (surface.type() == Surface::Plane) {
248 if (auto b = dynamic_cast<const RectangleBounds*>(&(surface.bounds()));
249 b != nullptr) {
250 return adjustBinUtility(bu, *b, surface.transform(gctx));
251 }
252 if (auto b = dynamic_cast<const TrapezoidBounds*>(&(surface.bounds()));
253 b != nullptr) {
254 return adjustBinUtility(bu, *b, surface.transform(gctx));
255 }
256 }
257
258 std::stringstream ss;
259 ss << surface.toStream({});
260 throw std::invalid_argument(
261 "Bin adjustment not implemented for this surface yet:\n" + ss.str());
262}
