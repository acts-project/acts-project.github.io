73template <typename Derived>
74double perp(const Eigen::MatrixBase<Derived>& v) noexcept {
75 constexpr int rows = Eigen::MatrixBase<Derived>::RowsAtCompileTime;
76 if constexpr (rows != -1) {
77 // static size, do compile time check
78 static_assert(rows >= 2,
79 "Perp function not valid for vectors not at least 2D");
80 } else {
81 // dynamic size
82 assert(v.rows() >= 2 &&
83 "Perp function not valid for vectors not at least 2D");
84 }
85 return v.template head<2>().norm();
86}

118template <typename Derived>
119double eta(const Eigen::MatrixBase<Derived>& v) noexcept
120 requires(Eigen::MatrixBase<Derived>::RowsAtCompileTime == 3)
121{
122 if (v[0] == 0. && v[1] == 0.) {
123 return std::copysign(std::numeric_limits<double>::infinity(), v[2]);
124 } else {
125 return std::asinh(v[2] / perp(v));
126 }
127}

134template <typename Derived>
135double eta(const Eigen::MatrixBase<Derived>& v) noexcept
136 requires(Eigen::MatrixBase<Derived>::RowsAtCompileTime == -1)
137{
138 assert(v.rows() == 3 && "Eta function not valid for non-3D vectors.");
139 if (v[0] == 0. && v[1] == 0.) {
140 return std::copysign(std::numeric_limits<double>::infinity(), v[2]);
141 } else {
142 return std::asinh(v[2] / perp(v));
143 }
144}

151inline std::array<double, 4> evaluateTrigonomics(const Vector3& direction) {
152 const double x = direction(0); // == cos(phi) * sin(theta)
153 const double y = direction(1); // == sin(phi) * sin(theta)
154 const double z = direction(2); // == cos(theta)
155 // can be turned into cosine/sine
156 const double cosTheta = z;
157 const double sinTheta = std::sqrt(1 - z * z);
158 assert(sinTheta != 0 &&
159 "VectorHelpers: Vector is parallel to the z-axis "
160 "which leads to division by zero");
161 const double invSinTheta = 1. / sinTheta;
162 const double cosPhi = x * invSinTheta;
163 const double sinPhi = y * invSinTheta;
164
165 return {cosPhi, sinPhi, cosTheta, sinTheta};
166}

177inline double cast(const Vector3& position, AxisDirection aDir) {
178 using enum AxisDirection;
179 switch (aDir) {
180 case AxisX:
181 return position[0];
182 case AxisY:
183 return position[1];
184 case AxisZ:
185 return position[2];
186 case AxisR:
187 return perp(position);
188 case AxisPhi:
189 return phi(position);
190 case AxisRPhi:
191 return perp(position) * phi(position);
192 case AxisTheta:
193 return theta(position);
194 case AxisEta:
195 return eta(position);
196 case AxisMag:
197 return position.norm();
198 default:
199 assert(false && "Invalid AxisDirection enum value");
200 return std::numeric_limits<double>::quiet_NaN();
201 }
202}

249inline std::pair<double, double> incidentAngles(
250 const Acts::Vector3& direction,
251 const Acts::RotationMatrix3& globalToLocal) {
252 Acts::Vector3 trfDir = globalToLocal * direction;
253 // The angles are defined with respect to the measurement axis
254 // i.e. "head-on" == pi/2, parallel = 0
255 double phi = std::atan2(trfDir[2], trfDir[0]);
256 double theta = std::atan2(trfDir[2], trfDir[1]);
257 return {phi, theta};
258}

266template <typename Derived>
267double deltaR(const Eigen::MatrixBase<Derived>& v1,
268 const Eigen::MatrixBase<Derived>& v2)
269 requires(Eigen::MatrixBase<Derived>::RowsAtCompileTime == 3)
270{
271 const double dphi =
272 detail::difference_periodic(phi(v1), phi(v2), 2 * std::numbers::pi);
273 const double deta = eta(v1) - eta(v2);
274 return fastHypot(dphi, deta);
275}
