33template <typename index_grid>
34nlohmann::json convertImpl(const index_grid& indexGrid, bool detray = false,
35 bool checkSwap = false) {
36 nlohmann::json jIndexedGrid;
37
38 // Axis swapping (detray version)
39 bool swapAxes = checkSwap;
40
41 // Fill the casts
42 nlohmann::json jCasts;
43 // 1D casts
44 if constexpr (index_grid::grid_type::DIM == 1u) {
45 jCasts.push_back(indexGrid.casts[0u]);
46 }
47 // 1D casts
48 if constexpr (index_grid::grid_type::DIM == 2u) {
49 jCasts.push_back(indexGrid.casts[0u]);
50 jCasts.push_back(indexGrid.casts[1u]);
51 // Check for axis swap (detray version)
52 swapAxes = checkSwap && (indexGrid.casts[0u] == AxisDirection::AxisZ &&
53 indexGrid.casts[1u] == AxisDirection::AxisPhi);
54 }
55 jIndexedGrid["casts"] = jCasts;
56 jIndexedGrid["transform"] =
57 Transform3JsonConverter::toJson(indexGrid.transform);
58 if (detray) {
59 jIndexedGrid = GridJsonConverter::toJsonDetray(indexGrid.grid, swapAxes);
60 } else {
61 jIndexedGrid["grid"] = GridJsonConverter::toJson(indexGrid.grid);
62 }
63 return jIndexedGrid;
64}

76template <typename updator_type, typename generator_type>
77updator_type generateFromJson(const nlohmann::json& jUpdater,
78 const std::string& jIndicator) {
79 generator_type generator;
80
81 using ValueType = typename generator_type::value_type;
82
85 auto eqExtractor = [](const nlohmann::json& jAxis)
86 -> std::tuple<std::array<double, 2u>, std::size_t> {
87 std::array<double, 2u> range = jAxis["range"];
88 std::size_t bins = jAxis["bins"];
89 return {range, bins};
90 };
91
94 auto vExtractor = [](const nlohmann::json& jAxis) -> std::vector<double> {
95 std::vector<double> vEx(jAxis["boundaries"]);
96 return vEx;
97 };
98
99 if (jUpdater["type"] != jIndicator ||
100 jUpdater.find("grid") == jUpdater.end()) {
101 // The return object
102 updator_type updator;
103 return updator;
104 }
105
106 // Peek into the json object to understand what to do
107 const Transform3 transform =
108 Transform3JsonConverter::fromJson(jUpdater["transform"]);
109 const auto jGrid = jUpdater["grid"];
110 const auto jCasts = jUpdater["casts"].get<std::vector<AxisDirection>>();
111 const auto jAxes = jGrid["axes"];
112
113 // 1D cases
114 if (jAxes.size() == 1u) {
115 AxisDirection bValue = jCasts[0u];
116 auto jAxis = jAxes[0u];
117
118 AxisType axisType = jAxis["type"];
119 AxisBoundaryType axisBoundaryType = jAxis["boundary_type"];
120
121 // Equidistant axis
122 if (axisType == AxisType::Equidistant) {
123 auto [range, bins] = eqExtractor(jAxis);
124 if (axisBoundaryType == AxisBoundaryType::Closed) {
125 EqClosed ecAG{range, bins};
126 auto grid =
127 GridJsonConverter::fromJson<EqClosed, ValueType>(jGrid, ecAG);
128 return generator.createUpdater(std::move(grid), {bValue}, transform);
129 } else {
130 EqBound ebAG{range, bins};
131 auto grid =
132 GridJsonConverter::fromJson<EqBound, ValueType>(jGrid, ebAG);
133 return generator.createUpdater(std::move(grid), {bValue}, transform);
134 }
135 } else {
136 // Variable type
137 if (axisBoundaryType == AxisBoundaryType::Closed) {
138 VarClosed vcAG{vExtractor(jAxis)};
139 auto grid =
140 GridJsonConverter::fromJson<VarClosed, ValueType>(jGrid, vcAG);
141 return generator.createUpdater(std::move(grid), {bValue}, transform);
142 } else {
143 VarBound vbAG{vExtractor(jAxis)};
144 auto grid =
145 GridJsonConverter::fromJson<VarBound, ValueType>(jGrid, vbAG);
146 return generator.createUpdater(std::move(grid), {bValue}, transform);
147 }
148 }
149 } else if (jAxes.size() == 2u) {
150 // This currently writes out only the main options of 2D grids
151 // nota bene: it assumes if one axis is closed, it is axis B
152
153 AxisDirection bValueA = jCasts[0u];
154 AxisDirection bValueB = jCasts[1u];
155 auto jAxisA = jAxes[0u];
156 auto jAxisB = jAxes[1u];
157
158 AxisType axisTypeA = jAxisA["type"];
159 AxisType axisTypeB = jAxisB["type"];
160 AxisBoundaryType axisBoundaryTypeB = jAxisB["boundary_type"];
161
162 if (axisBoundaryTypeB != AxisBoundaryType::Closed) {
163 // First axis equidistant
164 if (axisTypeA == AxisType::Equidistant) {
165 auto [rangeA, binsA] = eqExtractor(jAxisA);
166 if (axisTypeB == AxisType::Equidistant) {
167 auto [rangeB, binsB] = eqExtractor(jAxisB);
168 EqBoundEqBound ebebAG{rangeA, binsA, rangeB, binsB};
169 auto grid = GridJsonConverter::fromJson<EqBoundEqBound, ValueType>(
170 jGrid, ebebAG);
171 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
172 transform);
173 } else {
174 EqBoundVarBound ebvbAG{rangeA, binsA, vExtractor(jAxisB)};
175 auto grid = GridJsonConverter::fromJson<EqBoundVarBound, ValueType>(
176 jGrid, ebvbAG);
177 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
178 transform);
179 }
180 } else {
181 if (axisTypeB == AxisType::Equidistant) {
182 auto [rangeB, binsB] = eqExtractor(jAxisB);
183 VarBoundEqBound vbebAG{vExtractor(jAxisA), rangeB, binsB};
184 auto grid = GridJsonConverter::fromJson<VarBoundEqBound, ValueType>(
185 jGrid, vbebAG);
186 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
187 transform);
188 } else {
189 VarBoundVarBound vbvbAG{vExtractor(jAxisA), vExtractor(jAxisB)};
190 auto grid = GridJsonConverter::fromJson<VarBoundVarBound, ValueType>(
191 jGrid, vbvbAG);
192 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
193 transform);
194 }
195 }
196 } else {
197 // Closed cases
198 if (axisTypeA == AxisType::Equidistant) {
199 auto [rangeA, binsA] = eqExtractor(jAxisA);
200 if (axisTypeB == AxisType::Equidistant) {
201 auto [rangeB, binsB] = eqExtractor(jAxisB);
202 EqBoundEqClosed ebecAG{rangeA, binsA, rangeB, binsB};
203 auto grid = GridJsonConverter::fromJson<EqBoundEqClosed, ValueType>(
204 jGrid, ebecAG);
205 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
206 transform);
207 } else {
208 EqBoundVarClosed ebvcAG{rangeA, binsA, vExtractor(jAxisB)};
209 auto grid = GridJsonConverter::fromJson<EqBoundVarClosed, ValueType>(
210 jGrid, ebvcAG);
211 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
212 transform);
213 }
214 } else {
215 if (axisTypeB == AxisType::Equidistant) {
216 auto [rangeB, binsB] = eqExtractor(jAxisB);
217 VarBoundEqClosed vbecAG{vExtractor(jAxisA), rangeB, binsB};
218 auto grid = GridJsonConverter::fromJson<VarBoundEqClosed, ValueType>(
219 jGrid, vbecAG);
220 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
221 transform);
222 } else {
223 VarBoundVarClosed vbvcAG{vExtractor(jAxisA), vExtractor(jAxisB)};
224 auto grid = GridJsonConverter::fromJson<VarBoundVarClosed, ValueType>(
225 jGrid, vbvcAG);
226 return generator.createUpdater(std::move(grid), {bValueA, bValueB},
227 transform);
228 }
229 }
230 }
231 }
232
233 // Default return object if jAxes.size() != 1 or 2
234 updator_type updator;
235 return updator;
236}
