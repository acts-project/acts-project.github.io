62 template <bool EnsureConst, typename T>
63 static std::any component_impl(T& instance, HashedString key,
64 IndexType itrack) {
65 using namespace Acts::HashedStringLiteral;
66 if constexpr (EnsureConst) {
67 static_assert(std::is_const_v<std::remove_reference_t<T>>,
68 "Is not const");
69 }
70
71 using namespace Acts::HashedStringLiteral;
72 switch (key) {
73 case "tipIndex"_hash:
74 return &instance.m_tipIndex[itrack];
75 case "stemIndex"_hash:
76 return &instance.m_stemIndex[itrack];
77 case "params"_hash:
78 return &instance.m_params[itrack];
79 case "cov"_hash:
80 return &instance.m_cov[itrack];
81 case "nMeasurements"_hash:
82 return &instance.m_nMeasurements[itrack];
83 case "nHoles"_hash:
84 return &instance.m_nHoles[itrack];
85 case "chi2"_hash:
86 return &instance.m_chi2[itrack];
87 case "ndf"_hash:
88 return &instance.m_ndf[itrack];
89 case "nOutliers"_hash:
90 return &instance.m_nOutliers[itrack];
91 case "nSharedHits"_hash:
92 return &instance.m_nSharedHits[itrack];
93 default:
94 auto it = instance.m_dynamic.find(key);
95 if (it == instance.m_dynamic.end()) {
96 throw std::runtime_error("Unable to handle this component");
97 }
98
99 std::conditional_t<EnsureConst, const detail::DynamicColumnBase*,
100 detail::DynamicColumnBase*>
101 col = it->second.get();
102 assert(col && "Dynamic column is null");
103 return col->get(itrack);
104 }
105 }
106
107 bool checkConsistency() const {
108 std::size_t size = m_tipIndex.size();
109
110 bool result = true;
111 result = result && m_tipIndex.size() == size;
112 assert(result);
113 result = result && m_stemIndex.size() == size;
114 assert(result);
115 result = result && m_particleHypothesis.size() == size;
116 assert(result);
117 result = result && m_params.size() == size;
118 assert(result);
119 result = result && m_cov.size() == size;
120 assert(result);
121 result = result && m_referenceSurfaces.size() == size;
122 assert(result);
123 result = result && m_nMeasurements.size() == size;
124 assert(result);
125 result = result && m_nHoles.size() == size;
126 assert(result);
127 result = result && m_chi2.size() == size;
128 assert(result);
129 result = result && m_ndf.size() == size;
130 assert(result);
131 result = result && m_nOutliers.size() == size;
132 assert(result);
133 result = result && m_nSharedHits.size() == size;
134
135 for (const auto& [key, col] : m_dynamic) {
136 (void)key;
137 result = result && col->size() == size;
138 }
139
140 return result;
141 }
142
143 public:
144 constexpr bool hasColumn_impl(HashedString key) const {
145 using namespace Acts::HashedStringLiteral;
146 switch (key) {
147 case "tipIndex"_hash:
148 case "stemIndex"_hash:
149 case "params"_hash:
150 case "cov"_hash:
151 case "nMeasurements"_hash:
152 case "nHoles"_hash:
153 case "chi2"_hash:
154 case "ndf"_hash:
155 case "nOutliers"_hash:
156 case "nSharedHits"_hash:
157 return true;
158 default:
159 return m_dynamic.contains(key);
160 }
161 }
162
163 const Surface* referenceSurface_impl(IndexType itrack) const {

165 }
166
167 ParticleHypothesis particleHypothesis_impl(IndexType itrack) const {

169 }

171 std::size_t size_impl() const {
172 assert(checkConsistency());
173 return m_tipIndex.size();
174 }
175
176 detail::DynamicKeyRange<detail::DynamicColumnBase> dynamicKeys_impl() const {
177 return {m_dynamic.begin(), m_dynamic.end()};
178 }

180 // END INTERFACE HELPER

182 std::vector<IndexType> m_tipIndex;
183 std::vector<IndexType> m_stemIndex;
184 std::vector<ParticleHypothesis> m_particleHypothesis;
185 std::vector<typename detail_tsp::FixedSizeTypes<eBoundSize>::Coefficients>
186 m_params;
187 std::vector<typename detail_tsp::FixedSizeTypes<eBoundSize>::Covariance>
188 m_cov;

191 std::vector<unsigned int> m_nMeasurements;
192 std::vector<unsigned int> m_nHoles;
193 std::vector<float> m_chi2;
194 std::vector<unsigned int> m_ndf;
195 std::vector<unsigned int> m_nOutliers;
196 std::vector<unsigned int> m_nSharedHits;
197
198 std::unordered_map<HashedString, std::unique_ptr<detail::DynamicColumnBase>>
199 m_dynamic;
200 std::vector<HashedString> m_dynamicKeys;
201};
