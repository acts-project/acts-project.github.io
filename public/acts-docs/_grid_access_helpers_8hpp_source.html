30template <typename cast_container, typename Array, std::size_t... idx>
31void fillCasts(const Vector3& position, const cast_container& globalCasts,
32 Array& ra, std::index_sequence<idx...> /*indices*/) {
33 ((ra[idx] = VectorHelpers::cast(position, globalCasts[idx])), ...);
34}
35
48template <typename grid_type, typename cast_container>
49typename grid_type::point_t castPosition(const Vector3& position,
50 const cast_container& globalCasts) {
51 // Fill the grid point from global
52 typename grid_type::point_t casted{};
53 fillCasts(position, globalCasts, casted,
54 std::make_integer_sequence<std::size_t, grid_type::DIM>{});
55 return casted;
56}

65template <typename Array, typename local_indices, std::size_t... idx>
66void fillLocal(const Vector2& lposition, const local_indices& laccess,
67 Array& ra, std::index_sequence<idx...> /*indices*/) {
68 ((ra[idx] = lposition[laccess[idx]]), ...);
69}

80template <typename grid_type, typename local_indices>
81typename grid_type::point_t accessLocal(const Vector2& lposition,
82 const local_indices& laccess) {
83 if constexpr (grid_type::DIM > 2u) {
84 throw std::invalid_argument(
85 "GridAccessHelper: only 1-D and 2-D grids are possible for local "
86 "access.");
87 }
88 // Fill the grid point from local according to the accessors
89 typename grid_type::point_t accessed{};
90 fillLocal(lposition, laccess, accessed,
91 std::make_integer_sequence<std::size_t, grid_type::DIM>{});
92 return accessed;
93}

111template <typename global_to_grid_local_t>
112class Affine3Transformed : public IGlobalToGridLocal {
113 public:
114 using grid_local_t = typename global_to_grid_local_t::grid_local_t;
115
117 global_to_grid_local_t globalToGridLocal;
118
120 Transform3 transform;
121
126 Affine3Transformed(global_to_grid_local_t g2gl, const Transform3& t)
127 : globalToGridLocal(std::move(g2gl)), transform(t) {}
128
134 typename global_to_grid_local_t::grid_local_t toGridLocal(
135 const Vector3& position) const {
136 return globalToGridLocal.toGridLocal(transform * position);
137 }
138};

143template <AxisDirection... Args>
144class GlobalSubspace : public IGlobalToGridLocal {
145 public:
147 using grid_local_t = std::array<double, sizeof...(Args)>;
148
150 static_assert(sizeof...(Args) > 0,
151 "GlobalSubspace: cannot have an empty binning value list.");
152
154 static_assert(sizeof...(Args) <= 3,
155 "GlobalSubspace: cannot have more than 3 binning values.");
156
157 // Constructor
158 GlobalSubspace() = default;
159
161 static constexpr std::array<AxisDirection, sizeof...(Args)> axisDirs = {
162 Args...};
163
169 grid_local_t toGridLocal(const Vector3& position) const {
170 // Fill the grid point from global
171 grid_local_t glocal{};
172 GridAccessHelpers::fillCasts(
173 position, axisDirs, glocal,
174 std::make_integer_sequence<std::size_t, sizeof...(Args)>{});
175 return glocal;
176 }
177};

179// The bound to grid local transformation, if only access of a subspace
180// is requested
181template <std::size_t... Args>
182class LocalSubspace : public IBoundToGridLocal {
183 public:
184 using grid_local_t = std::array<double, sizeof...(Args)>;
185
187 static_assert(sizeof...(Args) == 1 || sizeof...(Args) == 2,
188 "LocalSubspace: only 1 or 2 accessors are allowed.");
189
191 static_assert(((Args < 2) && ...),
192 "LocalSubspace: local access needs to be 0u or 1u");
193
194 // Constructor
195 LocalSubspace() = default;
196
197 static constexpr std::array<std::size_t, sizeof...(Args)> accessors = {
198 Args...};
199
205 grid_local_t toGridLocal(const Vector2& lposition) const {
206 // Fill the grid point from local according to the accessors
207 grid_local_t accessed{};
208 GridAccessHelpers::fillLocal(
209 lposition, accessors, accessed,
210 std::make_integer_sequence<std::size_t, sizeof...(Args)>{});
211 return accessed;
212 }
213};
