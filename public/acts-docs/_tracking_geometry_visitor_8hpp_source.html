132template <typename Callable>
133consteval bool callableWithAnyMutable() {
134 return std::is_invocable_v<Callable, Surface&> ||
135 std::is_invocable_v<Callable, TrackingVolume&> ||
136 std::is_invocable_v<Callable, BoundarySurfaceT<TrackingVolume>&> ||
137 std::is_invocable_v<Callable, Layer&> ||
138 std::is_invocable_v<Callable, Portal&>;
139}

141template <typename Callable>
142consteval bool callableWithAnyConst() {
143 return std::is_invocable_v<Callable, const Surface&> ||
144 std::is_invocable_v<Callable, const TrackingVolume&> ||
145 std::is_invocable_v<Callable,
146 const BoundarySurfaceT<TrackingVolume>&> ||
147 std::is_invocable_v<Callable, const Layer&> ||
148 std::is_invocable_v<Callable, const Portal&>;
149}
150
151template <typename Callable>
152consteval bool callableWithAny() {
153 return callableWithAnyMutable<Callable>() || callableWithAnyConst<Callable>();
154}
