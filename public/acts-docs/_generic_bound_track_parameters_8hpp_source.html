63 static Result<GenericBoundTrackParameters> create(

65 const Vector4& pos4, const Vector3& dir, double qOverP,
66 std::optional<CovarianceMatrix> cov,

201 std::optional<ActsSquareMatrix<3>> impactParameterCovariance() const {
202 if (!m_cov.has_value()) {
203 return std::nullopt;
204 }
205
206 ActsSquareMatrix<3> ipCov;
207 ipCov.template topLeftCorner<2, 2>() =
208 m_cov.value().template topLeftCorner<2, 2>();
209 ipCov.template block<2, 1>(0, 2) =
210 m_cov.value().template block<2, 1>(0, eBoundTime);
211 ipCov.template block<1, 2>(2, 0) =
212 m_cov.value().template block<1, 2>(eBoundTime, 0);
213 ipCov(2, 2) = m_cov.value()(eBoundTime, eBoundTime);
214 return ipCov;
215 }
