357 [[nodiscard]] Result<void> initialize(State& state, const Vector3& position,
358 const Vector3& direction,
359 Direction propagationDirection) const {
360 (void)propagationDirection;
361
362 ACTS_VERBOSE(volInfo(state) << "Initialization.");
363
364 auto printGeometryVersion = [](auto ver) {
365 using enum TrackingGeometry::GeometryVersion;
366 switch (ver) {
367 case Gen1:
368 return "Gen1";
369 case Gen3:
370 return "Gen3";
371 default:
372 throw std::runtime_error("Unknown geometry version.");
373 }
374 };
375 ACTS_VERBOSE(volInfo(state) << "Geometry version is: "
376 << printGeometryVersion(m_geometryVersion));
377
378 state.reset();
379
380 if (m_geometryVersion == GeometryVersion::Gen3) {
381 // Empirical pre-allocation of candidates for the next navigation
382 // iteration.
383 // @TODO: Make this user configurable through the configuration
384 state.stream.candidates().reserve(50);
385 }
386
387 state.startSurface = state.options.startSurface;
388 state.targetSurface = state.options.targetSurface;
389
390 // @TODO: Implement fast initialization with Gen3. This requires the volume lookup to work properly
391
392 // Fast Navigation initialization for start condition:
393 // - short-cut through object association, saves navigation in the
394 // - geometry and volume tree search for the lowest volume
395 if (state.startSurface != nullptr &&
396 state.startSurface->associatedLayer() != nullptr) {
397 ACTS_VERBOSE(
398 volInfo(state)
399 << "Fast start initialization through association from Surface.");
400
401 state.startLayer = state.startSurface->associatedLayer();
402 state.startVolume = state.startLayer->trackingVolume();
403 } else if (state.startVolume != nullptr) {
404 ACTS_VERBOSE(
405 volInfo(state)
406 << "Fast start initialization through association from Volume.");
407
408 state.startLayer = state.startVolume->associatedLayer(
409 state.options.geoContext, position);
410 } else {
411 ACTS_VERBOSE(volInfo(state)
412 << "Slow start initialization through search.");
413 ACTS_VERBOSE(volInfo(state)
414 << "Starting from position " << toString(position)
415 << " and direction " << toString(direction));
416
417 // current volume and layer search through global search
418 state.startVolume = m_cfg.trackingGeometry->lowestTrackingVolume(
419 state.options.geoContext, position);
420
421 if (state.startVolume != nullptr) {
422 state.startLayer = state.startVolume->associatedLayer(
423 state.options.geoContext, position);
424 } else {
425 ACTS_ERROR(volInfo(state)
426 << "No start volume resolved. Nothing left to do.");
427 state.navigationBreak = true;
428 }
429 }
430
431 state.currentVolume = state.startVolume;
432 state.currentLayer = state.startLayer;
433 state.currentSurface = state.startSurface;
434
435 if (state.currentVolume != nullptr) {
436 ACTS_VERBOSE(volInfo(state) << "Start volume resolved "
437 << state.currentVolume->geometryId());
438
439 if (!state.currentVolume->inside(position,
440 state.options.surfaceTolerance)) {
441 ACTS_DEBUG(
442 volInfo(state)
443 << "We did not end up inside the expected volume. position = "
444 << position.transpose());
445
446 return Result<void>::failure(NavigatorError::NotInsideExpectedVolume);
447 }
448 }
449 if (state.currentLayer != nullptr) {
450 ACTS_VERBOSE(volInfo(state) << "Start layer resolved "
451 << state.currentLayer->geometryId());
452 }
453 if (state.currentSurface != nullptr) {
454 ACTS_VERBOSE(volInfo(state) << "Start surface resolved "
455 << state.currentSurface->geometryId());
456
457 if (!state.currentSurface->isOnSurface(
458 state.options.geoContext, position, direction,
459 BoundaryTolerance::Infinite(), state.options.surfaceTolerance)) {
460 ACTS_DEBUG(volInfo(state)
461 << "We did not end up on the expected surface. surface = "
462 << state.currentSurface->geometryId()
463 << " position = " << position.transpose()
464 << " direction = " << direction.transpose());
465
466 return Result<void>::failure(NavigatorError::NotOnExpectedSurface);
467 }
468 }
469
470 return Result<void>::success();
471 }

482 NavigationTarget nextTarget(State& state, const Vector3& position,
483 const Vector3& direction) const {
484 // Reset the current surface
485 state.currentSurface = nullptr;
486
487 if (inactive(state)) {
488 return NavigationTarget::None();
489 }
490
491 ACTS_VERBOSE(volInfo(state) << "Entering Navigator::nextTarget.");
492
493 NavigationTarget nextTarget = tryGetNextTarget(state, position, direction);
494 if (!nextTarget.isNone()) {
495 return nextTarget;
496 }
497
498 state.reset();
499 ++state.statistics.nRenavigations;
500
501 // We might have punched through a boundary and entered another volume
502 // so we have to reinitialize
503 state.currentVolume = m_cfg.trackingGeometry->lowestTrackingVolume(
504 state.options.geoContext, position);
505
506 if (state.currentVolume == nullptr) {
507 ACTS_VERBOSE(volInfo(state) << "No volume found, stop navigation.");
508 state.navigationBreak = true;
509 return NavigationTarget::None();
510 }
511
512 state.currentLayer = state.currentVolume->associatedLayer(
513 state.options.geoContext, position);
514
515 ACTS_VERBOSE(volInfo(state) << "Resolved volume and layer.");
516
517 // Rerun the targeting
518 nextTarget = tryGetNextTarget(state, position, direction);
519 if (!nextTarget.isNone()) {
520 return nextTarget;
521 }
522
523 ACTS_VERBOSE(
524 volInfo(state)
525 << "No targets found again, we got really lost! Stop navigation.");
526 state.navigationBreak = true;
527 return NavigationTarget::None();
528 }

555 void handleSurfaceReached(State& state, const Vector3& position,
556 const Vector3& direction,
557 const Surface& surface) const {
558 if (inactive(state)) {
559 return;
560 }
561
562 ACTS_VERBOSE(volInfo(state) << "Entering Navigator::handleSurfaceReached.");
563
564 state.currentSurface = &surface;
565
566 ACTS_VERBOSE(volInfo(state)
567 << "Current surface: " << state.currentSurface->geometryId());
568
569 // handling portals in gen3 configuration
570 if (m_geometryVersion == GeometryVersion::Gen3) {
571 if (state.navCandidate().isPortalTarget() &&
572 &state.navCandidate().surface() == &surface) {
573 ACTS_VERBOSE(volInfo(state) << "Handling portal status.");
574
575 // Switch to the next volume using the portal
576 const Portal* portal = &state.navCandidate().portal();
577 auto res = portal->resolveVolume(state.options.geoContext, position,
578 direction);
579 if (!res.ok()) {
580 ACTS_ERROR(volInfo(state)
581 << "Failed to resolve volume through portal: "
582 << res.error().message());
583 return;
584 }
585
586 state.currentVolume = res.value();
587
588 // partial reset
589 state.resetAfterVolumeSwitch();
590
591 if (state.currentVolume != nullptr) {
592 ACTS_VERBOSE(volInfo(state) << "Volume updated.");
593
594 // this is set only for the check target validity since gen3 does not
595 // care
596 state.navigationStage = Stage::surfaceTarget;
597 } else {
598 ACTS_VERBOSE(
599 volInfo(state)
600 << "No more volume to progress to, stopping navigation.");
601 state.navigationBreak = true;
602 }
603 }
604 return;
605 }
606
607 if (state.navigationStage == Stage::surfaceTarget &&
608 &state.navSurface().surface() == &surface) {
609 ACTS_VERBOSE(volInfo(state) << "Handling surface status.");
610
611 return;
612 }
613
614 if (state.navigationStage == Stage::layerTarget &&
615 &state.navLayer().surface() == &surface) {
616 ACTS_VERBOSE(volInfo(state) << "Handling layer status.");
617
618 // Switch to the next layer
619 state.currentLayer = &state.navLayer().layer();
620 state.navigationStage = Stage::surfaceTarget;
621
622 // partial reset
623 state.resetAfterLayerSwitch();
624
625 return;
626 }
627
628 if (state.navigationStage == Stage::boundaryTarget &&
629 &state.navBoundary().surface() == &surface) {
630 ACTS_VERBOSE(volInfo(state) << "Handling boundary status.");
631
632 // Switch to the next volume using the boundary
633 const BoundarySurface* boundary = &state.navBoundary().boundarySurface();
634 assert(boundary != nullptr && "Retrieved boundary surface is nullptr");
635 state.currentVolume = boundary->attachedVolume(state.options.geoContext,
636 position, direction);
637
638 // partial reset
639 state.resetAfterVolumeSwitch();
640
641 if (state.currentVolume != nullptr) {
642 ACTS_VERBOSE(volInfo(state) << "Volume updated.");
643 state.navigationStage = Stage::layerTarget;
644 } else {
645 ACTS_VERBOSE(volInfo(state)
646 << "No more volume to progress to, stopping navigation.");
647 state.navigationBreak = true;
648 }
649
650 return;
651 }
652
653 ACTS_ERROR(volInfo(state) << "Surface reached but unknown state.");
654 }

662 NavigationTarget getNextTargetGen1(State& state, const Vector3& position,
663 const Vector3& direction) const {
664 // Try targeting the surfaces - then layers - then boundaries
665 if (state.navigationStage == Stage::surfaceTarget) {
666 if (!state.navSurfaceIndex.has_value()) {
667 // First time, resolve the surfaces
668 resolveSurfaces(state, position, direction);
669 state.navSurfaceIndex = 0;
670 } else {
671 ++state.navSurfaceIndex.value();
672 }
673 if (state.navSurfaceIndex.value() < state.navSurfaces.size()) {
674 ACTS_VERBOSE(volInfo(state) << "Target set to next surface.");
675 return state.navSurface();
676 } else {
677 // This was the last surface, switch to layers
678 ACTS_VERBOSE(volInfo(state) << "Target layers.");
679 state.navigationStage = Stage::layerTarget;
680 }
681 }
682
683 if (state.navigationStage == Stage::layerTarget) {
684 if (!state.navLayerIndex.has_value()) {
685 // First time, resolve the layers
686 resolveLayers(state, position, direction);
687 state.navLayerIndex = 0;
688 } else {
689 ++state.navLayerIndex.value();
690 }
691 if (state.navLayerIndex.value() < state.navLayers.size()) {
692 ACTS_VERBOSE(volInfo(state) << "Target set to next layer.");
693 return state.navLayer();
694 } else {
695 // This was the last layer, switch to boundaries
696 ACTS_VERBOSE(volInfo(state) << "Target boundaries.");
697 state.navigationStage = Stage::boundaryTarget;
698 }
699 }
700
701 if (state.navigationStage == Stage::boundaryTarget) {
702 if (!state.navBoundaryIndex.has_value()) {
703 // First time, resolve the boundaries
704 resolveBoundaries(state, position, direction);
705 state.navBoundaryIndex = 0;
706 } else {
707 ++state.navBoundaryIndex.value();
708 }
709 if (state.navBoundaryIndex.value() < state.navBoundaries.size()) {
710 ACTS_VERBOSE(volInfo(state) << "Target set to next boundary.");
711 return state.navBoundary();
712 } else {
713 // This was the last boundary, we have to leave the volume somehow,
714 // renavigate
715 ACTS_VERBOSE(volInfo(state)
716 << "Boundary targets exhausted. Renavigate.");
717 return NavigationTarget::None();
718 }
719 }
720
721 ACTS_VERBOSE(volInfo(state)
722 << "Unknown state. No target found. Renavigate.");
723 return NavigationTarget::None();
724 }

757 NavigationTarget tryGetNextTarget(State& state, const Vector3& position,
758 const Vector3& direction) const {
759 // Try different approach to get navigation target for gen1 and gen3
760 // configuration
761
762 // This is common, in gen1 we start by surfaces and in gen3 we always look
763 // for surfaces
764 if (state.navigationStage == Stage::initial) {
765 ACTS_VERBOSE(volInfo(state) << "Target surfaces.");
766 state.navigationStage = Stage::surfaceTarget;
767 }
768
769 if (m_geometryVersion == GeometryVersion::Gen1) {
770 return getNextTargetGen1(state, position, direction);
771
772 } else { // gen3 handling of the next target
773
774 return getNextTargetGen3(state, position, direction);
775 }

786 void resolveCandidates(State& state, const Vector3& position,
787 const Vector3& direction) const {
788 if (state.currentVolume == nullptr) {
789 ACTS_VERBOSE(volInfo(state) << "No volume to resolve candidates.");
790 return;
791 }
792 ACTS_VERBOSE(volInfo(state) << "Searching for compatible candidates.");
793
794 state.stream.reset();
795 AppendOnlyNavigationStream appendOnly{state.stream};
796 NavigationArguments args;
797 args.position = position;
798 args.direction = direction;
799 state.currentVolume->initializeNavigationCandidates(
800 state.options.geoContext, args, appendOnly, logger());
801
802 ACTS_VERBOSE(volInfo(state) << "Found " << state.stream.candidates().size()
803 << " navigation candidates.");
804
805 state.stream.initialize(state.options.geoContext, {position, direction},
806 BoundaryTolerance::None(),
807 state.options.surfaceTolerance);
808
809 ACTS_VERBOSE(volInfo(state)
810 << "Now " << state.stream.candidates().size()
811 << " navigation candidates after initialization");
812
813 state.navCandidates.clear();
814
815 for (auto& candidate : state.stream.candidates()) {
816 if (!detail::checkPathLength(candidate.intersection().pathLength(),
817 state.options.nearLimit,
818 state.options.farLimit, logger())) {
819 continue;
820 }
821
822 state.navCandidates.emplace_back(candidate);
823 }
824
825 // Sort the candidates with the path length
826 std::ranges::sort(state.navCandidates, [](const auto& a, const auto& b) {
827 return a.intersection().pathLength() < b.intersection().pathLength();
828 });
829
830 // Print the navigation candidates
831
832 if (logger().doPrint(Logging::VERBOSE)) {
833 std::ostringstream os;
834 os << "Navigation candidates: " << state.navCandidates.size() << "\n";
835 for (auto& candidate : state.navCandidates) {
836 os << "Candidate: " << candidate.surface().geometryId()
837 << " at path length: " << candidate.intersection().pathLength()
838 << " ";
839 }
840
841 logger().log(Logging::VERBOSE, os.str());
842 }
843 }

852 void resolveSurfaces(State& state, const Vector3& position,
853 const Vector3& direction) const {
854 ACTS_VERBOSE(volInfo(state) << "Searching for compatible surfaces.");
855
856 const Layer* currentLayer = state.currentLayer;
857
858 if (currentLayer == nullptr) {
859 ACTS_VERBOSE(volInfo(state) << "No layer to resolve surfaces.");
860 return;
861 }
862
863 const Surface* layerSurface = &currentLayer->surfaceRepresentation();
864
865 NavigationOptions<Surface> navOpts;
866 navOpts.resolveSensitive = m_cfg.resolveSensitive;
867 navOpts.resolveMaterial = m_cfg.resolveMaterial;
868 navOpts.resolvePassive = m_cfg.resolvePassive;
869 navOpts.startObject = state.currentSurface;
870 navOpts.endObject = state.targetSurface;
871 navOpts.nearLimit = state.options.nearLimit;
872 navOpts.farLimit = state.options.farLimit;
873
874 if (!state.options.externalSurfaces.empty()) {
875 auto layerId = layerSurface->geometryId().layer();
876 auto externalSurfaceRange =
877 state.options.externalSurfaces.equal_range(layerId);
878 navOpts.externalSurfaces.reserve(
879 state.options.externalSurfaces.count(layerId));
880 for (auto itSurface = externalSurfaceRange.first;
881 itSurface != externalSurfaceRange.second; itSurface++) {
882 navOpts.externalSurfaces.push_back(itSurface->second);
883 }
884 }
885
886 // Request the compatible surfaces
887 state.navSurfaces = currentLayer->compatibleSurfaces(
888 state.options.geoContext, position, direction, navOpts);
889 // Sort the surfaces by path length.
890 // Special care is taken for the external surfaces which should always
891 // come first, so they are preferred to be targeted and hit first.
892 std::ranges::sort(state.navSurfaces, [&state](const NavigationTarget& a,
893 const NavigationTarget& b) {
894 // Prefer to sort by path length. We assume surfaces are at the same
895 // distance if the difference is smaller than the tolerance.
896 if (std::abs(a.pathLength() - b.pathLength()) >
897 state.options.surfaceTolerance) {
898 return NavigationTarget::pathLengthOrder(a, b);
899 }
900 // If the path length is practically the same, sort by geometry.
901 // First we check if one of the surfaces is external.
902 bool aIsExternal = a.boundaryTolerance().isInfinite();
903 bool bIsExternal = b.boundaryTolerance().isInfinite();
904 if (aIsExternal == bIsExternal) {
905 // If both are external or both are not external, sort by geometry
906 // identifier
907 return a.surface().geometryId() < b.surface().geometryId();
908 }
909 // If only one is external, it should come first
910 return aIsExternal;
911 });
912 // For now we implicitly remove overlapping surfaces.
913 // For track finding it might be useful to discover overlapping surfaces
914 // and check for compatible measurements. This is under investigation
915 // and might be implemented in the future.
916 auto toBeRemoved = std::ranges::unique(
917 state.navSurfaces, [&](const auto& a, const auto& b) {
918 return std::abs(a.pathLength() - b.pathLength()) <
919 state.options.surfaceTolerance;
920 });
921 if (toBeRemoved.begin() != toBeRemoved.end()) {
922 ACTS_VERBOSE(volInfo(state)
923 << "Removing "
924 << std::distance(toBeRemoved.begin(), toBeRemoved.end())
925 << " overlapping surfaces.");
926 }
927 state.navSurfaces.erase(toBeRemoved.begin(), toBeRemoved.end());
928
929 // Print surface information
930 if (logger().doPrint(Logging::VERBOSE)) {
931 std::ostringstream os;
932 os << state.navSurfaces.size();
933 os << " surface candidates found at path(s): ";
934 for (auto& sfc : state.navSurfaces) {
935 os << sfc.pathLength() << " ";
936 }
937 logger().log(Logging::VERBOSE, os.str());
938 }
939
940 if (state.navSurfaces.empty()) {
941 ACTS_VERBOSE(volInfo(state) << "No surface candidates found.");
942 }
943 }

952 void resolveLayers(State& state, const Vector3& position,
953 const Vector3& direction) const {
954 ACTS_VERBOSE(volInfo(state) << "Searching for compatible layers.");
955
956 NavigationOptions<Layer> navOpts;
957 navOpts.resolveSensitive = m_cfg.resolveSensitive;
