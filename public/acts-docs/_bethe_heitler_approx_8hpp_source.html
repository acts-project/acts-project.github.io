22struct GaussianComponent {
23 double weight = 0;
24 double mean = 0;
25 double var = 0;
26};
27
30inline GaussianComponent transformComponent(const GaussianComponent &cmp) {
31 GaussianComponent transformed_cmp;
32 transformed_cmp.weight = std::log(cmp.weight) - std::log(1 - cmp.weight);
33 transformed_cmp.mean = std::log(cmp.mean) - std::log(1 - cmp.mean);
34 transformed_cmp.var = std::log(cmp.var);
35 return transformed_cmp;
36}
37
40inline GaussianComponent inverseTransformComponent(
41 const GaussianComponent &transformed_cmp) {
42 GaussianComponent cmp;
43 cmp.weight = 1 / (1 + std::exp(-transformed_cmp.weight));
44 cmp.mean = 1 / (1 + std::exp(-transformed_cmp.mean));
45 cmp.var = std::exp(transformed_cmp.var);
46 return cmp;
47}

93 std::span<Component> mixture(
94 const double xOverX0, const std::span<Component> mixture) const override {
95 mixture[0].weight = 1.0;
96
97 const double c = xOverX0 / std::numbers::ln2;
98 mixture[0].mean = std::pow(2, -c);
99 mixture[0].var = std::pow(3, -c) - std::pow(4, -c);

101 return mixture;
