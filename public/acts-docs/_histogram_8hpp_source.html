156template <std::size_t Dim>
157class Efficiency {
158 public:
164 Efficiency(std::string name, std::string title,
165 const std::array<AxisVariant, Dim>& axes)
166 : m_name(std::move(name)),
167 m_title(std::move(title)),
168 m_accepted(boost::histogram::make_histogram(axes.begin(), axes.end())),
169 m_total(boost::histogram::make_histogram(axes.begin(), axes.end())) {}
170
175 void fill(const std::array<double, Dim>& values, bool accepted) {
176 std::apply(
177 [&](auto... v) {
178 m_total(v...);
179 if (accepted) {
180 m_accepted(v...);
181 }
182 },
183 std::tuple_cat(values));
184 }
185
187 const std::string& name() const { return m_name; }
188
190 const std::string& title() const { return m_title; }
191
193 static constexpr std::size_t rank() { return Dim; }
194
196 const BoostHist& acceptedHistogram() const { return m_accepted; }
197
199 const BoostHist& totalHistogram() const { return m_total; }
200
201 private:
202 std::string m_name;
203 std::string m_title;
204
205 BoostHist m_accepted;
206 BoostHist m_total;
207};
