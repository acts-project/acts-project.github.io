111inline double PhotonConversion::screenFunction1(double delta) const {
112 // Compute the value of the screening function 3*PHI1(delta) - PHI2(delta)
113 return (delta > 1.4) ? 42.038 - 8.29 * std::log(delta + 0.958)
114 : 42.184 - delta * (7.444 - 1.623 * delta);
115}
116
117inline double PhotonConversion::screenFunction2(double delta) const {
118 // Compute the value of the screening function 1.5*PHI1(delta)
119 // +0.5*PHI2(delta)
120 return (delta > 1.4) ? 42.038 - 8.29 * std::log(delta + 0.958)
121 : 41.326 - delta * (5.848 - 0.902 * delta);
122}

124template <typename generator_t>
125std::pair<double, double> PhotonConversion::generatePathLimits(
126 generator_t& generator, const Particle& particle) const {
128
129 // Fast exit if not a photon or the energy is too low
130 if (particle.pdg() != Acts::PdgParticle::eGamma ||
131 particle.absoluteMomentum() < (2 * kElectronMass)) {
132 return {std::numeric_limits<double>::infinity(),
133 std::numeric_limits<double>::infinity()};
134 }
135
136 // Use for the moment only Al data - Yung Tsai - Rev.Mod.Particle Physics Vol.
137 // 46, No.4, October 1974 optainef from a fit given in the momentum range 100
138 // 10 6 2 1 0.6 0.4 0.2 0.1 GeV
139
141 // Double_t fitFunction(Double_t *x, Double_t *par) {
142 // return par[0] + par[1]*pow(x[0],par[2]);
143 // }
144 // EXT PARAMETER STEP FIRST
145 // NO. NAME VALUE ERROR SIZE DERIVATIVE
146 // 1 p0 -7.01612e-03 8.43478e-01 1.62766e-04 1.11914e-05
147 // 2 p1 7.69040e-02 1.00059e+00 8.90718e-05 -8.41167e-07
148 // 3 p2 -6.07682e-01 5.13256e+00 6.07228e-04 -9.44448e-07
149 constexpr double p0 = -7.01612e-03;
150 constexpr double p1 = 7.69040e-02;
151 constexpr double p2 = -6.07682e-01;
152
153 // Calculate xi
154 const double xi = p0 + p1 * std::pow(particle.absoluteMomentum(), p2);
155
156 std::uniform_real_distribution<double> uniformDistribution{0., 1.};
157 // This is a transformation of eq. 3.75
158 return {-9. / 7. *
159 std::log(conversionProbScaleFactor *
160 (1 - uniformDistribution(generator))) /
161 (1. - xi),
162 std::numeric_limits<double>::infinity()};
163}

165template <typename generator_t>
166double PhotonConversion::generateFirstChildEnergyFraction(
167 generator_t& generator, double gammaMom) const {
169
171 //
172 // Compute Coulomb correction factor (Phys Rev. D50 3-1 (1994) page 1254)
173 constexpr double k1 = 0.0083;
174 constexpr double k2 = 0.20206;
175 constexpr double k3 = 0.0020; // This term is missing in Athena
176 constexpr double k4 = 0.0369;
177 constexpr double alphaEM = 1. / 137.;
178 constexpr double m_Z = 13.; // Aluminium
179 constexpr double az2 = (alphaEM * m_Z) * (alphaEM * m_Z);
180 constexpr double az4 = az2 * az2;
181 constexpr double coulombFactor =
182 (k1 * az4 + k2 + 1. / (1. + az2)) * az2 - (k3 * az4 + k4) * az4;
183
184 const double logZ13 = std::log(m_Z) * 1. / 3.;
185 const double FZ = 8. * (logZ13 + coulombFactor);
186 const double deltaMax = std::exp((42.038 - FZ) * 0.1206) - 0.958;
187
188 const double deltaPreFactor = 136. / std::pow(m_Z, 1. / 3.);
189 const double eps0 = kElectronMass / gammaMom;
190 const double deltaFactor = deltaPreFactor * eps0;
191 const double deltaMin = 4. * deltaFactor;
192
193 // Compute the limits of eps
194 const double epsMin =
195 std::max(eps0, 0.5 - 0.5 * std::sqrt(1. - deltaMin / deltaMax));
196 const double epsRange = 0.5 - epsMin;
197
198 // Sample the energy rate (eps) of the created electron (or positron)
199 const double F10 = screenFunction1(deltaMin) - FZ;
200 const double F20 = screenFunction2(deltaMin) - FZ;
201 const double NormF1 = F10 * epsRange * epsRange;
202 const double NormF2 = 1.5 * F20;
203
204 // We will need 3 uniform random number for each trial of sampling
205 double greject = 0.;
206 double eps = 0.;
207 std::uniform_real_distribution<double> rndmEngine;
208 do {
209 if (NormF1 > rndmEngine(generator) * (NormF1 + NormF2)) {
210 eps = 0.5 - epsRange * std::pow(rndmEngine(generator), 1. / 3.);
211 const double delta = deltaFactor / (eps * (1. - eps));
212 greject = (screenFunction1(delta) - FZ) / F10;
213 } else {
214 eps = epsMin + epsRange * rndmEngine(generator);
215 const double delta = deltaFactor / (eps * (1. - eps));
216 greject = (screenFunction2(delta) - FZ) / F20;
217 }
218 } while (greject < rndmEngine(generator));
219 // End of eps sampling
220 return eps * childEnergyScaleFactor;
221}

223template <typename generator_t>
224Acts::Vector3 PhotonConversion::generateChildDirection(
225 generator_t& generator, const Particle& particle) const {
227
228 // Following the Geant4 approximation from L. Urban
229 // the azimutal angle
230 double theta = kElectronMass / particle.energy();
231
232 std::uniform_real_distribution<double> uniformDistribution{0., 1.};
233 const double u = -std::log(uniformDistribution(generator) *
234 uniformDistribution(generator)) *
235 1.6;
236
237 theta *= (uniformDistribution(generator) < 0.25)
238 ? u
239 : u * 1. / 3.; // 9./(9.+27) = 0.25
240
241 // draw the random orientation angle
242 const auto psi = std::uniform_real_distribution<double>(
243 -std::numbers::pi, std::numbers::pi)(generator);
244
245 Acts::Vector3 direction = particle.direction();
246 // construct the combined rotation to the scattered direction
247 Acts::RotationMatrix3 rotation(
248 // rotation of the scattering deflector axis relative to the reference
249 Acts::AngleAxis3(psi, direction) *
250 // rotation by the scattering angle around the deflector axis
251 Acts::AngleAxis3(theta, Acts::createCurvilinearUnitU(direction)));
252 direction.applyOnTheLeft(rotation);
253 return direction;
254}

256inline std::array<Particle, 2> PhotonConversion::generateChildren(
257 const Particle& photon, double childEnergy,
258 const Acts::Vector3& childDirection) const {
259 using namespace Acts::UnitLiterals;
260
261 // Calculate the child momentum
262 const double massChild = kElectronMass;
263 const double momentum1 =
264 std::sqrt(childEnergy * childEnergy - massChild * massChild);
265
266 // Use energy-momentum conservation for the other child
267 const Acts::Vector3 vtmp =
268 photon.fourMomentum().template segment<3>(Acts::eMom0) -
269 momentum1 * childDirection;
270 const double momentum2 = vtmp.norm();
271
272 // The daughter particles are created with the explicit electron mass used in
273 // the calculations for consistency. Using the full Particle constructor with
274 // charge and mass also avoids an additional lookup in the internal data
275 // tables.
276 std::array<Particle, 2> children = {
277 Particle(photon.particleId().makeDescendant(0), Acts::eElectron, -1_e,
278 kElectronMass)
279 .setPosition4(photon.fourPosition())
280 .setDirection(childDirection)
281 .setAbsoluteMomentum(momentum1)
282 .setProcess(ProcessType::ePhotonConversion)
283 .setReferenceSurface(photon.referenceSurface()),
284 Particle(photon.particleId().makeDescendant(1), Acts::ePositron, 1_e,
285 kElectronMass)
286 .setPosition4(photon.fourPosition())
287 .setDirection(childDirection)
288 .setAbsoluteMomentum(momentum2)
289 .setProcess(ProcessType::ePhotonConversion)
290 .setReferenceSurface(photon.referenceSurface()),
291 };
292 return children;
293}

295template <typename generator_t>
296bool PhotonConversion::run(generator_t& generator, Particle& particle,
297 std::vector<Particle>& generated) const {
298 // Fast exit if particle is not a photon
299 if (particle.pdg() != Acts::PdgParticle::eGamma) {
300 return false;
301 }
302
303 // Fast exit if momentum is too low
304 const double p = particle.absoluteMomentum();
305 if (p < (2 * kElectronMass)) {
306 return false;
307 }
308
309 // Get one child energy
310 const double childEnergy = p * generateFirstChildEnergyFraction(generator, p);
311
312 // Now get the deflection
313 const Acts::Vector3 childDir = generateChildDirection(generator, particle);
314
315 // Produce the final state
316 const std::array<Particle, 2> finalState =
317 generateChildren(particle, childEnergy, childDir);
318 generated.insert(generated.end(), finalState.begin(), finalState.end());
319
320 return true;
321}
