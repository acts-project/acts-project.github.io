65template <typename traj_t>
66TrajectoryState trajectoryState(const traj_t& multiTraj, std::size_t tipIndex) {
67 TrajectoryState trajState;
68 multiTraj.visitBackwards(tipIndex, [&](const auto& state) {
69 // Get the volume Id of this surface
70 const auto& geoID = state.referenceSurface().geometryId();
71 const auto& volume = geoID.volume();
72 const auto& layer = geoID.layer();
73 trajState.nStates++;
74 auto typeFlags = state.typeFlags();
75 if (typeFlags.test(Acts::TrackStateFlag::HoleFlag)) {
76 trajState.nHoles++;
77 } else if (typeFlags.test(Acts::TrackStateFlag::OutlierFlag)) {
78 trajState.nOutliers++;
79 trajState.outlierChi2.push_back(state.chi2());
80 trajState.outlierVolume.push_back(volume);
81 trajState.outlierLayer.push_back(layer);
82 } else if (typeFlags.test(Acts::TrackStateFlag::MeasurementFlag)) {
83 if (typeFlags.test(Acts::TrackStateFlag::SharedHitFlag)) {
84 trajState.nSharedHits++;
85 }
86 trajState.nMeasurements++;
87 trajState.measurementChi2.push_back(state.chi2());
88 trajState.measurementVolume.push_back(volume);
89 trajState.measurementLayer.push_back(layer);
90 trajState.chi2Sum += state.chi2();
91 trajState.NDF += state.calibratedSize();
92 }
93 });
94 return trajState;
95}

108template <typename traj_t>
109VolumeTrajectoryStateContainer trajectoryState(
110 const traj_t& multiTraj, std::size_t tipIndex,
111 const std::vector<GeometryIdentifier::Value>& volumeIds) {
112 VolumeTrajectoryStateContainer trajStateContainer;
113 multiTraj.visitBackwards(tipIndex, [&](const auto& state) {
114 // Get the volume Id of this surface
115 const auto& geoID = state.referenceSurface().geometryId();
116 const auto& volume = geoID.volume();
117 const auto& layer = geoID.layer();
118 // Check if the track info for this sub-detector is requested
119 if (!rangeContainsValue(volumeIds, volume)) {
120 return true;
121 }
122 // The trajectory state for this volume
123 auto& trajState = trajStateContainer[volume];
124 trajState.nStates++;
125 trajState.NDF += state.calibratedSize();
126 auto typeFlags = state.typeFlags();
127 if (typeFlags.test(Acts::TrackStateFlag::HoleFlag)) {
128 trajState.nHoles++;
129 } else if (typeFlags.test(Acts::TrackStateFlag::OutlierFlag)) {
130 trajState.nOutliers++;
131 trajState.outlierChi2.push_back(state.chi2());
132 trajState.outlierVolume.push_back(volume);
133 trajState.outlierLayer.push_back(layer);
134 } else if (typeFlags.test(Acts::TrackStateFlag::MeasurementFlag)) {
135 if (typeFlags.test(Acts::TrackStateFlag::SharedHitFlag)) {
136 trajState.nSharedHits++;
137 }
138 trajState.nMeasurements++;
139 trajState.measurementChi2.push_back(state.chi2());
140 trajState.measurementVolume.push_back(volume);
141 trajState.measurementLayer.push_back(layer);
142 trajState.chi2Sum += state.chi2();
143 }
144 return true;
145 });
146 return trajStateContainer;
147}
