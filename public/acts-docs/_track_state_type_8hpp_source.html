40class TrackStateType {

53 explicit TrackStateType(raw_type& raw) : m_raw{&raw} {
54 assert(m_raw != nullptr);
55 }
56
57 // Delete copy constructor
58 TrackStateType(const TrackStateType&) = delete;
59
60 // Delete move constructor
61 TrackStateType(TrackStateType&&) = delete;
62
63 // Disable assignment
64 TrackStateType& operator=(const TrackStateType&) = delete;
65
66 // Disable move assignment
67 TrackStateType& operator=(TrackStateType&&) = delete;
68
72 TrackStateType& operator=(const ConstTrackStateType& other);
73
77 bool test(std::size_t pos) const {
78 std::bitset<kRawBits> bs{*m_raw};
79 return bs.test(pos);
80 }
81
85 void set(std::size_t pos, bool value = true) {
86 std::bitset<kRawBits> bs{*m_raw};
87 bs.set(pos, value);
88 *m_raw = bs.to_ullong();
89 }
90
93 void reset(std::size_t pos) { set(pos, false); }
94
95 friend std::ostream& operator<<(std::ostream& os, const TrackStateType& t) {
96 std::bitset<kRawBits> bs{*t.m_raw};
97 std::bitset<TrackStateFlag::NumTrackStateFlags> trunc;
98 for (std::size_t i = 0; i < TrackStateFlag::NumTrackStateFlags; i++) {
99 trunc[i] = bs[i];
100 }
101 // SharedhitMaterialHoleOutlierParameterMeasurement
102 os << "SMHOPM=" << trunc;
103 return os;
104 }

106 private:
107 raw_type* m_raw{nullptr};
108};
