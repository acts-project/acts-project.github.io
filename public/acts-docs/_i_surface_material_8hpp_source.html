142inline double ISurfaceMaterial::factor(Direction pDir,
143 MaterialUpdateStage mStage) const {
144 if (mStage == Acts::MaterialUpdateStage::FullUpdate) {
145 return 1.;
146 } else if (mStage == Acts::MaterialUpdateStage::PreUpdate) {
147 return pDir == Direction::Negative() ? m_splitFactor : 1 - m_splitFactor;
148 } else /*if (mStage == Acts::MaterialUpdateStage::PostUpdate)*/ {
149 return pDir == Direction::Positive() ? m_splitFactor : 1 - m_splitFactor;
150 }
151}

153inline MaterialSlab ISurfaceMaterial::materialSlab(
154 const Vector2& lp, Direction pDir, MaterialUpdateStage mStage) const {
155 // The plain material properties associated to this bin
156 MaterialSlab plainMatProp = materialSlab(lp);
157 // Scale if you have material to scale
158 if (!plainMatProp.isVacuum()) {
159 double scaleFactor = factor(pDir, mStage);
160 if (scaleFactor == 0.) {
161 return MaterialSlab::Nothing();
162 }
163 plainMatProp.scaleThickness(scaleFactor);
164 }
165 return plainMatProp;
166}
