43struct MaxMomentumComponent {
44 template <typename component_range_t>
45 auto operator()(const component_range_t& cmps) const {
46 return std::max_element(cmps.begin(), cmps.end(),
47 [&](const auto& a, const auto& b) {
48 return std::abs(a.state.pars[eFreeQOverP]) >
49 std::abs(b.state.pars[eFreeQOverP]);
50 });
51 }
52};
53
54struct MaxWeightComponent {
55 template <typename component_range_t>
56 auto operator()(const component_range_t& cmps) {
57 return std::max_element(
58 cmps.begin(), cmps.end(),
59 [&](const auto& a, const auto& b) { return a.weight < b.weight; });
60 }
61};

63template <typename component_chooser_t>
64struct SingleComponentReducer {
65 template <typename stepper_state_t>
66 static Vector3 position(const stepper_state_t& s) {
67 return component_chooser_t{}(s.components)
68 ->state.pars.template segment<3>(eFreePos0);
69 }
70
71 template <typename stepper_state_t>
72 static Vector3 direction(const stepper_state_t& s) {
73 return component_chooser_t{}(s.components)
74 ->state.pars.template segment<3>(eFreeDir0);
75 }
76
77 template <typename stepper_state_t>
78 static double qOverP(const stepper_state_t& s) {
79 const auto cmp = component_chooser_t{}(s.components);
80 return cmp->state.pars[eFreeQOverP];
81 }
82
83 template <typename stepper_state_t>
84 static double absoluteMomentum(const stepper_state_t& s) {
85 const auto cmp = component_chooser_t{}(s.components);
86 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]);
87 }
88
89 template <typename stepper_state_t>
90 static Vector3 momentum(const stepper_state_t& s) {
91 const auto cmp = component_chooser_t{}(s.components);
92 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]) *
93 cmp->state.pars.template segment<3>(eFreeDir0);
94 }
95
96 template <typename stepper_state_t>
97 static double charge(const stepper_state_t& s) {
98 const auto cmp = component_chooser_t{}(s.components);
99 return s.particleHypothesis.extractCharge(cmp->state.pars[eFreeQOverP]);
100 }
101
102 template <typename stepper_state_t>
103 static double time(const stepper_state_t& s) {
104 return component_chooser_t{}(s.components)->state.pars[eFreeTime];
105 }
106
107 template <typename stepper_state_t>
108 static FreeVector pars(const stepper_state_t& s) {
109 return component_chooser_t{}(s.components)->state.pars;
110 }
111
112 template <typename stepper_state_t>
113 static FreeVector cov(const stepper_state_t& s) {
114 return component_chooser_t{}(s.components)->state.cov;
115 }

269 void initialize(State& state,
270 const MultiComponentBoundTrackParameters& par) const {
271 if (par.components().empty()) {
272 throw std::invalid_argument(
273 "Cannot construct MultiEigenStepperLoop::State with empty "
274 "multi-component parameters");
275 }
276
277 state.particleHypothesis = par.particleHypothesis();
278
279 const auto surface = par.referenceSurface().getSharedPtr();
280
281 for (auto i = 0ul; i < par.components().size(); ++i) {
282 const auto& [weight, singlePars] = par[i];
283 auto& cmp =
284 state.components.emplace_back(SingleStepper::makeState(state.options),
285 weight, IntersectionStatus::onSurface);
286 SingleStepper::initialize(cmp.state, singlePars);
287 }
288
289 if (std::get<2>(par.components().front())) {
290 state.covTransport = true;
291 }
292 }

394 state.components.begin(), state.components.end(), [](const auto& cmp) {

404 void reweightComponents(State& state) const {
405 double sumOfWeights = 0.0;
406 for (const auto& cmp : state.components) {
407 sumOfWeights += cmp.weight;
408 }
409 for (auto& cmp : state.components) {
410 cmp.weight /= sumOfWeights;
411 }

528 IntersectionStatus updateSurfaceStatus(
529 State& state, const Surface& surface, std::uint8_t index,
530 Direction navDir, const BoundaryTolerance& boundaryTolerance,
531 double surfaceTolerance, ConstrainedStep::Type stype,
532 const Logger& logger = getDummyLogger()) const {
533 using Status = IntersectionStatus;
534
535 std::array<int, 3> counts = {0, 0, 0};
536
537 for (auto& component : state.components) {
538 component.status = detail::updateSingleSurfaceStatus<SingleStepper>(
539 *this, component.state, surface, index, navDir, boundaryTolerance,
540 surfaceTolerance, stype, logger);
541 ++counts[static_cast<std::size_t>(component.status)];
542 }
543
544 // If at least one component is on a surface, we can remove all missed
545 // components before the step. If not, we must keep them for the case that
546 // all components miss and we need to retarget
547 if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
548 removeMissedComponents(state);
549 reweightComponents(state);
550 }
551
552 ACTS_VERBOSE("Component status wrt "
553 << surface.geometryId() << " at {"
554 << surface.center(state.options.geoContext).transpose()
555 << "}:\t" << [&]() {
556 std::stringstream ss;
557 for (auto& component : state.components) {
558 ss << component.status << "\t";
559 }
560 return ss.str();
561 }());
562
563 // Switch on stepCounter if one or more components reached a surface, but
564 // some are still in progress of reaching the surface
565 if (!state.stepCounterAfterFirstComponentOnSurface &&
566 counts[static_cast<std::size_t>(Status::onSurface)] > 0 &&
567 counts[static_cast<std::size_t>(Status::reachable)] > 0) {
568 state.stepCounterAfterFirstComponentOnSurface = 0;
569 ACTS_VERBOSE("started stepCounterAfterFirstComponentOnSurface");
570 }
571
572 // If there are no components onSurface, but the counter is switched on
573 // (e.g., if the navigator changes the target surface), we need to switch it
574 // off again
575 if (state.stepCounterAfterFirstComponentOnSurface &&
576 counts[static_cast<std::size_t>(Status::onSurface)] == 0) {
577 state.stepCounterAfterFirstComponentOnSurface.reset();
578 ACTS_VERBOSE("switch off stepCounterAfterFirstComponentOnSurface");
579 }
580
581 // This is a 'any_of' criterium. As long as any of the components has a
582 // certain state, this determines the total state (in the order of a
583 // somewhat importance)
584 if (counts[static_cast<std::size_t>(Status::reachable)] > 0) {
585 return Status::reachable;
586 } else if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
587 state.stepCounterAfterFirstComponentOnSurface.reset();
588 return Status::onSurface;
589 } else {
590 return Status::unreachable;
591 }
592 }

605 template <typename object_intersection_t>
606 void updateStepSize(State& state, const object_intersection_t& oIntersection,
607 Direction direction, ConstrainedStep::Type stype) const {
608 const Surface& surface = *oIntersection.object();
609
610 for (auto& component : state.components) {
611 auto intersection = surface.intersect(
612 component.state.options.geoContext,
613 SingleStepper::position(component.state),
614 direction * SingleStepper::direction(component.state),
615 BoundaryTolerance::None())[oIntersection.index()];
616
617 SingleStepper::updateStepSize(component.state, intersection, direction,
618 stype);
619 }
620 }
