42template <Experimental::CompositeSpacePointPtr Point_t>
43SquareMatrix2 betaMatrix(const std::array<Point_t, 4>& layerQuartett) {
44 SquareMatrix2 bMatrix{SquareMatrix2::Identity()};
45
46 Vector3 b1Aterm = (layerQuartett[3]->sensorDirection().dot(
47 layerQuartett[1]->sensorDirection())) *
48 layerQuartett[1]->sensorDirection() -
49 layerQuartett[3]->sensorDirection();
50 Vector3 b1Gterm = (layerQuartett[3]->sensorDirection().dot(
51 layerQuartett[0]->sensorDirection())) *
52 layerQuartett[0]->sensorDirection() -
53 (layerQuartett[3]->sensorDirection().dot(
54 layerQuartett[1]->sensorDirection())) *
55 layerQuartett[1]->sensorDirection();
56
57 Vector3 b2Aterm = layerQuartett[2]->sensorDirection() -
58 (layerQuartett[2]->sensorDirection().dot(
59 layerQuartett[1]->sensorDirection())) *
60 layerQuartett[1]->sensorDirection();
61 Vector3 b2Kterm = (layerQuartett[2]->sensorDirection().dot(
62 layerQuartett[1]->sensorDirection())) *
63 layerQuartett[1]->sensorDirection() -
64 (layerQuartett[2]->sensorDirection().dot(
65 layerQuartett[0]->sensorDirection())) *
66 layerQuartett[0]->sensorDirection();
67
68 // get the distances of the layers along z direction
69 double A = (layerQuartett[0]->localPosition().z() -
70 layerQuartett[1]->localPosition().z());
71 double G = (layerQuartett[0]->localPosition().z() -
72 layerQuartett[2]->localPosition().z());
73 double K = (layerQuartett[0]->localPosition().z() -
74 layerQuartett[3]->localPosition().z());
75
76 // define B matrix
77 Vector3 b1 = A * b1Aterm + G * b1Gterm;
78 Vector3 b2 = K * b2Kterm + A * b2Aterm;
79
80 bMatrix.col(0) = Vector2(b1.x(), b1.y());
81 bMatrix.col(1) = Vector2(b2.x(), b2.y());
82
83 return bMatrix;
84}

91template <Experimental::CompositeSpacePointPtr Point_t>
92std::array<double, 4> defineParameters(
93 const SquareMatrix2& betaMatrix,
94 const std::array<Point_t, 4>& layerQuartett) {
95 double A = (layerQuartett[0]->localPosition().z() -
96 layerQuartett[1]->localPosition().z());
97 double G = (layerQuartett[0]->localPosition().z() -
98 layerQuartett[2]->localPosition().z());
99 double K = (layerQuartett[0]->localPosition().z() -
100 layerQuartett[3]->localPosition().z());
101
102 // Define y2 for the linear system
103 Vector3 y0 = K * (layerQuartett[2]->localPosition() -
104 layerQuartett[0]->localPosition()) +
105 G * (layerQuartett[0]->localPosition() -
106 layerQuartett[3]->localPosition());
107 Vector3 y1 = A * (layerQuartett[3]->localPosition() -
108 layerQuartett[2]->localPosition()) +
109 G * (layerQuartett[1]->localPosition() -
110 layerQuartett[3]->localPosition()) +
111 K * (layerQuartett[2]->localPosition() -
112 layerQuartett[1]->localPosition());
113 Vector3 y2 = (K - G) * (layerQuartett[0]->localPosition() -
114 layerQuartett[1]->localPosition()) -
115 (y1.dot(layerQuartett[1]->sensorDirection())) *
116 layerQuartett[1]->sensorDirection() +
117 (y0.dot(layerQuartett[0]->sensorDirection())) *
118 layerQuartett[0]->sensorDirection() +
119 A * (layerQuartett[3]->localPosition() -
120 layerQuartett[2]->localPosition());
121
122 Vector2 solution = betaMatrix.inverse() * y2.block<2, 1>(0, 0);
123 double kappa = solution.x();
124 double gamma = solution.y();
125
126 double lambda = (y0.dot(layerQuartett[0]->sensorDirection()) +
127 K * gamma *
128 (layerQuartett[0]->sensorDirection().dot(
129 layerQuartett[2]->sensorDirection())) -
130 G * kappa *
131 (layerQuartett[0]->sensorDirection().dot(
132 layerQuartett[3]->sensorDirection()))) /
133 (K - G);
134 double alpha = (y1.dot(layerQuartett[1]->sensorDirection()) +
135 (A - G) * kappa *
136 layerQuartett[3]->sensorDirection().dot(
137 layerQuartett[1]->sensorDirection()) +
138 (K - A) * gamma *
139 layerQuartett[2]->sensorDirection().dot(
140 layerQuartett[1]->sensorDirection())) /
141 (K - G);
142
143 return std::array<double, 4>({lambda, alpha, gamma, kappa});
144}

146// Solve the equations for the seed position and direction (M,DM)
152template <Experimental::CompositeSpacePointPtr Point_t>
153std::pair<Vector3, Vector3> seedSolution(
154 const std::array<Point_t, 4>& layerQuartett,
155 const std::array<double, 4>& parameters) {
156 // estimate the seed positionInChamber from the 1st equation of the system of
157 // the layer equations
158 Vector3 seedPosition = layerQuartett[0]->localPosition() +
159 parameters[0] * layerQuartett[0]->sensorDirection();
160
161 // estimate the seed direction from the 2nd equation of the system of the
162 // layer equations
163 Vector3 seedDirection =
164 ((layerQuartett[1]->localPosition() +
165 parameters[1] * layerQuartett[1]->sensorDirection() - seedPosition))
166 .normalized();
167
168 // calculate the position at z=0
169 Intersection3D intersectionZ0 = PlanarHelper::intersectPlane(
170 seedPosition, seedDirection, Vector3::UnitZ(), 0.0);
171 Vector3 seedPositionZ0 = intersectionZ0.position();
172
173 return std::make_pair(seedPositionZ0,
174 copySign(seedDirection, seedDirection.z()));
175};
