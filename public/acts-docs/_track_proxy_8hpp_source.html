482 template <TrackProxyConcept track_proxy_t>
483 void copyFrom(const track_proxy_t& other)
484 requires(!ReadOnly)
485 {
486 copyFromWithoutStates(other);
487
488 // append track states (cheap), but they're in the wrong order
489 for (const auto& srcTrackState : other.trackStatesReversed()) {
490 auto destTrackState = appendTrackState(srcTrackState.getMask());
491 destTrackState.copyFrom(srcTrackState, Acts::TrackStatePropMask::All,
492 true);
493 }
494
495 // reverse using standard linked list reversal algorithm
496 reverseTrackStates();
497 }

607 void reverseTrackStates(bool invertJacobians = false)
608 requires(!ReadOnly)
609 {
610 IndexType current = tipIndex();
611 IndexType next = kInvalid;
612 IndexType prev = kInvalid;
613
614 stemIndex() = tipIndex();
615
616 // @TODO: Maybe refactor to not need this variable if invertJacobians == false
617 BoundMatrix nextJacobian;
618
619 while (current != kInvalid) {
620 auto ts = m_container->trackStateContainer().getTrackState(current);
621 prev = ts.previous();
622 ts.template component<IndexType>(detail_tp::kNextKey) = prev;
623 ts.previous() = next;
624 if (invertJacobians) {
625 if (next != kInvalid) {
626 BoundMatrix curJacobian = ts.jacobian();
627 ts.jacobian() = nextJacobian.inverse();
628 nextJacobian = curJacobian;
629 } else {
630 nextJacobian = ts.jacobian();
631 ts.jacobian().setZero();
632 }
633 }
634 next = current;
635 tipIndex() = current;
636 current = prev;
637 }
638 }
