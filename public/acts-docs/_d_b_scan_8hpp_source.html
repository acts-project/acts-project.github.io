45template <std::size_t kDims, typename scalar_t = double,
46 std::size_t kLeafSize = 4>
47class DBScan {
48 public:

73 explicit DBScan(scalar_t epsilon = 1.0, std::size_t minPoints = 1,
74 bool onePointCluster = false)
75 : m_eps(epsilon),
76 m_minPoints(minPoints),
77 m_onePointCluster(onePointCluster) {}

89 int cluster(const VectorPoints& inputPoints,
90 std::vector<int>& clusteredPoints) {
91 // Transform the initial vector of input point to a vector of pairs
92 // with the index of the point in the initial vector.
93 VectorPairs inputPointsWithIndex;
94 for (std::size_t id = 0; id < inputPoints.size(); id++) {
95 inputPointsWithIndex.push_back(std::make_pair(inputPoints[id], id));
96 }
97 // Build the KDTree with the input points.
98 Tree tree = Tree(std::move(inputPointsWithIndex));
99
100 // Initialize the cluster ID to 0.
101 int clusterID = 0;
102 // By default all the points are considered as noise.
103 clusteredPoints = std::vector<int>(inputPoints.size(), -1);
104
105 // Loop over all the points
106 for (std::size_t id = 0; id < inputPoints.size(); id++) {
107 // If the point is already assigned to a cluster, skip it.
108 if (clusteredPoints[id] != -1) {
109 continue;
110 }
111 // If not we try to build a new cluster
112 std::vector<std::size_t> pointToProcess{id};
113 expandCluster(tree, inputPoints, pointToProcess, clusteredPoints,
114 clusterID);
115 // If the cluster has been created, increment the cluster ID.
116 if (clusteredPoints[id] != -1) {
117 clusterID++;
118 }
119 }
120 if (m_onePointCluster) {
121 // If noise is present and onePointCluster is true, all the noise points
122 // are considered as individual one point clusters. Loop over all the
123 // points in the KDTree.
124 for (auto& cluster : clusteredPoints) {
125 // If the point is assigned to noise, assign it to a new cluster.
126 if (cluster == -1) {
127 cluster = clusterID;
128 clusterID++;
129 }
130 }
131 }
132 return clusterID;
133 }
134
135 private:
150 void expandCluster(const Tree& tree, const VectorPoints& inputPoints,
151 const std::vector<std::size_t>& pointsToProcess,
152 std::vector<int>& clusteredPoints, const int clusterID) {
153 // Loop over all the points that need to be process.
154 for (const auto id : pointsToProcess) {
155 // Lets look for the neighbours of the current point.
156 const Point currentPoint = inputPoints[id];
157 std::vector<std::size_t> neighbours;
158 // We create the range in which we will look for the neighbours (an
159 // hypercube with a length of 2 epsilon).
160 typename Tree::range_t range;
161 for (std::size_t dim = 0; dim < kDims; dim++) {
162 range[dim] = Range1D<scalar_t>(currentPoint[dim] - m_eps,
163 currentPoint[dim] + m_eps);
164 }
165 // We use the KDTree to find the neighbours.
166 // An extra cut needs to be applied to only keep the neighbours that
167 // are within the epsilon radius.
168 tree.rangeSearchMapDiscard(
169 range, [this, &neighbours, currentPoint](
170 const typename Tree::coordinate_t& pos,
171 const typename Tree::value_t& val) {
172 scalar_t distance = 0;
173 for (std::size_t dim = 0; dim < kDims; dim++) {
174 distance += (pos[dim] - currentPoint[dim]) *
175 (pos[dim] - currentPoint[dim]);
176 }
177 if (distance <= m_eps * m_eps) {
178 neighbours.push_back(val);
179 }
180 });
181 std::size_t nNeighbours = neighbours.size();
182 // If a cluster has already been started we add the neighbours to it
183 if (clusteredPoints[id] != -1) {
184 updateNeighbours(neighbours, clusteredPoints, clusterID);
185 }
186 if (nNeighbours >= m_minPoints) {
187 // If the cluster has not been started yet and we have enough
188 // neighbours, we start the cluster and assign the current point and its
189 // neighbours to it.
190 if (clusteredPoints[id] == -1) {
191 clusteredPoints[id] = clusterID;
192 updateNeighbours(neighbours, clusteredPoints, clusterID);
193 }
194 // Try to extend the cluster with the neighbours.
195 expandCluster(tree, inputPoints, neighbours, clusteredPoints,
196 clusterID);
197 }
198 }
199 }
200
210 void updateNeighbours(std::vector<std::size_t>& neighbours,
211 std::vector<int>& clusteredPoints,
212 const int clusterID) {
213 neighbours.erase(std::remove_if(neighbours.begin(), neighbours.end(),
214 [&clusteredPoints](int i) {
215 return clusteredPoints[i] != -1;
216 }),
217 neighbours.end());
218
219 for (const auto& neighbour : neighbours) {
220 clusteredPoints[neighbour] = clusterID;
221 }
222 }
223
224 // The epsilon radius used to find the neighbours.
225 scalar_t m_eps;
226 // The minimum number of points to form a cluster.
227 std::size_t m_minPoints = 1;
228 // If true, all the noise points are considered as individual one point
229 // clusters.
230 bool m_onePointCluster = false;
231};
