49template <typename actor_t, typename propagator_state_t, typename stepper_t,
50 typename navigator_t, typename... Args>
51concept ActorHasOldVoidInterface =
52 // Check without result parameter (for actors without result_type)

70template <typename actor_t, typename propagator_state_t, typename stepper_t,
71 typename navigator_t, typename... Args>
72concept ActorHasAct =
73 ActorHasActWithoutResult<actor_t, propagator_state_t, stepper_t,
74 navigator_t, Args...> ||
75 ActorHasActWithResult<actor_t, propagator_state_t, stepper_t, navigator_t,
76 Args...>;

101template <typename actor_t, typename propagator_state_t, typename stepper_t,
102 typename navigator_t, typename... Args>
103concept ActorHasAbort =
104 ActorHasAbortWithoutResult<actor_t, propagator_state_t, stepper_t,
105 navigator_t, Args...> ||
106 ActorHasAbortWithResult<actor_t, propagator_state_t, stepper_t, navigator_t,

109template <typename actor_t, typename propagator_state_t, typename stepper_t,
110 typename navigator_t, typename... Args>
111concept Actor = (ActorHasAct<actor_t, propagator_state_t, stepper_t,
112 navigator_t, Args...> ||
113 ActorHasAbort<actor_t, propagator_state_t, stepper_t,
114 navigator_t, Args...>) &&
115 !ActorHasOldVoidInterface<actor_t, propagator_state_t,
116 stepper_t, navigator_t, Args...>;
