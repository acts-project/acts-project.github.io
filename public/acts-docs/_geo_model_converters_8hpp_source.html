87inline std::shared_ptr<const IGeoShapeConverter> geoShapesConverters(
88 int geoShapeId) {

90 std::shared_ptr<const IGeoShapeConverter>>
91 converters{
92 {GeoBox::getClassTypeID(), std::make_shared<GeoBoxConverter>()},
93 {GeoShapeIntersection::getClassTypeID(),
94 std::make_shared<GeoIntersectionAnnulusConverter>()},
95 {GeoShapeShift::getClassTypeID(),

97 {GeoTrd::getClassTypeID(), std::make_shared<GeoTrdConverter>()},
98 {GeoTube::getClassTypeID(), std::make_shared<GeoTubeConverter>()},
99 {GeoSimplePolygonBrep::getClassTypeID(),
100 std::make_shared<GeoPolygonConverter>()},
101 {GeoShapeSubtraction::getClassTypeID(),
102 std::make_shared<GeoSubtractionConverter>()},
103 {GeoShapeUnion::getClassTypeID(),
104 std::make_shared<GeoUnionDoubleTrdConverter>()}};
105 auto itr = converters.find(geoShapeId);
106
107 return itr != converters.end() ? itr->second : nullptr;
108};
