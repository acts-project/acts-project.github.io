82 constexpr double value() const {
83 double min = *std::min_element(m_values.begin(), m_values.end());
84 // accuracy is always positive and therefore handled separately
85 double result = std::min(std::abs(min), m_accuracy);
86 return std::signbit(min) ? -result : result;

116 constexpr void update(double v, Type type) {
117 // check the current value and set it if appropriate
118 // this will also allow signed values due to overstepping
119 if (std::abs(v) < std::abs(value(type))) {
120 // TODO enable assert; see

130 std::ostream& toStream(std::ostream& os) const {
131 // Helper method to avoid unreadable screen output
132 auto streamValue = [&](double v) {
133 os << std::setw(5);
134 if (std::abs(v) == kNotSet) {
135 os << (v > 0 ? "+∞" : "-∞");
136 } else {
137 os << v;
138 }
139 };
140
141 os << "(";
142 streamValue(m_accuracy);
143 os << ", ";
144 streamValue(value(Type::Navigator));
145 os << ", ";
146 streamValue(value(Type::Actor));
147 os << ", ";
148 streamValue(value(Type::User));
149 os << ")";
150
151 return os;
152 }
153
156 std::string toString() const {
157 std::stringstream dstream;
158 toStream(dstream);
159 return dstream.str();
160 }

162 private:
