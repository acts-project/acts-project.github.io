95 bool operator()(const std::shared_ptr<BoundsType_t>& a,
96 const std::shared_ptr<BoundsType_t>& b) const {
99 if (a->type() != b->type()) {
100 return toUnderlying(a->type()) < toUnderlying(b->type());
101 }
102 const std::vector<double> avalues{a->values()};
103 const std::vector<double> bvalues{b->values()};
105 if (avalues.size() != bvalues.size()) {
106 return avalues.size() < bvalues.size();
107 }
111 return std::ranges::lexicographical_compare(
112 avalues, bvalues,
113 [](double parA, double parB) { return parA < parB; });
114 }
115 };
