111 template <Acts::detail::SourceLinkGrid grid_t, typename container_t>
112 void findSeeds(const GeometryContext& gctx,
113 const std::unordered_map<GeometryIdentifier, grid_t>&
114 sourceLinkGridLookup,
115 container_t& seedCollection) const {
116 // Create the seeds
117 for (auto& refGeoId : m_cfg.refLayerIds) {
118 auto refGrid = sourceLinkGridLookup.at(refGeoId);
119
120 for (auto it = refGrid.begin(); it != refGrid.end(); it++) {
121 std::vector<SourceLink> pivotSourceLinks = *it;
122
123 for (const auto& pivot : pivotSourceLinks) {
124 // Get the IP parameters
125 auto [ipParameters, refLayerParameters] =
126 m_cfg.trackEstimator(gctx, pivot);
127
128 // Intersect with the surfaces
129 std::vector<std::pair<GeometryIdentifier, Vector2>> intersections =
130 m_cfg.intersectionFinder(gctx, refLayerParameters);
131
132 // Continue if no intersections
133 if (intersections.empty()) {
134 continue;
135 }
136
137 // Iterate over the intersections
138 // and get the source links
139 // in the subsequent layers
140 std::vector<SourceLink> seedSourceLinks;
141 for (auto& [geoId, refPoint] : intersections) {
142 // Get the path width
143 auto [pathWidth0, pathWidth1] =
144 m_cfg.pathWidthProvider(gctx, geoId);
145
146 // Get the bounds of the path
147 double top0 = refPoint[0] + pathWidth0;
148 double bot0 = refPoint[0] - pathWidth0;
149 double top1 = refPoint[1] + pathWidth1;
150 double bot1 = refPoint[1] - pathWidth1;
151
152 // Get the lookup table for the source links
153 auto grid = sourceLinkGridLookup.at(geoId);
154
155 // Get the range of bins to search for source links
156 auto botLeftBin = grid.localBinsFromPosition(Vector2(bot0, bot1));
157 auto topRightBin = grid.localBinsFromPosition(Vector2(top0, top1));
158
159 // Get the source links from the lookup table
160 // by iterating over the bin ranges
161 auto currentBin = botLeftBin;
162 while (currentBin.at(1) <= topRightBin.at(1)) {
163 while (currentBin.at(0) <= topRightBin.at(0)) {
164 auto sourceLinksToAdd = grid.atLocalBins(currentBin);
165
166 seedSourceLinks.insert(seedSourceLinks.end(),
167 sourceLinksToAdd.begin(),
168 sourceLinksToAdd.end());
169
170 currentBin.at(0)++;
171 }
172 currentBin.at(1)++;
173 currentBin.at(0) = botLeftBin.at(0);
174 }
175 }
176 PathSeed seed = {ipParameters, seedSourceLinks};
177
178 // Add the seed to the collection
179 Acts::detail::pushBackOrInsertAtEnd(seedCollection, seed);
180 }
181 }
182 }
183 }

186 Config m_cfg;
