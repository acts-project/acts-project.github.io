42inline Acts::AxisDirection stringToAxisDirection(
43 const std::string &binningString) {
44 using enum Acts::AxisDirection;
45 if (binningString == "x") {
46 return AxisX;
47 } else if (binningString == "y") {
48 return AxisY;
49 } else if (binningString == "z") {
50 return AxisZ;
51 } else if (binningString == "phi") {
52 return AxisPhi;
53 } else if (binningString == "r") {
54 return AxisR;
55 } else {
56 throw std::invalid_argument("DD4hepBinningHelpers: Binning value " +
57 binningString + " not allowed.");
58 }
59}

68inline std::vector<Acts::AxisDirection> stringToAxisDirections(
69 const std::string &binningString, const char &del = ',') {
70 if (binningString.empty()) {
71 return {};
72 }
73 std::vector<Acts::AxisDirection> bBinning;
74 std::stringstream s(binningString);
75 std::string b = "";
76 while (getline(s, b, del)) {
77 bBinning.push_back(stringToAxisDirection(b));
78 }
79 return bBinning;
80}
