45template <bool reverse, typename trajectory_t, std::size_t M, bool ReadOnly>
46class TrackStateRange {
47 using ProxyType = TrackStateProxy<trajectory_t, M, ReadOnly>;
48 using IndexType = typename ProxyType::IndexType;
49 static constexpr IndexType kInvalid = ProxyType::kInvalid;
50
51 public:
54 struct Iterator {
55 std::optional<ProxyType> proxy;
56
57 using iterator_category = std::forward_iterator_tag;
58 using value_type = ProxyType;
59 using difference_type = std::ptrdiff_t;
60 using pointer = void;
61 using reference = void;
62
63 Iterator& operator++() {
64 if (!proxy) {
65 return *this;
66 }
67 if constexpr (reverse) {
68 if (proxy->hasPrevious()) {
69 proxy = proxy->trajectory().getTrackState(proxy->previous());
70 return *this;
71 } else {
72 proxy = std::nullopt;
73 return *this;
74 }
75 } else {
76 IndexType next =
77 proxy->template component<IndexType, hashString("next")>();
78 if (next != kInvalid) {
79 proxy = proxy->trajectory().getTrackState(next);
80 return *this;
81 } else {
82 proxy = std::nullopt;
83 return *this;
84 }
85 }
86 }
87
88 Iterator operator++(int) {
89 Iterator tmp(*this);
90 operator++();
91 return tmp;
92 }
93
94 bool operator==(const Iterator& other) const {
95 if (!proxy && !other.proxy) {
96 return true;
97 }
98 if (proxy && other.proxy) {
99 return proxy->index() == other.proxy->index();
100 }
101 return false;
102 }
103
104 ProxyType operator*() const { return *proxy; }
105 ProxyType operator*() { return *proxy; }
106 };
107
108 explicit TrackStateRange(ProxyType _begin) : m_begin{_begin} {}
109 TrackStateRange() : m_begin{std::nullopt} {}
110
111 Iterator begin() { return Iterator{m_begin}; }
112 Iterator end() { return Iterator{std::nullopt}; }
113
114 Iterator cbegin() const { return Iterator{m_begin}; }
115 Iterator cend() const { return Iterator{std::nullopt}; }
116
117 private:
118 Iterator m_begin;
119};

306 template <typename F>
307 void applyBackwards(IndexType iendpoint, F&& callable)
308 requires(!ReadOnly) && detail_lt::VisitorConcept<F, TrackStateProxy>
309 {
310 if (iendpoint == kInvalid) {
311 throw std::runtime_error(
312 "Cannot apply backwards with kInvalid as endpoint");
313 }
314
315 while (true) {
316 auto ts = getTrackState(iendpoint);
317 if constexpr (std::is_same_v<std::invoke_result_t<F, TrackStateProxy>,
318 bool>) {
319 bool proceed = callable(ts);
320 // this point has no parent and ends the trajectory, or a break was
321 // requested
322 if (!proceed || !ts.hasPrevious()) {
323 break;
324 }
325 } else {
326 callable(ts);
327 // this point has no parent and ends the trajectory
328 if (!ts.hasPrevious()) {
329 break;
330 }
331 }
332 iendpoint = ts.previous();
333 }
334 }
