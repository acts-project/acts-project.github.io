376 template <typename propagator_state_t, typename stepper_t,
377 typename navigator_t>
378 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
379 const navigator_t& navigator, result_type& result,
380 const Logger& /*logger*/) const {
381 assert(result.fittedStates && "No MultiTrajectory set");
382
383 if (result.finished) {
384 return Result<void>::success();
385 }
386
387 ACTS_VERBOSE("KalmanFitter step at pos: "
388 << stepper.position(state.stepping).transpose()
389 << " dir: " << stepper.direction(state.stepping).transpose()
390 << " momentum: "
391 << stepper.absoluteMomentum(state.stepping));
392
393 // Initialize path limit reached aborter
394 if (result.pathLimitReached.internalLimit ==
395 std::numeric_limits<double>::max()) {
396 detail::setupLoopProtection(state, stepper, result.pathLimitReached,
397 true, logger());
398 }
399
400 // Update:
401 // - Waiting for a current surface
402 const Surface* surface = navigator.currentSurface(state.navigation);
403 if (surface != nullptr) {
404 // Check if the surface is in the measurement map
405 // -> Get the measurement / calibrate
406 // -> Create the predicted state
407 // -> Check outlier behavior, if non-outlier:
408 // -> Perform the kalman update
409 // -> Fill track state information & update stepper information
410
411 if (!result.smoothed && !result.reversed) {
412 ACTS_VERBOSE("Perform " << state.options.direction << " filter step");
413 auto res = filter(surface, state, stepper, navigator, result);
414 if (!res.ok()) {
415 ACTS_ERROR("Error in " << state.options.direction
416 << " filter: " << res.error());
417 return res.error();
418 }
419 }
420 if (result.reversed) {
421 ACTS_VERBOSE("Perform " << state.options.direction << " filter step");
422 auto res = reversedFilter(surface, state, stepper, navigator, result);
423 if (!res.ok()) {
424 ACTS_ERROR("Error in " << state.options.direction
425 << " filter: " << res.error());
426 return res.error();
427 }
428 }
429 }
430
431 // Finalization:
432 // when all track states have been handled or the navigation is breaked,
433 // reset navigation&stepping before run reversed filtering or
434 // proceed to run smoothing
435 if (!result.smoothed && !result.reversed) {
436 const bool isTrackComplete =
437 result.measurementStates == inputMeasurements->size();
438 const bool isEndOfWorldReached =
439 endOfWorldReached.checkAbort(state, stepper, navigator, logger());
440 const bool isVolumeConstraintReached =
441 volumeConstraintAborter.checkAbort(state, stepper, navigator,
442 logger());
443 const bool isPathLimitReached = result.pathLimitReached.checkAbort(
444 state, stepper, navigator, logger());
445 if (isTrackComplete || isEndOfWorldReached ||
446 isVolumeConstraintReached || isPathLimitReached) {
447 if (result.lastMeasurementIndex == MultiTrajectoryTraits::kInvalid) {
448 ACTS_INFO(
449 "No measurements were found on the track, cannot proceed "
450 "to smoothing or reversed filtering.");
451 result.finished = true;
452 return Result<void>::success();
453 }
454
455 // Remove the missing surfaces that occur after the last measurement
456 result.missedActiveSurfaces.resize(result.measurementHoles);
457 // now get track state proxy for the outlier logic
458 typename traj_t::ConstTrackStateProxy trackStateProxy{
459 result.fittedStates->getTrackState(result.lastMeasurementIndex)};
460 const bool doReverseFiltering =
461 reversedFiltering ||
462 extensions.reverseFilteringLogic(trackStateProxy);
463 if (doReverseFiltering) {
464 // Start to run reversed filtering:
465 // Reverse navigation direction and reset navigation and stepping
466 // state to last measurement
467 ACTS_VERBOSE("Reverse navigation direction.");
468 auto res = reverse(state, stepper, navigator, result);
469 if (!res.ok()) {
470 ACTS_ERROR("Error in reversing navigation: " << res.error());
471 return res.error();
472 }
473 } else {
474 // --> Search the starting state to run the smoothing
475 // --> Call the smoothing
476 // --> Set a stop condition when all track states have been handled
477 ACTS_VERBOSE("Finalize/run smoothing");
478 auto res = finalize(state, stepper, navigator, result);
479 if (!res.ok()) {
480 ACTS_ERROR("Error in finalize: " << res.error());
481 return res.error();
482 }
483 }
484 }
485 }
486
487 // Post-finalization:
488 // - Progress to target/reference surface and built the final track
489 // parameters
490 if (result.smoothed || result.reversed) {
491 if (result.smoothed) {
492 // Update state and stepper with material effects
493 // Only for smoothed as reverse filtering will handle this separately
494 materialInteractor(navigator.currentSurface(state.navigation), state,
495 stepper, navigator,
496 MaterialUpdateStage::FullUpdate);
497 }
498
499 if (targetReached.surface == nullptr) {
500 // If no target surface provided:
501 // -> Return an error when using reversed filtering mode
502 // -> Fitting is finished here
503 if (result.reversed) {
504 ACTS_ERROR(
505 "The target surface needed for aborting reversed propagation "
506 "is not provided");
507 return KalmanFitterError::ReversePropagationFailed;
508 } else {
509 ACTS_VERBOSE(
510 "No target surface set. Completing without fitted track "
511 "parameter");
512 // Remember the track fitting is done
513 result.finished = true;
514 }
515 } else if (targetReached.checkAbort(state, stepper, navigator,
516 logger())) {
517 ACTS_VERBOSE("Completing with fitted track parameter");
518 // Transport & bind the parameter to the final surface
519 auto res = stepper.boundState(state.stepping, *targetReached.surface,
520 true, freeToBoundCorrection);
521 if (!res.ok()) {
522 ACTS_ERROR("Error in " << state.options.direction
523 << " filter: " << res.error());
524 return res.error();
525 }
526 auto& fittedState = *res;
527 // Assign the fitted parameters
528 result.fittedParameters = std::get<BoundTrackParameters>(fittedState);
529
530 // Reset smoothed status of states missed in reversed filtering
531 if (result.reversed) {
532 result.fittedStates->applyBackwards(
533 result.lastMeasurementIndex, [&](auto trackState) {
534 auto fSurface = &trackState.referenceSurface();
535 if (!rangeContainsValue(result.passedAgainSurfaces,
536 fSurface)) {
537 // If reversed filtering missed this surface, then there is
538 // no smoothed parameter
539 trackState.unset(TrackStatePropMask::Smoothed);
540 if (trackState.typeFlags().test(
541 TrackStateFlag::MeasurementFlag)) {
542 trackState.typeFlags().set(TrackStateFlag::OutlierFlag);
543 }
544 }
545 });
546 }
547 // Remember the track fitting is done
548 result.finished = true;
549 }
550 }
551
552 return Result<void>::success();
553 }

558 const navigator_t& /*navigator*/, const result_type& result,
559 const Logger& /*logger*/) const {
560 return result.finished;

573 template <typename propagator_state_t, typename stepper_t,
574 typename navigator_t>
575 Result<void> reverse(propagator_state_t& state, const stepper_t& stepper,
576 const navigator_t& navigator,
577 result_type& result) const {
578 // Check if there is a measurement on track
579 if (result.lastMeasurementIndex == MultiTrajectoryTraits::kInvalid) {
580 ACTS_ERROR("No point to reverse for a track without measurements.");
581 return KalmanFitterError::ReversePropagationFailed;
582 }
583
584 // Remember the navigation direction has been reversed
585 result.reversed = true;
586
587 // Reverse navigation direction
588 state.options.direction = state.options.direction.invert();
589
590 // Reset propagator options
591 // TODO Not sure if reset of pathLimit during propagation makes any sense
592 state.options.pathLimit =
593 state.options.direction * std::abs(state.options.pathLimit);
594
595 // Get the last measurement state and reset navigation&stepping state
596 // based on information on this state
597 auto st = result.fittedStates->getTrackState(result.lastMeasurementIndex);
598
599 // Update the stepping state
600 stepper.initialize(
601 state.stepping, st.filtered(),
602 reversedFilteringCovarianceScaling * st.filteredCovariance(),
603 stepper.particleHypothesis(state.stepping), st.referenceSurface());
604
605 // For the last measurement state, smoothed is filtered
606 st.smoothed() = st.filtered();
607 st.smoothedCovariance() = st.filteredCovariance();
608 result.passedAgainSurfaces.push_back(&st.referenceSurface());
609
610 // Reset navigation state
611 // We do not need to specify a target here since this will be handled
612 // separately in the KF actor
613 state.navigation.options.startSurface = &st.referenceSurface();
614 state.navigation.options.targetSurface = nullptr;
615 auto navInitRes = navigator.initialize(
616 state.navigation, stepper.position(state.stepping),
617 stepper.direction(state.stepping), state.options.direction);
618 if (!navInitRes.ok()) {
619 return navInitRes.error();
620 }
621
622 // Update material effects for last measurement state in reversed
623 // direction
624 materialInteractor(navigator.currentSurface(state.navigation), state,
625 stepper, navigator, MaterialUpdateStage::FullUpdate);
626
627 // Set path limit based on loop protection
628 detail::setupLoopProtection(state, stepper, result.pathLimitReached, true,
629 logger());
630
631 // Set path limit based on target surface
632 targetReached.checkAbort(state, stepper, navigator, logger());
633
634 return Result<void>::success();
635 }

648 template <typename propagator_state_t, typename stepper_t,
649 typename navigator_t>
650 Result<void> filter(const Surface* surface, propagator_state_t& state,
651 const stepper_t& stepper, const navigator_t& navigator,
652 result_type& result) const {
653 const bool precedingMeasurementExists = result.measurementStates > 0;
654 const bool surfaceIsSensitive =
655 surface->associatedDetectorElement() != nullptr;
656 const bool surfaceHasMaterial = surface->surfaceMaterial() != nullptr;
657
658 // Try to find the surface in the measurement surfaces
659 auto sourceLinkIt = inputMeasurements->find(surface->geometryId());
660 if (sourceLinkIt != inputMeasurements->end()) {
661 // Screen output message
662 ACTS_VERBOSE("Measurement surface " << surface->geometryId()
663 << " detected.");
664 // Transport the covariance to the surface
665 stepper.transportCovarianceToBound(state.stepping, *surface,
666 freeToBoundCorrection);
667
668 // Update state and stepper with pre material effects
669 materialInteractor(surface, state, stepper, navigator,
670 MaterialUpdateStage::PreUpdate);
671
672 // do the kalman update (no need to perform covTransport here, hence no
673 // point in performing globalToLocal correction)
674 auto trackStateProxyRes = detail::kalmanHandleMeasurement(
675 *calibrationContext, state, stepper, extensions, *surface,
676 sourceLinkIt->second, *result.fittedStates, result.lastTrackIndex,
677 false, logger());
678
679 if (!trackStateProxyRes.ok()) {
680 return trackStateProxyRes.error();
681 }
682
683 const auto& trackStateProxy = *trackStateProxyRes;
684 result.lastTrackIndex = trackStateProxy.index();
685
686 // Update the stepper if it is not an outlier
687 if (trackStateProxy.typeFlags().test(TrackStateFlag::MeasurementFlag)) {
688 // Update the stepping state with filtered parameters
689 ACTS_VERBOSE("Filtering step successful, updated parameters are:\n"
690 << trackStateProxy.filtered().transpose());
691 // update stepping state using filtered parameters after kalman
692 stepper.update(state.stepping,
693 MultiTrajectoryHelpers::freeFiltered(
694 state.options.geoContext, trackStateProxy),
695 trackStateProxy.filtered(),
696 trackStateProxy.filteredCovariance(), *surface);
697 // We count the state with measurement
698 ++result.measurementStates;
699 }
700
701 // Update state and stepper with post material effects
702 materialInteractor(surface, state, stepper, navigator,
703 MaterialUpdateStage::PostUpdate);
704 // We count the processed state
705 ++result.processedStates;
706 // Update the number of holes count only when encountering a
707 // measurement
708 result.measurementHoles = result.missedActiveSurfaces.size();
709 // Since we encountered a measurement update the lastMeasurementIndex to
710 // the lastTrackIndex.
711 result.lastMeasurementIndex = result.lastTrackIndex;
712
713 } else if ((precedingMeasurementExists && surfaceIsSensitive) ||
714 surfaceHasMaterial) {
715 // We only create track states here if there is already measurement
716 // detected or if the surface has material (no holes before the first
717 // measurement)
718 auto trackStateProxyRes = detail::kalmanHandleNoMeasurement(
719 state.stepping, stepper, *surface, *result.fittedStates,
720 result.lastTrackIndex, true, logger(), precedingMeasurementExists,
721 freeToBoundCorrection);
722
723 if (!trackStateProxyRes.ok()) {
724 return trackStateProxyRes.error();
725 }
726
727 const auto& trackStateProxy = *trackStateProxyRes;
728 result.lastTrackIndex = trackStateProxy.index();
729
730 if (trackStateProxy.typeFlags().test(TrackStateFlag::HoleFlag)) {
731 // Count the missed surface
732 result.missedActiveSurfaces.push_back(surface);
733 }
734
735 ++result.processedStates;
736
737 // Update state and stepper with (possible) material effects
738 materialInteractor(surface, state, stepper, navigator,
739 MaterialUpdateStage::FullUpdate);
740 }
741 return Result<void>::success();
742 }

755 template <typename propagator_state_t, typename stepper_t,
756 typename navigator_t>
757 Result<void> reversedFilter(const Surface* surface,
758 propagator_state_t& state,
759 const stepper_t& stepper,
760 const navigator_t& navigator,
761 result_type& result) const {
762 // Try to find the surface in the measurement surfaces
763 auto sourceLinkIt = inputMeasurements->find(surface->geometryId());
764 if (sourceLinkIt != inputMeasurements->end()) {
765 // Screen output message
766 ACTS_VERBOSE("Measurement surface "
767 << surface->geometryId()
768 << " detected in reversed propagation.");
769
770 // No reversed filtering for last measurement state, but still update
771 // with material effects
772 if (result.reversed &&
773 surface == navigator.startSurface(state.navigation)) {
774 materialInteractor(surface, state, stepper, navigator,
775 MaterialUpdateStage::FullUpdate);
776 return Result<void>::success();
777 }
778
779 // Transport the covariance to the surface
780 stepper.transportCovarianceToBound(state.stepping, *surface,
781 freeToBoundCorrection);
782
783 // Update state and stepper with pre material effects
784 materialInteractor(surface, state, stepper, navigator,
785 MaterialUpdateStage::PreUpdate);
786
787 auto& fittedStates = *result.fittedStates;
788
789 // Add a <mask> TrackState entry multi trajectory. This allocates
790 // storage for all components, which we will set later.
791 TrackStatePropMask mask =
792 TrackStatePropMask::Predicted | TrackStatePropMask::Filtered |
793 TrackStatePropMask::Smoothed | TrackStatePropMask::Jacobian |
794 TrackStatePropMask::Calibrated;
795 const std::size_t currentTrackIndex =
796 fittedStates.addTrackState(mask, MultiTrajectoryTraits::kInvalid);
797
798 // now get track state proxy back
799 typename traj_t::TrackStateProxy trackStateProxy =
800 fittedStates.getTrackState(currentTrackIndex);
801
802 // Set the trackStateProxy components with the state from the ongoing
803 // propagation
804 {
805 trackStateProxy.setReferenceSurface(surface->getSharedPtr());
806 // Bind the transported state to the current surface
807 auto res = stepper.boundState(state.stepping, *surface, false,
808 freeToBoundCorrection);
809 if (!res.ok()) {
810 return res.error();
811 }
812 const auto& [boundParams, jacobian, pathLength] = *res;
813
814 // Fill the track state
815 trackStateProxy.predicted() = boundParams.parameters();
816 trackStateProxy.predictedCovariance() = state.stepping.cov;
817
818 trackStateProxy.jacobian() = jacobian;
819 trackStateProxy.pathLength() = pathLength;
820 }
821
822 // We have predicted parameters, so calibrate the uncalibrated input
823 // measurement
824 extensions.calibrator(state.geoContext, *calibrationContext,
825 sourceLinkIt->second, trackStateProxy);
826
827 // If the update is successful, set covariance and
828 auto updateRes =
829 extensions.updater(state.geoContext, trackStateProxy, logger());
830 if (!updateRes.ok()) {
831 ACTS_ERROR("Backward update step failed: " << updateRes.error());
832 return updateRes.error();
833 } else {
834 // Update the stepping state with filtered parameters
835 ACTS_VERBOSE(
836 "Backward filtering step successful, updated parameters are:\n"
837 << trackStateProxy.filtered().transpose());
838
839 // Fill the smoothed parameter for the existing track state
840 result.fittedStates->applyBackwards(
841 result.lastMeasurementIndex, [&](auto trackState) {
842 auto fSurface = &trackState.referenceSurface();
843 if (fSurface == surface) {
844 result.passedAgainSurfaces.push_back(surface);
845 trackState.smoothed() = trackStateProxy.filtered();
846 trackState.smoothedCovariance() =
847 trackStateProxy.filteredCovariance();
848 return false;
849 }
850 return true;
851 });
852
853 // update stepping state using filtered parameters after kalman
854 // update We need to (re-)construct a BoundTrackParameters instance
855 // here, which is a bit awkward.
856 stepper.update(state.stepping,
857 MultiTrajectoryHelpers::freeFiltered(
858 state.options.geoContext, trackStateProxy),
859 trackStateProxy.filtered(),
860 trackStateProxy.filteredCovariance(), *surface);
861
862 // Update state and stepper with post material effects
863 materialInteractor(surface, state, stepper, navigator,
864 MaterialUpdateStage::PostUpdate);
865 }
866 } else if (surface->associatedDetectorElement() != nullptr ||
867 surface->surfaceMaterial() != nullptr) {
868 // Transport covariance
869 if (surface->associatedDetectorElement() != nullptr) {
870 ACTS_VERBOSE("Detected hole on " << surface->geometryId()
871 << " in reversed filtering");
872 if (state.stepping.covTransport) {
873 stepper.transportCovarianceToBound(state.stepping, *surface);
874 }
875 } else if (surface->surfaceMaterial() != nullptr) {
876 ACTS_VERBOSE("Detected in-sensitive surface "
877 << surface->geometryId() << " in reversed filtering");
878 if (state.stepping.covTransport) {
879 stepper.transportCovarianceToCurvilinear(state.stepping);
880 }
881 }
882 // Not creating bound state here, so need manually reinitialize
883 // jacobian
884 stepper.setIdentityJacobian(state.stepping);
885 if (surface->surfaceMaterial() != nullptr) {
886 // Update state and stepper with material effects
887 materialInteractor(surface, state, stepper, navigator,
888 MaterialUpdateStage::FullUpdate);
889 }
890 }
891
892 return Result<void>::success();

907 template <typename propagator_state_t, typename stepper_t,
908 typename navigator_t>
909 void materialInteractor(const Surface* surface, propagator_state_t& state,
910 const stepper_t& stepper,
911 const navigator_t& navigator,
912 const MaterialUpdateStage& updateStage) const {
913 // Protect against null surface
914 if (!surface) {
915 ACTS_VERBOSE(
916 "Surface is nullptr. Cannot be used for material interaction");
917 return;
918 }
919
920 // Indicator if having material
921 bool hasMaterial = false;
922
923 if (surface && surface->surfaceMaterial()) {
924 // Prepare relevant input particle properties
925 detail::PointwiseMaterialInteraction interaction(surface, state,
926 stepper);
927 // Evaluate the material properties
928 if (interaction.evaluateMaterialSlab(state, navigator, updateStage)) {
929 // Surface has material at this stage
930 hasMaterial = true;
931
932 // Evaluate the material effects
933 interaction.evaluatePointwiseMaterialInteraction(multipleScattering,
934 energyLoss);
935
936 // Screen out material effects info
937 ACTS_VERBOSE("Material effects on surface: "
938 << surface->geometryId()
939 << " at update stage: " << updateStage << " are :");
940 ACTS_VERBOSE("eLoss = "
941 << interaction.Eloss << ", "
942 << "variancePhi = " << interaction.variancePhi << ", "
943 << "varianceTheta = " << interaction.varianceTheta
944 << ", "
945 << "varianceQoverP = " << interaction.varianceQoverP);
946
947 // Update the state and stepper with material effects
948 interaction.updateState(state, stepper, addNoise);
949 }
950 }
951
952 if (!hasMaterial) {
953 // Screen out message
954 ACTS_VERBOSE("No material effects on surface: " << surface->geometryId()
955 << " at update stage: "
956 << updateStage);
957 }
958 }

970 template <typename propagator_state_t, typename stepper_t,
971 typename navigator_t>
972 Result<void> finalize(propagator_state_t& state, const stepper_t& stepper,
973 const navigator_t& navigator,
974 result_type& result) const {
975 // Remember you smoothed the track states
976 result.smoothed = true;
977
978 // Get the indices of the first states (can be either a measurement or
979 // material);
980 std::size_t firstStateIndex = result.lastMeasurementIndex;
981 // Count track states to be smoothed
982 std::size_t nStates = 0;
983 result.fittedStates->applyBackwards(
984 result.lastMeasurementIndex, [&](auto st) {
985 bool isMeasurement =
986 st.typeFlags().test(TrackStateFlag::MeasurementFlag);
987 bool isOutlier = st.typeFlags().test(TrackStateFlag::OutlierFlag);
988 // We are excluding non measurement states and outlier here. Those
989 // can decrease resolution because only the smoothing corrected the
990 // very first prediction as filtering is not possible.
991 if (isMeasurement && !isOutlier) {
992 firstStateIndex = st.index();
993 }
994 nStates++;
995 });
996 // Return error if the track has no measurement states (but this should
997 // not happen)
998 if (nStates == 0) {
999 ACTS_ERROR("Smoothing for a track without measurements.");
1000 return KalmanFitterError::SmoothFailed;
1001 }
1002 // Screen output for debugging
1003 ACTS_VERBOSE("Apply smoothing on " << nStates
1004 << " filtered track states.");
1005
1006 // Smooth the track states
1007 auto smoothRes =
1008 extensions.smoother(state.geoContext, *result.fittedStates,
1009 result.lastMeasurementIndex, logger());
1010 if (!smoothRes.ok()) {
1011 ACTS_ERROR("Smoothing step failed: " << smoothRes.error());
1012 return smoothRes.error();
1013 }
1014
1015 // Return in case no target surface
1016 if (targetReached.surface == nullptr) {
1017 return Result<void>::success();
1018 }
1019
1020 // Obtain the smoothed parameters at first/last measurement state
1021 auto firstCreatedState =
1022 result.fittedStates->getTrackState(firstStateIndex);
1023 auto lastCreatedMeasurement =
1024 result.fittedStates->getTrackState(result.lastMeasurementIndex);
1025
1026 // Lambda to get the intersection of the free params on the target surface
1027 auto target = [&](const FreeVector& freeVector) -> Intersection3D {
1028 return targetReached.surface
1029 ->intersect(
1030 state.geoContext, freeVector.segment<3>(eFreePos0),
1031 state.options.direction * freeVector.segment<3>(eFreeDir0),
1032 BoundaryTolerance::None(), state.options.surfaceTolerance)
1033 .closest();
1034 };
1035
1036 // The smoothed free params at the first/last measurement state.
1037 // (the first state can also be a material state)
1038 auto firstParams = MultiTrajectoryHelpers::freeSmoothed(
1039 state.options.geoContext, firstCreatedState);
1040 auto lastParams = MultiTrajectoryHelpers::freeSmoothed(
1041 state.options.geoContext, lastCreatedMeasurement);
1042 // Get the intersections of the smoothed free parameters with the target
1043 // surface
1044 const auto firstIntersection = target(firstParams);
1045 const auto lastIntersection = target(lastParams);
1046
1047 // Update the stepping parameters - in order to progress to destination.
1048 // At the same time, reverse navigation direction for further stepping if
1049 // necessary.
1050 // @note The stepping parameters is updated to the smoothed parameters at
1051 // either the first measurement state or the last measurement state. It
1052 // assumes the target surface is not within the first and the last
1053 // smoothed measurement state. Also, whether the intersection is on
1054 // surface is not checked here.
1055 bool useFirstTrackState = true;
1056 switch (targetSurfaceStrategy) {
1057 case KalmanFitterTargetSurfaceStrategy::first:
1058 useFirstTrackState = true;
1059 break;
1060 case KalmanFitterTargetSurfaceStrategy::last:
1061 useFirstTrackState = false;
1062 break;
1063 case KalmanFitterTargetSurfaceStrategy::firstOrLast:
1064 useFirstTrackState = std::abs(firstIntersection.pathLength()) <=
1065 std::abs(lastIntersection.pathLength());
1066 break;
1067 default:
1068 ACTS_ERROR("Unknown target surface strategy");
1069 return KalmanFitterError::SmoothFailed;
1070 }
1071 bool reverseDirection = false;
1072 if (useFirstTrackState) {
1073 stepper.initialize(state.stepping, firstCreatedState.smoothed(),
1074 firstCreatedState.smoothedCovariance(),
1075 stepper.particleHypothesis(state.stepping),
1076 firstCreatedState.referenceSurface());
1077 reverseDirection = firstIntersection.pathLength() < 0;
1078 } else {
1079 stepper.initialize(state.stepping, lastCreatedMeasurement.smoothed(),
1080 lastCreatedMeasurement.smoothedCovariance(),
1081 stepper.particleHypothesis(state.stepping),
1082 lastCreatedMeasurement.referenceSurface());
1083 reverseDirection = lastIntersection.pathLength() < 0;
1084 }
1085 // Reverse the navigation direction if necessary
1086 if (reverseDirection) {
1087 ACTS_VERBOSE(
1088 "Reverse navigation direction after smoothing for reaching the "
1089 "target surface");
1090 state.options.direction = state.options.direction.invert();
1091 }
1092 const auto& surface = useFirstTrackState
1093 ? firstCreatedState.referenceSurface()
1094 : lastCreatedMeasurement.referenceSurface();
1095
1096 ACTS_VERBOSE(
1097 "Smoothing successful, updating stepping state to smoothed "
1098 "parameters at surface "
1099 << surface.geometryId() << ". Prepared to reach the target surface.");
1100
1101 // Reset the navigation state to enable propagation towards the target
1102 // surface
1103 // Set targetSurface to nullptr as it is handled manually in the actor
1104 auto navigationOptions = state.navigation.options;
1105 navigationOptions.startSurface = &surface;
1106 navigationOptions.targetSurface = nullptr;
1107 state.navigation = navigator.makeState(navigationOptions);
1108 auto navInitRes = navigator.initialize(
1109 state.navigation, stepper.position(state.stepping),
1110 stepper.direction(state.stepping), state.options.direction);
1111 if (!navInitRes.ok()) {
1112 return navInitRes.error();
1113 }
1114
1115 return Result<void>::success();
1116 }
1117 };

1298 typename kalman_result_t, TrackContainerFrontend track_container_t>
1299 auto fit_impl(const start_parameters_t& sParameters,
1300 const propagator_options_t& propagatorOptions,
1301 track_container_t& trackContainer) const
1302 -> Result<typename track_container_t::TrackProxy> {
1303 auto propagatorState = m_propagator.makeState(propagatorOptions);
1304
1305 auto propagatorInitResult =
1306 m_propagator.initialize(propagatorState, sParameters);
1307 if (!propagatorInitResult.ok()) {
1308 ACTS_ERROR("Propagation initialization failed: "
1309 << propagatorInitResult.error());
1310 return propagatorInitResult.error();
1311 }
1312
1313 auto& kalmanResult =
1314 propagatorState.template get<KalmanFitterResult<traj_t>>();
1315 kalmanResult.fittedStates = &trackContainer.trackStateContainer();
1316
1317 // Run the fitter
1318 auto result = m_propagator.propagate(propagatorState);
1319
1320 if (!result.ok()) {
1321 ACTS_ERROR("Propagation failed: " << result.error());
1322 return result.error();
1323 }
1324
1327 if (!kalmanResult.measurementStates) {
1328 ACTS_ERROR("KalmanFilter failed: No measurement states found");
1329 return KalmanFitterError::NoMeasurementFound;
1330 }
1331
1332 auto track = trackContainer.makeTrack();
1333 track.tipIndex() = kalmanResult.lastMeasurementIndex;
1334 if (kalmanResult.fittedParameters) {
1335 const auto& params = kalmanResult.fittedParameters.value();
1336 track.parameters() = params.parameters();
1337 track.covariance() = params.covariance().value();
1338 track.setReferenceSurface(params.referenceSurface().getSharedPtr());
1339 }
1340
1341 calculateTrackQuantities(track);
1342
1343 if (trackContainer.hasColumn(hashString("smoothed"))) {
1344 track.template component<bool, hashString("smoothed")>() =
1345 kalmanResult.smoothed;
1346 }
1347
1348 if (trackContainer.hasColumn(hashString("reversed"))) {
1349 track.template component<bool, hashString("reversed")>() =
1350 kalmanResult.reversed;
1351 }
1352
1353 // Return the converted Track
1354 return track;
