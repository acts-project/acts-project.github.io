223 explicit InterpolatedBFieldMap(Config cfg) : m_cfg{std::move(cfg)} {
224 typename Grid::index_t minBin{};
225 minBin.fill(1);
226 m_lowerLeft = m_cfg.grid.lowerLeftBinEdge(minBin);
227 m_upperRight = m_cfg.grid.lowerLeftBinEdge(m_cfg.grid.numLocalBins());

237 Result<FieldCell> getFieldCell(const Vector3& position) const {
238 const auto& gridPosition = m_cfg.transformPos(position);
239 const auto& indices = m_cfg.grid.localBinsFromPosition(gridPosition);
240 const auto& lowerLeft = m_cfg.grid.lowerLeftBinEdge(indices);
241 const auto& upperRight = m_cfg.grid.upperRightBinEdge(indices);
242
243 // loop through all corner points
244 constexpr std::size_t nCorners = 1 << DIM_POS;
245 std::array<Vector3, nCorners> neighbors{};
246 const auto& cornerIndices = m_cfg.grid.closestPointsIndices(gridPosition);
247
248 if (!isInsideLocal(gridPosition)) {
249 return MagneticFieldError::OutOfBounds;
250 }
251
252 std::size_t i = 0;
253 for (std::size_t index : cornerIndices) {
254 neighbors.at(i++) = m_cfg.transformBField(m_cfg.grid.at(index), position);
255 }
256
257 assert(i == nCorners);
258
259 return FieldCell(lowerLeft, upperRight, std::move(neighbors));
260 }

326 Result<Vector3> getField(const Vector3& position) const {
327 const auto gridPosition = m_cfg.transformPos(position);
328 if (!isInsideLocal(gridPosition)) {
329 return Result<Vector3>::failure(MagneticFieldError::OutOfBounds);
330 }
331
332 return Result<Vector3>::success(
333 m_cfg.transformBField(m_cfg.grid.interpolate(gridPosition), position));
334 }
335
343 Vector3 getFieldUnchecked(const Vector3& position) const final {
344 const auto gridPosition = m_cfg.transformPos(position);
345 return m_cfg.transformBField(m_cfg.grid.interpolate(gridPosition),
346 position);
347 }
348
350 Result<Vector3> getField(const Vector3& position,
351 MagneticFieldProvider::Cache& cache) const final {
352 Cache& lcache = cache.as<Cache>();
353 const auto gridPosition = m_cfg.transformPos(position);
354 if (!lcache.fieldCell || !(*lcache.fieldCell).isInside(gridPosition)) {
355 auto res = getFieldCell(position);
356 if (!res.ok()) {
357 return Result<Vector3>::failure(res.error());
358 }
359 lcache.fieldCell = *res;
360 }
361 return Result<Vector3>::success((*lcache.fieldCell).getField(gridPosition));
362 }

367 typename Grid::point_t m_lowerLeft;
368 typename Grid::point_t m_upperRight;
