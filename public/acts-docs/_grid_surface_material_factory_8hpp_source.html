30template <typename material_accessor_t>
31std::unique_ptr<
32 IGridSurfaceMaterial<typename material_accessor_t::grid_value_type>>
33create1D(
34 const ProtoAxis& pAxis, material_accessor_t&& materialAccessor,
35 GridAccess::BoundToGridLocal1DimDelegate boundToGridLocal,
36 GridAccess::GlobalToGridLocal1DimDelegate globalToGridLocal,
37 const std::vector<typename material_accessor_t::grid_value_type>& payload) {
38 // Visit the axis type and create the grid surface material
39 auto ism = pAxis.getAxis().visit(
40 [&]<typename AxisType>(const AxisType& axis)
41 -> std::unique_ptr<IGridSurfaceMaterial<
42 typename material_accessor_t::grid_value_type>> {
43 using GridType =
44 Grid<typename material_accessor_t::grid_value_type, AxisType>;
45 return std::make_unique<
46 GridSurfaceMaterialT<GridType, material_accessor_t>>(
47 GridType(axis), std::forward<material_accessor_t>(materialAccessor),
48 std::move(boundToGridLocal), std::move(globalToGridLocal));
49 });
50 // Fill it via the grid view
51 AnyGridView<typename material_accessor_t::grid_value_type> gv =
52 ism->gridView();
53 auto indices = gv.numLocalBins();
54 for (std::size_t i0 = 0; i0 < indices[0]; ++i0) {
55 // Offset comes from overflow/underflow bin
56 gv.atLocalBins({i0 + 1u}) = payload[i0];
57 }
58 return ism;
59}

72template <typename material_accessor_t>
73std::unique_ptr<
74 IGridSurfaceMaterial<typename material_accessor_t::grid_value_type>>
75create2D(
76 const ProtoAxis& pAxis0, const ProtoAxis& pAxis1,
77 material_accessor_t&& materialAccessor,
78 GridAccess::BoundToGridLocal2DimDelegate boundToGridLocal,
79 GridAccess::GlobalToGridLocal2DimDelegate globalToGridLocal,
80 const std::vector<
81 std::vector<typename material_accessor_t::grid_value_type>>& payload) {
82 // Validate axis compatibility
83 auto ism = pAxis0.getAxis().visit(
84 [&]<typename AxisTypeA>(const AxisTypeA& axisA)
85 -> std::unique_ptr<IGridSurfaceMaterial<
86 typename material_accessor_t::grid_value_type>> {
87 return pAxis1.getAxis().visit(
88 [&]<typename AxisTypeB>(const AxisTypeB& axisB)
89 -> std::unique_ptr<IGridSurfaceMaterial<
90 typename material_accessor_t::grid_value_type>> {
91 using GridType =
92 Grid<typename material_accessor_t::grid_value_type, AxisTypeA,
93 AxisTypeB>;
94 return std::make_unique<
95 GridSurfaceMaterialT<GridType, material_accessor_t>>(
96 GridType(axisA, axisB),
97 std::forward<material_accessor_t>(materialAccessor),
98 std::move(boundToGridLocal), std::move(globalToGridLocal));
99 });
100 });
101
102 // Fill it via the grid view
103 AnyGridView<typename material_accessor_t::grid_value_type> gv =
104 ism->gridView();
105 auto indices = gv.numLocalBins();
106 for (std::size_t i0 = 0; i0 < indices[0]; ++i0) {
107 for (std::size_t i1 = 0; i1 < indices[1]; ++i1) {
108 // Offset comes from overflow/underflow bin
109 gv.atLocalBins({i0 + 1, i1 + 1}) = payload[i0][i1];
110 }
111 }
112
113 return ism;
114}
