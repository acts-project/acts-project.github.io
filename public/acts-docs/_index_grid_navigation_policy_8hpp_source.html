43template <typename GridType>
44class IndexGridNavigationPolicy : public INavigationPolicy {
45 public:
46 using IndexGridType = IndexGrid<GridType>;
47
56 explicit IndexGridNavigationPolicy(const GeometryContext& gctx,
57 const TrackingVolume& volume,
58 const Logger& logger,
59 const IndexGridNavigationConfig& config,
60 const IndexGridType& grid)
61 : m_cfg(config), m_volume(volume), m_indexGrid(grid) {
62 ACTS_VERBOSE("Constructing IndexGridNavigationPolicy for volume '"
63 << m_volume.volumeName() << "'");
64
65 // Fill the grid with the surfaces from the volume
66 IndexGridFiller filler{m_cfg.binExpansion, m_cfg.referenceExpansion,
67 getDefaultLogger("IndexGridFiller", logger.level())};
68 const auto& surfaces = m_volume.surfaces();
69 // Fill the grid with surfaces
70 std::vector<std::shared_ptr<const Surface>> surfacePtrs = {};
71 surfacePtrs.reserve(surfaces.size());
72 for (const auto& surface : surfaces) {
73 surfacePtrs.push_back(surface.getSharedPtr());
74 }
75 filler.fill(gctx, m_indexGrid, surfacePtrs, *m_cfg.referenceGenerator);
76 }
77
83 void initializeCandidates(const GeometryContext& gctx,
84 const NavigationArguments& args,
85 AppendOnlyNavigationStream& stream,
86 const Logger& logger) const {
87 ACTS_VERBOSE(
88 "IndexGridNavigationPolicy: candidates initialization for volume '"
89 << m_volume.volumeName() << "'");
90
91 // Nominal or intersected position
92 Vector3 position = args.position;
93 if (m_cfg.surface) {
94 auto multiIntersection = m_cfg.surface->intersect(
95 gctx, position, args.direction, BoundaryTolerance::Infinite());
96 position = multiIntersection.closestForward().position();
97 }
98
99 // The indexing is guaranteed
100 const auto& surfaces = m_volume.surfaces();
101 const auto& indices =
102 m_indexGrid.grid.atPosition(GridAccessHelpers::castPosition<GridType>(
103 m_indexGrid.transform * position, m_indexGrid.casts));
104 // Fill the navigation stream with the container
105 for (const auto& idx : indices) {
106 stream.addSurfaceCandidate(surfaces[idx], args.tolerance);
107 }
108 }
109
112 void connect(NavigationDelegate& delegate) const override {
113 connectDefault<IndexGridNavigationPolicy>(delegate);
114 }
115
118 const IndexGridType& indexGrid() const { return m_indexGrid; }
119
120 private:
121 // Keep the config
122 IndexGridNavigationConfig m_cfg;
123 // The tracking volume
124 const TrackingVolume& m_volume;
125 // The index grid
126 IndexGridType m_indexGrid;
127};
