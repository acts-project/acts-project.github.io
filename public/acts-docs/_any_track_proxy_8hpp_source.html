274 template <TrackProxyConcept track_proxy_t>
275 requires(ReadOnly || !track_proxy_t::ReadOnly)
276 explicit AnyTrackProxy(track_proxy_t track)
277 : m_container(nullptr), m_index(track.m_index), m_handler(nullptr) {
278 using container_t = std::remove_const_t<
279 std::remove_reference_t<decltype(*track.m_container)>>;
280 auto* containerPtr = &(*track.m_container);
281 if constexpr (ReadOnly) {
282 m_container = static_cast<const void*>(containerPtr);
283 } else {
284 m_container = static_cast<void*>(containerPtr);
285 }
286 m_handler = &detail_anytrack::TrackHandler<container_t>::instance();
287 }
