23concept BasicTrackParameters = requires {
24 typename Parameters::ParametersVector;
25 typename Parameters::CovarianceMatrix;
26
27 requires requires(const Parameters &p) {
28 { p.time() } -> std::floating_point;
29 { p.direction() } -> std::same_as<Vector3>;
30 { p.absoluteMomentum() } -> std::floating_point;
31 { p.charge() } -> std::floating_point;
32 };

38template <typename Parameters>
39concept FreeTrackParametersConcept =
40 Concepts::BasicTrackParameters<Parameters> &&
41 requires(const Parameters &p) {
42 { p.parameters() } -> std::convertible_to<FreeVector>;

44 p.covariance()

52template <typename Parameters>
53concept BoundTrackParametersConcept =
54 Concepts::BasicTrackParameters<Parameters> &&
55 requires(const Parameters &p) {
56 { p.parameters() } -> std::convertible_to<BoundVector>;
57 {
58 p.covariance()
59 } -> std::convertible_to<std::optional<BoundSquareMatrix>>;
60 { p.referenceSurface() } -> std::same_as<const Surface &>;
61
62 requires requires(GeometryContext &c) {
63 { p.position(c) } -> std::same_as<Vector3>;
64 { p.fourPosition(c) } -> std::same_as<Vector4>;
65 { p.position(c) } -> std::same_as<Vector3>;
66 };
