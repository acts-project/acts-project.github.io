130template <typename value_type>
131std::vector<value_type> extractSeries(const dd4hep::DetElement& dd4hepElement,
132 const std::string& bname,
133 const value_type& unitConversion = 1) {
134 std::vector<value_type> series = {};
135
136 int fallBack = 0;
137 int nVals = getParamOr<int>(bname + "_n", dd4hepElement, fallBack);
138 series.reserve(nVals);
139 for (auto ib = 0; ib < nVals; ++ib) {
140 auto val = unitConversion *
141 getParamOr<value_type>(bname + "_" + std::to_string(ib),
142 dd4hepElement, 0.);
143 series.push_back(val);
144 }
145 return series;
146}

155inline Acts::Transform3 extractTransform(
156 const dd4hep::DetElement& dd4hepElement, const std::string& bname,
157 const double unitConversion = 1.) {
158 Acts::Transform3 transform = Acts::Transform3::Identity();
159 double x =
160 unitConversion * getParamOr<double>(bname + "_x", dd4hepElement, 0.);
161 double y =
162 unitConversion * getParamOr<double>(bname + "_y", dd4hepElement, 0.);
163 double z =
164 unitConversion * getParamOr<double>(bname + "_z", dd4hepElement, 0.);
165 transform.pretranslate(Acts::Vector3(x, y, z));
166 return transform;
167}
