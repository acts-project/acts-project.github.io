187 setUnchecked(IsSplitHit, false);

203 assertConsistency();

214 setUnchecked(HasMeasurement, true);

235 assertConsistency();

294 friend std::ostream& operator<<(std::ostream& os,
295 const TrackStateTypeBase& t) {
296 bool first = true;
297 const auto append = [&](const char* name, bool condition) {
298 if (condition) {
299 if (!first) {
300 os << ",";
301 }
302 os << name;
303 first = false;
304 }
305 };
306 append("HasParams", t.hasParameters());
307 if (t.isMaterial()) {
308 append("IsMat", true);
309 } else {
310 append("HasMat", t.hasMaterial());
311 }
312 if (t.isMeasurement()) {
313 append("IsMeas", true);
314 } else if (t.isOutlier()) {
315 append("IsOutlier", true);
316 } else if (t.isHole()) {
317 append("IsHole", true);
318 } else {
319 append("HasMeas", t.hasMeasurement());
320 }
321 append("IsSharedHit", t.isSharedHit());
322 append("IsSplitHit", t.isSplitHit());
323 append("NoExpectedHit", t.hasNoExpectedHit());
324 return os;
325 }

343 self().raw() = bits().set(pos, value).to_ullong();

349 void setUnchecked(TrackStateFlag flag, bool value = true)

352 setUnchecked(toUnderlying(flag), value);
353 }

355 protected:
356 Derived& self() { return static_cast<Derived&>(*this); }
357 const Derived& self() const { return static_cast<const Derived&>(*this); }
358
361 bitset_type bits() const { return bitset_type(self().raw()); }
362
364 void assertConsistency() const {
365 assert(!(test(HasNoExpectedHit) &&
366 (test(HasMeasurement) || test(IsOutlier) || test(IsHole) ||
367 test(IsSharedHit) || test(IsSplitHit))) &&
368 "TrackStateType - NoExpectedHit cannot be set with other "
369 "measurement flags");
370 assert(!(test(IsOutlier) && test(IsHole)) &&
371 "TrackStateType - Outlier and Hole cannot be set simultaneously");
372 assert(
373 !(test(IsHole) && test(HasMeasurement)) &&
374 "TrackStateType - Hole and Measurement cannot be set simultaneously");
375 assert(!(test(IsOutlier) && !test(HasMeasurement)) &&
376 "TrackStateType - Outlier flag requires Measurement flag to be set");
377 }
