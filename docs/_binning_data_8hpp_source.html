182 explicit BinningData(const DirectedProtoAxis& dpAxis)
183 : binvalue(dpAxis.getAxisDirection()), subBinningData(nullptr) {
184 const auto& axis = dpAxis.getAxis();
185 type = axis.getType() == AxisType::Equidistant ? equidistant : arbitrary;
186 option = axis.getBoundaryType() == AxisBoundaryType::Closed ? closed : open;
187 min = static_cast<float>(axis.getMin());
188 max = static_cast<float>(axis.getMax());
189 m_bins = axis.getNBins();
190 step = (max - min) / static_cast<float>(m_bins);
191 zdim = (m_bins == 1);
192 m_boundaries.reserve(axis.getBinEdges().size());
193 for (const auto& edge : axis.getBinEdges()) {
194 m_boundaries.push_back(static_cast<float>(edge));
195 }
196 m_totalBins = m_bins;
197 m_totalBoundaries = m_boundaries;
198 // Set the search function pointer based on axis type
199 m_functionPtr = (type == equidistant) ? &searchEquidistantWithBoundary
200 : &searchInVectorWithBoundary;
201 }

288 float value(const Vector3& position) const {
289 using VectorHelpers::eta;
290 using VectorHelpers::perp;
291 using VectorHelpers::phi;
292 // ordered after occurrence
293 if (binvalue == AxisDirection::AxisR ||
294 binvalue == AxisDirection::AxisTheta) {
295 return (perp(position));
296 }
297 if (binvalue == AxisDirection::AxisRPhi) {
298 return (perp(position) * phi(position));
299 }
300 if (binvalue == AxisDirection::AxisEta) {
301 return (eta(position));
302 }
303 if (toUnderlying(binvalue) < 3) {
304 return static_cast<float>(position[toUnderlying(binvalue)]);
305 }
306 // phi gauging
307 return phi(position);
308 }

316 const std::vector<float>& bvals = boundaries();
317 // take the center between bin boundaries

411 std::size_t searchWithSubStructure(float value) const {
412 // find the masterbin with the correct function pointer
413 std::size_t masterbin = (*m_functionPtr)(value, *this);
414 // additive sub binning -
415 if (subBinningAdditive) {
416 // no gauging done, for additive sub structure
417 return masterbin + subBinningData->search(value);
418 }
419 // gauge the value to the subBinData
420 float gvalue =
421 value - masterbin * (subBinningData->max - subBinningData->min);
422 // now go / additive or multiplicative
423 std::size_t subbin = subBinningData->search(gvalue);
424 // now return
425 return masterbin * subBinningData->bins() + subbin;
426 }

471 void checkSubStructure() {
472 // sub structure is only checked when sBinData is defined
473 if (subBinningData) {
474 m_totalBoundaries.clear();
475 // (A) additive sub structure
476 if (subBinningAdditive) {
477 // one bin is replaced by the sub bins
478 m_totalBins = m_bins + subBinningData->bins() - 1;
479 // the tricky one - exchange one bin by many others
480 m_totalBoundaries.reserve(m_totalBins + 1);
481 // get the sub bin boundaries
482 const std::vector<float>& subBinBoundaries =
483 subBinningData->boundaries();
484 float sBinMin = subBinBoundaries[0];
485 // get the min value of the sub bin boundaries
486 std::vector<float>::const_iterator mbvalue = m_boundaries.begin();
487 for (; mbvalue != m_boundaries.end(); ++mbvalue) {
488 // should define numerically stable
489 if (std::abs((*mbvalue) - sBinMin) < 10e-10) {
490 // copy the sub bin boundaries into the vector
491 m_totalBoundaries.insert(m_totalBoundaries.begin(),
492 subBinBoundaries.begin(),
493 subBinBoundaries.end());
494 ++mbvalue;
495 } else {
496 m_totalBoundaries.push_back(*mbvalue);
497 }
498 }
499 } else { // (B) multiplicative sub structure
500 // every bin is just replaced by the sub binning structure
501 m_totalBins = m_bins * subBinningData->bins();
502 m_totalBoundaries.reserve(m_totalBins + 1);
503 // get the sub bin boundaries if there are any
504 const std::vector<float>& subBinBoundaries =
505 subBinningData->boundaries();
506 // create the boundary vector
507 m_totalBoundaries.push_back(min);
508 for (std::size_t ib = 0; ib < m_bins; ++ib) {
509 float offset = ib * step;
510 for (std::size_t isb = 1; isb < subBinBoundaries.size(); ++isb) {
511 m_totalBoundaries.push_back(offset + subBinBoundaries[isb]);
512 }
513 }
514 }
515 // sort the total boundary vector
516 std::ranges::sort(m_totalBoundaries);
517 }
518 }

520 // Equidistant search
521 // - fastest method
522 static std::size_t searchEquidistantWithBoundary(float value,
523 const BinningData& bData) {
524 // vanilla
525
526 int bin = static_cast<int>((value - bData.min) / bData.step);
527 // special treatment of the 0 bin for closed
528 if (bData.option == closed) {
529 if (value < bData.min) {
530 return (bData.m_bins - 1);
531 }
532 if (value > bData.max) {
533 return 0;
534 }
535 }
536 // if outside boundary : return boundary for open, opposite bin for closed
537 bin = bin < 0 ? ((bData.option == open) ? 0 : (bData.m_bins - 1)) : bin;
538 return static_cast<std::size_t>(
539 (bin <= static_cast<int>(bData.m_bins - 1))
540 ? bin
541 : ((bData.option == open) ? (bData.m_bins - 1) : 0));
542 }
543
544 // Search in arbitrary boundary
545 static std::size_t searchInVectorWithBoundary(float value,
546 const BinningData& bData) {
547 // lower boundary
548 if (value <= bData.m_boundaries[0]) {
549 return (bData.option == closed) ? (bData.m_bins - 1) : 0;
550 }
551 // higher boundary
552 if (value >= bData.max) {
553 return (bData.option == closed) ? 0 : (bData.m_bins - 1);
554 }
555
556 auto lb = std::lower_bound(bData.m_boundaries.begin(),
557 bData.m_boundaries.end(), value);
558 return static_cast<std::size_t>(
559 std::distance(bData.m_boundaries.begin(), lb) - 1);
560 }
