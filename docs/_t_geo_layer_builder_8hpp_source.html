232inline void TGeoLayerBuilder::registerSplit(
233 std::vector<double>& parameters, double test, double tolerance,
234 std::pair<double, double>& range) const {
235 bool found = false;
236 // min/max setting
237 range.first = std::min(range.first, test);
238 range.second = std::max(range.second, test);
239 // Loop and find the split parameters
240 for (const auto& splitPar : parameters) {
241 if (std::abs(test - splitPar) < tolerance) {
242 found = true;
243 }
244 }
245 if (!found) {
246 parameters.push_back(test);
247 }
248}
