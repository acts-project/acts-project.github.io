22// Simple wrapper around boost::disjoint_sets. In theory, could use
23// boost::vector_property_map and use boost::disjoint_sets without
24// wrapping, but it's way slower
25class DisjointSets {
26 public:
27 explicit DisjointSets(std::size_t initial_size = 128)
28 : m_defaultSize(initial_size),
29 m_size(initial_size),
30 m_rank(m_size),
31 m_parent(m_size),
32 m_ds(&m_rank[0], &m_parent[0]) {}
33
34 Acts::Ccl::Label makeSet() {
35 // Empirically, m_size = 128 seems to be good default. If we
36 // exceed this, take a performance hit and do the right thing.
37 while (m_globalId >= m_size) {
38 m_size *= 2;
39 m_rank.resize(m_size);
40 m_parent.resize(m_size);
41 m_ds = boost::disjoint_sets<std::size_t*, std::size_t*>(&m_rank[0],
42 &m_parent[0]);
43 }
44 m_ds.make_set(m_globalId);
45 return static_cast<Acts::Ccl::Label>(m_globalId++);
46 }
47
48 void unionSet(std::size_t x, std::size_t y) { m_ds.union_set(x, y); }
49 Acts::Ccl::Label findSet(std::size_t x) {
50 return static_cast<Acts::Ccl::Label>(m_ds.find_set(x));
51 }
52
53 void clear() {
54 m_size = m_defaultSize;
55 m_rank.clear();
56 m_parent.clear();
57 m_rank.resize(m_size);
58 m_parent.resize(m_size);
59 m_globalId = 1;
60 m_ds = boost::disjoint_sets<std::size_t*, std::size_t*>(&m_rank[0],
61 &m_parent[0]);
62 }
63
64 private:
65 std::size_t m_defaultSize{128};
66 std::size_t m_globalId = 1;
67 std::size_t m_size{m_defaultSize};
68 std::vector<std::size_t> m_rank;
69 std::vector<std::size_t> m_parent;
70 boost::disjoint_sets<std::size_t*, std::size_t*> m_ds;
71};

105// When looking for a cell connected to a reference cluster, the code
106// always loops backward, starting from the reference cell. Since
107// the cells are globally sorted column-wise, the connection function
108// can therefore tell when the search should be stopped.
109enum class ConnectResult {
110 eNoConn, // No connections, keep looking
111 eNoConnStop, // No connections, stop looking
112 eConn, // Found connection
113 eDuplicate // Found duplicate cell, throw an exception
114};

116// Default connection type for 2-D grids: 4- or 8-cell connectivity
117template <typename Cell>
118 requires(Acts::Ccl::HasRetrievableColumnInfo<Cell> &&
119 Acts::Ccl::HasRetrievableRowInfo<Cell>)
120struct Connect2D {
121 bool conn8{true};
122 Connect2D() = default;
123 explicit Connect2D(bool commonCorner) : conn8{commonCorner} {}
124 virtual ConnectResult operator()(const Cell& ref, const Cell& iter) const;
125 virtual ~Connect2D() = default;
126};
127
128// Default connection type for 1-D grids: 2-cell connectivity
129template <Acts::Ccl::HasRetrievableColumnInfo Cell>
130struct Connect1D {
131 virtual ConnectResult operator()(const Cell& ref, const Cell& iter) const;
132 virtual ~Connect1D() = default;
133};

135// Default connection type based on GridDim
136template <typename Cell, std::size_t GridDim = 2>
137struct DefaultConnect {
138 static_assert(GridDim != 1 && GridDim != 2,
139 "Only grid dimensions of 1 or 2 are supported");
140};
141
142template <typename Cell>
143struct DefaultConnect<Cell, 1> : public Connect1D<Cell> {
144 ~DefaultConnect() override = default;
145};
146
147template <typename Cell>
148struct DefaultConnect<Cell, 2> : public Connect2D<Cell> {
149 explicit DefaultConnect(bool commonCorner) : Connect2D<Cell>(commonCorner) {}
150 DefaultConnect() = default;
151 ~DefaultConnect() override = default;
152};
