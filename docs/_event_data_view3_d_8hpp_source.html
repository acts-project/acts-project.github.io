40 static inline std::array<double, 3> decomposeCovariance(
41 const ActsSquareMatrix<2>& covariance) {
42 double c00 = covariance(eBoundLoc0, eBoundLoc0);
43 double c01 = covariance(eBoundLoc0, eBoundLoc1);
44 double c11 = covariance(eBoundLoc1, eBoundLoc1);
45
46 double cdsq = std::pow((c00 - c11), 2) / 4.;
47 double cosq = c01 * c01;
48
49 // Calculate the eigen values w.r.t reference frame
50 double lambda0 = (c00 + c11) / 2. + std::sqrt(cdsq + cosq);
51 double lambda1 = (c00 + c11) / 2. - std::sqrt(cdsq + cosq);
52 double theta = std::atan2(lambda0 - c00, c01);
53
54 return {lambda0, lambda1, theta};
55 }

66 static inline std::vector<Vector3> createEllipse(
67 double lambda0, double lambda1, double theta, std::size_t lseg,
68 double offset, const Vector2& lposition = Vector2(0., 0.),
69 const Transform3& transform = Transform3::Identity()) {
70 double ctheta = std::cos(theta);
71 double stheta = std::sin(theta);
72
73 double l1sq = std::sqrt(lambda0);
74 double l2sq = std::sqrt(lambda1);
75
76 // Now generate the ellipse points
77 std::vector<Vector3> ellipse;
78 ellipse.reserve(lseg);
79 double thetaStep = 2 * std::numbers::pi / lseg;
80 for (std::size_t it = 0; it < lseg; ++it) {
81 double phi = -std::numbers::pi + it * thetaStep;
82 double cphi = std::cos(phi);
83 double sphi = std::sin(phi);
84 double x = lposition.x() + (l1sq * ctheta * cphi - l2sq * stheta * sphi);
85 double y = lposition.y() + (l1sq * stheta * cphi + l2sq * ctheta * sphi);
86 ellipse.push_back(transform * Vector3(x, y, offset));
87 }
88 return ellipse;
89 }

130 template <typename parameters_t>
131 static inline void drawBoundTrackParameters(
132 IVisualization3D& helper, const parameters_t& parameters,
133 const GeometryContext& gctx =
134 GeometryContext::dangerouslyDefaultConstruct(),
135 double momentumScale = 1., double locErrorScale = 1.,
136 double angularErrorScale = 1.,
137 const ViewConfig& parConfig = s_viewParameter,
138 const ViewConfig& covConfig = s_viewParameter,
139 const ViewConfig& surfConfig = s_viewSensitive) {
140 if (surfConfig.visible) {
141 GeometryView3D::drawSurface(helper, parameters.referenceSurface(), gctx,
142 Transform3::Identity(), surfConfig);
143 }
144
145 // Draw the parameter shaft and cone
146 auto position = parameters.position(gctx);
147 auto direction = parameters.direction();
148 double p = parameters.absoluteMomentum();
149
150 ViewConfig lparConfig = parConfig;
151 lparConfig.lineThickness = 0.05;
152 Vector3 parLength = p * momentumScale * direction;
153
154 GeometryView3D::drawArrowBackward(
155 helper, position, position + 0.5 * parLength, 100., 1.0, lparConfig);
156
157 GeometryView3D::drawArrowForward(helper, position + 0.5 * parLength,
158 position + parLength, 4., 2.5, lparConfig);
159
160 if (parameters.covariance().has_value()) {
161 auto paramVec = parameters.parameters();
162 auto lposition = paramVec.template block<2, 1>(0, 0);
163
164 // Draw the local covariance
165 const auto& covariance = *parameters.covariance();
166 drawCovarianceCartesian(
167 helper, lposition, covariance.template block<2, 2>(0, 0),
168 parameters.referenceSurface().localToGlobalTransform(gctx),
169 locErrorScale, covConfig);
170
171 drawCovarianceAngular(
172 helper, position, direction, covariance.template block<2, 2>(2, 2),
173 0.9 * p * momentumScale, angularErrorScale, covConfig);
174 }
175 }

218 template <typename traj_t>
219 static void drawMultiTrajectory(
220 IVisualization3D& helper, const traj_t& multiTraj,
221 const std::size_t& entryIndex,
222 const GeometryContext& gctx =
223 GeometryContext::dangerouslyDefaultConstruct(),
224 double momentumScale = 1., double locErrorScale = 1.,
225 double angularErrorScale = 1.,
226 const ViewConfig& surfaceConfig = s_viewSensitive,
227 const ViewConfig& measurementConfig = s_viewMeasurement,
228 const ViewConfig& predictedConfig = s_viewPredicted,
229 const ViewConfig& filteredConfig = s_viewFiltered,
230 const ViewConfig& smoothedConfig = s_viewSmoothed) {
231 // @TODO: Refactor based on Track class
232
233 // TODO get particle hypothesis from track
234 ParticleHypothesis particleHypothesis = ParticleHypothesis::pion();
235
236 // Visit the track states on the trajectory
237 multiTraj.visitBackwards(entryIndex, [&](const auto& state) {
238 // Only draw the measurement states
239 if (!state.typeFlags().hasMeasurement()) {
240 return true;
241 }
242
243 // Use smaller scaling factors for the first state
244 // @Todo: add parameter for the first state error scaling
245 if (state.index() == 0) {
246 locErrorScale = locErrorScale * 0.1;
247 angularErrorScale = angularErrorScale * 0.1;
248 }
249
250 // First, if necessary, draw the surface
251 if (surfaceConfig.visible) {
252 GeometryView3D::drawSurface(helper, state.referenceSurface(), gctx,
253 Transform3::Identity(), surfaceConfig);
254 }
255
256 // Second, if necessary and present, draw the calibrated measurement (only
257 // draw 2D measurement here)
258 // @Todo: how to draw 1D measurement?
259 if (state.hasCalibrated() && state.calibratedSize() == 2) {
260 const Vector2& lposition = state.template calibrated<2>();
261 const SquareMatrix2 covariance =
262 state.template calibratedCovariance<2>();
263 drawMeasurement(helper, lposition, covariance,
264 state.referenceSurface().localToGlobalTransform(gctx),
265 locErrorScale, measurementConfig);
266 }
267
268 // Last, if necessary and present, draw the track parameters
269 // (a) predicted track parameters
270 if (predictedConfig.visible && state.hasPredicted()) {
271 drawBoundTrackParameters(
272 helper,
273 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
274 state.predicted(), state.predictedCovariance(),
275 particleHypothesis),
276 gctx, momentumScale, locErrorScale, angularErrorScale,
277 predictedConfig, predictedConfig, {.visible = false});
278 }
279 // (b) filtered track parameters
280 if (filteredConfig.visible && state.hasFiltered()) {
281 drawBoundTrackParameters(
282 helper,
283 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
284 state.filtered(), state.filteredCovariance(),
285 particleHypothesis),
286 gctx, momentumScale, locErrorScale, angularErrorScale,
287 filteredConfig, filteredConfig, {.visible = false});
288 }
289 // (c) smoothed track parameters
290 if (smoothedConfig.visible && state.hasSmoothed()) {
291 drawBoundTrackParameters(
292 helper,
293 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
294 state.smoothed(), state.smoothedCovariance(),
295 particleHypothesis),
296 gctx, momentumScale, locErrorScale, angularErrorScale,
297 smoothedConfig, smoothedConfig, {.visible = false});
298 }
299 return true;
300 });
301 }
