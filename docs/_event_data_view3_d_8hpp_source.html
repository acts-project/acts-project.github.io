46 static inline std::array<double, 3> decomposeCovariance(
47 const ActsSquareMatrix<2>& covariance) {
48 double c00 = covariance(eBoundLoc0, eBoundLoc0);
49 double c01 = covariance(eBoundLoc0, eBoundLoc1);
50 double c11 = covariance(eBoundLoc1, eBoundLoc1);
51
52 double cdsq = std::pow((c00 - c11), 2) / 4.;
53 double cosq = c01 * c01;
54
55 // Calculate the eigen values w.r.t reference frame
56 double lambda0 = (c00 + c11) / 2. + std::sqrt(cdsq + cosq);
57 double lambda1 = (c00 + c11) / 2. - std::sqrt(cdsq + cosq);
58 double theta = std::atan2(lambda0 - c00, c01);
59
60 return {lambda0, lambda1, theta};
61 }

72 static inline std::vector<Vector3> createEllipse(
73 double lambda0, double lambda1, double theta, std::size_t lseg,
74 double offset, const Vector2& lposition = Vector2(0., 0.),
75 const Transform3& transform = Transform3::Identity()) {
76 double ctheta = std::cos(theta);
77 double stheta = std::sin(theta);
78
79 double l1sq = std::sqrt(lambda0);
80 double l2sq = std::sqrt(lambda1);
81
82 // Now generate the ellipse points
83 std::vector<Vector3> ellipse;
84 ellipse.reserve(lseg);
85 double thetaStep = 2 * std::numbers::pi / lseg;
86 for (std::size_t it = 0; it < lseg; ++it) {
87 double phi = -std::numbers::pi + it * thetaStep;
88 double cphi = std::cos(phi);
89 double sphi = std::sin(phi);
90 double x = lposition.x() + (l1sq * ctheta * cphi - l2sq * stheta * sphi);
91 double y = lposition.y() + (l1sq * stheta * cphi + l2sq * ctheta * sphi);
92 ellipse.push_back(transform * Vector3(x, y, offset));
93 }
94 return ellipse;
95 }

136 template <typename parameters_t>
137 static inline void drawBoundTrackParameters(
138 IVisualization3D& helper, const parameters_t& parameters,
139 const GeometryContext& gctx = GeometryContext(),
140 double momentumScale = 1., double locErrorScale = 1.,
141 double angularErrorScale = 1.,
142 const ViewConfig& parConfig = s_viewParameter,
143 const ViewConfig& covConfig = s_viewParameter,
144 const ViewConfig& surfConfig = s_viewSensitive) {
145 if (surfConfig.visible) {
146 GeometryView3D::drawSurface(helper, parameters.referenceSurface(), gctx,
147 Transform3::Identity(), surfConfig);
148 }
149
150 // Draw the parameter shaft and cone
151 auto position = parameters.position(gctx);
152 auto direction = parameters.direction();
153 double p = parameters.absoluteMomentum();
154
155 ViewConfig lparConfig = parConfig;
156 lparConfig.lineThickness = 0.05;
157 Vector3 parLength = p * momentumScale * direction;
158
159 GeometryView3D::drawArrowBackward(
160 helper, position, position + 0.5 * parLength, 100., 1.0, lparConfig);
161
162 GeometryView3D::drawArrowForward(helper, position + 0.5 * parLength,
163 position + parLength, 4., 2.5, lparConfig);
164
165 if (parameters.covariance().has_value()) {
166 auto paramVec = parameters.parameters();
167 auto lposition = paramVec.template block<2, 1>(0, 0);
168
169 // Draw the local covariance
170 const auto& covariance = *parameters.covariance();
171 drawCovarianceCartesian(
172 helper, lposition, covariance.template block<2, 2>(0, 0),
173 parameters.referenceSurface().localToGlobalTransform(gctx),
174 locErrorScale, covConfig);
175
176 drawCovarianceAngular(
177 helper, position, direction, covariance.template block<2, 2>(2, 2),
178 0.9 * p * momentumScale, angularErrorScale, covConfig);
179 }
180 }

223 template <typename traj_t>
224 static void drawMultiTrajectory(
225 IVisualization3D& helper, const traj_t& multiTraj,
226 const std::size_t& entryIndex,
227 const GeometryContext& gctx = GeometryContext(),
228 double momentumScale = 1., double locErrorScale = 1.,
229 double angularErrorScale = 1.,
230 const ViewConfig& surfaceConfig = s_viewSensitive,
231 const ViewConfig& measurementConfig = s_viewMeasurement,
232 const ViewConfig& predictedConfig = s_viewPredicted,
233 const ViewConfig& filteredConfig = s_viewFiltered,
234 const ViewConfig& smoothedConfig = s_viewSmoothed) {
235 // @TODO: Refactor based on Track class
236
237 // TODO get particle hypothesis from track
238 ParticleHypothesis particleHypothesis = ParticleHypothesis::pion();
239
240 // Visit the track states on the trajectory
241 multiTraj.visitBackwards(entryIndex, [&](const auto& state) {
242 // Only draw the measurement states
243 if (!state.typeFlags().test(Acts::TrackStateFlag::MeasurementFlag)) {
244 return true;
245 }
246
247 // Use smaller scaling factors for the first state
248 // @Todo: add parameter for the first state error scaling
249 if (state.index() == 0) {
250 locErrorScale = locErrorScale * 0.1;
251 angularErrorScale = angularErrorScale * 0.1;
252 }
253
254 // First, if necessary, draw the surface
255 if (surfaceConfig.visible) {
256 GeometryView3D::drawSurface(helper, state.referenceSurface(), gctx,
257 Transform3::Identity(), surfaceConfig);
258 }
259
260 // Second, if necessary and present, draw the calibrated measurement (only
261 // draw 2D measurement here)
262 // @Todo: how to draw 1D measurement?
263 if (state.hasCalibrated() && state.calibratedSize() == 2) {
264 const Vector2& lposition = state.template calibrated<2>();
265 const SquareMatrix2 covariance =
266 state.template calibratedCovariance<2>();
267 drawMeasurement(helper, lposition, covariance,
268 state.referenceSurface().localToGlobalTransform(gctx),
269 locErrorScale, measurementConfig);
270 }
271
272 // Last, if necessary and present, draw the track parameters
273 // (a) predicted track parameters
274 if (predictedConfig.visible && state.hasPredicted()) {
275 drawBoundTrackParameters(
276 helper,
277 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
278 state.predicted(), state.predictedCovariance(),
279 particleHypothesis),
280 gctx, momentumScale, locErrorScale, angularErrorScale,
281 predictedConfig, predictedConfig, {.visible = false});
282 }
283 // (b) filtered track parameters
284 if (filteredConfig.visible && state.hasFiltered()) {
285 drawBoundTrackParameters(
286 helper,
287 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
288 state.filtered(), state.filteredCovariance(),
289 particleHypothesis),
290 gctx, momentumScale, locErrorScale, angularErrorScale,
291 filteredConfig, filteredConfig, {.visible = false});
292 }
293 // (c) smoothed track parameters
294 if (smoothedConfig.visible && state.hasSmoothed()) {
295 drawBoundTrackParameters(
296 helper,
297 BoundTrackParameters(state.referenceSurface().getSharedPtr(),
298 state.smoothed(), state.smoothedCovariance(),
299 particleHypothesis),
300 gctx, momentumScale, locErrorScale, angularErrorScale,
301 smoothedConfig, smoothedConfig, {.visible = false});
302 }
303 return true;
304 });
305 }
