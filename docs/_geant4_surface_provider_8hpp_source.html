38template <std::size_t kDim = 2u, std::size_t bSize = 100u>
39class Geant4SurfaceProvider : public Acts::ISurfacesProvider {
40 public:
42 struct Config {
44 const G4VPhysicalVolume* g4World = nullptr;
45
47 double scaleConversion = 1.;
48
50 bool convertMaterial = true;
51
53 double convertedMaterialThickness = -1;
54
57 G4Transform3D worldTransform = G4Transform3D();
58
60 std::shared_ptr<IGeant4PhysicalVolumeSelector> surfacePreselector =
61 std::make_shared<Geant4PhysicalVolumeSelectors::AllSelector>();
62 };
63
65 struct KdtOptions {
67 Acts::RangeXD<kDim, double> range;
68
70 std::array<Acts::AxisDirection, kDim> binningValues;
71
73 std::size_t leafSize = bSize;
74
76 std::shared_ptr<IReferenceGenerator> referenceGenerator =
77 std::make_shared<PolyhedronReferenceGenerator>();
78
80 KdtOptions() {
81 for (std::size_t i = 0; i < kDim; ++i) {
82 range[i].set(1, -1);
83 }
84 }
85 };

90 explicit Geant4SurfaceProvider(const Config& config,
91 const KdtOptions& options = KdtOptions()) {
92 if (config.g4World == nullptr) {
93 throw std::invalid_argument(
94 "Geant4SurfaceProvider: No World volume provided");
95 }
96 if (config.surfacePreselector == nullptr) {
97 throw std::invalid_argument(
98 "Geant4SurfaceProvider: no preselection criteria provided");
99 }
100
101 m_cfg = config;
102 m_kdtOptions = options;
103 m_g4World = m_cfg.g4World;
104 m_g4ToWorld = m_cfg.worldTransform;
105 };
106
108 ~Geant4SurfaceProvider() override = default;
109
110 std::vector<std::shared_ptr<Acts::Surface>> surfaces(
111 [[maybe_unused]] const Acts::GeometryContext& gctx) const override {
113 Geant4DetectorSurfaceFactory::Options g4SurfaceOptions;
114
118 g4SurfaceOptions.scaleConversion = m_cfg.scaleConversion;
119 g4SurfaceOptions.convertMaterial = m_cfg.convertMaterial;
120 g4SurfaceOptions.convertedMaterialThickness =
121 m_cfg.convertedMaterialThickness;
122 g4SurfaceOptions.passiveSurfaceSelector = m_cfg.surfacePreselector;
123
125 ActsPlugins::Geant4DetectorSurfaceFactory::Cache g4SurfaceCache;
126
128 auto g4ToWorldConsistent = G4Transform3D(
129 m_g4ToWorld.getRotation().inverse(), m_g4ToWorld.getTranslation());
130
131 Geant4DetectorSurfaceFactory::Config surfaceConfig;
132 Geant4DetectorSurfaceFactory(surfaceConfig)
133 .construct(g4SurfaceCache, g4ToWorldConsistent, *m_g4World,
134 g4SurfaceOptions);
135
136 auto surfaces = g4SurfaceCache.passiveSurfaces;
137
139 if (m_kdtOptions.range.degenerate()) {
140 return surfaces;
141 }
142
144 auto kdtSurfaces =
145 KdtSurfaces<kDim, bSize>(gctx, surfaces, m_kdtOptions.binningValues,
146 m_kdtOptions.referenceGenerator);
147
148 return kdtSurfaces.surfaces(m_kdtOptions.range);
149 };
150
151 private:
152 Config m_cfg;
153
154 KdtOptions m_kdtOptions;
155
156 const G4VPhysicalVolume* m_g4World;
157
158 G4Transform3D m_g4ToWorld;
159};
