30template <typename MatrixType>
31MatrixType bitsetToMatrix(const std::bitset<MatrixType::RowsAtCompileTime *
32 MatrixType::ColsAtCompileTime>
33 bs) {
34 constexpr int rows = MatrixType::RowsAtCompileTime;
35 constexpr int cols = MatrixType::ColsAtCompileTime;
36
37 static_assert(rows != -1 && cols != -1,
38 "bitsetToMatrix does not support dynamic matrices");
39
40 MatrixType m;
41 auto* p = m.data();
42 for (std::size_t i = 0; i < rows * cols; i++) {
43 p[i] = bs[rows * cols - 1 - i];
44 }
45 return m;
46}

54template <typename Derived>
55auto matrixToBitset(const Eigen::PlainObjectBase<Derived>& m) {
56 using MatrixType = Eigen::PlainObjectBase<Derived>;
57 constexpr std::size_t rows = MatrixType::RowsAtCompileTime;
58 constexpr std::size_t cols = MatrixType::ColsAtCompileTime;
59
60 std::bitset<rows * cols> res;
61
62 auto* p = m.data();
63 for (std::size_t i = 0; i < rows * cols; i++) {
64 res[rows * cols - 1 - i] = static_cast<bool>(p[i]);
65 }
66
67 return res;
68}

80template <typename A, typename B>
81inline ActsMatrix<A::RowsAtCompileTime, B::ColsAtCompileTime> blockedMult(
82 const A& a, const B& b) {
83 // Extract the sizes of the matrix types that we receive as template
84 // parameters.
85 constexpr int M = A::RowsAtCompileTime;
86 constexpr int N = A::ColsAtCompileTime;
87 constexpr int P = B::ColsAtCompileTime;
88
89 // Ensure that the second dimension of our first matrix equals the first
90 // dimension of the second matrix, otherwise we cannot multiply.
91 static_assert(N == B::RowsAtCompileTime);
92
93 if constexpr (M <= 4 && N <= 4 && P <= 4) {
94 // In cases where the matrices being multiplied are small, we can rely on
95 // Eigen do to a good job, and we don't really need to do any blocking.
96 return a * b;
97 } else {
98 // Here, we want to calculate the expression: C = AB, Eigen, natively,
99 // doesn't do a great job at this if the matrices A and B are large
100 // (roughly M >= 8, N >= 8, or P >= 8), and applies a slow GEMM operation.
101 // We apply a blocked matrix multiplication operation to decompose the
102 // multiplication into smaller operations, relying on the fact that:
103 //
104 // ┌ ┐ ┌ ┐ ┌ ┐
105 // │ C₁₁ C₁₂ │ = │ A₁₁ A₁₂ │ │ B₁₁ B₁₂ │
106 // │ C₂₁ C₂₂ │ = │ A₂₁ A₂₂ │ │ B₂₁ B₂₂ │
107 // └ ┘ └ ┘ └ ┘
108 //
109 // where:
110 //
111 // C₁₁ = A₁₁ * B₁₁ + A₁₂ * B₂₁
112 // C₁₂ = A₁₁ * B₁₂ + A₁₂ * B₂₂
113 // C₂₁ = A₂₁ * B₁₁ + A₂₂ * B₂₁
114 // C₂₂ = A₂₁ * B₁₂ + A₂₂ * B₂₂
115 //
116 // The sizes of these submatrices are roughly half (in each dimension) that
117 // of the parent matrix. If the size of the parent matrix is even, we can
118 // divide it exactly, If the size of the parent matrix is odd, then some
119 // of the submatrices will be one larger than the others. In general, for
120 // any matrix Q, the sizes of the submatrices are (where / denotes integer
121 // division):
122 //
123 // Q₁₁ : M / 2 × P / 2
124 // Q₁₂ : M / 2 × (P + 1) / 2
125 // Q₂₁ : (M + 1) / 2 × P / 2
126 // Q₂₂ : (M + 1) / 2 × (P + 1) / 2
127 //
128 // See https://csapp.cs.cmu.edu/public/waside/waside-blocking.pdf for a
129 // more in-depth explanation of blocked matrix multiplication.
130 constexpr int M1 = M / 2;
131 constexpr int M2 = (M + 1) / 2;
132 constexpr int N1 = N / 2;
133 constexpr int N2 = (N + 1) / 2;
134 constexpr int P1 = P / 2;
135 constexpr int P2 = (P + 1) / 2;
136
137 // Construct the end result in this matrix, which destroys a few of Eigen's
138 // built-in optimization techniques, but sadly this is necessary.
139 ActsMatrix<M, P> r;
140
141 // C₁₁ = A₁₁ * B₁₁ + A₁₂ * B₂₁
142 r.template topLeftCorner<M1, P1>().noalias() =
143 a.template topLeftCorner<M1, N1>() *
144 b.template topLeftCorner<N1, P1>() +
145 a.template topRightCorner<M1, N2>() *
146 b.template bottomLeftCorner<N2, P1>();
147
148 // C₁₂ = A₁₁ * B₁₂ + A₁₂ * B₂₂
149 r.template topRightCorner<M1, P2>().noalias() =
150 a.template topLeftCorner<M1, N1>() *
151 b.template topRightCorner<N1, P2>() +
152 a.template topRightCorner<M1, N2>() *
153 b.template bottomRightCorner<N2, P2>();
154
155 // C₂₁ = A₂₁ * B₁₁ + A₂₂ * B₂₁
156 r.template bottomLeftCorner<M2, P1>().noalias() =
157 a.template bottomLeftCorner<M2, N1>() *
158 b.template topLeftCorner<N1, P1>() +
159 a.template bottomRightCorner<M2, N2>() *
160 b.template bottomLeftCorner<N2, P1>();
161
162 // C₂₂ = A₂₁ * B₁₂ + A₂₂ * B₂₂
163 r.template bottomRightCorner<M2, P2>().noalias() =
164 a.template bottomLeftCorner<M2, N1>() *
165 b.template topRightCorner<N1, P2>() +
166 a.template bottomRightCorner<M2, N2>() *
167 b.template bottomRightCorner<N2, P2>();
168
169 return r;
170 }
171}

192template <typename MatrixType, typename ResultType = MatrixType>
193std::optional<ResultType> safeInverse(const MatrixType& m) noexcept {
194 constexpr int rows = MatrixType::RowsAtCompileTime;
195 constexpr int cols = MatrixType::ColsAtCompileTime;
196
197 static_assert(rows == cols);
198
199 ResultType result;
200 bool invertible = false;
201
202 if constexpr (rows > 4 || rows == -1) {
203 Eigen::FullPivLU<MatrixType> mFullPivLU(m);
204 if (mFullPivLU.isInvertible()) {
205 invertible = true;
206 result = mFullPivLU.inverse();
207 }
208 } else {
209 m.computeInverseWithCheck(result, invertible);
210 }
211
212 if (invertible) {
213 return result;
214 }
215
216 return std::nullopt;
217}

239template <typename T>
240constexpr T safeExp(T val) noexcept {
241 constexpr T maxExponent = ExpSafeLimit<T>::value;
242 constexpr T minExponent = -maxExponent;
243 if (val < minExponent) {
244 return 0.0;
245 }
246
247 if (val > maxExponent) {
248 return std::numeric_limits<T>::infinity();
249 }
250
251 return std::exp(val);
252}

259template <std::size_t N>
260constexpr std::size_t vecIdxFromSymMat(const std::size_t i, const std::size_t k)
261 requires(N > 0)
262{
263 assert(i < N);
264 assert(k < N);
265 if (k > i) {
266 return vecIdxFromSymMat<N>(k, i);
267 }
268 return sumUpToN(i) + k;
269}

276template <std::size_t N>
277constexpr std::array<std::size_t, 2> symMatIndices(const std::size_t k)
278 requires(N > 1)
279{
280 assert(k < sumUpToN(N));
281 constexpr std::size_t bound = sumUpToN(N - 1);
282 if (k >= bound) {
283 return std::array<std::size_t, 2>{N - 1, k - bound};
284 }
285 if constexpr (N > 2) {
286 return symMatIndices<N - 1>(k);
287 }
288 return filledArray<std::size_t, 2>(0);
289}
