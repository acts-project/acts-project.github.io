78 template <typename T>
79 static constexpr bool has_impl(T& instance, Acts::HashedString key,
80 Acts::TrackIndexType istate) {
81 constexpr auto kInvalid = Acts::kTrackIndexInvalid;
82 using namespace Acts::HashedStringLiteral;
83 auto trackState = instance.m_collection->at(istate);
84 const auto& data = trackState.getData();
85 switch (key) {
86 case "predicted"_hash:
87 return data.ipredicted != kInvalid;
88 case "filtered"_hash:
89 return data.ifiltered != kInvalid;
90 case "smoothed"_hash:
91 return data.ismoothed != kInvalid;
92 case "calibrated"_hash:
93 return data.measdim != kInvalid;
94 case "calibratedCov"_hash:
95 return data.measdim != kInvalid;
96 case "jacobian"_hash:
97 return data.ijacobian != kInvalid;
98 case "projector"_hash:
99 return data.hasProjector;
100 case "uncalibratedSourceLink"_hash:
101 return data.uncalibratedIdentifier != PodioUtil::kNoIdentifier;
102 case "previous"_hash:
103 case "next"_hash:
104 case "measdim"_hash:
105 case "referenceSurface"_hash:
106 case "chi2"_hash:
107 case "pathLength"_hash:
108 case "typeFlags"_hash:
109 return true;
110 default:
111 return instance.m_dynamic.contains(key);
112 }
113
114 return false;
115 }

117 template <bool EnsureConst, typename T>
118 static std::any component_impl(T& instance, Acts::HashedString key,
119 Acts::TrackIndexType istate) {
120 if constexpr (EnsureConst) {
121 static_assert(std::is_const_v<std::remove_reference_t<T>>,
122 "Is not const");
123 }
124 using namespace Acts::HashedStringLiteral;
125 auto trackState = instance.m_collection->at(istate);
126 std::conditional_t<EnsureConst, const ActsPodioEdm::TrackStateInfo*,
127 ActsPodioEdm::TrackStateInfo*>
128 dataPtr;
129 if constexpr (EnsureConst) {
130 dataPtr = &trackState.getData();
131 } else {
132 dataPtr = &PodioUtil::getDataMutable(trackState);
133 }
134 auto& data = *dataPtr;
135 switch (key) {
136 case "previous"_hash:
137 return &data.previous;
138 case "next"_hash:
139 return &data.next;
140 case "predicted"_hash:
141 return &data.ipredicted;
142 case "filtered"_hash:
143 return &data.ifiltered;
144 case "smoothed"_hash:
145 return &data.ismoothed;
146 case "projector"_hash:
147 return &data.projector;
148 case "measdim"_hash:
149 return &data.measdim;
150 case "chi2"_hash:
151 return &data.chi2;
152 case "pathLength"_hash:
153 return &data.pathLength;
154 case "typeFlags"_hash:
155 return &data.typeFlags;
156 default:
157 auto it = instance.m_dynamic.find(key);
158 if (it == instance.m_dynamic.end()) {
159 throw std::runtime_error("Unable to handle this component");
160 }
161 std::conditional_t<EnsureConst,
162 const podio_detail::ConstDynamicColumnBase*,
163 podio_detail::DynamicColumnBase*>
164 col = it->second.get();
165 assert(col && "Dynamic column is null");
166 return col->get(istate);
167 }
168 }

177 case "jacobian"_hash:

193 static void populateSurfaceBuffer(
194 const PodioUtil::ConversionHelper& helper,
195 const ActsPodioEdm::TrackStateCollection& collection,
196 std::vector<std::shared_ptr<const Acts::Surface>>& surfaces) noexcept {
197 surfaces.reserve(collection.size());
198 for (ActsPodioEdm::TrackState trackState : collection) {
199 surfaces.push_back(PodioUtil::convertSurfaceFromPodio(
200 helper, trackState.getReferenceSurface()));
201 }
202 }

214 : m_helper{helper},

226 explicit ConstPodioTrackStateContainer(
227 const MutablePodioTrackStateContainer& other);
228
229 ConstPodioTrackStateContainer(const PodioUtil::ConversionHelper& helper,
230 const podio::Frame& frame,
231 const std::string& suffix = "")
232 : m_helper{helper},
233 m_collection{nullptr},
234 m_params{nullptr},
235 m_jacs{nullptr} {
236 std::string s = suffix.empty() ? suffix : "_" + suffix;
237
238 std::vector<std::string> available = frame.getAvailableCollections();
239
240 std::string trackStatesKey = "trackStates" + s;
241 std::string paramsKey = "trackStateParameters" + s;
242 std::string jacsKey = "trackStateJacobians" + s;
243
244 if (!Acts::rangeContainsValue(available, trackStatesKey)) {
245 throw std::runtime_error{"Track state collection '" + trackStatesKey +
246 "' not found in frame"};
247 }
248
249 if (!Acts::rangeContainsValue(available, paramsKey)) {
250 throw std::runtime_error{"Track state parameters collection '" +
251 paramsKey + "' not found in frame"};
252 }
253
254 if (!Acts::rangeContainsValue(available, jacsKey)) {
255 throw std::runtime_error{"Track state jacobian collection '" + jacsKey +
256 "' not found in frame"};
257 }
258
259 loadCollection<ActsPodioEdm::TrackStateCollection>(m_collection, frame,
260 trackStatesKey);
261 loadCollection<ActsPodioEdm::BoundParametersCollection>(m_params, frame,
262 paramsKey);
263 loadCollection<ActsPodioEdm::JacobianCollection>(m_jacs, frame, jacsKey);
264
265 populateSurfaceBuffer(m_helper, *m_collection, m_surfaces);
266
267 podio_detail::recoverDynamicColumns(frame, trackStatesKey, m_dynamic);
268 }

459 IndexType addTrackState_impl(
460 Acts::TrackStatePropMask mask = Acts::TrackStatePropMask::All,
461 Acts::TrackIndexType iprevious = Acts::kTrackIndexInvalid) {
462 auto trackState = m_collection->create();
463 auto& data = PodioUtil::getDataMutable(trackState);
464 data.previous = iprevious;
465 data.ipredicted = kInvalid;
466 data.ifiltered = kInvalid;
467 data.ismoothed = kInvalid;
468 data.ijacobian = kInvalid;
469
470 PodioUtil::getReferenceSurfaceMutable(trackState).surfaceType =
471 PodioUtil::kNoSurface;
472
473 using enum Acts::TrackStatePropMask;
474
475 if (ACTS_CHECK_BIT(mask, Predicted)) {
476 m_params->create();
477 data.ipredicted = m_params->size() - 1;
478 }
479 if (ACTS_CHECK_BIT(mask, Filtered)) {
480 m_params->create();
481 data.ifiltered = m_params->size() - 1;
482 }
483 if (ACTS_CHECK_BIT(mask, Smoothed)) {
484 m_params->create();
485 data.ismoothed = m_params->size() - 1;
486 }
487 if (ACTS_CHECK_BIT(mask, Jacobian)) {
488 m_jacs->create();
489 data.ijacobian = m_jacs->size() - 1;
490 }
491 data.measdim = kInvalid;
492 data.hasProjector = false;
493 if (ACTS_CHECK_BIT(mask, Calibrated)) {
494 data.hasProjector = true;
495 }
496 m_surfaces.emplace_back();
497
498 data.uncalibratedIdentifier = PodioUtil::kNoIdentifier;
499 assert(m_collection->size() == m_surfaces.size() &&
500 "Inconsistent surface buffer");
501
502 for (const auto& [key, vec] : m_dynamic) {
503 vec->add();
504 }
505
506 return m_collection->size() - 1;
507 }

509 void addTrackStateComponents_impl(IndexType istate,
510 Acts::TrackStatePropMask mask) {
511 auto& data = PodioUtil::getDataMutable(m_collection->at(istate));
512
513 using enum Acts::TrackStatePropMask;
514
515 if (ACTS_CHECK_BIT(mask, Predicted) && data.ipredicted == kInvalid) {
516 m_params->create();
517 data.ipredicted = m_params->size() - 1;
518 }
519
520 if (ACTS_CHECK_BIT(mask, Filtered) && data.ifiltered == kInvalid) {
521 m_params->create();
522 data.ifiltered = m_params->size() - 1;
523 }
524
525 if (ACTS_CHECK_BIT(mask, Smoothed) && data.ismoothed == kInvalid) {
526 m_params->create();
527 data.ismoothed = m_params->size() - 1;
528 }
529
530 if (ACTS_CHECK_BIT(mask, Jacobian) && data.ijacobian == kInvalid) {
531 m_jacs->create();
532 data.ijacobian = m_jacs->size() - 1;
533 }
534
535 if (ACTS_CHECK_BIT(mask, Calibrated) && !data.hasProjector) {
536 data.hasProjector = true;
537 }
538 }

540 void shareFrom_impl(Acts::TrackIndexType iself, Acts::TrackIndexType iother,
541 Acts::TrackStatePropMask shareSource,
542 Acts::TrackStatePropMask shareTarget) {
543 auto& self = PodioUtil::getDataMutable(m_collection->at(iself));
544 auto& other = PodioUtil::getDataMutable(m_collection->at(iother));
545
546 assert(ACTS_CHECK_BIT(getTrackState(iother).getMask(), shareSource) &&
547 "Source has incompatible allocation");
548
549 using enum Acts::TrackStatePropMask;
550
551 IndexType sourceIndex{kInvalid};
552 switch (shareSource) {
553 case Predicted:
554 sourceIndex = other.ipredicted;
555 break;
556 case Filtered:
557 sourceIndex = other.ifiltered;
558 break;
559 case Smoothed:
560 sourceIndex = other.ismoothed;
561 break;
562 case Jacobian:
563 sourceIndex = other.ijacobian;
564 break;
565 default:
566 throw std::domain_error{"Unable to share this component"};
567 }
568
569 assert(sourceIndex != kInvalid);
570
571 switch (shareTarget) {
572 case Predicted:
573 assert(shareSource != Jacobian);
574 self.ipredicted = sourceIndex;
575 break;
576 case Filtered:
577 assert(shareSource != Jacobian);
578 self.ifiltered = sourceIndex;
579 break;
580 case Smoothed:
581 assert(shareSource != Jacobian);
582 self.ismoothed = sourceIndex;
583 break;
584 case Jacobian:
585 assert(shareSource == Jacobian);
586 self.ijacobian = sourceIndex;
587 break;
588 default:
589 throw std::domain_error{"Unable to share this component"};
590 }
591 }

636 template <typename val_t, typename cov_t>
637 void allocateCalibrated_impl(IndexType istate,
638 const Eigen::DenseBase<val_t>& val,
639 const Eigen::DenseBase<cov_t>& cov)
640 requires(Acts::Concepts::eigen_base_is_fixed_size<val_t> &&
641 Eigen::PlainObjectBase<val_t>::RowsAtCompileTime <=
642 toUnderlying(Acts::eBoundSize) &&
643 Acts::Concepts::eigen_bases_have_same_num_rows<val_t, cov_t> &&
644 Acts::Concepts::eigen_base_is_square<cov_t>)
645 {
646 constexpr std::size_t measdim = val_t::RowsAtCompileTime;
647
648 auto& data = PodioUtil::getDataMutable(m_collection->at(istate));
649
650 if (data.measdim != kInvalid && data.measdim != measdim) {
651 throw std::invalid_argument{
652 "Measurement dimension does not match the allocated dimension"};
653 }
654
655 data.measdim = measdim;
656
657 Eigen::Map<Acts::ActsVector<measdim>> valMap(data.measurement.data());
658 valMap = val;
659
660 Eigen::Map<Acts::ActsSquareMatrix<measdim>> covMap(
661 data.measurementCovariance.data());
662 covMap = cov;
663 }

694 void releaseInto(podio::Frame& frame, const std::string& suffix = "") {
695 std::string s = suffix;
696 if (!s.empty()) {
697 s = "_" + s;
698 }
699 frame.put(std::move(m_collection), "trackStates" + s);
700 frame.put(std::move(m_params), "trackStateParameters" + s);
701 frame.put(std::move(m_jacs), "trackStateJacobians" + s);
702 m_surfaces.clear();
703

705 col->releaseInto(frame, "trackStates" + s + "_extra__");
706 }
707
708 m_dynamic.clear();
709 }
