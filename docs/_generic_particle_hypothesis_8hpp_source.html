25template <ChargeConcept charge_t>
26class GenericParticleHypothesis {
27 public:
29 using ChargeType = charge_t;
30
37 constexpr GenericParticleHypothesis(PdgParticle absPdg, float mass,
38 ChargeType chargeType)
39 : m_absPdg{absPdg}, m_mass{mass}, m_chargeType{std::move(chargeType)} {
40 assert(absPdg == makeAbsolutePdgParticle(absPdg) &&
41 "pdg is expected to be absolute");
42 }
43
49 explicit GenericParticleHypothesis(PdgParticle absPdg)
50 : m_absPdg{absPdg},
51 m_mass{findMass(absPdg).value()},
52 m_chargeType{std::abs(findCharge(absPdg).value())} {
53 assert(absPdg == makeAbsolutePdgParticle(absPdg) &&
54 "pdg is expected to be absolute");
55 }
56
59 template <typename other_charge_t>
60 explicit constexpr GenericParticleHypothesis(
61 const GenericParticleHypothesis<other_charge_t>& other)
62 : m_absPdg{other.absolutePdg()},
63 m_mass{other.mass()},
64 m_chargeType{other.chargeType()} {}
65
68 constexpr PdgParticle absolutePdg() const noexcept { return m_absPdg; }
69
72 constexpr float mass() const noexcept { return m_mass; }
73
76 constexpr float absoluteCharge() const noexcept {
77 return m_chargeType.absQ();
78 }

85 template <typename T>
86 constexpr auto extractCharge(T qOverP) const noexcept {

88 }
89
95 template <typename T>
96 constexpr auto extractMomentum(T qOverP) const noexcept {
97 return m_chargeType.extractMomentum(qOverP);

99
106 template <typename P, typename Q>
107 constexpr auto qOverP(P momentum, Q signedQ) const noexcept {
108 return m_chargeType.qOverP(momentum, signedQ);
109 }
110
113 constexpr const ChargeType& chargeType() const noexcept {
114 return m_chargeType;
115 }
116
120 std::ostream& toStream(std::ostream& os) const {
121 os << "ParticleHypothesis{absPdg=";
122 if (auto shortString = pdgToShortAbsString(absolutePdg())) {
123 os << *shortString;
124 } else {
125 os << absolutePdg();
126 }
127 os << ", mass=" << mass() << ", absCharge=" << absoluteCharge() << "}";
128 return os;
129 }
130
135 friend std::ostream& operator<<(
136 std::ostream& os, const GenericParticleHypothesis& particleHypothesis) {
137 return particleHypothesis.toStream(os);
138 }
139
140 private:
141 PdgParticle m_absPdg;
142 float m_mass;
143 ChargeType m_chargeType;
144
145 friend bool operator==(const GenericParticleHypothesis<ChargeType>& lhs,
146 const GenericParticleHypothesis<ChargeType>& rhs) {
147 return (lhs.m_absPdg == rhs.m_absPdg) && (lhs.m_mass == rhs.m_mass) &&
148 (lhs.m_chargeType == rhs.m_chargeType);
149 }
150};
