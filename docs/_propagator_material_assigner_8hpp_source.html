66 template <typename propagator_state_t, typename stepper_t,
67 typename navigator_t>
68 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
69 const navigator_t& navigator, result_type& result,
70 const Logger& /*logger*/) const {
71 // Retrieve the current volume
72 auto currentVolume = navigator.currentVolume(state.navigation);
73
74 // The current volume has been assigned by the navigator
75 if (currentVolume != nullptr) {
76 auto collIt = result.collected.find(currentVolume->geometryId());
77 // Check if the volume has been collected and if it has material
78 if (collIt == result.collected.end() &&
79 currentVolume->volumeMaterial() != nullptr) {
80 Vector3 entryPosition = stepper.position(state.stepping);
81 Vector3 exitPosition = entryPosition;
82 IAssignmentFinder::VolumeAssignment vAssignment{
83 InteractionVolume(currentVolume), entryPosition, exitPosition};
84 result.collected.emplace(currentVolume->geometryId(), vAssignment);
85 } else if (collIt != result.collected.end()) {
86 // Update the exit position
87 (collIt->second).exit = stepper.position(state.stepping);
88 }
89 }
90 return Result<void>::success();
91 }
92};

101template <typename propagator_t>
102class PropagatorMaterialAssigner final : public IAssignmentFinder {
103 public:
106 explicit PropagatorMaterialAssigner(propagator_t propagator)
107 : m_propagator(std::move(propagator)) {}
108
118 std::pair<std::vector<IAssignmentFinder::SurfaceAssignment>,
119 std::vector<IAssignmentFinder::VolumeAssignment>>
120 assignmentCandidates(const GeometryContext& gctx,
121 const MagneticFieldContext& mctx,
122 const Vector3& position,
123 const Vector3& direction) const final {
124 // Return container
125 std::pair<std::vector<IAssignmentFinder::SurfaceAssignment>,
126 std::vector<IAssignmentFinder::VolumeAssignment>>
127 candidates;
128
129 using VectorHelpers::makeVector4;
130 // Neutral curvilinear parameters
131 NeutralBoundTrackParameters start =
132 NeutralBoundTrackParameters::createCurvilinear(
133 makeVector4(position, 0), direction, 1, std::nullopt,
134 NeutralParticleHypothesis::geantino());
135
136 // Prepare Action list and abort list
137 using MaterialSurfaceCollector =
138 SurfaceCollector<MaterialSurfaceIdentifier>;
139 using ActorList = ActorList<MaterialSurfaceCollector,
140 InteractionVolumeCollector, EndOfWorldReached>;
141 using PropagatorOptions =
142 typename propagator_t::template Options<ActorList>;
143
144 PropagatorOptions options(gctx, mctx);
145
146 const auto& result = m_propagator.propagate(start, options).value();
147
148 // The surface collection results
149 auto scResult =
150 result.template get<MaterialSurfaceCollector::result_type>();
151 auto mappingSurfaces = scResult.collected;
152
153 for (auto& mSurface : mappingSurfaces) {
154 candidates.first.push_back(IAssignmentFinder::SurfaceAssignment{
155 mSurface.surface, mSurface.position, direction});
156 }
157
158 // The volume collection results
159 auto vcResult =
160 result.template get<InteractionVolumeCollector::result_type>();
161 for (const auto& [geoId, vIntersection] : vcResult.collected) {
162 candidates.second.push_back(vIntersection);
163 }
164
165 // Return the candidates
166 return candidates;
167 }
168
169 private:
170 propagator_t m_propagator;
171};
