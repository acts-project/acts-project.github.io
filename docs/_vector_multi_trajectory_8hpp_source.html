93 template <typename T>
94 Statistics statistics(T& instance) const {
95 using namespace boost::histogram;
96 using cat = axis::category<std::string>;
97
98 Statistics::axes_t axes;
99 axes.emplace_back(cat({
100 "count",
101 "index",
102 "parPred",
103 "covPred",
104 "parFilt",
105 "covFilt",
106 "parSmth",
107 "covSmth",
108 "meas",
109 "measOffset",
110 "measCov",
111 "measCovOffset",
112 "jac",
113 "sourceLinks",
114 "projectors",
115 }));
116
117 axes.emplace_back(axis::category<>({0, 1}));
118
119 auto h = make_histogram(axes);
120
121 for (IndexType i = 0; i < instance.size(); i++) {
122 auto ts = instance.getTrackState(i);
123
124 bool isMeas = ts.typeFlags().hasMeasurement();
125
126 h("count", isMeas);
127
128 h("index", isMeas, weight(sizeof(IndexData)));
129
130 using scalar = typename decltype(ts.predicted())::Scalar;
131 std::size_t par_size = eBoundSize * sizeof(scalar);
132 std::size_t cov_size = eBoundSize * eBoundSize * sizeof(scalar);
133
134 const IndexData& index = m_index[i];
135 if (ts.hasPredicted() &&
136 ACTS_CHECK_BIT(index.allocMask, TrackStatePropMask::Predicted)) {
137 h("parPred", isMeas, weight(par_size));
138 h("covPred", isMeas, weight(cov_size));
139 }
140 if (ts.hasFiltered() &&
141 ACTS_CHECK_BIT(index.allocMask, TrackStatePropMask::Filtered)) {
142 h("parFilt", isMeas, weight(par_size));
143 h("covFilt", isMeas, weight(cov_size));
144 }
145 if (ts.hasSmoothed() &&
146 ACTS_CHECK_BIT(index.allocMask, TrackStatePropMask::Smoothed)) {
147 h("parSmth", isMeas, weight(par_size));
148 h("covSmth", isMeas, weight(cov_size));
149 }
150 h("sourceLinks", isMeas, weight(sizeof(SourceLink)));
151 h("measOffset", isMeas,
152 weight(sizeof(decltype(m_measOffset)::value_type)));
153 h("measCovOffset", isMeas,
154 weight(sizeof(decltype(m_measCovOffset)::value_type)));
155 if (ts.hasCalibrated() &&
156 ACTS_CHECK_BIT(index.allocMask, TrackStatePropMask::Calibrated)) {
157 std::size_t meas_size = ts.calibratedSize() * sizeof(scalar);
158 std::size_t meas_cov_size =
159 ts.calibratedSize() * ts.calibratedSize() * sizeof(scalar);
160
161 h("meas", isMeas, weight(meas_size));
162 h("measCov", isMeas, weight(meas_cov_size));
163 h("sourceLinks", isMeas, weight(sizeof(const SourceLink)));
164 h("projectors", isMeas, weight(sizeof(SerializedSubspaceIndices)));
165 }
166
167 if (ts.hasJacobian() &&
168 ACTS_CHECK_BIT(index.allocMask, TrackStatePropMask::Jacobian)) {
169 h("jac", isMeas, weight(cov_size));
170 }
171 }
172
173 return Statistics{h};
174 }

221 static constexpr bool has_impl(T& instance, HashedString key,
222 IndexType istate) {
223 using namespace Acts::HashedStringLiteral;
224 switch (key) {
225 case "predicted"_hash:
226 return instance.m_index[istate].ipredicted != kInvalid;
227 case "filtered"_hash:
228 return instance.m_index[istate].ifiltered != kInvalid;
229 case "smoothed"_hash:
230 return instance.m_index[istate].ismoothed != kInvalid;
231 case "calibrated"_hash:
232 return instance.m_measOffset[istate] != kInvalid;
233 case "calibratedCov"_hash:
234 return instance.m_measCovOffset[istate] != kInvalid;
235 case "jacobian"_hash:
236 return instance.m_index[istate].ijacobian != kInvalid;
237 case "projector"_hash:
238 return instance.m_index[istate].iprojector != kInvalid;
239 case "uncalibratedSourceLink"_hash:

242 case "previous"_hash:
243 case "next"_hash:
244 case "referenceSurface"_hash:
245 case "measdim"_hash:
246 case "chi2"_hash:

248 case "typeFlags"_hash:

252 }

255 template <bool EnsureConst, typename T>
256 static std::any component_impl(T& instance, HashedString key,
257 IndexType istate) {
258 if constexpr (EnsureConst) {
259 static_assert(std::is_const_v<std::remove_reference_t<T>>,
260 "Is not const");
261 }
262 using namespace Acts::HashedStringLiteral;
263 switch (key) {
264 case "previous"_hash:
265 return &instance.m_previous[istate];
266 case "next"_hash:
267 return &instance.m_next[istate];
268 case "predicted"_hash:
269 return &instance.m_index[istate].ipredicted;
270 case "filtered"_hash:
271 return &instance.m_index[istate].ifiltered;
272 case "smoothed"_hash:
273 return &instance.m_index[istate].ismoothed;
274 case "projector"_hash:
275 return &instance.m_projectors[instance.m_index[istate].iprojector];
276 case "measdim"_hash:
277 return &instance.m_index[istate].measdim;
278 case "chi2"_hash:
279 return &instance.m_index[istate].chi2;
280 case "pathLength"_hash:
281 return &instance.m_index[istate].pathLength;
282 case "typeFlags"_hash:
283 return &instance.m_index[istate].typeFlags;
284 default:
285 auto it = instance.m_dynamic.find(key);
286 if (it == instance.m_dynamic.end()) {
287 throw std::runtime_error("Unable to handle this component");
288 }
289 std::conditional_t<EnsureConst, const detail::DynamicColumnBase*,
290 detail::DynamicColumnBase*>
291 col = it->second.get();
292 assert(col && "Dynamic column is null");
293 return col->get(istate);
294 }
295 }

500 template <typename val_t, typename cov_t>
501 void allocateCalibrated_impl(IndexType istate,
502 const Eigen::DenseBase<val_t>& val,
503 const Eigen::DenseBase<cov_t>& cov)
504 requires(Concepts::eigen_base_is_fixed_size<val_t> &&
505 Concepts::eigen_bases_have_same_num_rows<val_t, cov_t> &&
506 Concepts::eigen_base_is_square<cov_t> &&
507 Eigen::PlainObjectBase<val_t>::RowsAtCompileTime <=
508 toUnderlying(eBoundSize))
509 {
510 constexpr std::size_t measdim = val_t::RowsAtCompileTime;
511
512 if (m_index[istate].measdim != kInvalid &&
513 m_index[istate].measdim != measdim) {
514 throw std::invalid_argument{
515 "Measurement dimension does not match the allocated dimension"};
516 }
517
518 if (m_measOffset[istate] == kInvalid ||
519 m_measCovOffset[istate] == kInvalid) {
520 m_measOffset[istate] = static_cast<IndexType>(m_meas.size());
521 m_meas.resize(m_meas.size() + measdim);
522
523 m_measCovOffset[istate] = static_cast<IndexType>(m_measCov.size());
524 m_measCov.resize(m_measCov.size() + measdim * measdim);
525 }
526
527 m_index[istate].measdim = measdim;
528
529 double* measPtr = &m_meas[m_measOffset[istate]];
530 Eigen::Map<ActsVector<measdim>> valMap(measPtr);
531 valMap = val;
532
533 double* covPtr = &m_measCov[m_measCovOffset[istate]];
534 Eigen::Map<ActsSquareMatrix<measdim>> covMap(covPtr);
535 covMap = cov;
536 }
