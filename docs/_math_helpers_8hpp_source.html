20template <typename T>
21constexpr T abs(const T n) {
22 if (std::is_constant_evaluated()) {
23 if constexpr (std::is_signed_v<T>) {
24 if (n < 0) {
25 return -n;
26 }
27 }
28 return n;
29 } else {
30 return std::abs(n);
31 }
32}

39template <typename out_t, typename sign_t>
40constexpr out_t copySign(const out_t& copyTo, const sign_t& sign) {
41 if constexpr (std::is_enum_v<sign_t>) {
42 return copySign(copyTo, static_cast<std::underlying_type_t<sign_t>>(sign));
43 } else {
44 constexpr sign_t zero = 0;
45 return sign >= zero ? copyTo : -copyTo;
46 }
47}

53template <typename T, std::integral P>
54constexpr T pow(T x, P p) {
55 if (std::is_constant_evaluated()) {
56 constexpr T one = 1;
57 if constexpr (std::is_signed_v<P>) {
58 if (p < 0 && abs(x) > std::numeric_limits<T>::epsilon()) {
59 x = one / x;
60 p = -p;
61 }
62 }
63 using unsigned_p = std::make_unsigned_t<P>;
64 return p == 0 ? one : x * pow(x, static_cast<unsigned_p>(p) - 1);
65 } else {
66 return static_cast<T>(std::pow(x, static_cast<T>(p)));
67 }
68}

107template <std::integral T>
108constexpr T factorial(const T upperN, const T lowerN = 1) {
109 constexpr T one = 1;
110 const T& limit = std::max(one, lowerN);
111 return upperN >= limit ? upperN * factorial(upperN - 1, limit) : one;
112}

121constexpr T binomial(const T n, const T k) {
