33 virtual ~IGridMaterialAccessor() = default;

38struct GridMaterialAccessor : public IGridMaterialAccessor {
40 using grid_value_type = MaterialSlab;
47 template <typename grid_type>
48 inline const MaterialSlab& slab(
49 grid_type& grid, const typename grid_type::point_t& point) const {
50 return grid.atPosition(point);
51 }
52
59 template <typename grid_type>
60 void scale(grid_type& grid, double scale) {
61 // Loop through the grid bins, get the indices and scale the material
62 for (std::size_t ib = 0; ib < grid.size(); ++ib) {
63 grid.at(ib).scaleThickness(static_cast<float>(scale));
64 }
65 }
66};

70struct IndexedMaterialAccessor : public IGridMaterialAccessor {
72 using grid_value_type = std::size_t;
73
76 explicit IndexedMaterialAccessor(std::vector<MaterialSlab>&& mmaterial)
77 : IGridMaterialAccessor(), material(std::move(mmaterial)) {}
78
80 std::vector<MaterialSlab> material;
87 template <typename grid_type>
88 inline const MaterialSlab& slab(
89 const grid_type& grid, const typename grid_type::point_t& point) const {
90 auto index = grid.atPosition(point);
91 return material[index];
92 }
93
97 template <typename grid_type>
98 void scale(grid_type& /*grid*/, double scale) {
99 for (auto& m : material) {
100 m.scaleThickness(static_cast<float>(scale));
101 }
102 }
103};

107struct GloballyIndexedMaterialAccessor : public IGridMaterialAccessor {
111 explicit GloballyIndexedMaterialAccessor(
112 std::shared_ptr<std::vector<MaterialSlab>> gMaterial, bool shared = false)
113 : IGridMaterialAccessor(),
114 globalMaterial(std::move(gMaterial)),
115 sharedEntries(shared) {}
116
118 using grid_value_type = std::size_t;
119
121 std::shared_ptr<std::vector<MaterialSlab>> globalMaterial = nullptr;
122
127 bool sharedEntries = false;
128
137 template <typename grid_type>
138 inline const MaterialSlab& slab(
139 const grid_type& grid, const typename grid_type::point_t& point) const {
140 auto index = grid.atPosition(point);
141 return (*globalMaterial)[index];
142 }
143
153 template <typename grid_type>
154 void scale(grid_type& grid, double scale) {
155 if (sharedEntries) {
156 throw std::invalid_argument(
157 "GloballyIndexedMaterialAccessor: shared entry scaling is not "
158 "supported.");
159 }
160 // Loop through the grid bins, get the indices and scale the material
161 for (std::size_t ib = 0; ib < grid.size(); ++ib) {
162 auto index = grid.at(ib);
163 (*globalMaterial)[index].scaleThickness(static_cast<float>(scale));
164 }
165 }
166};

212template <typename grid_t, typename material_accessor_t>
213class GridSurfaceMaterialT
214 : public IGridSurfaceMaterial<
215 typename material_accessor_t::grid_value_type> {
216 public:
219 using BoundToGridLocalDelegate =
220 OwningDelegate<typename grid_t::point_t(const Vector2&),
221 GridAccess::IBoundToGridLocal>;
222
225 using GlobalToGridLocalDelegate =
226 OwningDelegate<typename grid_t::point_t(const Vector3&),
227 GridAccess::IGlobalToGridLocal>;
228
230 using grid_type = grid_t;
231
233 using material_accessor_type = material_accessor_t;
234
241 GridSurfaceMaterialT(grid_type&& grid,
242 material_accessor_type&& materialAccessor,
243 BoundToGridLocalDelegate boundToGridLocal,
244 GlobalToGridLocalDelegate globalToGridLocal)
245 : m_grid(std::move(grid)),
246 m_materialAccessor(std::move(materialAccessor)),
247 m_globalToGridLocal(std::move(globalToGridLocal)),
248 m_boundToGridLocal(std::move(boundToGridLocal)) {
249 if (!m_globalToGridLocal.connected()) {
250 throw std::invalid_argument(
251 "GridSurfaceMaterialT: GlobalToGridLocalDelegate is not connected.");
252 }
253 if (!m_boundToGridLocal.connected()) {
254 throw std::invalid_argument(
255 "GridSurfaceMaterialT: BoundToGridLocalDelegate is not connected.");
256 }
257 }
258
260 const MaterialSlab& materialSlab(const Vector2& lp) const final {
261 return m_materialAccessor.slab(m_grid, m_boundToGridLocal(lp));
262 }
263
265 const MaterialSlab& materialSlab(const Vector3& gp) const final {
266 return m_materialAccessor.slab(m_grid, m_globalToGridLocal(gp));
267 }
268
273 ISurfaceMaterial& scale(double factor) final {
274 m_materialAccessor.scale(m_grid, factor);
275 return (*this);
276 }
277
281 std::ostream& toStream(std::ostream& sl) const final {
282 sl << "GridSurfaceMaterial - material access via accessor.";
283 return sl;
284 }
285
288 const grid_type& grid() const final { return m_grid; }
289
290 // Return a type-erased indexed grid view
292 AnyGridView<typename material_accessor_t::grid_value_type> gridView() final {
293 return AnyGridView<typename material_accessor_t::grid_value_type>(m_grid);
294 }
295
296 // Return a type-erased indexed const grid view
298 AnyGridConstView<typename material_accessor_t::grid_value_type>
299 gridConstView() const final {
300 return AnyGridConstView<typename material_accessor_t::grid_value_type>(
301 m_grid);
302 }
303
306 const material_accessor_type& materialAccessor() const final {
307 return m_materialAccessor;
308 }
309
312 const GridAccess::IBoundToGridLocal& boundToGridLocal() const final {
313 return *(m_boundToGridLocal.instance());
314 }
315
318 const BoundToGridLocalDelegate& boundToGridLocalDelegate() const {
319 return m_boundToGridLocal;
320 }
321
324 const GridAccess::IGlobalToGridLocal& globalToGridLocal() const final {
325 return *(m_globalToGridLocal.instance());
326 }
327
330 const GlobalToGridLocalDelegate& globalToGridLocalDelegate() const {
331 return m_globalToGridLocal;
332 }
333
334 private:
336 grid_type m_grid;
337
339 material_accessor_type m_materialAccessor;
340
342 GlobalToGridLocalDelegate m_globalToGridLocal;
343
345 BoundToGridLocalDelegate m_boundToGridLocal;
346};
