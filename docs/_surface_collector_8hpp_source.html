102 template <typename propagator_state_t, typename stepper_t,
103 typename navigator_t>
104 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
105 const navigator_t& navigator, result_type& result,
106 const Logger& logger) const {
107 if (state.stage == PropagatorStage::postPropagation) {
108 return {};
109 }
110
111 auto currentSurface = navigator.currentSurface(state.navigation);
112
113 // The current surface has been assigned by the navigator
114 if (currentSurface && selector(*currentSurface)) {
115 // Create for recording
116 SurfaceHit surface_hit;
117 surface_hit.surface = currentSurface;
118 surface_hit.position = stepper.position(state.stepping);
119 surface_hit.direction = stepper.direction(state.stepping);
120 // Save if in the result
121 result.collected.push_back(surface_hit);
122 // Screen output
123 ACTS_VERBOSE("Collect surface " << currentSurface->geometryId());
124 }
125
126 return {};
127 }
128};
