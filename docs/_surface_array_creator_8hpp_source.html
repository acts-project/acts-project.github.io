77 struct ProtoAxis {
78 BinningType bType = BinningType::equidistant;
79 AxisDirection axisDir = AxisDirection::AxisX;
80 std::size_t nBins = 0;
81 AxisScalar min = 0;
82 AxisScalar max = 0;
83 std::vector<AxisScalar> binEdges;
84
85 std::size_t getBin(AxisScalar x) const {
86 if (binEdges.empty()) {
87 // equidistant
88 AxisScalar w = (max - min) / nBins;
89 return static_cast<std::size_t>(std::floor((x - min) / w));
90 } else {
91 // variable
92 const auto it =
93 std::upper_bound(std::begin(binEdges), std::end(binEdges), x);
94 return std::distance(std::begin(binEdges), it) - 1;
95 }
96 }
97 };

260 static bool isSurfaceEquivalent(const GeometryContext& gctx,
261 AxisDirection aDir, const Surface* a,
262 const Surface* b) {
263 using namespace UnitLiterals;
264 using VectorHelpers::perp;
265
266 if (aDir == AxisDirection::AxisPhi) {
267 // Take the two binning positions
268 Vector3 pos1 = a->referencePosition(gctx, AxisDirection::AxisR);
269 Vector3 pos2 = b->referencePosition(gctx, AxisDirection::AxisR);
270
271 // Project them on the (x, y) plane, where Phi angles are calculated
272 auto proj1 = pos1.head<2>(), proj2 = pos2.head<2>();
273
274 // Basic dot and cross products identities give us the cosine and sine
275 // of these angles, time the squared vector norm
276 auto cos_dPhi_n2 = proj1.dot(proj2);
277 auto sin_dPhi_n2 = proj1.x() * proj2.y() - proj2.x() * proj1.y();
278
279 // ...so by injecting them into atan2, we get the angle between them
280 auto dPhi = std::atan2(sin_dPhi_n2, cos_dPhi_n2);
281 return std::abs(dPhi) < std::numbers::pi / 180.;
282 }
283
284 if (aDir == AxisDirection::AxisZ) {
285 return (std::abs(a->referencePosition(gctx, AxisDirection::AxisR).z() -
286 b->referencePosition(gctx, AxisDirection::AxisR).z()) <
287 1_um);
288 }
289
290 if (aDir == AxisDirection::AxisR) {
291 return (std::abs(perp(a->referencePosition(gctx, AxisDirection::AxisR)) -
292 perp(b->referencePosition(gctx, AxisDirection::AxisR))) <
293 1_um);
294 }
295
296 return false;
297 }

388 double layerTolerance, ProtoAxis pAxisA,

390 using ISGL = SurfaceArray::ISurfaceGridLookup;
391 std::unique_ptr<ISGL> ptr;
392
393 if (pAxisA.bType == equidistant && pAxisB.bType == equidistant) {
394 Axis<AxisType::Equidistant, bdtA> axisA(pAxisA.min, pAxisA.max,
395 pAxisA.nBins);
396 Axis<AxisType::Equidistant, bdtB> axisB(pAxisB.min, pAxisB.max,
397 pAxisB.nBins);
398
399 using SGL =
400 SurfaceArray::SurfaceGridLookup<decltype(axisA), decltype(axisB)>;
401 ptr = std::make_unique<SGL>(std::move(surface), layerTolerance,
402 std::pair{axisA, axisB},
403 std::vector{pAxisA.axisDir, pAxisB.axisDir});
404
405 } else if (pAxisA.bType == equidistant && pAxisB.bType == arbitrary) {
406 Axis<AxisType::Equidistant, bdtA> axisA(pAxisA.min, pAxisA.max,
407 pAxisA.nBins);
408 Axis<AxisType::Variable, bdtB> axisB(pAxisB.binEdges);
409
410 using SGL =
411 SurfaceArray::SurfaceGridLookup<decltype(axisA), decltype(axisB)>;
412 ptr = std::make_unique<SGL>(std::move(surface), layerTolerance,
413 std::pair{axisA, axisB},
