189 double transverseMomentum() const {
190 // direction vector w/ arbitrary normalization can be parametrized as
191 // [f*sin(theta)*cos(phi), f*sin(theta)*sin(phi), f*cos(theta)]
192 // w/ f,sin(theta) positive, the transverse magnitude is then
193 // sqrt(f^2*sin^2(theta)) = f*sin(theta)
194 double transverseMagnitude2 =
195 square(m_params[eFreeDir0]) + square(m_params[eFreeDir1]);
196 // absolute magnitude is f by construction
197 double magnitude2 = transverseMagnitude2 + square(m_params[eFreeDir2]);
198 // such that we can extract sin(theta) = f*sin(theta) / f
199 return std::sqrt(transverseMagnitude2 / magnitude2) * absoluteMomentum();
200 }
201
203 Vector3 momentum() const { return absoluteMomentum() * direction(); }

209 }
