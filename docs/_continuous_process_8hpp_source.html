42template <detail::ContinuousProcessConcept physics_t,
43 typename input_particle_selector_t,
44 typename output_particle_selector_t,
45 typename child_particle_selector_t = output_particle_selector_t>
46struct ContinuousProcess {
48 physics_t physics;
50 input_particle_selector_t selectInputParticle;
52 output_particle_selector_t selectOutputParticle;
54 child_particle_selector_t selectChildParticle;
55
65 template <typename generator_t>
66 bool operator()(generator_t &generator, const Acts::MaterialSlab &slab,
67 Particle &particle, std::vector<Particle> &generated) const {
68 // not selecting this process is not a break condition
69 if (!selectInputParticle(particle)) {
70 return false;
71 }
72 // modify particle according to the physics process
73 auto children = physics(generator, slab, particle);
74 // move selected child particles to the output container
75 std::copy_if(std::begin(children), std::end(children),
76 std::back_inserter(generated), selectChildParticle);
77 // break condition is defined by whether the output particle is still valid
78 // or not e.g. because it has fallen below a momentum threshold.
79 return !selectOutputParticle(particle);
80 }
81};
