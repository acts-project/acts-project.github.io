654 template <TrackProxyConcept track_proxy_t>
655 void copyFrom(const track_proxy_t& other)
656 requires(!ReadOnly)
657 {
658 copyFromWithoutStates(other);
659
660 // append track states (cheap), but they're in the wrong order
661 for (const auto& srcTrackState : other.trackStatesReversed()) {
662 auto destTrackState = appendTrackState(srcTrackState.getMask());
663 destTrackState.copyFrom(srcTrackState, Acts::TrackStatePropMask::All,
664 true);
665 }
666
667 // reverse using standard linked list reversal algorithm
668 reverseTrackStates();
669 }

700 template <TrackProxyConcept track_proxy_t>
701 void copyFromWithoutStates(const track_proxy_t& other)
702 requires(!ReadOnly)
703 {
704 setParticleHypothesis(other.particleHypothesis());
705
706 if (other.hasReferenceSurface()) {
707 setReferenceSurface(other.referenceSurface().getSharedPtr());
708 parameters() = other.parameters();
709 covariance() = other.covariance();
710 } else {
711 setReferenceSurface(nullptr);
712 }
713
714 nMeasurements() = other.nMeasurements();
715 nHoles() = other.nHoles();
716 nOutliers() = other.nOutliers();
717 nSharedHits() = other.nSharedHits();
718 chi2() = other.chi2();
719 nDoF() = other.nDoF();
720
721 m_container->copyDynamicFrom(m_index, other.m_container->container(),
722 other.m_index);
723
724 tipIndex() = kInvalid;
725 stemIndex() = kInvalid;
726 }

779 void reverseTrackStates(bool invertJacobians = false)
780 requires(!ReadOnly)
781 {
782 IndexType current = tipIndex();
783 IndexType next = kInvalid;
784 IndexType prev = kInvalid;
785
786 stemIndex() = tipIndex();
787
788 // @TODO: Maybe refactor to not need this variable if invertJacobians == false
789 BoundMatrix nextJacobian;
790
791 while (current != kInvalid) {
792 auto ts = m_container->trackStateContainer().getTrackState(current);
793 prev = ts.previous();
794 ts.template component<IndexType>(detail_tp::kNextKey) = prev;
795 ts.previous() = next;
796 if (invertJacobians) {
797 if (next != kInvalid) {
798 BoundMatrix curJacobian = ts.jacobian();
799 ts.jacobian() = nextJacobian.inverse();
800 nextJacobian = curJacobian;
801 } else {
802 nextJacobian = ts.jacobian();
803 ts.jacobian().setZero();
804 }
805 }
806 next = current;
807 tipIndex() = current;
808 current = prev;
809 }
810 }
