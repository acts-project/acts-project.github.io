484 template <TrackProxyConcept track_proxy_t>
485 void copyFrom(const track_proxy_t& other)
486 requires(!ReadOnly)
487 {
488 copyFromWithoutStates(other);
489
490 // append track states (cheap), but they're in the wrong order
491 for (const auto& srcTrackState : other.trackStatesReversed()) {
492 auto destTrackState = appendTrackState(srcTrackState.getMask());
493 destTrackState.copyFrom(srcTrackState, Acts::TrackStatePropMask::All,
494 true);
495 }
496
497 // reverse using standard linked list reversal algorithm
498 reverseTrackStates();
499 }

530 template <TrackProxyConcept track_proxy_t>
531 void copyFromWithoutStates(const track_proxy_t& other)
532 requires(!ReadOnly)
533 {
534 setParticleHypothesis(other.particleHypothesis());
535
536 if (other.hasReferenceSurface()) {
537 setReferenceSurface(other.referenceSurface().getSharedPtr());
538 parameters() = other.parameters();
539 covariance() = other.covariance();
540 } else {
541 setReferenceSurface(nullptr);
542 }
543
544 nMeasurements() = other.nMeasurements();
545 nHoles() = other.nHoles();
546 nOutliers() = other.nOutliers();
547 nSharedHits() = other.nSharedHits();
548 chi2() = other.chi2();
549 nDoF() = other.nDoF();
550
551 m_container->copyDynamicFrom(m_index, other.m_container->container(),
552 other.m_index);
553
554 tipIndex() = kInvalid;
555 stemIndex() = kInvalid;
556 }

609 void reverseTrackStates(bool invertJacobians = false)
610 requires(!ReadOnly)
611 {
612 IndexType current = tipIndex();
613 IndexType next = kInvalid;
614 IndexType prev = kInvalid;
615
616 stemIndex() = tipIndex();
617
618 // @TODO: Maybe refactor to not need this variable if invertJacobians == false
619 BoundMatrix nextJacobian;
620
621 while (current != kInvalid) {
622 auto ts = m_container->trackStateContainer().getTrackState(current);
623 prev = ts.previous();
624 ts.template component<IndexType>(detail_tp::kNextKey) = prev;
625 ts.previous() = next;
626 if (invertJacobians) {
627 if (next != kInvalid) {
628 BoundMatrix curJacobian = ts.jacobian();
629 ts.jacobian() = nextJacobian.inverse();
630 nextJacobian = curJacobian;
631 } else {
632 nextJacobian = ts.jacobian();
633 ts.jacobian().setZero();
634 }
635 }
636 next = current;
637 tipIndex() = current;
638 current = prev;
639 }
640 }
