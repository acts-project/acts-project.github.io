267 void resetAfterVolumeSwitch() {
268 resetAfterLayerSwitch();

270 navLayers.clear();
271 navLayerIndex.reset();
272 navBoundaries.clear();
273 navBoundaryIndex.reset();
274 navCandidates.clear();
275 navCandidateIndex.reset();

277 currentLayer = nullptr;

281 void reset() {
282 resetAfterVolumeSwitch();
283
284 currentVolume = nullptr;
285 currentSurface = nullptr;
286
287 navigationBreak = false;
288 navigationStage = Stage::initial;
289 // Set the surface reached switches back to false
290 std::ranges::for_each(freeCandidates,
291 [](std::pair<const Surface*, bool>& freeSurface) {
292 freeSurface.second = false;
293 });
294 }
