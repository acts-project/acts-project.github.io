96template <Acts::TrackProxyConcept track_proxy_t>
97void writeTrack(const Acts::GeometryContext& gctx, track_proxy_t track,
98 edm4hep::MutableTrack to, double Bz,
99 const Acts::Logger& logger = Acts::getDummyLogger()) {
100 ACTS_VERBOSE("Converting track to EDM4hep");
101 to.setChi2(track.chi2());
102 to.setNdf(track.nDoF());
103
104 std::vector<edm4hep::TrackState> outTrackStates;
105 outTrackStates.reserve(track.nTrackStates());
106
107 auto setParameters = [](edm4hep::TrackState& trackState,
108 const detail::Parameters& params) {
109 trackState.D0 = params.values[0];
110 trackState.Z0 = params.values[1];
111 trackState.phi = params.values[2];
112 trackState.tanLambda = params.values[3];
113 trackState.omega = params.values[4];
114 trackState.time = params.values[5];
115
116 if (params.covariance) {
117 detail::packCovariance(params.covariance.value(),
118 trackState.covMatrix.data());
119 }
120 };
121
122 ACTS_VERBOSE("Converting " << track.nTrackStates() << " track states");
123
124 for (const auto& state : track.trackStatesReversed()) {
125 auto typeFlags = state.typeFlags();
126 if (!typeFlags.test(Acts::TrackStateFlag::MeasurementFlag)) {
127 continue;
128 }
129
130 edm4hep::TrackState& trackState = outTrackStates.emplace_back();
131 trackState.location = edm4hep::TrackState::AtOther;
132
133 Acts::BoundTrackParameters params{state.referenceSurface().getSharedPtr(),
134 state.parameters(), state.covariance(),
135 track.particleHypothesis()};
136
137 // Convert to LCIO track parametrization expected by EDM4hep
138 detail::Parameters converted =
139 detail::convertTrackParametersToEdm4hep(gctx, Bz, params);
140
141 // Write the converted parameters to the EDM4hep track state
142 setParameters(trackState, converted);
143 ACTS_VERBOSE("- parameters: " << state.parameters().transpose() << " -> "
144 << converted.values.transpose());
145 ACTS_VERBOSE("- covariance: \n"
146 << state.covariance() << "\n->\n"
147 << converted.covariance.value());
148
149 // Converted parameters are relative to an ad-hoc perigee surface created at
150 // the hit location
151 auto center = converted.surface->center(gctx);
152 trackState.referencePoint.x = center.x();
153 trackState.referencePoint.y = center.y();
154 trackState.referencePoint.z = center.z();
155 ACTS_VERBOSE("- ref surface ctr: " << center.transpose());
156 }
157 outTrackStates.front().location = edm4hep::TrackState::AtLastHit;
158 outTrackStates.back().location = edm4hep::TrackState::AtFirstHit;
159
160 // Add a track state that represents the IP parameters
161 auto& ipState = outTrackStates.emplace_back();
162
163 // Convert the track parameters at the IP
164 Acts::BoundTrackParameters trackParams{
165 track.referenceSurface().getSharedPtr(), track.parameters(),
166 track.covariance(), track.particleHypothesis()};
167
168 // Convert to LCIO track parametrization expected by EDM4hep
169 auto converted =
170 detail::convertTrackParametersToEdm4hep(gctx, Bz, trackParams);
171 setParameters(ipState, converted);
172 ipState.location = edm4hep::TrackState::AtIP;
173 ACTS_VERBOSE("Writing track level quantities as IP track state");
174 ACTS_VERBOSE("- parameters: " << track.parameters().transpose());
175 ACTS_VERBOSE(" -> " << converted.values.transpose());
176 ACTS_VERBOSE("- covariance: \n"
177 << track.covariance() << "\n->\n"
178 << converted.covariance.value());
179
180 // Write the converted parameters to the EDM4hep track state
181 // The reference point is at the location of the reference surface of the
182 // track itself, but if that's not a perigee surface, another ad-hoc perigee
183 // at the position will be created.
184 auto center = converted.surface->center(gctx);
185 ipState.referencePoint.x = center.x();
186 ipState.referencePoint.y = center.y();
187 ipState.referencePoint.z = center.z();
188
189 ACTS_VERBOSE("- ref surface ctr: " << center.transpose());
190
191 for (auto& trackState : outTrackStates) {
192 to.addToTrackStates(trackState);
193 }
194}

196template <Acts::TrackProxyConcept track_proxy_t>
197void readTrack(const edm4hep::Track& from, track_proxy_t& track, double Bz,
198 const Acts::Logger& logger = Acts::getDummyLogger()) {
199 using namespace Acts;
200 ACTS_VERBOSE("Reading track from EDM4hep");
201 TrackStatePropMask mask = TrackStatePropMask::Smoothed;
202
203 std::optional<edm4hep::TrackState> ipState;
204
205 auto unpack =
206 [](const edm4hep::TrackState& trackState) -> detail::Parameters {
207 detail::Parameters params;
208 params.covariance = BoundMatrix::Zero();
209 params.values = BoundVector::Zero();
210 detail::unpackCovariance(trackState.covMatrix.data(),
211 params.covariance.value());
212 params.values[0] = trackState.D0;
213 params.values[1] = trackState.Z0;
214 params.values[2] = trackState.phi;
215 params.values[3] = trackState.tanLambda;
216 params.values[4] = trackState.omega;
217 params.values[5] = trackState.time;
218
219 Vector3 center = {
220 trackState.referencePoint.x,
221 trackState.referencePoint.y,
222 trackState.referencePoint.z,
223 };
224 params.surface = Acts::Surface::makeShared<PerigeeSurface>(center);
225
226 return params;
227 };
228
229 ACTS_VERBOSE("Reading " << from.trackStates_size()
230 << " track states (including IP state)");
231 // We write the trackstates out outside in, need to reverse iterate to get the
232 // same order
233 for (std::size_t i = from.trackStates_size() - 1;
234 i <= from.trackStates_size(); i--) {
235 auto trackState = from.getTrackStates(i);
236 if (trackState.location == edm4hep::TrackState::AtIP) {
237 ipState = trackState;
238 continue;
239 }
240
241 auto params = unpack(trackState);
242
243 auto ts = track.appendTrackState(mask);
244 ts.typeFlags().set(MeasurementFlag);
245
246 auto converted = detail::convertTrackParametersFromEdm4hep(Bz, params);
247
248 ts.smoothed() = converted.parameters();
249 ts.smoothedCovariance() =
250 converted.covariance().value_or(BoundMatrix::Zero());
251 ts.setReferenceSurface(params.surface);
252 }
253
254 if (!ipState.has_value()) {
255 ACTS_ERROR("Did not find IP state in edm4hep input");
256 throw std::runtime_error{"Did not find IP state in edm4hep input"};
257 }
258
259 detail::Parameters params = unpack(ipState.value());
260
261 auto converted = detail::convertTrackParametersFromEdm4hep(Bz, params);
262
263 ACTS_VERBOSE("IP state parameters: " << converted.parameters().transpose());
264 ACTS_VERBOSE("-> covariance:\n"
265 << converted.covariance().value_or(BoundMatrix::Zero()));
266
267 track.parameters() = converted.parameters();
268 track.covariance() = converted.covariance().value_or(BoundMatrix::Zero());
269 track.setReferenceSurface(params.surface);
270
271 track.chi2() = from.getChi2();
272 track.nDoF() = from.getNdf();
273 track.nMeasurements() = track.nTrackStates();
274}
