77template <typename surface_t, typename bounds_t>
78std::shared_ptr<surface_t> surfaceFromJsonT(const nlohmann::json& j) {
79 nlohmann::json jTransform = j["transform"];
80 Transform3 sTransform = Transform3JsonConverter::fromJson(jTransform);
81 if constexpr (std::is_same_v<bounds_t, void>) {
82 return Surface::makeShared<surface_t>(sTransform);
83 } else {
84 nlohmann::json jBounds = j["bounds"];
85 auto sBounds = SurfaceBoundsJsonConverter::fromJson<bounds_t>(jBounds);
86 return Surface::makeShared<surface_t>(sTransform, std::move(sBounds));
87 }
88}
