29template <typename F, typename... Args>
30concept NavigationPolicyIsolatedFactoryConcept = requires(
31 F f, const GeometryContext& gctx, const TrackingVolume& volume,
32 const Logger& logger, Args&&... args) {
33 { f(gctx, volume, logger, args...) } -> std::derived_from<INavigationPolicy>;

35 requires NavigationPolicyConcept<decltype(f(gctx, volume, logger, args...))>;

37 requires(std::is_copy_constructible_v<Args> && ...);

51class NavigationPolicyFactory {
52 private:
54 using factory_type = std::function<std::unique_ptr<INavigationPolicy>(
55 const GeometryContext&, const TrackingVolume&, const Logger&)>;
56
59 explicit NavigationPolicyFactory(std::vector<factory_type>&& factories)
60 : m_factories(std::move(factories)) {}
61
62 public:
64 NavigationPolicyFactory() = default;
65
73 template <NavigationPolicyConcept P, typename... Args>
74 requires(std::is_constructible_v<P, const GeometryContext&,
75 const TrackingVolume&, const Logger&,
76 Args...> &&
77 (std::is_copy_constructible_v<Args> && ...))
78 constexpr NavigationPolicyFactory add(Args&&... args) && {
79 auto factory = [=](const GeometryContext& gctx,
80 const TrackingVolume& volume, const Logger& logger) {
81 return std::make_unique<P>(gctx, volume, logger, args...);
82 };
83
84 m_factories.push_back(std::move(factory));
85 return std::move(*this);
86 }
87
96 template <typename Fn, typename... Args>
97 requires(detail::NavigationPolicyIsolatedFactoryConcept<Fn, Args...>)
98 constexpr NavigationPolicyFactory add(Fn&& fn, Args&&... args) && {
99 auto factory = [=](const GeometryContext& gctx,
100 const TrackingVolume& volume, const Logger& logger) {
101 using policy_type = decltype(fn(gctx, volume, logger, args...));
102 return std::make_unique<policy_type>(fn(gctx, volume, logger, args...));
103 };
104
105 m_factories.push_back(std::move(factory));
106 return std::move(*this);
107 }
108
111 std::unique_ptr<NavigationPolicyFactory> asUniquePtr() && {
112 return std::make_unique<NavigationPolicyFactory>(std::move(*this));
113 }
114
121 std::unique_ptr<MultiNavigationPolicy> operator()(
122 const GeometryContext& gctx, const TrackingVolume& volume,
123 const Logger& logger) const {
124 if (m_factories.empty()) {
125 throw std::runtime_error(
126 "No factories registered in the navigation policy factory");
127 }
128
129 std::vector<std::unique_ptr<INavigationPolicy>> policies;
130 policies.reserve(m_factories.size());
131 for (auto& factory : m_factories) {
132 policies.push_back(factory(gctx, volume, logger));
133 }
134
135 return std::make_unique<MultiNavigationPolicy>(std::move(policies));
136 }

143 std::unique_ptr<INavigationPolicy> build(const GeometryContext& gctx,
144 const TrackingVolume& volume,
145 const Logger& logger) const {
146 return operator()(gctx, volume, logger);
147 }
148
149 private:
151 std::vector<factory_type> m_factories;
152};
