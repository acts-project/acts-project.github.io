63inline int binIndex(double min, double max, unsigned nSteps, double val) {
64 return static_cast<int>((val - min) / (max - min) * nSteps);
65}
66// Returns the lower bound of the bin specified by step
74inline double lowerBinEdge(double min, double max, unsigned nSteps,
75 std::size_t binIndex) {
76 return min + (max - min) * binIndex / nSteps;
77}
78// Returns the lower bound of the bin specified by step
86inline double binCenter(double min, double max, unsigned nSteps,
87 std::size_t binIndex) {
88 return min + (max - min) * 0.5 * (2 * binIndex + 1) / nSteps;
89}
