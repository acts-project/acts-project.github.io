62inline int binIndex(double min, double max, unsigned nSteps, double val) {
63 return static_cast<int>((val - min) / (max - min) * nSteps);
64}
65// Returns the lower bound of the bin specified by step
73inline double lowerBinEdge(double min, double max, unsigned nSteps,
74 std::size_t binIndex) {
75 return min + (max - min) * binIndex / nSteps;
76}
77// Returns the lower bound of the bin specified by step
85inline double binCenter(double min, double max, unsigned nSteps,
86 std::size_t binIndex) {
87 return min + (max - min) * 0.5 * (2 * binIndex + 1) / nSteps;
88}
