98class Barcode {
99 public:
100 using PrimaryVertexId = std::uint16_t;
101 using SecondaryVertexId = std::uint16_t;
102 using ParticleId = std::uint32_t;
103 using GenerationId = std::uint8_t;
104 using SubParticleId = std::uint32_t;
105
106 // Construct an invalid barcode with all levels set to zero.
107 static constexpr Barcode Invalid() { return Barcode(); }
108
110 constexpr Barcode() = default;
112 constexpr Barcode(const Barcode&) = default;
114 constexpr Barcode(Barcode&&) = default;
117 Barcode& operator=(const Barcode&) = default;
120 Barcode& operator=(Barcode&&) = default;
121
123 bool operator==(const Barcode&) const = default;
124 friend constexpr auto operator<=>(Barcode lhs, Barcode rhs) {
125 return lhs.asVector() <=> rhs.asVector();
126 }
127
129 static constexpr bool isValid(const Barcode& b) { return b != Invalid(); }
130 constexpr bool isValid() const { return isValid(*this); }
131
134 constexpr PrimaryVertexId vertexPrimary() const { return vertexPrimaryID; }
135
138 constexpr SecondaryVertexId vertexSecondary() const {
139 return vertexSecondaryID;
140 }
141
144 constexpr ParticleId particle() const { return particleID; }
145
148 constexpr GenerationId generation() const { return generationID; }
149
152 constexpr SubParticleId subParticle() const { return subParticleID; }
153
155 constexpr std::vector<std::uint32_t> asVector() const {
156 return {vertexPrimary(), vertexSecondary(), particle(), generation(),
157 subParticle()};
158 }
159
163 [[nodiscard]]
164 constexpr Barcode withVertexPrimary(PrimaryVertexId id) const {
165 Barcode barcode = *this;
166 barcode.vertexPrimaryID = id;
167 return barcode;
168 }

173 [[nodiscard]]

183 [[nodiscard]]

187 return barcode;
188 }
189

196 barcode.generationID = id;
197 return barcode;
198 }
199
203 [[nodiscard]]
204 constexpr Barcode withSubParticle(SubParticleId id) const {
205 Barcode barcode = *this;
206 barcode.subParticleID = id;
207 return barcode;
208 }
209
213 [[nodiscard]]
214 constexpr Barcode withData(std::span<std::uint32_t> data) {
215 if (data.size() != 5) {
216 throw std::invalid_argument(
217 "Size of the data is " + std::to_string(data.size()) +
218 " but Barcode requires data vector to have exactly 5 elements");
219 }
220
221 Barcode barcode = *this;
222 barcode.vertexPrimaryID = data[0];
223 barcode.vertexSecondaryID = data[1];
224 barcode.particleID = data[2];
225 barcode.generationID = data[3];
226 barcode.subParticleID = data[4];
227 return barcode;
228 }
229
234 Barcode makeDescendant(SubParticleId sub = 0u) const {
235 Barcode barcode = *this;
236 barcode.generationID += 1;
237 barcode.subParticleID = sub;
238 return barcode;
239 }
240
243 constexpr Barcode vertexId() const {
244 // The vertex is identified by primary vertex, secondary vertex, and
245 // generation. The other components are set to 0 so two particle originating
246 // from the same vertex will have the same vertex ID.
247 Barcode barcode = *this;
248 barcode.particleID = 0u;
249 barcode.subParticleID = 0u;
250 return barcode;
251 }
252
254 constexpr Barcode withoutSubparticle() const {
255 // Provide a pseudo-barcode that contains all fields but not the
256 // subparticle counter. This can be used as key in a map to store the
257 // subparticle information
258 Barcode barcode = *this;
259 barcode.subParticleID = 0u;
260 return barcode;
261 }
262
264 friend inline std::ostream& operator<<(std::ostream& os, Barcode barcode) {
265 // extra "+" to ensure printing as a number and not as a character
266 os << "vp=" << +barcode.vertexPrimary()
267 << "|vs=" << +barcode.vertexSecondary() << "|p=" << +barcode.particle()
268 << "|g=" << +barcode.generation() << "|sp=" << +barcode.subParticle();
269 return os;
270 }
271
273 std::size_t hash() const {
274 std::size_t seed = 0;
275 boost::hash_combine(seed, vertexPrimary());
276 boost::hash_combine(seed, vertexSecondary());
277 boost::hash_combine(seed, particle());
278 boost::hash_combine(seed, generation());
279 boost::hash_combine(seed, subParticle());
280
281 return seed;
282 }
283
284 private:
285 PrimaryVertexId vertexPrimaryID = 0u;
286 SecondaryVertexId vertexSecondaryID = 0u;
287 ParticleId particleID = 0u;
288 GenerationId generationID = 0u;
289 SubParticleId subParticleID = 0u;
290};
