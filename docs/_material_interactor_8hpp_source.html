27struct MaterialInteractor {
29 bool multipleScattering = true;
31 bool energyLoss = true;
33 bool recordInteractions = false;
35 NoiseUpdateMode noiseUpdateMode = NoiseUpdateMode::addNoise;
36
38 using result_type = RecordedMaterial;

57 template <typename propagator_state_t, typename stepper_t,
58 typename navigator_t>
59 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
60 const navigator_t& navigator, result_type& result,
61 const Logger& logger) const {
62 if (state.stage == PropagatorStage::postPropagation) {
63 return Result<void>::success();
64 }
65
66 // Do nothing if nothing is what is requested.
67 if (!(multipleScattering || energyLoss || recordInteractions)) {
68 return Result<void>::success();
69 }
70
71 // Handle surface material
72
73 // Note that start and target surface conditions are handled in the
74 // interaction code
75 const Surface* surface = navigator.currentSurface(state.navigation);
76
77 // We only have material interactions if there is potential material
78 if (surface && surface->surfaceMaterial()) {
79 ACTS_VERBOSE("MaterialInteractor | " << "Found material on surface "
80 << surface->geometryId());
81
82 // Prepare relevant input particle properties
83 detail::PointwiseMaterialInteraction interaction(surface, state, stepper);
84
85 // Determine the effective traversed material and its properties
86 // Material exists but it's not real, i.e. vacuum; there is nothing to do
87 if (interaction.evaluateMaterialSlab(state, navigator)) {
88 // Evaluate the material effects
89 interaction.evaluatePointwiseMaterialInteraction(multipleScattering,
90 energyLoss);
91
92 if (energyLoss) {
93 using namespace UnitLiterals;
94 ACTS_VERBOSE("MaterialInteractor | "
95 << interaction.slab << " absPdg=" << interaction.absPdg
96 << " mass=" << interaction.mass / 1_MeV << "MeV"
97 << " momentum=" << interaction.momentum / 1_GeV << "GeV"
98 << " energyloss=" << interaction.Eloss / 1_MeV << "MeV");
99 }
100
101 // To integrate process noise, we need to transport
102 // the covariance to the current position in space
103 if (interaction.performCovarianceTransport) {
104 stepper.transportCovarianceToCurvilinear(state.stepping);
105 }
106 // Apply the material interactions
107 interaction.updateState(state, stepper, noiseUpdateMode);
108
109 // Record the result
110 recordResult(interaction, result);
111 }
112 }
113
114 // Handle volume material
115
116 // In case of Volume material update the result of the previous step
117 if (!result.materialInteractions.empty() &&
118 !result.materialInteractions.back().volume.empty() &&
119 result.materialInteractions.back().updatedVolumeStep == false) {
120 updateResult(state, stepper, result);
121 }
122
123 auto volume = navigator.currentVolume(state.navigation);
124
125 // We only have material interactions if there is potential material
126 if (volume && volume->volumeMaterial()) {
127 ACTS_VERBOSE("MaterialInteractor | " << "Found material in volume "
128 << volume->geometryId());
129
130 // Prepare relevant input particle properties
131 detail::VolumeMaterialInteraction interaction(volume, state, stepper);
132 // Determine the effective traversed material and its properties
133 // Material exists but it's not real, i.e. vacuum; there is nothing to do
134 if (interaction.evaluateMaterialSlab(state, navigator)) {
135 // Record the result
136 recordResult(interaction, result);
137 }
138 }
139 return Result<void>::success();
140 }
141
142 private:
147 void recordResult(const detail::PointwiseMaterialInteraction& interaction,
148 result_type& result) const {
149 result.materialInX0 += interaction.slab.thicknessInX0();
150 result.materialInL0 += interaction.slab.thicknessInL0();
151
152 // Record the interaction if requested
153 if (!recordInteractions) {
154 return;
155 }
156
157 MaterialInteraction mi;
158 mi.position = interaction.pos;
159 mi.time = interaction.time;
160 mi.direction = interaction.dir;
161 mi.deltaP = interaction.nextP - interaction.momentum;
162 mi.sigmaPhi2 = interaction.variancePhi;
163 mi.sigmaTheta2 = interaction.varianceTheta;
164 mi.sigmaQoP2 = interaction.varianceQoverP;
165 mi.surface = interaction.surface;
166 mi.volume = InteractionVolume();
167 mi.pathCorrection = interaction.pathCorrection;
168 mi.materialSlab = interaction.slab;
169 result.materialInteractions.push_back(std::move(mi));
170 }
171
176 void recordResult(const detail::VolumeMaterialInteraction& interaction,
177 result_type& result) const {
178 // Record the interaction if requested
179 if (!recordInteractions) {
180 return;
181 }
182
183 MaterialInteraction mi;
184 mi.position = interaction.pos;
185 mi.time = interaction.time;
186 mi.direction = interaction.dir;
187 mi.surface = nullptr;
188 mi.volume = interaction.volume;
189 mi.pathCorrection = interaction.pathCorrection;
190 mi.materialSlab = interaction.slab;
191 result.materialInteractions.push_back(std::move(mi));
192 }

199 template <typename propagator_state_t, typename stepper_t>
200 void updateResult(propagator_state_t& state, const stepper_t& stepper,
201 result_type& result) const {
202 // Update the previous interaction if requested
203 if (!recordInteractions) {
204 return;
205 }
206
207 Vector3 shift = stepper.position(state.stepping) -
208 result.materialInteractions.back().position;
209 double momentum = stepper.direction(state.stepping).norm();
210 result.materialInteractions.back().deltaP =
211 momentum - result.materialInteractions.back().direction.norm();
212 result.materialInteractions.back().materialSlab.scaleThickness(
213 shift.norm());
214 result.materialInteractions.back().updatedVolumeStep = true;
215 result.materialInX0 +=
216 result.materialInteractions.back().materialSlab.thicknessInX0();
217 result.materialInL0 +=
218 result.materialInteractions.back().materialSlab.thicknessInL0();
219 }
220};
