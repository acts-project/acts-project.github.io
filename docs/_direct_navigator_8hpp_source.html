228 [[nodiscard]] Result<void> initialize(State& state, const Vector3& position,
229 const Vector3& direction,
230 Direction propagationDirection) const {
231 static_cast<void>(position);
232 static_cast<void>(direction);
233
234 ACTS_VERBOSE("Initialize. Surface sequence for navigation:");
235 for (const Surface* surface : state.options.surfaces) {
236 ACTS_VERBOSE(surface->geometryId()
237 << " - "
238 << surface->center(state.options.geoContext).transpose());
239 }
240
241 state.direction = propagationDirection;
242 ACTS_VERBOSE("Navigation direction is " << propagationDirection);
243
244 // We set the current surface to the start surface
245 state.currentSurface = state.options.startSurface;
246 if (state.currentSurface != nullptr) {
247 ACTS_VERBOSE("Current surface set to start surface "
248 << state.currentSurface->geometryId());
249 } else {
250 ACTS_VERBOSE("Current surface set to nullptr");
251 }
252
253 // Find initial index.
254 auto found =
255 std::ranges::find(state.options.surfaces, state.options.startSurface);
256
257 if (found != state.options.surfaces.end()) {
258 // The index should be the index before the start surface, depending on
259 // the direction
260 state.surfaceIndex = std::distance(state.options.surfaces.begin(), found);
261 state.surfaceIndex += state.direction == Direction::Backward() ? 1 : -1;
262 } else {
263 ACTS_DEBUG(
264 "Did not find the start surface in the sequence. Assuming it is not "
265 "part of the sequence. Trusting the correctness of the input "
266 "sequence. Resetting the surface index.");
267 state.resetSurfaceIndex();
268 }
269
270 state.navigationBreak = false;
271
272 return Result<void>::success();
273 }

285 NavigationTarget nextTarget(State& state, const Vector3& position,
286 const Vector3& direction) const {
287 // Navigator target always resets the current surface
288 state.currentSurface = nullptr;
289
290 if (state.navigationBreak) {
291 return NavigationTarget::None();
292 }
293
294 ACTS_VERBOSE("DirectNavigator::nextTarget");
295
296 // Move the sequence to the next surface
297 state.nextSurface();
298
299 while (!state.endOfSurfaces()) {
300 ACTS_VERBOSE("Next surface candidate is "
301 << state.navSurface().geometryId() << ". "
302 << state.remainingSurfaces() << " out of "
303 << state.options.surfaces.size()
304 << " surfaces remain to try.");
305
306 // Establish & update the surface status
307 // TODO we do not know the intersection index - passing the closer one
308 const Surface& surface = state.navSurface();
309 const double farLimit = std::numeric_limits<double>::max();
310 const NavigationTarget target = chooseIntersection(
311 state.options.geoContext, surface, position, direction,
312 BoundaryTolerance::Infinite(), state.options.nearLimit, farLimit,
313 state.options.surfaceTolerance);
314 if (target.isValid()) {
315 return target;
316 }
317
318 ACTS_VERBOSE("No valid intersection found with surface "
319 << surface.geometryId() << ", trying next surface.");
320 state.nextSurface();
321 }
322
323 ACTS_VERBOSE("End of surfaces reached, navigation break.");
324 state.navigationBreak = true;
325 return NavigationTarget::None();
326 }

375 private:
376 NavigationTarget chooseIntersection(
377 const GeometryContext& gctx, const Surface& surface,
378 const Vector3& position, const Vector3& direction,
379 const BoundaryTolerance& boundaryTolerance, double nearLimit,
380 double farLimit, double tolerance) const {
381 auto intersections = surface.intersect(gctx, position, direction,
382 boundaryTolerance, tolerance);
383
384 for (auto [intersectionIndex, intersection] :
385 Acts::enumerate(intersections)) {
386 if (intersection.isValid() &&
387 detail::checkPathLength(intersection.pathLength(), nearLimit,
388 farLimit, logger())) {
389 return NavigationTarget(intersection, intersectionIndex, surface,
390 boundaryTolerance);
391 }
392 }
393
394 return NavigationTarget::None();
395 }
