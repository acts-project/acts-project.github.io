43struct MaxMomentumComponent {
44 template <typename component_range_t>
45 auto operator()(const component_range_t& cmps) const {
46 return std::ranges::max_element(cmps, [&](const auto& a, const auto& b) {
47 return std::abs(a.state.pars[eFreeQOverP]) >
48 std::abs(b.state.pars[eFreeQOverP]);
49 });
50 }
51};
52
53struct MaxWeightComponent {
54 template <typename component_range_t>
55 auto operator()(const component_range_t& cmps) {
56 return std::ranges::max_element(cmps, [&](const auto& a, const auto& b) {
57 return a.weight < b.weight;
58 });
59 }
60};

62template <typename component_chooser_t>
63struct SingleComponentReducer {
64 template <typename stepper_state_t>
65 static Vector3 position(const stepper_state_t& s) {
66 return component_chooser_t{}(s.components)
67 ->state.pars.template segment<3>(eFreePos0);
68 }
69
70 template <typename stepper_state_t>
71 static Vector3 direction(const stepper_state_t& s) {
72 return component_chooser_t{}(s.components)
73 ->state.pars.template segment<3>(eFreeDir0);
74 }
75
76 template <typename stepper_state_t>
77 static double qOverP(const stepper_state_t& s) {
78 const auto cmp = component_chooser_t{}(s.components);
79 return cmp->state.pars[eFreeQOverP];
80 }
81
82 template <typename stepper_state_t>
83 static double absoluteMomentum(const stepper_state_t& s) {
84 const auto cmp = component_chooser_t{}(s.components);
85 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]);
86 }
87
88 template <typename stepper_state_t>
89 static Vector3 momentum(const stepper_state_t& s) {
90 const auto cmp = component_chooser_t{}(s.components);
91 return s.particleHypothesis.extractMomentum(cmp->state.pars[eFreeQOverP]) *
92 cmp->state.pars.template segment<3>(eFreeDir0);
93 }
94
95 template <typename stepper_state_t>
96 static double charge(const stepper_state_t& s) {
97 const auto cmp = component_chooser_t{}(s.components);
98 return s.particleHypothesis.extractCharge(cmp->state.pars[eFreeQOverP]);
99 }
100
101 template <typename stepper_state_t>
102 static double time(const stepper_state_t& s) {
103 return component_chooser_t{}(s.components)->state.pars[eFreeTime];
104 }
105
106 template <typename stepper_state_t>
107 static FreeVector pars(const stepper_state_t& s) {
108 return component_chooser_t{}(s.components)->state.pars;
109 }
110
111 template <typename stepper_state_t>
112 static FreeVector cov(const stepper_state_t& s) {
113 return component_chooser_t{}(s.components)->state.cov;
114 }

263 void initialize(State& state,
264 const MultiComponentBoundTrackParameters& par) const {
265 if (par.components().empty()) {
266 throw std::invalid_argument(
267 "Cannot construct MultiEigenStepperLoop::State with empty "
268 "multi-component parameters");
269 }
270
271 state.particleHypothesis = par.particleHypothesis();
272
273 const auto surface = par.referenceSurface().getSharedPtr();
274
275 for (auto i = 0ul; i < par.components().size(); ++i) {
276 const auto& [weight, singlePars] = par[i];
277 auto& cmp =
278 state.components.emplace_back(SingleStepper::makeState(state.options),
279 weight, IntersectionStatus::onSurface);
280 SingleStepper::initialize(cmp.state, singlePars);
281 }
282
283 if (std::get<2>(par.components().front())) {
284 state.covTransport = true;
285 }
286 }

398 void reweightComponents(State& state) const {
399 double sumOfWeights = 0.0;
400 for (const auto& cmp : state.components) {
401 sumOfWeights += cmp.weight;
402 }
403 for (auto& cmp : state.components) {
404 cmp.weight /= sumOfWeights;
405 }

522 IntersectionStatus updateSurfaceStatus(
523 State& state, const Surface& surface, std::uint8_t index,
524 Direction navDir, const BoundaryTolerance& boundaryTolerance,
525 double surfaceTolerance, ConstrainedStep::Type stype,
526 const Logger& logger = getDummyLogger()) const {
527 using Status = IntersectionStatus;
528
529 std::array<int, 3> counts = {0, 0, 0};
530
531 for (auto& component : state.components) {
532 component.status = detail::updateSingleSurfaceStatus<SingleStepper>(
533 *this, component.state, surface, index, navDir, boundaryTolerance,
534 surfaceTolerance, stype, logger);
535 ++counts[static_cast<std::size_t>(component.status)];
536 }
537
538 // If at least one component is on a surface, we can remove all missed
539 // components before the step. If not, we must keep them for the case that
540 // all components miss and we need to retarget
541 if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
542 removeMissedComponents(state);
543 reweightComponents(state);
544 }
545
546 ACTS_VERBOSE("Component status wrt "
547 << surface.geometryId() << " at {"
548 << surface.center(state.options.geoContext).transpose()
549 << "}:\t" << [&]() {
550 std::stringstream ss;
551 for (auto& component : state.components) {
552 ss << component.status << "\t";
553 }
554 return ss.str();
555 }());
556
557 // Switch on stepCounter if one or more components reached a surface, but
558 // some are still in progress of reaching the surface
559 if (!state.stepCounterAfterFirstComponentOnSurface &&
560 counts[static_cast<std::size_t>(Status::onSurface)] > 0 &&
561 counts[static_cast<std::size_t>(Status::reachable)] > 0) {
562 state.stepCounterAfterFirstComponentOnSurface = 0;
563 ACTS_VERBOSE("started stepCounterAfterFirstComponentOnSurface");
564 }
565
566 // If there are no components onSurface, but the counter is switched on
567 // (e.g., if the navigator changes the target surface), we need to switch it
568 // off again
569 if (state.stepCounterAfterFirstComponentOnSurface &&
570 counts[static_cast<std::size_t>(Status::onSurface)] == 0) {
571 state.stepCounterAfterFirstComponentOnSurface.reset();
572 ACTS_VERBOSE("switch off stepCounterAfterFirstComponentOnSurface");
573 }
574
575 // This is a 'any_of' criterium. As long as any of the components has a
576 // certain state, this determines the total state (in the order of a
577 // somewhat importance)
578 if (counts[static_cast<std::size_t>(Status::reachable)] > 0) {
579 return Status::reachable;
580 } else if (counts[static_cast<std::size_t>(Status::onSurface)] > 0) {
581 state.stepCounterAfterFirstComponentOnSurface.reset();
582 return Status::onSurface;
583 } else {
584 return Status::unreachable;
585 }
586 }

599 template <typename object_intersection_t>
600 void updateStepSize(State& state, const object_intersection_t& oIntersection,
601 Direction direction, ConstrainedStep::Type stype) const {
602 const Surface& surface = *oIntersection.object();
603
604 for (auto& component : state.components) {
605 auto intersection = surface.intersect(
606 component.state.options.geoContext,
607 SingleStepper::position(component.state),
608 direction * SingleStepper::direction(component.state),
609 BoundaryTolerance::None())[oIntersection.index()];
610
611 SingleStepper::updateStepSize(component.state, intersection, direction,
612 stype);
613 }
614 }
