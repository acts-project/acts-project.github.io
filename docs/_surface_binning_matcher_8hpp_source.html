20struct SurfaceBinningMatcher {
22 using Range = std::pair<double, double>;
23 std::vector<Range> tolerances{static_cast<int>(numAxisDirections()),
24 {0., 0.}};
25
26 SurfaceBinningMatcher() = default;
27
28 explicit SurfaceBinningMatcher(const std::vector<Range>& tolpars)
29 : tolerances(tolpars) {}
30
37 bool operator()(const Acts::GeometryContext& gctx, Acts::AxisDirection aDir,
38 const Acts::Surface* one, const Acts::Surface* other) const {
39 // Fast exit
40 if (one == other) {
41 return true;
42 }
43
44 auto oneExt = one->polyhedronRepresentation(gctx, 1).extent();
45 auto otherExt = other->polyhedronRepresentation(gctx, 1).extent();
46
47 double oneMin = oneExt.min(aDir);
48 double oneMax = oneExt.max(aDir);
49
50 double otherMin = otherExt.min(aDir);
51 double otherMax = otherExt.max(aDir);
52
53 return (
54 std::abs(oneMin - otherMin) <= tolerances[toUnderlying(aDir)].first &&
55 std::abs(oneMax - otherMax) <= tolerances[toUnderlying(aDir)].second);
56 }
57};
