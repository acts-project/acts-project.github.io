50 template <typename traj_t>
51 Result<void> operator()(const GeometryContext& gctx, traj_t& trajectory,
52 std::size_t entryIndex,
53 const Logger& logger = getDummyLogger()) const {
54 static_cast<void>(gctx);
55 static_cast<void>(logger);
56
57 using TrackStateProxy = typename traj_t::TrackStateProxy;
58
59 TrackStateProxy startTs = trajectory.getTrackState(entryIndex);
60
61 // Notation consistent with the Wikipedia article
62 // https://en.wikipedia.org/wiki/Kalman_filter
63 BoundMatrix bigLambdaHat = BoundMatrix::Zero();
64 BoundVector smallLambdaHat = BoundVector::Zero();
65
66 trajectory.applyBackwards(startTs.index(), [&](TrackStateProxy ts) {
67 // ensure the track state has a smoothed component
68 ts.addComponents(TrackStatePropMask::Smoothed);
69
70 AnyMutableTrackStateProxy internalTrackState(ts);
71
72 // Smoothe the current state
73 calculateSmoothed(internalTrackState, bigLambdaHat, smallLambdaHat);
74
75 // We smoothed the last state - no need to update the lambdas
76 if (!ts.hasPrevious()) {
77 return;
78 }
79
80 // Update the lambdas depending on the type of track state
81 if (ts.typeFlags().test(TrackStateFlag::MeasurementFlag)) {
82 visitMeasurement(AnyConstTrackStateProxy{ts}, bigLambdaHat,
83 smallLambdaHat);
84 } else {
85 visitNonMeasurement(std::as_const(ts).jacobian(), bigLambdaHat,
86 smallLambdaHat);
87 }
88 });
89
90 return Result<void>::success();
91 }
92
93 private:
95 void calculateSmoothed(AnyMutableTrackStateProxy& ts,
96 const BoundMatrix& bigLambdaHat,
97 const BoundVector& smallLambdaHat) const;
98
100 void visitNonMeasurement(
101 const AnyConstTrackStateProxy::ConstCovarianceMap& jacobian,
102 BoundMatrix& bigLambdaHat, BoundVector& smallLambdaHat) const;
103
105 void visitMeasurement(const AnyConstTrackStateProxy& ts,
106 BoundMatrix& bigLambdaHat,
107 BoundVector& smallLambdaHat) const;
108};
