101template <Acts::Concepts::arithmetic T>
102class Tensor {
103 public:
105 using Shape = std::array<std::size_t, 2>;
106
111 static Tensor Create(Shape shape, const ExecutionContext &execContext) {
112 auto ptr = detail::createTensorMemory(shape[0] * shape[1] * sizeof(T),
113 execContext);
114 return Tensor(shape, std::move(ptr), execContext);
115 }
116
122 Tensor clone(const ExecutionContext &to) const {
123 auto clonedPtr = detail::cloneTensorMemory(m_ptr, nbytes(), m_device, to);
124 return Tensor(m_shape, std::move(clonedPtr), to);
125 }
126
129 T *data() { return static_cast<T *>(m_ptr.get()); }
130
133 const T *data() const { return static_cast<const T *>(m_ptr.get()); }
134
137 Shape shape() const { return m_shape; }
138
141 std::size_t size() const { return m_shape[0] * m_shape[1]; }
142
145 std::size_t nbytes() const { return size() * sizeof(T); }
146
148 Device device() const { return m_device; }
150
151 private:
152 Tensor(Shape shape, detail::TensorPtr ptr, const ExecutionContext &ctx)
153 : m_shape(shape), m_ptr(std::move(ptr)), m_device(ctx.device) {}
154
155 Shape m_shape{};
156 detail::TensorPtr m_ptr;
157 Device m_device{};
158};
