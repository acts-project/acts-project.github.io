45template <typename R, typename H, DelegateType O, typename... Args>
46class Delegate<R(Args...), H, O> {
47 public:
49 static constexpr DelegateType kOwnership = O;
50
52 using return_type = R;
54 using holder_type = H;
56 using function_type = return_type (*)(const holder_type *, Args...);
58 using function_ptr_type = return_type (*)(Args...);
60 using signature_type = R(Args...);
61
63 using deleter_type = void (*)(const holder_type *);
64
65 private:
66 template <typename T, typename C>
67 using isSignatureCompatible =
68 decltype(std::declval<T &>() = std::declval<C>());
69
70 using OwningDelegate =
71 Delegate<R(Args...), holder_type, DelegateType::Owning>;
72 using NonOwningDelegate =
73 Delegate<R(Args...), holder_type, DelegateType::NonOwning>;
74
75 template <typename T>
76 using isNoFunPtr = std::conjunction<
77 std::negation<std::is_convertible<std::decay_t<T>, function_type>>,
78 std::negation<std::is_same<std::decay_t<T>, OwningDelegate>>,
79 std::negation<std::is_same<std::decay_t<T>, NonOwningDelegate>>>;
80
81 public:
82 Delegate() = default;
83
85 Delegate(Delegate &&) noexcept = default;
88 Delegate &operator=(Delegate &&) noexcept = default;
90 Delegate(const Delegate &) noexcept = default;
93 Delegate &operator=(const Delegate &) noexcept = default;

101 explicit Delegate(function_type callable) { connect(callable); }

136 }

143 = delete;

159 void operator=(Callable &callable)
160 requires(isNoFunPtr<Callable>::value)

162 connect(callable);

177 template <auto Callable>
178 void connect()
179 requires(
180 Concepts::invocable_and_returns<Callable, return_type, Args && ...>)
181 {
182 m_payload.payload = nullptr;
183
184 static_assert(
185 Concepts::invocable_and_returns<Callable, return_type, Args &&...>,
186 "Callable given does not correspond exactly to required call "
187 "signature");
188
189 m_function = [](const holder_type * /*payload*/,
190 Args... args) -> return_type {
191 return std::invoke(Callable, std::forward<Args>(args)...);
192 };
193 }
195
201 template <typename Callable>
202 void connect(Callable &callable)
203 requires(isNoFunPtr<Callable>::value)
204 {
205 connect<&Callable::operator(), Callable>(&callable);
206 }
207
210 template <typename Callable>
211 void connect(Callable &&)
212 requires(isNoFunPtr<Callable>::value)
213 = delete;
214
221 void connect(function_type callable) {
222 if constexpr (kOwnership == DelegateType::NonOwning) {
223 m_payload.payload = nullptr;
224 }
225 m_function = callable;
226 }
228
233 template <typename Type>
234 void connect(function_type callable, const Type *instance)
235 requires(kOwnership == DelegateType::NonOwning)
236 {
237 m_payload.payload = instance;
238 m_function = callable;
239 }
240
247 template <auto Callable, typename Type>
248 void connect(const Type *instance)
249 requires(kOwnership == DelegateType::NonOwning &&
250 Concepts::invocable_and_returns<Callable, return_type, Type,
251 Args && ...>)
252
253 {
254 m_payload.payload = instance;
255
256 m_function = [](const holder_type *payload, Args... args) -> return_type {
257 assert(payload != nullptr && "Payload is required, but not set");
258 const auto *concretePayload = static_cast<const Type *>(payload);
259 return std::invoke(Callable, concretePayload,
260 std::forward<Args>(args)...);
261 };
262 }
263
269 template <auto Callable, typename Type>
270 void connect(std::unique_ptr<const Type> instance)
271 requires(kOwnership == DelegateType::Owning &&
272 Concepts::invocable_and_returns<Callable, return_type, Type,
273 Args && ...>)
274 {
275 m_payload.payload = std::unique_ptr<const holder_type, deleter_type>(
276 instance.release(), [](const holder_type *payload) {
277 const auto *concretePayload = static_cast<const Type *>(payload);
278 delete concretePayload;
279 });
280
281 m_function = [](const holder_type *payload, Args &&...args) -> return_type {
282 assert(payload != nullptr && "Payload is required, but not set");
283 const auto *concretePayload = static_cast<const Type *>(payload);
284 return std::invoke(Callable, concretePayload,
285 std::forward<Args>(args)...);
286 };
287 }
288
292 template <typename... Ts>
293 return_type operator()(Ts &&...args) const
294 requires(std::is_invocable_v<function_type, const holder_type *, Ts...>)
295 {
296 assert(connected() && "Delegate is not connected");
297 return std::invoke(m_function, m_payload.ptr(), std::forward<Ts>(args)...);
298 }
299
302 bool connected() const { return m_function != nullptr; }
303
306 explicit operator bool() const { return connected(); }
307
309 void disconnect() {
310 m_payload.clear();
311 m_function = nullptr;
312 }
313
316 const holder_type *instance() const
317 requires(!std::same_as<holder_type, void>)
318 {
319 return m_payload.ptr();
320 }
321
322 private:
323 // Deleter that does not do anything
324 static void noopDeleter(const holder_type * /*unused*/) {
325 // we do not own the payload
326 }
327
329
330 // Payload object without a deleter
331 struct NonOwningPayload {
332 void clear() { payload = nullptr; }
333
334 const holder_type *ptr() const { return payload; }
335
336 const holder_type *payload{nullptr};
337 };
338
339 // Payload object with a deleter
340 struct OwningPayload {
341 void clear() { payload.reset(); }
342
343 const holder_type *ptr() const { return payload.get(); }
344
345 std::unique_ptr<const holder_type, deleter_type> payload{nullptr,
346 &noopDeleter};
347 };
348
350 std::conditional_t<kOwnership == DelegateType::NonOwning, NonOwningPayload,
351 OwningPayload>
352 m_payload;
353
355
357 function_type m_function{nullptr};
358};
