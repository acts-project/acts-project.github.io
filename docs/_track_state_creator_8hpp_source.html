108 Result<CkfTypes::BranchVector<TrackIndexType>> createTrackStates(
109 const GeometryContext& gctx, const CalibrationContext& calibrationContext,
110 [[maybe_unused]] const Surface& surface, const BoundState& boundState,
111 TrackIndexType prevTip,
112 std::vector<TrackStateProxy>& trackStateCandidates,
113 TrackStateContainerBackend& trajectory, const Logger& logger) const {
114 TrackStatesResult tsRes = TrackStatesResult::success({});
115 using SourceLinkRange = decltype(sourceLinkAccessor(surface));
116 SourceLinkRange slRange = sourceLinkAccessor(surface);
117 if (slRange.first != slRange.second) {
118 auto [slBegin, slEnd] = slRange;
119 tsRes = createSourceLinkTrackStates(
120 gctx, calibrationContext, surface, boundState, slBegin, slEnd,
121 prevTip, trackStateCandidates, trajectory, logger);
122 }

140 Result<CkfTypes::BranchVector<TrackIndexType>> createSourceLinkTrackStates(
141 const GeometryContext& gctx, const CalibrationContext& calibrationContext,
142 [[maybe_unused]] const Surface& surface, const BoundState& boundState,
143 source_link_iterator_t slBegin, source_link_iterator_t slEnd,
144 TrackIndexType prevTip,
145 std::vector<TrackStateProxy>& trackStateCandidates,
146 TrackStateContainerBackend& trajectory, const Logger& logger) const {
147 using PM = TrackStatePropMask;
148
149 using ResultTrackStateList =
150 Acts::Result<CkfTypes::BranchVector<TrackIndexType>>;
151 ResultTrackStateList resultTrackStateList{
152 CkfTypes::BranchVector<TrackIndexType>()};
153 const auto& [boundParams, jacobian, pathLength] = boundState;
154
155 trackStateCandidates.clear();
156 if constexpr (std::ranges::random_access_range<source_link_iterator_t>) {
157 trackStateCandidates.reserve(std::distance(slBegin, slEnd));
158 }
159
160 // Calibrate all the source links on the surface since the selection has
161 // to be done based on calibrated measurement
162 for (auto it = slBegin; it != slEnd; ++it) {
163 // get the source link
164 const auto sourceLink = *it;
165
166 // prepare the track state
167 PM mask = PM::Predicted | PM::Jacobian | PM::Calibrated;
168 if (it != slBegin) {
169 // not the first TrackState, only need uncalibrated and calibrated
170 mask = PM::Calibrated;
171 }
172
173 ACTS_VERBOSE("Create temp track state with mask: " << mask);
174 // Temporary and final track states are created in the same
175 // trajectory, which could lead to very large containers.
176
177 // CAREFUL! This trackstate has a previous index that is not in this
178 // MultiTrajectory Visiting backwards from this track state will
179 // fail!
180 auto ts = trajectory.makeTrackState(mask, prevTip);
181
182 if (it == slBegin) {
183 // only set these for first
184 ts.predicted() = boundParams.parameters();
185 if (boundParams.covariance()) {
186 ts.predictedCovariance() = *boundParams.covariance();
187 }
188 ts.jacobian() = jacobian;
189 } else {
190 // subsequent track states can reuse
191 auto& first = trackStateCandidates.front();
192 ts.shareFrom(first, PM::Predicted);
193 ts.shareFrom(first, PM::Jacobian);
194 }
195
196 ts.pathLength() = pathLength;
197 ts.setReferenceSurface(boundParams.referenceSurface().getSharedPtr());
198
199 // now calibrate the track state
200 calibrator(gctx, calibrationContext, sourceLink, ts);
201
202 trackStateCandidates.push_back(ts);
203 }
204
205 bool isOutlier = false;
206 Result<std::pair<typename std::vector<TrackStateProxy>::iterator,
207 typename std::vector<TrackStateProxy>::iterator>>
208 selectorResult =
209 measurementSelector(trackStateCandidates, isOutlier, logger);
210 if (!selectorResult.ok()) {
211 ACTS_ERROR("Selection of calibrated measurements failed: "
212 << selectorResult.error().message());
213 resultTrackStateList =
214 ResultTrackStateList::failure(selectorResult.error());
215 } else {
216 auto selectedTrackStateRange = *selectorResult;
217 resultTrackStateList = processSelectedTrackStates(
218 selectedTrackStateRange.first, selectedTrackStateRange.second,
219 trajectory, isOutlier, logger);
220 }
221
222 return resultTrackStateList;
223 }

233 Result<CkfTypes::BranchVector<TrackIndexType>> processSelectedTrackStates(
234 typename std::vector<TrackStateProxy>::const_iterator begin,
235 typename std::vector<TrackStateProxy>::const_iterator end,
236 TrackStateContainerBackend& trackStates, bool isOutlier,
237 const Logger& logger) const {
238 using PM = TrackStatePropMask;
239
240 using ResultTrackStateList =
241 Acts::Result<CkfTypes::BranchVector<TrackIndexType>>;
242 ResultTrackStateList resultTrackStateList{
243 CkfTypes::BranchVector<TrackIndexType>()};
244 CkfTypes::BranchVector<TrackIndexType>& trackStateList =
245 *resultTrackStateList;
246 trackStateList.reserve(end - begin);
247
248 std::optional<TrackStateProxy> firstTrackState{std::nullopt};
249 for (auto it = begin; it != end; ++it) {
250 auto& candidateTrackState = *it;
251
252 PM mask = PM::Predicted | PM::Filtered | PM::Jacobian | PM::Calibrated;
253 if (it != begin) {
254 // subsequent track states don't need storage for these as they will
255 // be shared
256 mask &= ~PM::Predicted & ~PM::Jacobian;
257 }
258 if (isOutlier) {
259 // outlier won't have separate filtered parameters
260 mask &= ~PM::Filtered;
261 }
262
263 // copy this trackstate into fitted states MultiTrajectory
264 auto trackState =
265 trackStates.makeTrackState(mask, candidateTrackState.previous());
266 ACTS_VERBOSE("Create SourceLink output track state #"
267 << trackState.index() << " with mask: " << mask);
268
269 if (it != begin) {
270 // assign indices pointing to first track state
271 trackState.shareFrom(*firstTrackState, PM::Predicted);
272 trackState.shareFrom(*firstTrackState, PM::Jacobian);
273 } else {
274 firstTrackState = trackState;
275 }
276
277 // either copy ALL or everything except for predicted and jacobian
278 trackState.copyFrom(candidateTrackState, mask, false);
279
280 auto typeFlags = trackState.typeFlags();
281 typeFlags.set(TrackStateFlag::ParameterFlag);
282 typeFlags.set(TrackStateFlag::MeasurementFlag);
283 if (trackState.referenceSurface().surfaceMaterial() != nullptr) {
284 typeFlags.set(TrackStateFlag::MaterialFlag);
285 }
286 if (isOutlier) {
287 // propagate information that this is an outlier state
288 ACTS_VERBOSE(
289 "Creating outlier track state with tip = " << trackState.index());
290 typeFlags.set(TrackStateFlag::OutlierFlag);
291 }
292
293 trackStateList.push_back(trackState.index());
294 }
295 return resultTrackStateList;
296 }
