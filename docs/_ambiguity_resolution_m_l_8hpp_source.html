60 template <TrackContainerFrontend track_container_t,
61 typename source_link_hash_t, typename source_link_equality_t>
62 std::multimap<int, std::pair<std::size_t, std::vector<std::size_t>>>
63 mapTrackHits(const track_container_t& tracks,
64 const source_link_hash_t& sourceLinkHash,
65 const source_link_equality_t& sourceLinkEquality) const {
66 // A map to store (and generate) the measurement index for each source link
67 auto measurementIndexMap =
68 std::unordered_map<SourceLink, std::size_t, source_link_hash_t,
69 source_link_equality_t>(0, sourceLinkHash,
70 sourceLinkEquality);
71
72 // A map to store the track Id and their associated measurements ID, a
73 // multimap is used to automatically sort the tracks by the number of
74 // measurements
75 std::multimap<int, std::pair<std::size_t, std::vector<std::size_t>>>
76 trackMap;
77 std::size_t trackIndex = 0;
78 std::vector<std::size_t> measurements;
79 // Loop over all the trajectories in the events
80 for (const auto& track : tracks) {
81 // Kick out tracks that do not fulfill our initial requirements
82 if (track.nMeasurements() < m_cfg.nMeasurementsMin) {
83 continue;
84 }
85 measurements.clear();
86 for (auto ts : track.trackStatesReversed()) {
87 if (ts.typeFlags().test(Acts::TrackStateFlag::MeasurementFlag)) {
88 SourceLink sourceLink = ts.getUncalibratedSourceLink();
89 // assign a new measurement index if the source link was not seen yet
90 auto emplace = measurementIndexMap.try_emplace(
91 sourceLink, measurementIndexMap.size());
92 measurements.push_back(emplace.first->second);
93 }
94 }
95 trackMap.emplace(track.nMeasurements(),
96 std::make_pair(trackIndex, measurements));
97 ++trackIndex;
98 }
99 return trackMap;
100 }

110 template <TrackContainerFrontend track_container_t>
111 std::vector<std::size_t> solveAmbiguity(
112 std::unordered_map<std::size_t, std::vector<std::size_t>>& clusters,
113 const track_container_t& tracks) const {
114 std::vector<std::vector<float>> outputTensor =
115 m_duplicateClassifier.inferScores(clusters, tracks);
116 std::vector<std::size_t> goodTracks =
117 m_duplicateClassifier.trackSelection(clusters, outputTensor);
118
119 return goodTracks;
120 }
