72 trackToVerticesMultiMap.emplace(trk, &vtx);

78 void removeVertexFromMultiMap(Vertex& vtx) {
79 for (auto iter = trackToVerticesMultiMap.begin();
80 iter != trackToVerticesMultiMap.end();) {
81 if (iter->second == &vtx) {
82 iter = trackToVerticesMultiMap.erase(iter);
83 } else {
84 ++iter;
85 }
86 }
87 }

93 Result<void> removeVertexFromCollection(Vertex& vtxToRemove,
94 const Logger& logger) {
95 auto it = std::ranges::find(vertexCollection, &vtxToRemove);
96 // Check if the value was found before erasing
97 if (it == vertexCollection.end()) {
98 ACTS_ERROR("vtxToRemove is not part of vertexCollection.");
99 return VertexingError::ElementNotFound;
100 }
101 // Erase the element if found
102 vertexCollection.erase(it);
103 return {};
104 }
