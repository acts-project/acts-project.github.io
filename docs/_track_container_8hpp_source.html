474 return m_traj->forwardTrackStateRange(stem);

479 void copyDynamicFrom(IndexType dstIdx, const T& src, IndexType srcIdx)
480 requires(!ReadOnly)
481 {
482 const auto& dynamicKeys = src.dynamicKeys_impl();
483 for (const auto key : dynamicKeys) {
484 std::any srcPtr = src.component_impl(key, srcIdx);
485 container().copyDynamicFrom_impl(dstIdx, key, srcPtr);
486 }
487 }
