30template <typename G>
31struct MaterialMapLookup {
32 public:
34 using Grid_t = G;
36 static constexpr std::size_t DIM_POS = Grid_t::DIM;
37
42 struct MaterialCell {
44 static constexpr unsigned int N = 1 << DIM_POS;
45
58 MaterialCell(
59 std::function<ActsVector<DIM_POS>(const Vector3&)> transformPos,
60 std::array<double, DIM_POS> lowerLeft,
61 std::array<double, DIM_POS> upperRight,
62 std::array<Material::ParametersVector, N> materialValues)
63 : m_transformPos(std::move(transformPos)),
64 m_lowerLeft(std::move(lowerLeft)),
65 m_upperRight(std::move(upperRight)),
66 m_materialValues(std::move(materialValues)) {}
67
74 Material getMaterial(const Vector3& position) const {
75 // defined in Interpolation.hpp
76 return Material(interpolate(m_transformPos(position), m_lowerLeft,
77 m_upperRight, m_materialValues));
78 }
79
85 bool isInside(const Vector3& position) const {
86 const auto& gridCoordinates = m_transformPos(position);
87 for (unsigned int i = 0; i < DIM_POS; ++i) {
88 if (gridCoordinates[i] < m_lowerLeft.at(i) ||
89 gridCoordinates[i] >= m_upperRight.at(i)) {
90 return false;
91 }
92 }
93 return true;
94 }
95
96 private:
98 std::function<ActsVector<DIM_POS>(const Vector3&)> m_transformPos;
99
101 std::array<double, DIM_POS> m_lowerLeft;
102
104 std::array<double, DIM_POS> m_upperRight;
105
110 std::array<Material::ParametersVector, N> m_materialValues;
111 };
112
118 MaterialMapLookup(
119 std::function<ActsVector<DIM_POS>(const Vector3&)> transformPos,
120 Grid_t grid)
121 : m_transformPos(std::move(transformPos)), m_grid(std::move(grid)) {}
122
130 Material material(const Vector3& position) const {
131 return Material(m_grid.atLocalBins(
132 m_grid.localBinsFromLowerLeftEdge(m_transformPos(position))));
133 }
134
142 Material getMaterial(const Vector3& position) const {
143 return Material(m_grid.interpolate(m_transformPos(position)));
144 }
145
153 MaterialCell getMaterialCell(const Vector3& position) const {
154 const auto& gridPosition = m_transformPos(position);
155 std::size_t bin = m_grid.globalBinFromPosition(gridPosition);
156 const auto& indices = m_grid.localBinsFromPosition(bin);
157 const auto& lowerLeft = m_grid.lowerLeftBinEdge(indices);
158 const auto& upperRight = m_grid.upperRightBinEdge(indices);
159
160 // Loop through all corner points
161 constexpr std::size_t nCorners = 1 << DIM_POS;
162 std::array<Material::ParametersVector, nCorners> neighbors{};
163 const auto& cornerIndices = m_grid.closestPointsIndices(gridPosition);
164
165 std::size_t i = 0;
166 for (std::size_t index : cornerIndices) {
167 neighbors.at(i++) = m_grid.at(index);
168 }
169
170 return MaterialCell(m_transformPos, lowerLeft, upperRight,
171 std::move(neighbors));
172 }
173
177 std::vector<std::size_t> getNBins() const {
178 auto nBinsArray = m_grid.numLocalBins();
179 return std::vector<std::size_t>(nBinsArray.begin(), nBinsArray.end());
180 }
181
185 std::vector<double> getMin() const {
186 auto minArray = m_grid.minPosition();
187 return std::vector<double>(minArray.begin(), minArray.end());
188 }
189
193 std::vector<double> getMax() const {
194 auto maxArray = m_grid.maxPosition();
195 return std::vector<double>(maxArray.begin(), maxArray.end());
196 }
197
203 bool isInside(const Vector3& position) const {
204 return m_grid.isInside(m_transformPos(position));
205 }
206
210 const Grid_t& getGrid() const { return m_grid; }
211
212 private:
214 std::function<ActsVector<DIM_POS>(const Vector3&)> m_transformPos;
216 Grid_t m_grid;
217};

289 Material getMaterial(const Vector3& position, Cache& cache) const {
290 if (!cache.initialized || !(*cache.matCell).isInside(position)) {

293 }
294 return (*cache.matCell).getMaterial(position);

350 }

352 private:

362 return m_mapper.getMaterialCell(position);
363 }
