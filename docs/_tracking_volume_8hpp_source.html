171 template <SurfaceVisitor visitor_t>
172 void visitSurfaces(visitor_t&& visitor, bool restrictToSensitives) const {
173 auto sensitive = [&visitor](const Surface& surface) {
174 if (surface.geometryId().sensitive() == 0) {
175 return;
176 }
177 visitor(&surface);
178 };
179
180 if (restrictToSensitives) {
181 apply(sensitive);
182 } else {
183 apply(overloaded{
184 sensitive,
185 [&visitor](const Portal& portal) { visitor(&portal.surface()); },
186 [&visitor](const BoundarySurface& bs) {
187 visitor(&bs.surfaceRepresentation());
188 }});
189 }
190 }
191
201 template <SurfaceVisitor visitor_t>
202 void visitSurfaces(visitor_t&& visitor) const {
203 visitSurfaces(std::forward<visitor_t>(visitor), true);
204 }
