96 template <typename callable_t>
97 decltype(auto) visit(const callable_t& callable) const {
98 auto switchOnType =
99 [this, &callable]<AxisBoundaryType bdt>(AxisBoundaryTypeTag<bdt>) {
100 switch (getType()) {
101 using enum AxisType;
102 case Equidistant:
103 return callable(
104 dynamic_cast<const Axis<AxisType::Equidistant, bdt>&>(*this));
105 case Variable:
106 return callable(
107 dynamic_cast<const Axis<AxisType::Variable, bdt>&>(*this));
108 default:
109 throw std::logic_error("Unknown axis type");
110 }
111 };
112
113 switch (getBoundaryType()) {
114 using enum AxisBoundaryType;
115 case Open:
116 return switchOnType(AxisOpen);
117 case Bound:
118 return switchOnType(AxisBound);
119 case Closed:
120 return switchOnType(AxisClosed);
121 default:
122 throw std::logic_error("Unknown axis type");
123 }
124 }
