59template <typename grid_type>
60ProtoGrid convert(const grid_type& grid,
61 const std::array<Acts::AxisDirection, grid_type::DIM>& aDirs,
62 const GridConverter::Options& cOptions) {
63 // The return object
64 ProtoGrid pGrid;
65
66 // Grid axes
67 auto axes = grid.axes();
68
69 // The edge values - these need to follow the ACTSVG convention,
70 // so there could be swapping when necessary
71 std::vector<double> edges0;
72 std::vector<double> edges1;
73
74 // 1D case (more to be filled in later)
75 if constexpr (grid_type::DIM == 1u) {
76 if (aDirs[0u] == Acts::AxisDirection::AxisPhi &&
77 axes[0]->getBoundaryType() == Acts::AxisBoundaryType::Closed) {
78 // swap needed
79 edges1 = axes[0]->getBinEdges();
80 pGrid._type = actsvg::proto::grid::e_r_phi;
81 }
82 if (cOptions.optionalBound.has_value()) {
83 auto [boundRange, boundValue] = cOptions.optionalBound.value();
84 if (boundValue == Acts::AxisDirection::AxisR) {
85 // good - no swap needed
86 edges0 = {boundRange[0u], boundRange[1u]};
87 }
88 }
89 }
90 // 2D cases
91 if constexpr (grid_type::DIM == 2u) {
92 // Assign
93 edges0 = axes[0]->getBinEdges();
94 edges1 = axes[1]->getBinEdges();
95 if (aDirs[0] == Acts::AxisDirection::AxisPhi &&
96 aDirs[1] == Acts::AxisDirection::AxisZ) {
97 // swap needed
98 std::swap(edges0, edges1);
99 pGrid._type = actsvg::proto::grid::e_z_phi;
100 } else if (aDirs[0] == Acts::AxisDirection::AxisPhi &&
101 aDirs[1] == Acts::AxisDirection::AxisR) {
102 // swap needed
103 std::swap(edges0, edges1);
104 pGrid._type = actsvg::proto::grid::e_r_phi;
105 } else if (aDirs[0] == Acts::AxisDirection::AxisZ &&
106 aDirs[1] == Acts::AxisDirection::AxisPhi) {
107 // good - no swap needed
108 pGrid._type = actsvg::proto::grid::e_z_phi;
109 } else if (aDirs[0] == Acts::AxisDirection::AxisR &&
110 aDirs[1] == Acts::AxisDirection::AxisPhi) {
111 // good - no swap needed
112 pGrid._type = actsvg::proto::grid::e_r_phi;
113 } else if (aDirs[0] == Acts::AxisDirection::AxisX &&
114 aDirs[1] == Acts::AxisDirection::AxisY) {
115 // good - no swap needed
116 pGrid._type = actsvg::proto::grid::e_x_y;
117 }
118 }
119
120 // Assign grid edges
121 pGrid._edges_0 = std::vector<actsvg::scalar>(edges0.begin(), edges0.end());
122 pGrid._edges_1 = std::vector<actsvg::scalar>(edges1.begin(), edges1.end());
123
124 auto [fill, stroke] = cOptions.style.fillAndStroke();
125 pGrid._fill = fill;
126 pGrid._stroke = stroke;
127
128 return pGrid;
129}
