64template <Acts::detail::isBoundOrFreeTrackParams parameters_t>
65struct adl_serializer<parameters_t> {
67 using CovarianceMatrix = typename parameters_t::CovarianceMatrix;
68
73 static void to_json(nlohmann::json& j, const parameters_t& t) {
74 // Serialize parameters
75 // common to all track parameters
76 j["direction"] = t.direction();
77 j["qOverP"] = t.qOverP();
78 j["particleHypothesis"] = t.particleHypothesis().absolutePdg();
79
80 // Covariance is optional
81 j["covariance"];
82 if (t.covariance().has_value()) {
83 // Extract covariance matrix
84 // parameters and serialize
85 auto cov = t.covariance().value();
86 constexpr unsigned int size = cov.rows();
87 std::array<double, size * size> covData{};
88 for (std::size_t n = 0; n < size; ++n) {
89 for (std::size_t m = 0; m < size; ++m) {
90 covData[n * size + m] = cov(n, m);
91 }
92 }
93 j["covariance"] = covData;
94 }
95 // Bound track parameters have
96 // reference surface attached
97 // and position takes a geometry context
98 if constexpr (Acts::detail::isGenericBoundTrackParams<parameters_t>) {
99 Acts::GeometryContext gctx;
100 j["position"] = t.fourPosition(gctx);
101
102 j["referenceSurface"] =
103 Acts::SurfaceJsonConverter::toJson(gctx, t.referenceSurface());
104 } else {
105 j["position"] = t.fourPosition();
106 }
107 }
108
113 static parameters_t from_json(const nlohmann::json& j) {
114 // Extract common parameters
115 std::array<double, 4> posData = j.at("position");
116 Acts::Vector4 position(posData[0], posData[1], posData[2], posData[3]);
117
118 std::array<double, 3> dirData = j.at("direction");
119 Acts::Vector3 direction(dirData[0], dirData[1], dirData[2]);
120
121 double qOverP = j.at("qOverP");
122 Acts::PdgParticle absPdg = j.at("particleHypothesis");
123
124 // Covariance is optional
125 std::optional<CovarianceMatrix> cov;
126 if (j.at("covariance").is_null()) {
127 cov = std::nullopt;
128 } else {
129 // Extract covariance matrix
130 // parameters and deserialize
131 CovarianceMatrix mat;
132 constexpr unsigned int size = mat.rows();
133 std::array<double, size * size> covData = j.at("covariance");
134 for (std::size_t n = 0; n < size; ++n) {
135 for (std::size_t m = 0; m < size; ++m) {
136 mat(n, m) = covData[n * size + m];
137 }
138 }
139 cov.emplace(std::move(mat));
140 }
141
142 // Create particle hypothesis
143 typename parameters_t::ParticleHypothesis particle(absPdg);
144
145 // Bound track parameters have
146 // reference surface attached
147 // and constructor is hidden
148 // behind a factory method
149 if constexpr (Acts::detail::isGenericBoundTrackParams<parameters_t>) {
150 Acts::GeometryContext gctx;
151 auto referenceSurface =
152 Acts::SurfaceJsonConverter::fromJson(j.at("referenceSurface"));
153
154 auto res = parameters_t::create(gctx, referenceSurface, position,
155 direction, qOverP, cov, particle);
156
157 if (!res.ok()) {
158 throw std::invalid_argument("Invalid bound track parameters");
159 }
160 return res.value();
161 } else {
162 return parameters_t(position, direction, qOverP, cov, particle);
163 }
164 }
165};

176struct adl_serializer<std::shared_ptr<parameters_t>> {

178 static void to_json(nlohmann::json& j,
