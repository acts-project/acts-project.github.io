106 static double toRadian(double deg) {
107 double d = Acts::detail::wrap_periodic(deg, -180.0, 360.0);
108
109 // If the angle was mapped to 0 from a non-0 multiple of 360, set it to 360
110 if (constexpr double eps = 1e-6; std::abs(d) < eps && std::abs(deg) > eps) {
111 d = 360.;
112 }
113
114 // Convert to rads
115 return d * Acts::UnitConstants::degree;
116 }
