74static constexpr PdgParticle makeNucleusGroundState(PdgParticle pdg) {
75 if (!isNucleus(pdg)) {
76 throw std::invalid_argument("PDG must represent a nucleus");
77 }
78 // set isomer level to zero
79 const auto value = static_cast<std::int32_t>(pdg);
80 return static_cast<PdgParticle>((value / 10) * 10);
81}

88static constexpr std::pair<std::int32_t, std::int32_t> extractNucleusZandA(
89 PdgParticle pdg) {
90 if (!isNucleus(pdg)) {
91 throw std::invalid_argument("PDG must represent a nucleus");
92 }
93 const auto value = static_cast<std::int32_t>(pdg);
94 // proton number respects the charge
95 int Z = (value / 10000) % 1000;
96 // atomic number is always positive
97 int A = std::abs((value / 10) % 1000);
98 return std::make_pair(Z, A);
99}
