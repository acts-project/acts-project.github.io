83 template <bool EnsureConst, typename T>
84 static std::any component_impl(T& instance, Acts::HashedString key,
85 IndexType itrack) {
86 using namespace Acts::HashedStringLiteral;
87 if constexpr (EnsureConst) {
88 static_assert(std::is_const_v<std::remove_reference_t<T>>,
89 "Is not const");
90 }
91
92 using namespace Acts::HashedStringLiteral;
93 auto track = instance.m_collection->at(itrack);
94 std::conditional_t<EnsureConst, const ActsPodioEdm::TrackInfo*,
95 ActsPodioEdm::TrackInfo*>
96 dataPtr;
97 if constexpr (EnsureConst) {
98 dataPtr = &track.getData();
99 } else {
100 dataPtr = &PodioUtil::getDataMutable(track);
101 }
102 auto& data = *dataPtr;
103 switch (key) {
104 case "tipIndex"_hash:
105 return &data.tipIndex;
106 case "stemIndex"_hash:
107 return &data.stemIndex;
108 case "params"_hash:
109 return data.parameters.data();
110 case "cov"_hash:
111 return data.covariance.data();
112 case "nMeasurements"_hash:
113 return &data.nMeasurements;
114 case "nHoles"_hash:
115 return &data.nHoles;
116 case "chi2"_hash:
117 return &data.chi2;
118 case "ndf"_hash:
119 return &data.ndf;
120 case "nOutliers"_hash:
121 return &data.nOutliers;
122 case "nSharedHits"_hash:
123 return &data.nSharedHits;
124 default:
125 auto it = instance.m_dynamic.find(key);
126 if (it == instance.m_dynamic.end()) {
127 throw std::runtime_error("Unable to handle this component");
128 }
129
130 std::conditional_t<EnsureConst,
131 const podio_detail::ConstDynamicColumnBase*,
132 podio_detail::DynamicColumnBase*>
133 col = it->second.get();
134 assert(col && "Dynamic column is null");
135 return col->get(itrack);
136 }
137 }
138
139 template <typename T>
140 static auto dynamicKeys_impl(T& instance) {
141 using column_type =
142 typename decltype(instance.m_dynamic)::mapped_type::element_type;
143 return Acts::detail::DynamicKeyRange<column_type>{
144 instance.m_dynamic.begin(), instance.m_dynamic.end()};
145 }
146
147 template <typename T>
148 static Acts::ParticleHypothesis particleHypothesis_impl(T& instance,
149 IndexType itrack) {
150 auto track = instance.m_collection->at(itrack);
151 const auto& src = track.getParticleHypothesis();
152 return Acts::ParticleHypothesis{static_cast<Acts::PdgParticle>(src.absPdg),
153 src.mass, Acts::AnyCharge{src.absQ}};
154 }
155
156 static void populateSurfaceBuffer(
157 const PodioUtil::ConversionHelper& helper,
158 const ActsPodioEdm::TrackCollection& collection,
159 std::vector<std::shared_ptr<const Acts::Surface>>& surfaces) noexcept {
160 surfaces.reserve(collection.size());
161 for (ActsPodioEdm::Track track : collection) {
162 surfaces.push_back(PodioUtil::convertSurfaceFromPodio(
163 helper, track.getReferenceSurface()));
164 }
165 }
166
167 std::reference_wrapper<const PodioUtil::ConversionHelper> m_helper;
168 std::vector<std::shared_ptr<const Acts::Surface>> m_surfaces;
169};

298 void releaseInto(podio::Frame& frame, const std::string& suffix = "") {
299 std::string s = suffix;

341class ConstPodioTrackContainer : public PodioTrackContainerBase {
342 public:
343 ConstPodioTrackContainer(const PodioUtil::ConversionHelper& helper,
344 const ActsPodioEdm::TrackCollection& collection)
345 : PodioTrackContainerBase{helper}, m_collection{&collection} {
346 // Not much we can do to recover dynamic columns here
347 populateSurfaceBuffer(m_helper, *m_collection, m_surfaces);
348 }
349
350 ConstPodioTrackContainer(const PodioUtil::ConversionHelper& helper,
351 const podio::Frame& frame,
352 const std::string& suffix = "")
353 : PodioTrackContainerBase{helper} {
354 std::string s = suffix.empty() ? suffix : "_" + suffix;
355 std::string tracksKey = "tracks" + s;
356
357 std::vector<std::string> available = frame.getAvailableCollections();
358 if (!Acts::rangeContainsValue(available, tracksKey)) {
359 throw std::runtime_error{"Track collection '" + tracksKey +
360 "' not found in frame"};
361 }
362
363 const auto* collection = frame.get(tracksKey);
364
365 if (const auto* d =
366 dynamic_cast<const ActsPodioEdm::TrackCollection*>(collection);
367 d != nullptr) {
368 m_collection = d;
369 } else {
370 throw std::runtime_error{"Unable to get collection " + tracksKey};
371 }
372
373 populateSurfaceBuffer(m_helper, *m_collection, m_surfaces);
374
375 podio_detail::recoverDynamicColumns(frame, tracksKey, m_dynamic);
376 }
