62 template <AxisConcept axis_t>
63 static std::unique_ptr<GridPortalLinkT<axis_t>> make(
64 std::shared_ptr<RegularSurface> surface, AxisDirection direction,
65 axis_t&& axis) {
66 using enum AxisDirection;
67 if (dynamic_cast<const CylinderSurface*>(surface.get()) != nullptr) {
68 if (direction != AxisZ && direction != AxisRPhi) {
69 throw std::invalid_argument{"Invalid binning direction"};
70 }
71 } else if (dynamic_cast<const DiscSurface*>(surface.get()) != nullptr &&
72 direction != AxisR && direction != AxisPhi) {
73 throw std::invalid_argument{"Invalid binning direction"};
74 } else if (dynamic_cast<const PlaneSurface*>(surface.get()) != nullptr &&
75 direction != AxisX && direction != AxisY) {
76 throw std::invalid_argument{"Invalid binning direction"};
77 }
78
79 return std::make_unique<GridPortalLinkT<axis_t>>(
80 surface, direction, std::forward<axis_t>(axis));
81 }

94 std::shared_ptr<RegularSurface> surface, axis_1_t axis1, axis_2_t axis2) {
95 std::optional<AxisDirection> direction;
96 if (dynamic_cast<const CylinderSurface*>(surface.get()) != nullptr) {
97 direction = AxisDirection::AxisRPhi;

99 direction = AxisDirection::AxisR;
100 } else if (dynamic_cast<const PlaneSurface*>(surface.get()) != nullptr) {
101 direction = AxisDirection::AxisX;
102 }
103
104 return std::make_unique<GridPortalLinkT<axis_1_t, axis_2_t>>(
105 surface, direction.value(), std::move(axis1), std::move(axis2));

429template <typename... Axes>
430 requires(sizeof...(Axes) <= 2)
431class GridPortalLinkT : public GridPortalLink {
432 public:
434 using GridType = Grid<const TrackingVolume*, Axes...>;
435
437 static constexpr std::size_t DIM = sizeof...(Axes);
438
444 GridPortalLinkT(std::shared_ptr<RegularSurface> surface,
445 AxisDirection direction, Axes&&... axes)
446 : GridPortalLink(std::move(surface), direction),
447 m_grid(std::tuple{std::move(axes)...}) {
448 using enum AxisDirection;
449
450 if (const auto* cylinder =
451 dynamic_cast<const CylinderSurface*>(m_surface.get())) {
452 checkConsistency(*cylinder);
453
454 if (direction == AxisRPhi) {
455 m_projection = &projection<CylinderSurface, AxisRPhi>;
456 } else if (direction == AxisZ) {
457 m_projection = &projection<CylinderSurface, AxisZ>;
458 } else {
459 throw std::invalid_argument{"Invalid binning direction"};
460 }
461
462 } else if (const auto* disc =
463 dynamic_cast<const DiscSurface*>(m_surface.get())) {
464 checkConsistency(*disc);
465
466 if (direction == AxisR) {
467 m_projection = &projection<DiscSurface, AxisR>;
468 } else if (direction == AxisDirection::AxisPhi) {
469 m_projection = &projection<DiscSurface, AxisPhi>;
470 } else {
471 throw std::invalid_argument{"Invalid binning direction"};
472 }
473 } else if (const auto* plane =
474 dynamic_cast<const PlaneSurface*>(m_surface.get())) {
475 checkConsistency(*plane);
476
477 if (direction == AxisX) {
478 m_projection = &projection<PlaneSurface, AxisX>;
479 } else if (direction == AxisDirection::AxisY) {
480 m_projection = &projection<PlaneSurface, AxisY>;
481 } else {
482 throw std::invalid_argument{"Invalid binning direction"};
483 }
484
485 } else {
486 throw std::logic_error{"Surface type is not supported"};
487 }
488 }
489
492 const GridType& grid() const override { return m_grid; }
493
496 GridType& grid() override { return m_grid; }
497
500 unsigned int dim() const override { return DIM; }
501
504 void toStream(std::ostream& os) const override {
505 os << "GridPortalLink<dim=" << dim() << ">";
506 }

512 std::unique_ptr<GridPortalLink> extendTo2d(
513 const IAxis* other) const override {
514 if constexpr (DIM == 2) {
515 return std::make_unique<GridPortalLinkT<Axes...>>(*this);
516 } else {
517 if (auto cylinder =
518 std::dynamic_pointer_cast<CylinderSurface>(m_surface)) {
519 return extendTo2dImpl(cylinder, other);
520 } else if (auto disc =
521 std::dynamic_pointer_cast<DiscSurface>(m_surface)) {
522 return extendTo2dImpl(disc, other);
523 } else if (auto plane =
524 std::dynamic_pointer_cast<PlaneSurface>(m_surface)) {
525 return extendTo2dImpl(plane, other);
526 } else {
527 throw std::logic_error{
528 "Surface type is not supported (this should not happen)"};
529 }
530 }
531 }

535 void setVolume(TrackingVolume* volume) override {
536 auto loc = m_grid.numLocalBins();
537 if constexpr (GridType::DIM == 1) {
538 for (std::size_t i = 1; i <= loc[0]; i++) {
539 m_grid.atLocalBins({i}) = volume;
540 }
541 } else {
542 for (std::size_t i = 1; i <= loc[0]; i++) {
543 for (std::size_t j = 1; j <= loc[1]; j++) {
544 m_grid.atLocalBins({i, j}) = volume;
545 }
546 }
547 }

556 Result<const TrackingVolume*> resolveVolume(
557 const GeometryContext& gctx, const Vector3& position,
558 double tolerance = s_onSurfaceTolerance) const override {
559 auto res = m_surface->globalToLocal(gctx, position, tolerance);
560 if (!res.ok()) {
561 return res.error();
562 }
563
564 const Vector2& local = *res;
565 return resolveVolume(gctx, local, tolerance);

572 Result<const TrackingVolume*> resolveVolume(
573 const GeometryContext& /*gctx*/, const Vector2& position,
574 double /*tolerance*/ = s_onSurfaceTolerance) const override {
575 throw_assert(surface().insideBounds(position, BoundaryTolerance::None()),
576 "Checking volume outside of bounds");
577 return m_grid.atPosition(m_projection(position));
578 }

580 private:
583 template <class surface_t, AxisDirection direction>
584 static ActsVector<DIM> projection(const Vector2& position) {
585 using enum AxisDirection;
586 if constexpr (DIM == 2) {
587 return position;
588 } else {
589 if constexpr (std::is_same_v<surface_t, CylinderSurface>) {
590 static_assert(direction == AxisRPhi || direction == AxisZ,
591 "Invalid binning direction");
592
593 if constexpr (direction == AxisRPhi) {
594 return ActsVector<1>{position[0]};
595 } else if constexpr (direction == AxisZ) {
596 return ActsVector<1>{position[1]};
597 }
598 } else if constexpr (std::is_same_v<surface_t, DiscSurface>) {
599 static_assert(direction == AxisR || direction == AxisPhi,
600 "Invalid binning direction");
601
602 if constexpr (direction == AxisR) {
603 return ActsVector<1>{position[0]};
604 } else if constexpr (direction == AxisPhi) {
605 return ActsVector<1>{position[1]};
606 }
607 } else if constexpr (std::is_same_v<surface_t, PlaneSurface>) {
608 static_assert(direction == AxisX || direction == AxisY,
609 "Invalid binning direction");
610
611 if constexpr (direction == AxisX) {
612 return ActsVector<1>{position[0]};
613 } else if constexpr (direction == AxisY) {
614 return ActsVector<1>{position[1]};
615 }
616 }
617 }
618 }

621
623 ActsVector<DIM> (*m_projection)(const Vector2& position);
