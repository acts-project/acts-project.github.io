23inline double roundWithPrecision(double val, int precision) {
24 if (val < 0 && std::abs(val) * std::pow(10, precision) < 1.) {
25 return -val;
26 }
27 return val;
28}

45template <typename derived_t>
46inline std::string toString(const Eigen::MatrixBase<derived_t>& matrix,
47 int precision = 4, const std::string& offset = "") {
48 std::ostringstream sout;
49
50 sout << std::setiosflags(std::ios::fixed) << std::setprecision(precision);
51 if (matrix.cols() == 1) {
52 sout << "(";
53 for (int i = 0; i < matrix.rows(); ++i) {
54 double val = detail::roundWithPrecision(matrix(i, 0), precision);
55 sout << val;
56 if (i != matrix.rows() - 1) {
57 sout << ", ";
58 }
59 }
60 sout << ")";
61 } else {
62 for (int i = 0; i < matrix.rows(); ++i) {
63 for (int j = 0; j < matrix.cols(); ++j) {
64 if (j == 0) {
65 sout << "(";
66 }
67 double val = detail::roundWithPrecision(matrix(i, j), precision);
68 sout << val;
69 if (j == matrix.cols() - 1) {
70 sout << ")";
71 } else {
72 sout << ", ";
73 }
74 }
75 if (i != matrix.rows() -
76 1) { // make the end line and the offset in the next line
77 sout << std::endl;
78 sout << offset;
79 }
80 }
81 }
82 return sout.str();
83}

103inline std::string toString(const Acts::Transform3& transform,
104 int precision = 4, const std::string& offset = "") {
105 std::ostringstream sout;
106 sout << "Translation : " << toString(transform.translation(), precision)
107 << std::endl;
108 std::string rotationOffset = offset + " ";
109 sout << offset << "Rotation : "
110 << toString(transform.rotation(), precision + 2, rotationOffset);
111 return sout.str();
112}

118inline std::string toString(const std::vector<double>& pVector,
119 int precision = 4) {
120 std::ostringstream sout;
121 sout << std::setiosflags(std::ios::fixed) << std::setprecision(precision);
122 sout << "(";
123 for (const auto [i, val] : enumerate(pVector)) {
124 sout << val;
125 if (i != pVector.size() - 1) {
126 sout << ", ";
127 }
128 }
129 sout << ")";
130 return sout.str();
131}
