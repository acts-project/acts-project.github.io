44 template <Acts::TrackContainerFrontend track_container_t>
45 std::vector<std::vector<float>> inferScores(
46 std::unordered_map<std::size_t, std::vector<std::size_t>>& clusters,
47 const track_container_t& tracks) const {
48 // Compute the number of entry (since it is smaller than the number of
49 // track)
50 int trackNb = 0;
51 for (const auto& [_, val] : clusters) {
52 trackNb += val.size();
53 }
54 // Input of the neural network
55 NetworkBatchInput networkInput(trackNb, 8);
56 std::size_t inputID = 0;
57 // Get the input feature of the network for all the tracks
58 for (const auto& [key, val] : clusters) {
59 for (const auto& trackID : val) {
60 auto track = tracks.getTrack(trackID);
61 auto trajState = Acts::MultiTrajectoryHelpers::trajectoryState(
62 tracks.trackStateContainer(), track.tipIndex());
63 networkInput(inputID, 0) = trajState.nStates;
64 networkInput(inputID, 1) = trajState.nMeasurements;
65 networkInput(inputID, 2) = trajState.nOutliers;
66 networkInput(inputID, 3) = trajState.nHoles;
67 networkInput(inputID, 4) = trajState.NDF;
68 networkInput(inputID, 5) = (trajState.chi2Sum * 1.0) /
69 (trajState.NDF != 0 ? trajState.NDF : 1);
70 networkInput(inputID, 6) = Acts::VectorHelpers::eta(track.momentum());
71 networkInput(inputID, 7) = Acts::VectorHelpers::phi(track.momentum());
72 inputID++;
73 }
74 }
75 // Use the network to compute a score for all the tracks.
76 std::vector<std::vector<float>> outputTensor =
77 m_duplicateClassifier.runONNXInference(networkInput);
78 return outputTensor;
79 }

86 std::vector<std::size_t> trackSelection(
87 std::unordered_map<std::size_t, std::vector<std::size_t>>& clusters,
88 std::vector<std::vector<float>>& outputTensor) const {
89 std::vector<std::size_t> goodTracks;
90 std::size_t iOut = 0;
91 // Loop over all the cluster and only keep the track with the highest score
92 // in each cluster
93 for (const auto& [key, val] : clusters) {
94 std::size_t bestTrackID = 0;
95 float bestTrackScore = 0;
96 for (const auto& track : val) {
97 if (outputTensor[iOut][0] > bestTrackScore) {
98 bestTrackScore = outputTensor[iOut][0];
99 bestTrackID = track;
100 }
101 iOut++;
102 }
103 goodTracks.push_back(bestTrackID);
104 }
105 return goodTracks;
106 }
