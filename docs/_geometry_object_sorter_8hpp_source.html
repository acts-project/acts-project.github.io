79template <class T>
80class DistanceSorterT {
81 public:
86 DistanceSorterT(AxisDirection aDir, Vector3 reference)
87 : m_sortingDirection(aDir),
88 m_reference(reference),
89 m_refR(VectorHelpers::perp(reference)),
90 m_refPhi(VectorHelpers::phi(reference)),
91 m_refEta(VectorHelpers::eta(reference)) {}
92
98 bool operator()(T one, T two) const {
99 using Acts::VectorHelpers::eta;
100 using Acts::VectorHelpers::perp;
101 using Acts::VectorHelpers::phi;
102 // switch the sorting value
103 // - AxisX, AxisY, AxisZ, AxisR, AxisPhi, AxisRPhi, AxisTheta, AxisEta
104 switch (m_sortingDirection) {
105 // compare on diff x
106 case AxisDirection::AxisX: {
107 double diffOneX = one.x() - m_reference.x();
108 double diffTwoX = two.x() - m_reference.x();
109 return std::abs(diffOneX) < std::abs(diffTwoX);
110 }
111 // compare on diff y
112 case AxisDirection::AxisY: {
113 double diffOneY = one.y() - m_reference.y();
114 double diffTwoY = two.y() - m_reference.y();
115 return std::abs(diffOneY) < std::abs(diffTwoY);
116 }
117 // compare on diff z
118 case AxisDirection::AxisZ: {
119 double diffOneZ = one.z() - m_reference.z();
120 double diffTwoZ = two.z() - m_reference.z();
121 return std::abs(diffOneZ) < std::abs(diffTwoZ);
122 }
123 // compare on r
124 case AxisDirection::AxisR: {
125 double diffOneR = perp(one) - m_refR;
126 double diffTwoR = perp(two) - m_refR;
127 return std::abs(diffOneR) < std::abs(diffTwoR);
128 }
129 // compare on phi /// @todo add cyclic value
130 case AxisDirection::AxisPhi: {
131 double diffOnePhi = phi(one) - m_refPhi;
132 double diffTwoPhi = phi(two) - m_refPhi;
133 return std::abs(diffOnePhi) < std::abs(diffTwoPhi);
134 }
135 // compare on eta
136 case AxisDirection::AxisEta: {
137 double diffOneEta = eta(one) - m_refEta;
138 double diffTwoEta = eta(two) - m_refEta;
139 return std::abs(diffOneEta) < std::abs(diffTwoEta);
140 }
141 // default for the moment
142 default: {
143 T diffOne(one - m_reference);
144 T diffTwo(two - m_reference);
145 return diffOne.mag2() < diffTwo.mag2();
146 }
147 }
148 }
149
150 private:
151 AxisDirection m_sortingDirection;
152 T m_reference;
153 double m_refR;
154 double m_refPhi;
155 double m_refEta;
156};

158template <class T>
159class GeometryObjectSorterT {
160 public:
166 GeometryObjectSorterT(const GeometryContext& gctx, AxisDirection aDir,
167 std::shared_ptr<const Transform3> transform = nullptr)
168 : m_context(gctx),
169 m_objectSorter(aDir),
170 m_transform(std::move(transform)) {}
171
178 bool operator()(T one, T two) const {
179 // get the pos one / pos two
180 Vector3 posOne = m_transform
181 ? m_transform->inverse() *
182 one->referencePosition(
183 m_context, m_objectSorter.sortingDirection())
184 : one->referencePosition(
185 m_context, m_objectSorter.sortingDirection());
186 Vector3 posTwo = m_transform
187 ? m_transform->inverse() *
188 two->referencePosition(
189 m_context, m_objectSorter.sortingDirection())
190 : two->referencePosition(
191 m_context, m_objectSorter.sortingDirection());
192 // now call the distance sorter
193 return m_objectSorter.operator()(posOne, posTwo);
194 }
195
196 protected:
197 std::reference_wrapper<const GeometryContext> m_context;
198 ObjectSorterT<Vector3> m_objectSorter;
199 std::shared_ptr<const Transform3> m_transform;
200};
