192 [[nodiscard]] Result<void> initialize(State& state, const Vector3& position,
193 const Vector3& direction,
194 Direction propagationDirection) const {
195 static_cast<void>(propagationDirection);
196
197 ACTS_VERBOSE("initialize");
198
199 state.startSurface = state.options.startSurface;
200 state.targetSurface = state.options.targetSurface;
201
202 const TrackingVolume* startVolume = nullptr;
203
204 if (state.startSurface != nullptr &&
205 state.startSurface->associatedLayer() != nullptr) {
206 ACTS_VERBOSE(
207 "Fast start initialization through association from Surface.");
208 const auto* startLayer = state.startSurface->associatedLayer();
209 startVolume = startLayer->trackingVolume();
210 } else {
211 ACTS_VERBOSE("Slow start initialization through search.");
212 ACTS_VERBOSE("Starting from position " << toString(position)
213 << " and direction "
214 << toString(direction));
215 startVolume = m_cfg.trackingGeometry->lowestTrackingVolume(
216 state.options.geoContext, position);
217 }
218
219 // Initialize current volume, layer and surface
220 {
221 state.currentVolume = startVolume;
222 if (state.currentVolume != nullptr) {
223 ACTS_VERBOSE(volInfo(state) << "Start volume resolved.");
224 } else {
225 ACTS_DEBUG("Start volume not resolved.");
226 state.navigationBreak = true;
227 return NavigatorError::NoStartVolume;
228 }
229
230 state.currentSurface = state.startSurface;
231 if (state.currentSurface != nullptr) {
232 ACTS_VERBOSE(volInfo(state) << "Current surface set to start surface "
233 << state.currentSurface->geometryId());
234 } else {
235 ACTS_VERBOSE(volInfo(state) << "No start surface set.");
236 }
237 }
238
239 return Result<void>::success();
240 }
241
242 protected:
245 void initializeVolumeCandidates(State& state) const {
246 const TrackingVolume* volume = state.currentVolume;

257 m_cfg.resolveMaterial, m_cfg.resolvePassive,
258 m_cfg.boundaryToleranceSurfaceApproach, logger());
259 }

267 " | ";
268 }

369 NavigationTarget nextTarget(State& state, const Vector3& position,
370 const Vector3& direction) const {
371 // Navigator preStep always resets the current surface
372 state.currentSurface = nullptr;
373
374 // Check if the navigator is inactive
375 if (state.navigationBreak) {
376 return NavigationTarget::None();
377 }
378
379 ACTS_VERBOSE(volInfo(state) << "nextTarget");
380
381 double nearLimit = state.options.nearLimit;
382 double farLimit = state.options.farLimit;
383
384 // handle overstepping
385 if (!state.currentTargets.empty()) {
386 const NavigationTarget& previousTarget = state.currentTargets.front();
387
388 const Surface& surface = previousTarget.surface();
389 IntersectionIndex index = previousTarget.intersectionIndex();
390 BoundaryTolerance boundaryTolerance = previousTarget.boundaryTolerance();
391
392 auto intersection =
393 surface
394 .intersect(state.options.geoContext, position, direction,
395 boundaryTolerance, state.options.surfaceTolerance)
396 .at(index);
397
398 if (intersection.pathLength() < 0) {
399 nearLimit = std::min(nearLimit, intersection.pathLength() -
400 state.options.surfaceTolerance);
401 farLimit = -state.options.surfaceTolerance;
402
403 ACTS_VERBOSE(volInfo(state)
404 << "handle overstepping with nearLimit " << nearLimit
405 << " and farLimit " << farLimit);
406 }
407 }
408
409 std::vector<NavigationTarget> nextTargets;
410
411 // Find intersections with all candidates
412 for (const auto& candidate : state.navigationCandidates) {
413 auto intersections =
414 candidate.intersect(state.options.geoContext, position, direction,
415 state.options.surfaceTolerance);
416 for (auto [intersectionIndex, intersection] :
417 Acts::enumerate(intersections)) {
418 // exclude invalid intersections
419 if (!intersection.isValid() ||
420 !detail::checkPathLength(intersection.pathLength(), nearLimit,
421 farLimit)) {
422 continue;
423 }
424 // store candidate
425 nextTargets.emplace_back(
426 candidate.target(intersection, intersectionIndex));
427 }
428 }
429
430 std::ranges::sort(nextTargets, NavigationTarget::pathLengthOrder);
431
432 ACTS_VERBOSE(volInfo(state)
433 << "found " << nextTargets.size() << " intersections");
434
435 NavigationTarget nextTarget = NavigationTarget::None();
436 state.currentTargets.clear();
437
438 for (const auto& target : nextTargets) {
439 const Intersection3D& intersection = target.intersection();
440
441 if (intersection.status() == IntersectionStatus::onSurface) {
442 ACTS_ERROR(volInfo(state)
443 << "We are on surface " << target.surface().geometryId()
444 << " before trying to reach it. This should not happen. "
445 "Good luck.");
446 continue;
447 }
448
449 if (intersection.status() == IntersectionStatus::reachable) {
450 nextTarget = target;
451 break;
452 }
453 }
454
455 state.currentTargets = std::move(nextTargets);
456
457 if (nextTarget.isNone()) {
458 ACTS_VERBOSE(volInfo(state) << "no target found");
459 } else {
460 ACTS_VERBOSE(volInfo(state)
461 << "next target is " << nextTarget.surface().geometryId());
462 }
463
464 return nextTarget;
465 }

497 void handleSurfaceReached(State& state, const Vector3& position,
498 const Vector3& direction,
499 const Surface& /*surface*/) const {
500 // Check if the navigator is inactive
501 if (state.navigationBreak) {
502 return;
503 }
504
505 ACTS_VERBOSE(volInfo(state) << "handleSurfaceReached");
506
507 if (state.currentTargets.empty()) {
508 ACTS_VERBOSE(volInfo(state) << "No current target set.");
509 return;
510 }
511
512 assert(state.currentSurface == nullptr && "Current surface must be reset.");
513
514 // handle multiple surface intersections due to increased bounds
515
516 std::vector<NavigationTarget> hitTargets;
517
518 for (const auto& target : state.currentTargets) {
519 std::uint8_t index = target.intersectionIndex();
520 const Surface& surface = target.surface();
521 BoundaryTolerance boundaryTolerance = BoundaryTolerance::None();
522
523 Intersection3D intersection =
524 surface
525 .intersect(state.options.geoContext, position, direction,
526 boundaryTolerance, state.options.surfaceTolerance)
527 .at(index);
528
529 if (intersection.status() == IntersectionStatus::onSurface) {
530 hitTargets.emplace_back(target);
531 }
532 }
533
534 state.currentTargets.clear();
535
536 ACTS_VERBOSE(volInfo(state)
537 << "Found " << hitTargets.size()
538 << " intersections on surface with bounds check.");
539
540 if (hitTargets.empty()) {
541 ACTS_VERBOSE(volInfo(state) << "No hit targets found.");
542 return;
543 }
544
545 if (hitTargets.size() > 1) {
546 ACTS_VERBOSE(volInfo(state)
547 << "Only using first intersection within bounds.");
548 }
549
550 // we can only handle a single surface hit so we pick the first one
551 const auto target = hitTargets.front();
552 const Surface& surface = target.surface();
553
554 ACTS_VERBOSE(volInfo(state) << "Surface " << surface.geometryId()
555 << " successfully hit, storing it.");
556 state.currentSurface = &surface;
557
558 if (target.isSurfaceTarget()) {
559 ACTS_VERBOSE(volInfo(state) << "This is a surface");
560 } else if (target.isLayerTarget()) {
561 ACTS_VERBOSE(volInfo(state) << "This is a layer");
562 } else if (target.isPortalTarget()) {
563 ACTS_VERBOSE(volInfo(state)
564 << "This is a boundary. Reinitialize navigation");
565
566 const BoundarySurface& boundary = target.boundarySurface();
567
568 state.currentVolume = boundary.attachedVolume(state.options.geoContext,
569 position, direction);
570
571 ACTS_VERBOSE(volInfo(state) << "Switched volume");
572
573 reinitializeCandidates(state);
574 } else {
575 ACTS_ERROR(volInfo(state) << "Unknown intersection type");
576 }
577 }
578
579 private:
581 void reinitializeCandidates(State& state) const {
582 state.navigationCandidates.clear();
583 state.currentTargets.clear();
584
585 initializeVolumeCandidates(state);
586 }

705 NavigationTarget nextTarget(State& state, const Vector3& position,
706 const Vector3& direction) const {
707 static_cast<void>(direction);
708
709 // Navigator preStep always resets the current surface
710 state.currentSurface = nullptr;
711
712 // Check if the navigator is inactive
713 if (state.navigationBreak) {
714 return NavigationTarget::None();
715 }
716
717 ACTS_VERBOSE(volInfo(state) << "nextTarget");
718
719 // We cannot do anything without a last position
720 if (!state.lastPosition.has_value() && state.endOfTargets()) {
721 ACTS_VERBOSE(
722 volInfo(state)
723 << "Initial position, nothing to do, blindly stepping forward.");
724 state.lastPosition = position;
725 return NavigationTarget::None();
726 }
727
728 if (state.endOfTargets()) {
729 ACTS_VERBOSE(volInfo(state) << "evaluate blind step");
730
731 Vector3 stepStart = state.lastPosition.value();
732 Vector3 stepEnd = position;
733 Vector3 step = stepEnd - stepStart;
734 double stepDistance = step.norm();
735 if (stepDistance < std::numeric_limits<double>::epsilon()) {
736 ACTS_ERROR(volInfo(state) << "Step distance is zero. " << stepDistance);
737 }
738 Vector3 stepDirection = step.normalized();
739
740 double nearLimit = -stepDistance + state.options.surfaceTolerance;
741 double farLimit = 0;
742
743 state.lastPosition.reset();
744 state.activeTargets.clear();
745 state.activeTargetIndex = -1;
746
747 // Find intersections with all candidates
748 for (const auto& candidate : state.navigationCandidates) {
749 auto intersections =
750 candidate.intersect(state.options.geoContext, stepEnd,
751 stepDirection, state.options.surfaceTolerance);
752 for (auto [intersectionIndex, intersection] :
753 Acts::enumerate(intersections)) {
754 // exclude invalid intersections
755 if (!intersection.isValid() ||
756 !detail::checkPathLength(intersection.pathLength(), nearLimit,
757 farLimit)) {
758 continue;
759 }
760 // store candidate
761 state.activeTargets.emplace_back(
762 candidate.target(intersection, intersectionIndex));
763 }
764 }
765
766 std::ranges::sort(state.activeTargets, NavigationTarget::pathLengthOrder);
767
768 ACTS_VERBOSE(volInfo(state) << "Found " << state.activeTargets.size()
769 << " intersections");
770
771 for (const auto& target : state.activeTargets) {
772 ACTS_VERBOSE("found target " << target.surface().geometryId());
773 }
774 }
775
776 ++state.activeTargetIndex;
777
778 if (state.endOfTargets()) {
779 ACTS_VERBOSE(volInfo(state)
780 << "No target found, blindly stepping forward.");
781 state.lastPosition = position;
782 return NavigationTarget::None();
783 }
784
785 ACTS_VERBOSE(volInfo(state) << "handle active candidates");
786
787 ACTS_VERBOSE(volInfo(state)
788 << (state.activeTargets.size() - state.activeTargetIndex)
789 << " out of " << state.activeTargets.size()
790 << " surfaces remain to try.");
791
792 const auto& target = state.activeTarget();
793
794 ACTS_VERBOSE(volInfo(state) << "Next surface candidate will be "
795 << target.surface().geometryId());
796
797 return target;
798 }

827 void handleSurfaceReached(State& state, const Vector3& position,
828 const Vector3& direction,
829 const Surface& /*surface*/) const {
830 if (state.navigationBreak) {
831 return;
832 }
833
834 ACTS_VERBOSE(volInfo(state) << "handleSurfaceReached");
835
836 assert(state.currentSurface == nullptr && "Current surface must be reset.");
837
838 if (state.endOfTargets()) {
839 ACTS_VERBOSE(volInfo(state) << "No active candidate set.");
840 return;
841 }
842
843 std::vector<NavigationTarget> hitTargets;
844
845 while (!state.endOfTargets()) {
846 const auto& candidate = state.activeTarget();
847 IntersectionIndex index = candidate.intersectionIndex();
848 const Surface& surface = candidate.surface();
849 BoundaryTolerance boundaryTolerance = candidate.boundaryTolerance();
850
851 // first with boundary tolerance
852 IntersectionStatus surfaceStatus =
853 surface
854 .intersect(state.options.geoContext, position, direction,
855 boundaryTolerance, state.options.surfaceTolerance)
856 .at(index)
857 .status();
858
859 if (surfaceStatus != IntersectionStatus::onSurface) {
860 break;
861 }
862
863 // now without boundary tolerance
864 boundaryTolerance = BoundaryTolerance::None();
865 surfaceStatus =
866 surface
867 .intersect(state.options.geoContext, position, direction,
868 boundaryTolerance, state.options.surfaceTolerance)
869 .at(index)
870 .status();
871
872 if (surfaceStatus == IntersectionStatus::onSurface) {
873 hitTargets.emplace_back(candidate);
874 }
875
876 ++state.activeTargetIndex;
877 ACTS_VERBOSE("skip target " << surface.geometryId());
878 }
879
880 // we increased the target index one too many times
881 --state.activeTargetIndex;
882
883 ACTS_VERBOSE(volInfo(state)
884 << "Found " << hitTargets.size()
885 << " intersections on surface with bounds check.");
886
887 if (hitTargets.empty()) {
888 ACTS_VERBOSE(volInfo(state)
889 << "Surface successfully hit, but outside bounds.");
890 return;
891 }
892
893 if (hitTargets.size() > 1) {
894 ACTS_VERBOSE(volInfo(state)
895 << "Only using first intersection within bounds.");
896 }
897
898 // we can only handle a single surface hit so we pick the first one
899 const auto& candidate = hitTargets.front();
900 const Surface& surface = candidate.surface();
901
902 ACTS_VERBOSE(volInfo(state) << "Surface successfully hit, storing it.");
903 state.currentSurface = &surface;
904
905 if (state.currentSurface != nullptr) {
906 ACTS_VERBOSE(volInfo(state) << "Current surface set to surface "
907 << surface.geometryId());
908 }
909
910 if (candidate.isSurfaceTarget()) {
911 ACTS_VERBOSE(volInfo(state) << "This is a surface");
912 } else if (candidate.isLayerTarget()) {
913 ACTS_VERBOSE(volInfo(state) << "This is a layer");
914 } else if (candidate.isPortalTarget()) {
915 ACTS_VERBOSE(volInfo(state)
916 << "This is a portal. Reinitialize navigation");
917
918 const BoundarySurface& boundary = candidate.boundarySurface();
919
920 state.currentVolume = boundary.attachedVolume(state.options.geoContext,
921 position, direction);
922
923 ACTS_VERBOSE(volInfo(state) << "Switched volume");
924
925 reinitializeCandidates(state);
926 } else {
927 ACTS_ERROR(volInfo(state) << "Unknown intersection type");
928 }
929 }
930
931 private:
933 void reinitializeCandidates(State& state) const {
934 state.navigationCandidates.clear();
935 state.activeTargets.clear();
936 state.activeTargetIndex = -1;
937
938 initializeVolumeCandidates(state);
939 }
