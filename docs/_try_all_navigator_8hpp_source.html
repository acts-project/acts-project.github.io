191 [[nodiscard]] Result<void> initialize(State& state, const Vector3& position,
192 const Vector3& direction,
193 Direction propagationDirection) const {
194 (void)propagationDirection;
195
196 ACTS_VERBOSE("initialize");
197
198 state.startSurface = state.options.startSurface;
199 state.targetSurface = state.options.targetSurface;
200
201 const TrackingVolume* startVolume = nullptr;
202
203 if (state.startSurface != nullptr &&
204 state.startSurface->associatedLayer() != nullptr) {
205 ACTS_VERBOSE(
206 "Fast start initialization through association from Surface.");
207 const auto* startLayer = state.startSurface->associatedLayer();
208 startVolume = startLayer->trackingVolume();
209 } else {
210 ACTS_VERBOSE("Slow start initialization through search.");
211 ACTS_VERBOSE("Starting from position " << toString(position)
212 << " and direction "
213 << toString(direction));
214 startVolume = m_cfg.trackingGeometry->lowestTrackingVolume(
215 state.options.geoContext, position);
216 }
217
218 // Initialize current volume, layer and surface
219 {
220 state.currentVolume = startVolume;
221 if (state.currentVolume != nullptr) {
222 ACTS_VERBOSE(volInfo(state) << "Start volume resolved.");
223 } else {
224 ACTS_ERROR("Start volume not resolved.");
225 }
226
227 state.currentSurface = state.startSurface;
228 if (state.currentSurface != nullptr) {
229 ACTS_VERBOSE(volInfo(state) << "Current surface set to start surface "
230 << state.currentSurface->geometryId());
231 } else {
232 ACTS_VERBOSE(volInfo(state) << "No start surface set.");
233 }
234 }
235
236 return Result<void>::success();
237 }

366 NavigationTarget nextTarget(State& state, const Vector3& position,
367 const Vector3& direction) const {
368 // Navigator preStep always resets the current surface
369 state.currentSurface = nullptr;
370
371 // Check if the navigator is inactive
372 if (state.navigationBreak) {
373 return NavigationTarget::None();
374 }
375
376 ACTS_VERBOSE(volInfo(state) << "nextTarget");
377
378 double nearLimit = state.options.nearLimit;
379 double farLimit = state.options.farLimit;
380
381 // handle overstepping
382 if (!state.currentTargets.empty()) {
383 const NavigationTarget& previousTarget = state.currentTargets.front();
384
385 const Surface& surface = previousTarget.surface();
386 IntersectionIndex index = previousTarget.intersectionIndex();
387 BoundaryTolerance boundaryTolerance = previousTarget.boundaryTolerance();
388
389 auto intersection =
390 surface
391 .intersect(state.options.geoContext, position, direction,
392 boundaryTolerance, state.options.surfaceTolerance)
393 .at(index);
394
395 if (intersection.pathLength() < 0) {
396 nearLimit = std::min(nearLimit, intersection.pathLength() -
397 state.options.surfaceTolerance);
398 farLimit = -state.options.surfaceTolerance;
399
400 ACTS_VERBOSE(volInfo(state)
401 << "handle overstepping with nearLimit " << nearLimit
402 << " and farLimit " << farLimit);
403 }
404 }
405
406 std::vector<NavigationTarget> nextTargets;
407
408 // Find intersections with all candidates
409 for (const auto& candidate : state.navigationCandidates) {
410 auto intersections =
411 candidate.intersect(state.options.geoContext, position, direction,
412 state.options.surfaceTolerance);
413 for (auto [intersectionIndex, intersection] :
414 Acts::enumerate(intersections)) {
415 // exclude invalid intersections
416 if (!intersection.isValid() ||
417 !detail::checkPathLength(intersection.pathLength(), nearLimit,
418 farLimit)) {
419 continue;
420 }
421 // store candidate
422 nextTargets.emplace_back(
423 candidate.target(intersection, intersectionIndex));
424 }
425 }
426
427 std::ranges::sort(nextTargets, NavigationTarget::pathLengthOrder);
428
429 ACTS_VERBOSE(volInfo(state)
430 << "found " << nextTargets.size() << " intersections");
431
432 NavigationTarget nextTarget = NavigationTarget::None();
433 state.currentTargets.clear();
434
435 for (const auto& target : nextTargets) {
436 const Intersection3D& intersection = target.intersection();
437
438 if (intersection.status() == IntersectionStatus::onSurface) {
439 ACTS_ERROR(volInfo(state)
440 << "We are on surface " << target.surface().geometryId()
441 << " before trying to reach it. This should not happen. "
442 "Good luck.");
443 continue;
444 }
445
446 if (intersection.status() == IntersectionStatus::reachable) {
447 nextTarget = target;
448 break;
449 }
450 }
451
452 state.currentTargets = std::move(nextTargets);
453
454 if (nextTarget.isNone()) {
455 ACTS_VERBOSE(volInfo(state) << "no target found");
456 } else {
457 ACTS_VERBOSE(volInfo(state)
458 << "next target is " << nextTarget.surface().geometryId());
459 }
460
461 return nextTarget;
462 }

494 void handleSurfaceReached(State& state, const Vector3& position,
495 const Vector3& direction,
496 const Surface& /*surface*/) const {
497 // Check if the navigator is inactive
498 if (state.navigationBreak) {
499 return;
500 }
501
502 ACTS_VERBOSE(volInfo(state) << "handleSurfaceReached");
503
504 if (state.currentTargets.empty()) {
505 ACTS_VERBOSE(volInfo(state) << "No current target set.");
506 return;
507 }
508
509 assert(state.currentSurface == nullptr && "Current surface must be reset.");
510
511 // handle multiple surface intersections due to increased bounds
512
513 std::vector<NavigationTarget> hitTargets;
514
515 for (const auto& target : state.currentTargets) {
516 std::uint8_t index = target.intersectionIndex();
517 const Surface& surface = target.surface();
518 BoundaryTolerance boundaryTolerance = BoundaryTolerance::None();
519
520 Intersection3D intersection =
521 surface
522 .intersect(state.options.geoContext, position, direction,
523 boundaryTolerance, state.options.surfaceTolerance)
524 .at(index);
525
526 if (intersection.status() == IntersectionStatus::onSurface) {
527 hitTargets.emplace_back(target);
528 }
529 }
530
531 state.currentTargets.clear();
532
533 ACTS_VERBOSE(volInfo(state)
534 << "Found " << hitTargets.size()
535 << " intersections on surface with bounds check.");
536
537 if (hitTargets.empty()) {
538 ACTS_VERBOSE(volInfo(state) << "No hit targets found.");
539 return;
540 }
541
542 if (hitTargets.size() > 1) {
543 ACTS_VERBOSE(volInfo(state)
544 << "Only using first intersection within bounds.");
545 }
546
547 // we can only handle a single surface hit so we pick the first one
548 const auto target = hitTargets.front();
549 const Surface& surface = target.surface();
550
551 ACTS_VERBOSE(volInfo(state) << "Surface " << surface.geometryId()
552 << " successfully hit, storing it.");
553 state.currentSurface = &surface;
554
555 if (target.isSurfaceTarget()) {
556 ACTS_VERBOSE(volInfo(state) << "This is a surface");
557 } else if (target.isLayerTarget()) {
558 ACTS_VERBOSE(volInfo(state) << "This is a layer");
559 } else if (target.isPortalTarget()) {
560 ACTS_VERBOSE(volInfo(state)
561 << "This is a boundary. Reinitialize navigation");
562
563 const BoundarySurface& boundary = target.boundarySurface();
564
565 state.currentVolume = boundary.attachedVolume(state.options.geoContext,
566 position, direction);
567
568 ACTS_VERBOSE(volInfo(state) << "Switched volume");
569
570 reinitializeCandidates(state);
571 } else {
572 ACTS_ERROR(volInfo(state) << "Unknown intersection type");
573 }
574 }
575
576 private:
578 void reinitializeCandidates(State& state) const {
579 state.navigationCandidates.clear();
580 state.currentTargets.clear();
581
582 initializeVolumeCandidates(state);
583 }

702 NavigationTarget nextTarget(State& state, const Vector3& position,
703 const Vector3& direction) const {
704 (void)direction;
705
706 // Navigator preStep always resets the current surface
707 state.currentSurface = nullptr;
708
709 // Check if the navigator is inactive
710 if (state.navigationBreak) {
711 return NavigationTarget::None();
712 }
713
714 ACTS_VERBOSE(volInfo(state) << "nextTarget");
715
716 // We cannot do anything without a last position
717 if (!state.lastPosition.has_value() && state.endOfTargets()) {
718 ACTS_VERBOSE(
719 volInfo(state)
720 << "Initial position, nothing to do, blindly stepping forward.");
721 state.lastPosition = position;
722 return NavigationTarget::None();
723 }
724
725 if (state.endOfTargets()) {
726 ACTS_VERBOSE(volInfo(state) << "evaluate blind step");
727
728 Vector3 stepStart = state.lastPosition.value();
729 Vector3 stepEnd = position;
730 Vector3 step = stepEnd - stepStart;
731 double stepDistance = step.norm();
732 if (stepDistance < std::numeric_limits<double>::epsilon()) {
733 ACTS_ERROR(volInfo(state) << "Step distance is zero. " << stepDistance);
734 }
735 Vector3 stepDirection = step.normalized();
736
737 double nearLimit = -stepDistance + state.options.surfaceTolerance;
738 double farLimit = 0;
739
740 state.lastPosition.reset();
741 state.activeTargets.clear();
742 state.activeTargetIndex = -1;
743
744 // Find intersections with all candidates
745 for (const auto& candidate : state.navigationCandidates) {
746 auto intersections =
747 candidate.intersect(state.options.geoContext, stepEnd,
748 stepDirection, state.options.surfaceTolerance);
749 for (auto [intersectionIndex, intersection] :
750 Acts::enumerate(intersections)) {
751 // exclude invalid intersections
752 if (!intersection.isValid() ||
753 !detail::checkPathLength(intersection.pathLength(), nearLimit,
754 farLimit)) {
755 continue;
756 }
757 // store candidate
758 state.activeTargets.emplace_back(
759 candidate.target(intersection, intersectionIndex));
760 }
761 }
762
763 std::ranges::sort(state.activeTargets, NavigationTarget::pathLengthOrder);
764
765 ACTS_VERBOSE(volInfo(state) << "Found " << state.activeTargets.size()
766 << " intersections");
767
768 for (const auto& target : state.activeTargets) {
769 ACTS_VERBOSE("found target " << target.surface().geometryId());
770 }
771 }
772
773 ++state.activeTargetIndex;
774
775 if (state.endOfTargets()) {
776 ACTS_VERBOSE(volInfo(state)
777 << "No target found, blindly stepping forward.");
778 state.lastPosition = position;
779 return NavigationTarget::None();
780 }
781
782 ACTS_VERBOSE(volInfo(state) << "handle active candidates");
783
784 ACTS_VERBOSE(volInfo(state)
785 << (state.activeTargets.size() - state.activeTargetIndex)
786 << " out of " << state.activeTargets.size()
787 << " surfaces remain to try.");
788
789 const auto& target = state.activeTarget();
790
791 ACTS_VERBOSE(volInfo(state) << "Next surface candidate will be "
792 << target.surface().geometryId());
793
794 return target;
795 }

824 void handleSurfaceReached(State& state, const Vector3& position,
825 const Vector3& direction,
826 const Surface& /*surface*/) const {
827 if (state.navigationBreak) {
828 return;
829 }
830
831 ACTS_VERBOSE(volInfo(state) << "handleSurfaceReached");
832
833 assert(state.currentSurface == nullptr && "Current surface must be reset.");
834
835 if (state.endOfTargets()) {
836 ACTS_VERBOSE(volInfo(state) << "No active candidate set.");
837 return;
838 }
839
840 std::vector<NavigationTarget> hitTargets;
841
842 while (!state.endOfTargets()) {
843 const auto& candidate = state.activeTarget();
844 IntersectionIndex index = candidate.intersectionIndex();
845 const Surface& surface = candidate.surface();
846 BoundaryTolerance boundaryTolerance = candidate.boundaryTolerance();
847
848 // first with boundary tolerance
849 IntersectionStatus surfaceStatus =
850 surface
851 .intersect(state.options.geoContext, position, direction,
852 boundaryTolerance, state.options.surfaceTolerance)
853 .at(index)
854 .status();
855
856 if (surfaceStatus != IntersectionStatus::onSurface) {
857 break;
858 }
859
860 // now without boundary tolerance
861 boundaryTolerance = BoundaryTolerance::None();
862 surfaceStatus =
863 surface
864 .intersect(state.options.geoContext, position, direction,
865 boundaryTolerance, state.options.surfaceTolerance)
866 .at(index)
867 .status();
868
869 if (surfaceStatus == IntersectionStatus::onSurface) {
870 hitTargets.emplace_back(candidate);
871 }
872
873 ++state.activeTargetIndex;
874 ACTS_VERBOSE("skip target " << surface.geometryId());
875 }
876
877 // we increased the target index one too many times
878 --state.activeTargetIndex;
879
880 ACTS_VERBOSE(volInfo(state)
881 << "Found " << hitTargets.size()
882 << " intersections on surface with bounds check.");
883
884 if (hitTargets.empty()) {
885 ACTS_VERBOSE(volInfo(state)
886 << "Surface successfully hit, but outside bounds.");
887 return;
888 }
889
890 if (hitTargets.size() > 1) {
891 ACTS_VERBOSE(volInfo(state)
892 << "Only using first intersection within bounds.");
893 }
894
895 // we can only handle a single surface hit so we pick the first one
896 const auto& candidate = hitTargets.front();
897 const Surface& surface = candidate.surface();
898
899 ACTS_VERBOSE(volInfo(state) << "Surface successfully hit, storing it.");
900 state.currentSurface = &surface;
901
902 if (state.currentSurface != nullptr) {
903 ACTS_VERBOSE(volInfo(state) << "Current surface set to surface "
904 << surface.geometryId());
905 }
906
907 if (candidate.isSurfaceTarget()) {
908 ACTS_VERBOSE(volInfo(state) << "This is a surface");
909 } else if (candidate.isLayerTarget()) {
910 ACTS_VERBOSE(volInfo(state) << "This is a layer");
911 } else if (candidate.isPortalTarget()) {
912 ACTS_VERBOSE(volInfo(state)
913 << "This is a portal. Reinitialize navigation");
914
915 const BoundarySurface& boundary = candidate.boundarySurface();
916
917 state.currentVolume = boundary.attachedVolume(state.options.geoContext,
918 position, direction);
919
920 ACTS_VERBOSE(volInfo(state) << "Switched volume");
921
922 reinitializeCandidates(state);
923 } else {
924 ACTS_ERROR(volInfo(state) << "Unknown intersection type");
925 }
926 }
927
928 private:
930 void reinitializeCandidates(State& state) const {
931 state.navigationCandidates.clear();
932 state.activeTargets.clear();
933 state.activeTargetIndex = -1;
934
935 initializeVolumeCandidates(state);
936 }
