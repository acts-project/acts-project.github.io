199 Value m_value = 0;
200
202 static constexpr int extractShift(Value mask) {
203 // use compiler builtin to extract the number of trailing bits from the
204 // mask. the builtin should be available on all supported compilers.
205 // need unsigned long long version (...ll) to ensure std::uint64_t
206 // compatibility.
207 // WARNING undefined behaviour for mask == 0 which we should not have.
208 return __builtin_ctzll(mask);
209 }
210
211 constexpr static Value getMaxValue(Value mask) {
212 return mask >> extractShift(mask);
213 }
214
216 constexpr Value getBits(Value mask) const {
217 return (m_value & mask) >> extractShift(mask);
218 }
220 constexpr GeometryIdentifier& setBits(Value mask, Value id) {
221 if (id > getMaxValue(mask)) {
222 throw std::invalid_argument(
223 "Value " + std::to_string(id) + " exceeds maximum value " +
224 std::to_string(getMaxValue(mask)) + " for this field");
225 }
226
227 m_value = (m_value & ~mask) | ((id << extractShift(mask)) & mask);
228 // return *this here that we need to write fewer lines in the setXXX
229 // methods
230 return *this;
231 }
232
233 friend constexpr bool operator==(GeometryIdentifier lhs,
234 GeometryIdentifier rhs) {
235 return lhs.m_value == rhs.m_value;
236 }
237
238 friend constexpr bool operator<(GeometryIdentifier lhs,
239 GeometryIdentifier rhs) {
240 return lhs.m_value < rhs.m_value;
241 }
