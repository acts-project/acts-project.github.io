49 std::tuple<actsvg::style::fill, actsvg::style::stroke> fillAndStroke() const {
50 actsvg::style::fill fll;
51 fll._fc._rgb = fillColor;
52 fll._fc._opacity = static_cast<actsvg::scalar>(fillOpacity);
53 fll._fc._hl_rgb = highlightColor;
54 fll._fc._highlight = highlights;
55
56 actsvg::style::stroke str;
57 str._sc._rgb = strokeColor;
58 str._sc._hl_rgb = highlightStrokeColor;
59 str._width = static_cast<actsvg::scalar>(strokeWidth);
60 str._hl_width = static_cast<actsvg::scalar>(highlightStrokeWidth);
61 str._dasharray = strokeDasharray;
62
63 return {fll, str};
64 }
65
68 std::tuple<actsvg::style::fill, actsvg::style::stroke, actsvg::style::font>
69 fillStrokeFont() const {
70 auto [fll, str] = fillAndStroke();
71
72 actsvg::style::font fnt;
73 fnt._size = fontSize;
74 fnt._fc._rgb = fontColor;
75
76 return std::tie(fll, str, fnt);
77 }
78};

136inline static actsvg::svg::object measure(double xStart, double yStart,
137 double xEnd, double yEnd,
138 const std::string& variable = "",
139 double value = 0.,
140 const std::string& unit = "") {
141 std::string mlabel = "";
142 if (!variable.empty()) {
143 mlabel = variable + " = ";
144 }
145 if (value != 0.) {
146 mlabel += actsvg::utils::to_string(static_cast<actsvg::scalar>(value));
147 }
148 if (!unit.empty()) {
149 mlabel += " ";
150 mlabel += unit;
151 }
152 return actsvg::draw::measure(
153 "measure",
154 {static_cast<actsvg::scalar>(xStart),
155 static_cast<actsvg::scalar>(yStart)},
156 {static_cast<actsvg::scalar>(xEnd), static_cast<actsvg::scalar>(yEnd)},
157 actsvg::style::stroke(), actsvg::style::marker({"o"}),
158 actsvg::style::marker({"|<<"}), actsvg::style::font(), mlabel);
159}

189inline static actsvg::svg::object infoBox(
190 double xPos, double yPos, const std::string& title, const Style& titleStyle,
191 const std::vector<std::string>& info, const Style& infoStyle,
192 const actsvg::svg::object& object,
193 const std::vector<std::string>& highlights = {"mouseover", "mouseout"}) {
194 auto [titleFill, titleStroke, titleFont] = titleStyle.fillStrokeFont();
195 auto [infoFill, infoStroke, infoFont] = infoStyle.fillStrokeFont();
196
197 actsvg::style::stroke stroke;
198
199 return actsvg::draw::connected_info_box(
200 object._id + "_infoBox",
201 {static_cast<actsvg::scalar>(xPos), static_cast<actsvg::scalar>(yPos)},
202 title, titleFill, titleFont, info, infoFill, infoFont, stroke, object,
203 highlights);
204}
