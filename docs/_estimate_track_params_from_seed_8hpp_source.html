69template <std::ranges::range spacepoint_range_t>
70FreeVector estimateTrackParamsFromSeed(spacepoint_range_t spRange,
71 const Vector3& bField) {
72 // Check the number of provided space points
73 if (spRange.size() != 3) {
74 throw std::invalid_argument(
75 "There should be exactly three space points provided.");
76 }
77
78 // The global positions of the bottom, middle and space points
79 std::array<Vector3, 3> spPositions = {Vector3::Zero(), Vector3::Zero(),
80 Vector3::Zero()};
81 std::array<std::optional<double>, 3> spTimes = {std::nullopt, std::nullopt,
82 std::nullopt};
83 // The first, second and third space point are assumed to be bottom, middle
84 // and top space point, respectively
85 for (auto [sp, spPosition, spTime] :
86 Acts::zip(spRange, spPositions, spTimes)) {
87 if (sp == nullptr) {
88 throw std::invalid_argument("Empty space point found.");
89 }
90 spPosition = Vector3(sp->x(), sp->y(), sp->z());
91 spTime = sp->t();
92 }
93
94 FreeVector params = estimateTrackParamsFromSeed(
95 spPositions[0], spPositions[1], spPositions[2], bField);
96 params[eFreeTime] = spTimes[0].value_or(0);
97 return params;
98}

123template <std::ranges::range spacepoint_range_t>
124Result<BoundVector> estimateTrackParamsFromSeed(const GeometryContext& gctx,
125 spacepoint_range_t spRange,
126 const Surface& surface,
127 const Vector3& bField) {
128 FreeVector freeParams = estimateTrackParamsFromSeed(spRange, bField);
129
130 const auto* sp0 = *spRange.begin();
131 Vector3 origin = Vector3(sp0->x(), sp0->y(), sp0->z());
132 Vector3 direction = freeParams.segment<3>(eFreeDir0);
133
134 BoundVector params = BoundVector::Zero();
135 params[eBoundPhi] = VectorHelpers::phi(direction);
136 params[eBoundTheta] = VectorHelpers::theta(direction);
137 params[eBoundQOverP] = freeParams[eFreeQOverP];
138
139 // Transform the bottom space point to local coordinates of the provided
140 // surface
141 auto lpResult = surface.globalToLocal(gctx, origin, direction);
142 if (!lpResult.ok()) {
143 return Result<BoundVector>::failure(lpResult.error());
144 }
145 Vector2 bottomLocalPos = lpResult.value();
146 // The estimated loc0 and loc1
147 params[eBoundLoc0] = bottomLocalPos.x();
148 params[eBoundLoc1] = bottomLocalPos.y();
149 params[eBoundTime] = sp0->t().value_or(0);
150
151 return Result<BoundVector>::success(params);
152}
