112 template <typename stepper_t>
113 void propagateTime(typename stepper_t::State& state, const stepper_t& stepper,
114 const double h) const {
118 auto m = stepper.particleHypothesis(state).mass();
119 auto p = stepper.absoluteMomentum(state);
120 auto dtds = std::sqrt(1 + m * m / (p * p));
121 state.pars[eFreeTime] += h * dtds;
122 if (state.covTransport) {
123 state.derivative(3) = dtds;
124 }
125 }

137 template <typename stepper_t>
138 bool transportMatrix(typename stepper_t::State& state,
139 const stepper_t& stepper, const double h,
140 FreeMatrix& D) const {
159
160 auto m = state.particleHypothesis.mass();
161 auto& sd = state.stepData;
162 auto dir = stepper.direction(state);
163 auto qop = stepper.qOverP(state);
164 auto p = stepper.absoluteMomentum(state);
165 auto dtds = std::sqrt(1 + m * m / (p * p));
166
167 D = FreeMatrix::Identity();
168
169 double half_h = h * 0.5;
170 // This sets the reference to the sub matrices
171 // dFdx is already initialised as (3x3) identity
172 auto dFdT = D.block<3, 3>(0, 4);
173 auto dFdL = D.block<3, 1>(0, 7);
174 // dGdx is already initialised as (3x3) zero
175 auto dGdT = D.block<3, 3>(4, 4);
176 auto dGdL = D.block<3, 1>(4, 7);
177
178 SquareMatrix3 dk1dT = SquareMatrix3::Zero();
179 SquareMatrix3 dk2dT = SquareMatrix3::Identity();
180 SquareMatrix3 dk3dT = SquareMatrix3::Identity();
181 SquareMatrix3 dk4dT = SquareMatrix3::Identity();
182
183 Vector3 dk1dL = Vector3::Zero();
184 Vector3 dk2dL = Vector3::Zero();
185 Vector3 dk3dL = Vector3::Zero();
186 Vector3 dk4dL = Vector3::Zero();
187
188 // For the case without energy loss
189 dk1dL = dir.cross(sd.B_first);
190 dk2dL = (dir + half_h * sd.k1).cross(sd.B_middle) +
191 qop * half_h * dk1dL.cross(sd.B_middle);
192 dk3dL = (dir + half_h * sd.k2).cross(sd.B_middle) +
193 qop * half_h * dk2dL.cross(sd.B_middle);
194 dk4dL =
195 (dir + h * sd.k3).cross(sd.B_last) + qop * h * dk3dL.cross(sd.B_last);
196
197 dk1dT(0, 1) = sd.B_first.z();
198 dk1dT(0, 2) = -sd.B_first.y();
199 dk1dT(1, 0) = -sd.B_first.z();
200 dk1dT(1, 2) = sd.B_first.x();
201 dk1dT(2, 0) = sd.B_first.y();
202 dk1dT(2, 1) = -sd.B_first.x();
203 dk1dT *= qop;
204
205 dk2dT += half_h * dk1dT;
206 dk2dT = qop * VectorHelpers::cross(dk2dT, sd.B_middle);
207
208 dk3dT += half_h * dk2dT;
209 dk3dT = qop * VectorHelpers::cross(dk3dT, sd.B_middle);
210
211 dk4dT += h * dk3dT;
212 dk4dT = qop * VectorHelpers::cross(dk4dT, sd.B_last);
213
214 dFdT.setIdentity();
215 dFdT += h / 6. * (dk1dT + dk2dT + dk3dT);
216 dFdT *= h;
217
218 dFdL = (h * h) / 6. * (dk1dL + dk2dL + dk3dL);
219
220 dGdT += h / 6. * (dk1dT + 2. * (dk2dT + dk3dT) + dk4dT);
221
222 dGdL = h / 6. * (dk1dL + 2. * (dk2dL + dk3dL) + dk4dL);
223
224 D(3, 7) = h * m * m * qop / dtds;
225 return true;
226 }
