454template <typename track_state_t>
455void addMaterialToGx2fSums(
456 Gx2fSystem& extendedSystem, const std::size_t nMaterialsHandled,
457 const std::unordered_map<GeometryIdentifier, ScatteringProperties>&
458 scatteringMap,
459 const track_state_t& trackState, const Logger& logger) {
460 // Get and store geoId for the current material surface
461 const GeometryIdentifier geoId = trackState.referenceSurface().geometryId();
462 const auto scatteringMapId = scatteringMap.find(geoId);
463 if (scatteringMapId == scatteringMap.end()) {
464 ACTS_ERROR("No scattering angles found for material surface " << geoId);
465 throw std::runtime_error(
466 "No scattering angles found for material surface.");
467 }
468
469 const double sinThetaLoc = std::sin(trackState.smoothed()[eBoundTheta]);
470
471 // The position, where we need to insert the values in aMatrix and bVector
472 const std::size_t deltaPosition = eBoundSize + 2 * nMaterialsHandled;
473
474 const BoundVector& scatteringAngles =
475 scatteringMapId->second.scatteringAngles();
476
477 const double invCov = scatteringMapId->second.invCovarianceMaterial();
478
479 // Phi contribution
480 extendedSystem.aMatrix()(deltaPosition, deltaPosition) +=
481 invCov * sinThetaLoc * sinThetaLoc;
482 extendedSystem.bVector()(deltaPosition, 0) -=
483 invCov * scatteringAngles[eBoundPhi] * sinThetaLoc;
484 extendedSystem.chi2() += invCov * scatteringAngles[eBoundPhi] * sinThetaLoc *
485 scatteringAngles[eBoundPhi] * sinThetaLoc;
486
487 // Theta Contribution
488 extendedSystem.aMatrix()(deltaPosition + 1, deltaPosition + 1) += invCov;
489 extendedSystem.bVector()(deltaPosition + 1, 0) -=
490 invCov * scatteringAngles[eBoundTheta];
491 extendedSystem.chi2() +=
492 invCov * scatteringAngles[eBoundTheta] * scatteringAngles[eBoundTheta];
493
494 ACTS_VERBOSE(
495 "Contributions in addMaterialToGx2fSums:\n"
496 << " invCov: " << invCov << "\n"
497 << " sinThetaLoc: " << sinThetaLoc << "\n"
498 << " deltaPosition: " << deltaPosition << "\n"
499 << " Phi:\n"
500 << " scattering angle: " << scatteringAngles[eBoundPhi] << "\n"
501 << " aMatrix contribution: " << invCov * sinThetaLoc * sinThetaLoc
502 << "\n"
503 << " bVector contribution: "
504 << invCov * scatteringAngles[eBoundPhi] * sinThetaLoc << "\n"
505 << " chi2sum contribution: "
506 << invCov * scatteringAngles[eBoundPhi] * sinThetaLoc *
507 scatteringAngles[eBoundPhi] * sinThetaLoc
508 << "\n"
509 << " Theta:\n"
510 << " scattering angle: " << scatteringAngles[eBoundTheta]
511 << "\n"
512 << " aMatrix contribution: " << invCov << "\n"
513 << " bVector contribution: "
514 << invCov * scatteringAngles[eBoundTheta] << "\n"
515 << " chi2sum contribution: "
516 << invCov * scatteringAngles[eBoundTheta] * scatteringAngles[eBoundTheta]
517 << "\n");
518
519 return;
520}

537template <TrackProxyConcept track_proxy_t>
538void fillGx2fSystem(
539 const track_proxy_t track, Gx2fSystem& extendedSystem,
540 const bool multipleScattering,
541 const std::unordered_map<GeometryIdentifier, ScatteringProperties>&
542 scatteringMap,
543 std::vector<GeometryIdentifier>& geoIdVector, const Logger& logger) {
544 std::vector<BoundMatrix> jacobianFromStart;
545 jacobianFromStart.emplace_back(BoundMatrix::Identity());
546
547 for (const auto& trackState : track.trackStates()) {
548 // Get and store geoId for the current surface
549 const GeometryIdentifier geoId = trackState.referenceSurface().geometryId();
550 ACTS_DEBUG("Start to investigate trackState on surface " << geoId);
551 const auto typeFlags = trackState.typeFlags();
552 const bool stateHasMeasurement =
553 typeFlags.test(TrackStateFlag::MeasurementFlag);
554 const bool stateHasMaterial = typeFlags.test(TrackStateFlag::MaterialFlag);
555
556 // First we figure out, if we would need to look into material
557 // surfaces at all. Later, we also check, if the material slab is
558 // valid, otherwise we modify this flag to ignore the material
559 // completely.
560 bool doMaterial = multipleScattering && stateHasMaterial;
561 if (doMaterial) {
562 const auto scatteringMapId = scatteringMap.find(geoId);
563 assert(scatteringMapId != scatteringMap.end() &&
564 "No scattering angles found for material surface.");
565 doMaterial = doMaterial && scatteringMapId->second.materialIsValid();
566 }
567
568 // We only consider states with a measurement (and/or material)
569 if (!stateHasMeasurement && !doMaterial) {
570 ACTS_DEBUG(" Skip state.");
571 continue;
572 }
573
574 // update all Jacobians from start
575 for (auto& jac : jacobianFromStart) {
576 jac = trackState.jacobian() * jac;
577 }
578
579 // Handle measurement
580 if (stateHasMeasurement) {
581 ACTS_DEBUG(" Handle measurement.");
582
583 const auto measDim = trackState.calibratedSize();
584
585 if (measDim < 1 || 6 < measDim) {
586 ACTS_ERROR("Can not process state with measurement with "
587 << measDim << " dimensions.");
588 throw std::domain_error(
589 "Found measurement with less than 1 or more than 6 dimension(s).");
590 }
591
592 extendedSystem.ndf() += measDim;
593
594 visit_measurement(measDim, [&](auto N) {
595 addMeasurementToGx2fSums<N>(extendedSystem, jacobianFromStart,
596 trackState, logger);
597 });
598 }
599
600 // Handle material
601 if (doMaterial) {
602 ACTS_DEBUG(" Handle material");
603 // Add for this material a new Jacobian, starting from this surface.
604 jacobianFromStart.emplace_back(BoundMatrix::Identity());
605
606 // Add the material contribution to the system
607 addMaterialToGx2fSums(extendedSystem, geoIdVector.size(), scatteringMap,
608 trackState, logger);
609
610 geoIdVector.emplace_back(geoId);
611 }
612 }

631 const std::unordered_map<GeometryIdentifier, ScatteringProperties>&
632 scatteringMap,

650 if (!scatteringMapId->second.materialIsValid()) {
651 continue;
652 }
653
654 nMaterialSurfaces++;
655 }
656
657 return nMaterialSurfaces;

804 template <typename propagator_state_t, typename stepper_t,
805 typename navigator_t>
806 Result<void> act(propagator_state_t& state, const stepper_t& stepper,
807 const navigator_t& navigator, result_type& result,
808 const Logger& /*logger*/) const {
809 assert(result.fittedStates && "No MultiTrajectory set");
810
811 // Check if we can stop to propagate
812 if (result.measurementStates == inputMeasurements->size()) {
813 ACTS_DEBUG("Actor: finish: All measurements have been found.");
814 result.finished = true;
815 } else if (state.navigation.navigationBreak) {
816 ACTS_DEBUG("Actor: finish: navigationBreak.");
817 result.finished = true;
818 }
819
820 // End the propagation and return to the fitter
821 if (result.finished) {
822 // Remove the missing surfaces that occur after the last measurement
823 if (result.measurementStates > 0) {
824 result.missedActiveSurfaces.resize(result.measurementHoles);
825 }
826
827 return Result<void>::success();
828 }
829
830 // We are only interested in surfaces. If we are not on a surface, we
831 // continue the navigation
832 auto surface = navigator.currentSurface(state.navigation);
833 if (surface == nullptr) {
834 return Result<void>::success();
835 }
836
837 ++result.surfaceCount;
838 const GeometryIdentifier geoId = surface->geometryId();
839 ACTS_DEBUG("Surface " << geoId << " detected.");
840
841 const bool surfaceIsSensitive = surface->isSensitive();
842 const bool surfaceHasMaterial = (surface->surfaceMaterial() != nullptr);
843 // First we figure out, if we would need to look into material surfaces at
844 // all. Later, we also check, if the material slab is valid, otherwise we
845 // modify this flag to ignore the material completely.
846 bool doMaterial = multipleScattering && surfaceHasMaterial;
847
848 // Found material - add a scatteringAngles entry if not done yet.
849 // Handling will happen later
850 if (doMaterial) {
851 ACTS_DEBUG(" The surface contains material, ...");
852
853 auto scatteringMapId = scatteringMap->find(geoId);
854 if (scatteringMapId == scatteringMap->end()) {
855 ACTS_DEBUG(" ... create entry in scattering map.");
856
857 Acts::detail::PointwiseMaterialInteraction interaction(surface, state,
858 stepper);
859 // We need to evaluate the material to create the correct slab
860 const bool slabIsValid = interaction.evaluateMaterialSlab(
861 state, navigator, MaterialUpdateStage::FullUpdate);
862 double invSigma2 = 0.;
863 if (slabIsValid) {
864 const auto& particle =
865 parametersWithHypothesis->particleHypothesis();
866
867 const double sigma =
868 static_cast<double>(Acts::computeMultipleScatteringTheta0(
869 interaction.slab, particle.absolutePdg(), particle.mass(),
870 static_cast<float>(
871 parametersWithHypothesis->parameters()[eBoundQOverP]),
872 particle.absoluteCharge()));
873 ACTS_VERBOSE(
874 " The Highland formula gives sigma = " << sigma);
875 invSigma2 = 1. / std::pow(sigma, 2);
876 } else {
877 ACTS_VERBOSE(" Material slab is not valid.");
878 }
879
880 scatteringMap->emplace(
881 geoId, ScatteringProperties{BoundVector::Zero(), invSigma2,
882 slabIsValid});
883 scatteringMapId = scatteringMap->find(geoId);
884 } else {
885 ACTS_DEBUG(" ... found entry in scattering map.");
886 }
887
888 doMaterial = doMaterial && scatteringMapId->second.materialIsValid();
889 }
890
891 // Here we handle all measurements
892 if (auto sourceLinkIt = inputMeasurements->find(geoId);
893 sourceLinkIt != inputMeasurements->end()) {
894 ACTS_DEBUG(" The surface contains a measurement.");
895
896 // Transport the covariance to the surface
897 stepper.transportCovarianceToBound(state.stepping, *surface,
898 freeToBoundCorrection);
899
900 // TODO generalize the update of the currentTrackIndex
901 auto& fittedStates = *result.fittedStates;
902
903 // Add a <trackStateMask> TrackState entry multi trajectory. This
904 // allocates storage for all components, which we will set later.
905 typename traj_t::TrackStateProxy trackStateProxy =
906 fittedStates.makeTrackState(Gx2fConstants::trackStateMask,
907 result.lastTrackIndex);
908 const std::size_t currentTrackIndex = trackStateProxy.index();
909
910 // Set the trackStateProxy components with the state from the ongoing
911 // propagation
912 {
913 trackStateProxy.setReferenceSurface(surface->getSharedPtr());
914 // Bind the transported state to the current surface
915 auto res = stepper.boundState(state.stepping, *surface, false,
916 freeToBoundCorrection);
917 if (!res.ok()) {
918 return res.error();
919 }
920 // Not const since, we might need to update with scattering angles
921 auto& [boundParams, jacobian, pathLength] = *res;
922
923 // For material surfaces, we also update the angles with the
924 // available scattering information
925 if (doMaterial) {
926 ACTS_DEBUG(" Update parameters with scattering angles.");
927 const auto scatteringMapId = scatteringMap->find(geoId);
928 ACTS_VERBOSE(
929 " scatteringAngles: "
930 << scatteringMapId->second.scatteringAngles().transpose());
931 ACTS_VERBOSE(" boundParams before the update: "
932 << boundParams.parameters().transpose());
933 boundParams.parameters() +=
934 scatteringMapId->second.scatteringAngles();
935 ACTS_VERBOSE(" boundParams after the update: "
936 << boundParams.parameters().transpose());
937 }
938
939 // Fill the track state
940 trackStateProxy.smoothed() = boundParams.parameters();
941 trackStateProxy.smoothedCovariance() = state.stepping.cov;
942
943 trackStateProxy.jacobian() = jacobian;
944 trackStateProxy.pathLength() = pathLength;
945
946 if (doMaterial) {
947 stepper.update(state.stepping,
948 transformBoundToFreeParameters(
949 trackStateProxy.referenceSurface(),

1594 if (!propagatorInitResult.ok()) {
1595 ACTS_ERROR("Propagation initialization failed: "
1596 << propagatorInitResult.error());
1597 return propagatorInitResult.error();
1598 }
1599
1600 auto& r = propagatorState.template get<Gx2FitterResult<traj_t>>();
1601 r.fittedStates = &trackContainer.trackStateContainer();
1602
1603 auto propagationResult = m_propagator.propagate(propagatorState);
1604
1605 // Run the fitter
1606 auto result =
1607 m_propagator.makeResult(std::move(propagatorState), propagationResult,
1608 propagatorOptions, false);
1609
1610 if (!result.ok()) {
1611 ACTS_ERROR("Propagation failed: " << result.error());
1612 return result.error();
1613 }
1614
1615 auto& propRes = *result;
1616 GX2FResult gx2fResult = std::move(propRes.template get<GX2FResult>());
1617
1618 if (tipIndex != gx2fResult.lastMeasurementIndex) {
1619 ACTS_INFO("Final fit used unreachable measurements.");
1620 tipIndex = gx2fResult.lastMeasurementIndex;
1621
1622 // It could happen, that no measurements were found. Then the track
1623 // would be empty and the following operations would be invalid.
1624 if (tipIndex == kInvalid) {
1625 ACTS_INFO("Did not find any measurements in final propagation.");
1626 return Experimental::GlobalChiSquareFitterError::
1627 NotEnoughMeasurements;
1628 }
1629 }
1630 }
1631
1632 if (!trackContainer.hasColumn(
1633 Acts::hashString(Gx2fConstants::gx2fnUpdateColumn))) {
1634 trackContainer.template addColumn<std::uint32_t>("Gx2fnUpdateColumn");
1635 }
1636
1637 // Prepare track for return
1638 auto track = trackContainer.makeTrack();
1639 track.tipIndex() = tipIndex;
1640 track.parameters() = params.parameters();
1641 track.covariance() = fullCovariancePredicted;
1642 track.setReferenceSurface(params.referenceSurface().getSharedPtr());
1643
1644 if (trackContainer.hasColumn(
1645 Acts::hashString(Gx2fConstants::gx2fnUpdateColumn))) {
1646 ACTS_DEBUG("Add nUpdate to track");
1647 track.template component<std::uint32_t>("Gx2fnUpdateColumn") =
1648 static_cast<std::uint32_t>(nUpdate);
1649 }
1650
1651 // TODO write test for calculateTrackQuantities
1652 calculateTrackQuantities(track);
1653
1654 // Set the chi2sum for the track summary manually, since we don't calculate
1655 // it for each state
1656 track.chi2() = chi2sum;
1657
1658 // Return the converted Track
1659 return track;
1660 }
1661};
