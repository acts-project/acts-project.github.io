53 template <typename projector_t>
54 auto reduce(projector_t&& proj) const {
55 using Ret = std::decay_t<decltype(proj(std::declval<Parameters>()))>;
56
57 Ret ret;
58
59 if constexpr (std::is_floating_point_v<Ret>) {
60 ret = 0.0;
61 } else {
62 ret = Ret::Zero();
63 }
64
65 for (auto i = 0ul; i < m_components.size(); ++i) {
66 const auto [weight, single_pars] = (*this)[i];
67 ret += weight * proj(single_pars);
68 }
69
70 return ret;
71 }

76 using ConstructionTuple = std::tuple<double, Acts::Vector4, Acts::Vector3,
77 double, CovarianceMatrix>;

84 static MultiComponentBoundTrackParameters createCurvilinear(
85 const GeometryContext& geoCtx,
86 const std::vector<ConstructionTuple>& curvi,
87 ParticleHypothesis particleHypothesis) {
88 // Construct and average surface
89 Acts::Vector3 avgPos = Acts::Vector3::Zero();
90 Acts::Vector3 avgDir = Acts::Vector3::Zero();
91 for (const auto& [w, pos4, dir, qop, cov] : curvi) {
92 avgPos += w * pos4.template segment<3>(0);
93 avgDir += w * dir;
94 }
95
96 std::shared_ptr<PlaneSurface> s =
97 CurvilinearSurface(avgPos, avgDir).planeSurface();
98
99 std::vector<std::tuple<double, ParametersVector, CovarianceMatrix>> bound;
100 bound.reserve(curvi.size());
101
102 // Project the position onto the surface, keep everything else as is
103 for (const auto& [w, pos4, dir, qop, cov] : curvi) {
104 Intersection3D closestIntersection =
105 s->intersect(geoCtx, pos4.template segment<3>(eFreePos0), dir,
106 BoundaryTolerance::Infinite())
107 .closest();
108 const Vector3& newPos = closestIntersection.position();
109
110 ParametersVector bv =
111 transformFreeToCurvilinearParameters(pos4[eTime], dir, qop);
112
113 // Because of the projection this should never fail
114 bv.template segment<2>(eBoundLoc0) =
115 *(s->globalToLocal(geoCtx, newPos, dir));
116
117 bound.emplace_back(w, bv, cov);
118 }
119
120 return MultiComponentBoundTrackParameters(s, bound, particleHypothesis);
121 }

127 template <typename covariance_t>
128 MultiComponentBoundTrackParameters(
129 std::shared_ptr<const Surface> surface,
130 const std::vector<std::tuple<double, ParametersVector, covariance_t>>&
131 cmps,
132 ParticleHypothesis particleHypothesis)
133 : m_surface(std::move(surface)),
134 m_particleHypothesis(particleHypothesis) {
135 static_assert(
136 std::is_same_v<BoundSquareMatrix, covariance_t> ||
137 std::is_same_v<std::optional<BoundSquareMatrix>, covariance_t>);
138 if constexpr (std::is_same_v<BoundSquareMatrix, covariance_t>) {
139 for (const auto& [weight, params, cov] : cmps) {
140 m_components.push_back({weight, params, cov});
141 }
142 } else {
143 m_components = cmps;
144 }
145 }
