76 template <typename... Args>
77 void addRow(Args&&... args) {
78 if (sizeof...(Args) != m_columns.size()) {
79 throw std::runtime_error("Number of arguments (" +
80 std::to_string(sizeof...(Args)) +
81 ") does not match number of columns (" +
82 std::to_string(m_columns.size()) + ")");
83 }
84
85 std::vector<std::string> row;
86 std::size_t colIndex = 0;
87
88 auto addCell = [&](auto&& arg) {
89 std::string formatted =
90 std::vformat(m_columns[colIndex].format, std::make_format_args(arg));
91 row.push_back(formatted);
92 m_columns[colIndex].width =
93 std::max(m_columns[colIndex].width, formatted.length());
94 ++colIndex;
95 };
96
97 (addCell(args), ...);
98
99 m_rows.push_back(std::move(row));
100 }

104 std::string toString() const {
105 if (m_columns.empty()) {
106 return "";
107 }
108
109 std::string result;
110
111 // Build header row
112 result += "|";
113 for (std::size_t i = 0; i < m_columns.size(); ++i) {
114 result += std::format(
115 " {} |", formatAligned(m_columns[i].header, m_columns[i].width,
116 m_columns[i].alignment));
117 }
118 result += "\n";
119
120 // Build separator row
121 result += "|";
122 for (std::size_t i = 0; i < m_columns.size(); ++i) {
123 std::size_t contentWidth = m_columns[i].width;
124
125 if (m_includeMarkdownMarkers) {
126 switch (m_columns[i].alignment) {
127 case Alignment::Left:
128 result += std::format(":{:-<{}}", "", contentWidth + 1);
129 break;
130 case Alignment::Right:
131 result += std::format("{:-<{}}:", "", contentWidth + 1);
132 break;
133 case Alignment::Center:
134 result += std::format(":{:-<{}}:", "", contentWidth);
135 break;
136 }
137 } else {
138 result += std::format("{:-<{}}", "", contentWidth + 2);
139 }
140 result += "|";
141 }
142 result += "\n";
143
144 // Build data rows
145 for (const auto& row : m_rows) {
146 result += "|";
147 for (std::size_t i = 0; i < row.size(); ++i) {
148 result += std::format(" {} |", formatAligned(row[i], m_columns[i].width,
149 m_columns[i].alignment));
150 }
151 result += "\n";
152 }
153
154 return result;
155 }

165 static Alignment parseAlignment(const std::string& alignment) {
166 std::string lower = alignment;
167 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
168
169 if (lower == "left" || lower == "l") {
170 return Alignment::Left;
171 } else if (lower == "right" || lower == "r") {
172 return Alignment::Right;
173 } else if (lower == "center" || lower == "c") {
174 return Alignment::Center;
175 } else {
176 throw std::invalid_argument(
177 "Invalid alignment string: '" + alignment +
178 "'. Valid options: 'left'/'l', 'right'/'r', 'center'/'c'");
179 }
180 }
181
182 std::string formatAligned(const std::string& text, std::size_t width,
183 Alignment alignment) const {
184 switch (alignment) {
185 case Alignment::Left:
186 return std::format("{:<{}}", text, width);
187 case Alignment::Right:
188 return std::format("{:>{}}", text, width);
189 case Alignment::Center:
190 return std::format("{:^{}}", text, width);
191 }
192 return text;
193 }
