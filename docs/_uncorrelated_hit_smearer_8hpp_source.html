43template <typename generator_t, std::size_t kSize>
44struct BoundParametersSmearer {
46 using ParametersVector = Acts::ActsVector<kSize>;
48 using CovarianceMatrix = Acts::ActsSquareMatrix<kSize>;
50 using Result = Acts::Result<std::pair<ParametersVector, CovarianceMatrix>>;
51
53 std::array<Acts::BoundIndices, kSize> indices{};
55 std::array<SingleParameterSmearFunction<generator_t>, kSize> smearFunctions{};
57 std::array<bool, kSize> forcePositive = {};
59 std::size_t maxRetries = 0;
60
63 static constexpr std::size_t size() { return kSize; }
64
73 Result operator()(generator_t& rng, const Hit& hit,
74 const Acts::Surface& surface,
75 const Acts::GeometryContext& geoCtx) const {
76 // We use the thickness of the detector element as tolerance, because Geant4
77 // treats the Surfaces as volumes and thus it is not ensured, that each hit
78 // lies exactly on the Acts::Surface
79 const auto tolerance =
80 surface.isSensitive() ? surface.associatedDetectorElement()->thickness()
81 : Acts::s_onSurfaceTolerance;
82
83 // construct full bound parameters. they are probably not all needed, but it
84 // is easier to just create them all and then select the requested ones.
85 Acts::Result<Acts::BoundVector> boundParamsRes =
86 Acts::transformFreeToBoundParameters(hit.position(), hit.time(),
87 hit.direction(), 0, surface,
88 geoCtx, tolerance);
89
90 if (!boundParamsRes.ok()) {
91 return boundParamsRes.error();
92 }
93
94 const auto& boundParams = *boundParamsRes;
95 Acts::BoundVector smearedBoundParams = boundParams;
96
97 for (std::size_t k = 0; k < maxRetries + 1; ++k) {
98 ParametersVector par = ParametersVector::Zero();
99 CovarianceMatrix cov = CovarianceMatrix::Zero();
100 for (std::size_t i = 0; i < kSize; ++i) {
101 auto res = smearFunctions[i](boundParams[indices[i]], rng);
102 if (!res.ok()) {
103 return Result::failure(res.error());
104 }
105 auto [value, stddev] = res.value();
106 par[i] = value;
107 if (forcePositive[i]) {
108 par[i] = std::abs(value);
109 }
110 smearedBoundParams[indices[i]] = par[i];
111 cov(i, i) = stddev * stddev;
112 }
113
114 if (!surface.insideBounds(smearedBoundParams.head<2>())) {
115 continue;
116 }
117
118 return Result::success(std::make_pair(par, cov));
119 }
120
121 return Result::failure(DigitizationError::MaximumRetriesExceeded);
122 }
123};

135template <typename generator_t, std::size_t kSize>
136struct FreeParametersSmearer {
138 using ParametersVector = Acts::ActsVector<kSize>;
140 using CovarianceMatrix = Acts::ActsSquareMatrix<kSize>;
142 using Result = Acts::Result<std::pair<ParametersVector, CovarianceMatrix>>;
143
145 std::array<Acts::FreeIndices, kSize> indices{};
147 std::array<SingleParameterSmearFunction<generator_t>, kSize> smearFunctions;
148
151 static constexpr std::size_t size() { return kSize; }
152
160 Result operator()(generator_t& rng, const Hit& hit) const {
161 // construct full free parameters. they are probably not all needed, but it
162 // is easier to just create them all and then select the requested ones.
163 Acts::FreeVector freeParams;
164 freeParams.segment<3>(Acts::eFreePos0) = hit.position();
165 freeParams[Acts::eFreeTime] = hit.time();
166 freeParams.segment<3>(Acts::eFreeDir0) = hit.direction();
167 freeParams[Acts::eFreeQOverP] = 0;
168
169 ParametersVector par = ParametersVector::Zero();
170 CovarianceMatrix cov = CovarianceMatrix::Zero();
171 for (std::size_t i = 0; i < kSize; ++i) {
172 auto res = smearFunctions[i](freeParams[indices[i]], rng);
173 if (!res.ok()) {
174 return Result::failure(res.error());
175 }
176 auto [value, stddev] = res.value();
177 par[i] = value;
178 cov(i, i) = stddev * stddev;
179 }
180
181 return Result::success(std::make_pair(par, cov));
182 }
183};
