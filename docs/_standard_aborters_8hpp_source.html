24struct PathLimitReached {
26 double internalLimit = std::numeric_limits<double>::max();
27
39 template <typename propagator_state_t, typename stepper_t,
40 typename navigator_t>
41 bool checkAbort(propagator_state_t& state, const stepper_t& stepper,
42 const navigator_t& navigator, const Logger& logger) const {
43 (void)navigator;
44
45 // Check if the maximum allowed step size has to be updated
46 double distance =
47 std::abs(internalLimit) - std::abs(state.stepping.pathAccumulated);
48 double tolerance = state.options.surfaceTolerance;
49 bool limitReached = (std::abs(distance) < std::abs(tolerance));
50 if (limitReached) {
51 ACTS_VERBOSE("PathLimit aborter | " << "Path limit reached at distance "
52 << distance);
53 return true;
54 }
55 stepper.updateStepSize(state.stepping, distance,
56 ConstrainedStep::Type::Actor);
57 ACTS_VERBOSE("PathLimit aborter | "
58 << "Target stepSize (path limit) updated to "
59 << stepper.outputStepSize(state.stepping));
60 return false;
61 }
62};

66struct SurfaceReached {
68 const Surface* surface = nullptr;
70 BoundaryTolerance boundaryTolerance = BoundaryTolerance::None();

76 double nearLimit = -100 * UnitConstants::um;

78 SurfaceReached() = default;
81 explicit SurfaceReached(double nLimit) : nearLimit(nLimit) {}

94 template <typename propagator_state_t, typename stepper_t,
95 typename navigator_t>
96 bool checkAbort(propagator_state_t& state, const stepper_t& stepper,
97 const navigator_t& navigator, const Logger& logger) const {
98 if (surface == nullptr) {
99 ACTS_VERBOSE("SurfaceReached aborter | Target surface not set.");
100 return false;
101 }
102
103 if (navigator.currentSurface(state.navigation) == surface) {
104 ACTS_VERBOSE("SurfaceReached aborter | Target surface reached.");
105 return true;
106 }
107
108 // not using the stepper overstep limit here because it does not always work
109 // for perigee surfaces
110 // note: the near limit is necessary for surfaces with more than one
111 // intersection in order to discard the ones which are behind us
112 const double farLimit = std::numeric_limits<double>::max();
113 const double tolerance = state.options.surfaceTolerance;
114
115 const MultiIntersection3D multiIntersection = surface->intersect(
116 state.geoContext, stepper.position(state.stepping),
117 state.options.direction * stepper.direction(state.stepping),
118 boundaryTolerance, tolerance);
119 const Intersection3D closestIntersection = multiIntersection.closest();
120
121 bool reached = false;
122
123 if (closestIntersection.status() == IntersectionStatus::onSurface) {
124 const double distance = closestIntersection.pathLength();
125 ACTS_VERBOSE(
126 "SurfaceReached aborter | "
127 "Target surface reached at distance (tolerance) "
128 << distance << " (" << tolerance << ")");
129 reached = true;
130 }
131
132 bool intersectionFound = false;
133
134 for (auto [intersectionIndex, intersection] :
135 Acts::enumerate(multiIntersection)) {
136 if (intersection.isValid() &&
137 detail::checkPathLength(intersection.pathLength(), nearLimit,
138 farLimit, logger)) {
139 stepper.updateStepSize(state.stepping, intersection.pathLength(),
140 ConstrainedStep::Type::Actor);
141 ACTS_VERBOSE(
142 "SurfaceReached aborter | "
143 "Target stepSize (surface) updated to "
144 << stepper.outputStepSize(state.stepping));
145 intersectionFound = true;
146 break;
147 }
148 }
149
150 if (!intersectionFound) {
151 ACTS_VERBOSE(
152 "SurfaceReached aborter | "
153 "Target intersection not found. Maybe next time?");
154 }
155 return reached;
156 }
157};

190struct VolumeConstraintAborter {
200 template <typename propagator_state_t, typename stepper_t,
201 typename navigator_t>
202 bool checkAbort(propagator_state_t& state, const stepper_t& /*stepper*/,
203 const navigator_t& navigator, const Logger& logger) const {
204 const auto& constrainToVolumeIds = state.options.constrainToVolumeIds;
205 const auto& endOfWorldVolumeIds = state.options.endOfWorldVolumeIds;
206
207 if (constrainToVolumeIds.empty() && endOfWorldVolumeIds.empty()) {
208 return false;
209 }
210 const auto* currentVolume = navigator.currentVolume(state.navigation);
211
212 // We need a volume to check its ID
213 if (currentVolume == nullptr) {
214 return false;
215 }
216
217 const auto currentVolumeId =
218 static_cast<std::uint32_t>(currentVolume->geometryId().volume());
219
220 if (!constrainToVolumeIds.empty() &&
221 !rangeContainsValue(constrainToVolumeIds, currentVolumeId)) {
222 ACTS_VERBOSE(
223 "VolumeConstraintAborter aborter | Abort with volume constrain "
224 << currentVolumeId);
225 return true;
226 }
227
228 if (!endOfWorldVolumeIds.empty() &&
229 rangeContainsValue(endOfWorldVolumeIds, currentVolumeId)) {
230 ACTS_VERBOSE(
231 "VolumeConstraintAborter aborter | Abort with additional end of "
232 "world volume "
233 << currentVolumeId);
234 return true;
235 }
236
237 return false;
238 }
239};

242struct AnySurfaceReached {
252 template <typename propagator_state_t, typename stepper_t,
253 typename navigator_t>
254 bool checkAbort(propagator_state_t& state, const stepper_t& stepper,
255 const navigator_t& navigator, const Logger& logger) const {
256 (void)stepper;
257 (void)logger;
258
259 const Surface* startSurface = navigator.startSurface(state.navigation);
260 const Surface* targetSurface = navigator.targetSurface(state.navigation);
261 const Surface* currentSurface = navigator.currentSurface(state.navigation);
262
263 // `startSurface` is excluded because we want to reach a new surface
264 // `targetSurface` is excluded because another aborter should handle it
265 if (currentSurface != nullptr && currentSurface != startSurface &&
266 currentSurface != targetSurface) {
267 return true;
268 }
269
270 return false;
271 }
272};
