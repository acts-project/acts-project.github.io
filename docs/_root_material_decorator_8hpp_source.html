63 void decorate(Acts::Surface& surface) const final {
64 // Null out the material for this surface
65 if (m_clearSurfaceMaterial) {
66 surface.assignSurfaceMaterial(nullptr);
67 }
68 // Try to find the surface in the map
69 auto sMaterial = m_surfaceMaterialMap.find(surface.geometryId());
70 if (sMaterial != m_surfaceMaterialMap.end()) {
71 surface.assignSurfaceMaterial(sMaterial->second);
72 }
73 }
74
78 void decorate(Acts::TrackingVolume& volume) const final {
79 // Null out the material for this volume
80 if (m_clearSurfaceMaterial) {
81 volume.assignVolumeMaterial(nullptr);
82 }
83 // Try to find the surface in the map
84 auto vMaterial = m_volumeMaterialMap.find(volume.geometryId());
85 if (vMaterial != m_volumeMaterialMap.end()) {
86 volume.assignVolumeMaterial(vMaterial->second);
87 }
88 }
89
91 Acts::TrackingGeometryMaterial materialMaps() const {
92 return {m_surfaceMaterialMap, m_volumeMaterialMap};
93 }
