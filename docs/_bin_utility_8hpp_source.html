152 BinUtility& operator+=(const BinUtility& gbu) {
153 const std::vector<BinningData>& bData = gbu.binningData();
154
155 m_transform = m_transform * gbu.transform();
156 m_itransform = m_transform.inverse();
157 if (m_binningData.size() + bData.size() > 3) {
158 throw std::runtime_error{"BinUtility does not support dim > 3"};
159 }
160 m_binningData.insert(m_binningData.end(), bData.begin(), bData.end());
161 return (*this);
162 }

190 std::array<std::size_t, 3> binTriple(const Vector3& position) const {
192 const Vector3 bPosition = m_itransform * position;
193 // get the dimension
194 std::size_t mdim = m_binningData.size();
196 std::size_t bin0 = m_binningData[0].searchGlobal(bPosition);
197 std::size_t bin1 = mdim > 1 ? m_binningData[1].searchGlobal(bPosition) : 0;
198 std::size_t bin2 = mdim > 2 ? m_binningData[2].searchGlobal(bPosition) : 0;
200 return {{bin0, bin1, bin2}};
201 }
202
209 std::size_t bin(const Vector3& position, std::size_t ba = 0) const {
210 if (ba >= m_binningData.size()) {
211 return 0;
212 }
213 std::size_t bEval = m_binningData[ba].searchGlobal(m_itransform * position);
214 return bEval;
215 }

226 int nextDirection(const Vector3& position, const Vector3& direction,
227 std::size_t ba = 0) const {

245 std::size_t bin(const Vector2& lposition, std::size_t ba = 0) const {
246 if (ba >= m_binningData.size()) {
247 return 0;
248 }
249 return m_binningData[ba].searchLocal(lposition);
250 }
251
255 bool inside(const Vector3& position) const {
257 const Vector3& bPosition = m_itransform * position;
258 // loop and break
259 for (auto& bData : m_binningData) {
260 if (!(bData.inside(bPosition))) {
261 return false;
262 }
263 }
264 // survived all the checks
265 return true;
266 }
267
270 std::size_t dimensions() const { return m_binningData.size(); }

277 std::size_t max(std::size_t ba = 0) const {

318 std::size_t serialize(const std::array<std::size_t, 3>& bin) const {
319 std::size_t serializedBin = bin[0];
320 if (m_binningData.size() == 2) {
321 serializedBin += bin[1] * m_binningData[0].bins();
322 } else if (m_binningData.size() == 3) {
323 serializedBin +=
324 (bin[1] * m_binningData[0].bins() * bin[2] * m_binningData[1].bins());
325 }
326 return serializedBin;
327 }
