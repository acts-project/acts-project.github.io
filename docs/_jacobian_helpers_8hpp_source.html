26inline ActsMatrix<3, 2> sphericalToFreeDirectionJacobian(
27 const Vector3& direction) {
28 auto [cosPhi, sinPhi, cosTheta, sinTheta] =
29 VectorHelpers::evaluateTrigonomics(direction);
30
31 // clang-format off
32 ActsMatrix<3, 2> jacobian;
33 jacobian <<
34 -direction.y(), cosTheta * cosPhi,
35 direction.x(), cosTheta * sinPhi,
36 0, -sinTheta;
37 // clang-format on
38
39 return jacobian;

52inline ActsMatrix<2, 3> freeToSphericalDirectionJacobian(
53 const Vector3& direction) {
54 auto [cosPhi, sinPhi, cosTheta, sinTheta] =
55 VectorHelpers::evaluateTrigonomics(direction);
56 double invSinTheta = 1. / sinTheta;
57
58 // clang-format off
59 ActsMatrix<2, 3> jacobian;
60 jacobian <<
61 -sinPhi * invSinTheta, cosPhi * invSinTheta, 0,
62 cosPhi * cosTheta, sinPhi * cosTheta, -sinTheta;
63 // clang-format on
64
65 return jacobian;
66}
