69 template <typename generator_t>
70 Acts::Result<SimulationResult> simulate(
71 const Acts::GeometryContext &geoCtx,
72 const Acts::MagneticFieldContext &magCtx, generator_t &generator,
73 const Particle &particle) const {
74 // propagator-related additional types
75 using Actor = detail::SimulationActor<generator_t, decay_t, interactions_t,
76 hit_surface_selector_t>;
77 using Result = typename Actor::result_type;
78 using ActorList = Acts::ActorList<Actor>;
79 using PropagatorOptions =
80 typename propagator_t::template Options<ActorList>;
81
82 // Construct per-call options.
83 PropagatorOptions options(geoCtx, magCtx);
84 options.stepping.maxStepSize = maxStepSize;
85 options.pathLimit = pathLimit;
86 // setup the interactor as part of the propagator options
87 auto &actor = options.actorList.template get<Actor>();
88 actor.generator = &generator;
89 actor.decay = decay;
90 actor.interactions = interactions;
91 actor.selectHitSurface = selectHitSurface;
92 actor.initialParticle = particle;
93
94 if (particle.hasReferenceSurface()) {
95 auto result = propagator.propagate(
96 particle.boundParameters(geoCtx).value(), options);
97 if (!result.ok()) {
98 return result.error();
99 }
100 auto &value = result.value().template get<Result>();
101 return std::move(value);
102 }
103
104 auto result =
105 propagator.propagate(particle.curvilinearParameters(), options);
106 if (!result.ok()) {
107 return result.error();
108 }
109 auto &value = result.value().template get<Result>();
110 return std::move(value);
111 }
112};

190 template <typename generator_t, typename input_particles_t,
191 typename output_particles_t, typename hits_t>
192 Acts::Result<std::vector<FailedParticle>> simulate(
193 const Acts::GeometryContext &geoCtx,
194 const Acts::MagneticFieldContext &magCtx, generator_t &generator,
195 const input_particles_t &inputParticles,
196 output_particles_t &simulatedParticlesInitial,
197 output_particles_t &simulatedParticlesFinal, hits_t &hits) const {
198 assert(
199 (simulatedParticlesInitial.size() == simulatedParticlesFinal.size()) &&
200 "Inconsistent initial sizes of the simulated particle containers");
201
202 using SingleParticleSimulationResult = Acts::Result<SimulationResult>;
203
204 std::vector<FailedParticle> failedParticles;
205
206 for (const Particle &inputParticle : inputParticles) {
207 // only consider simulatable particles
208 if (!selectParticle(inputParticle)) {
209 continue;
210 }
211 // required to allow correct particle id numbering for secondaries later
212 if ((inputParticle.particleId().generation() != 0u) ||
213 (inputParticle.particleId().subParticle() != 0u)) {
214 return detail::SimulationError::InvalidInputParticleId;
215 }
216
217 // Do a *depth-first* simulation of the particle and its secondaries,
218 // i.e. we simulate all secondaries, tertiaries, ... before simulating
219 // the next primary particle. Use the end of the output container as
220 // a queue to store particles that should be simulated.
221 //
222 // WARNING the initial particle state output container will be modified
223 // during iteration. New secondaries are added to and failed
224 // particles might be removed. To avoid issues, access must always
225 // occur via indices.
226 auto iinitial = simulatedParticlesInitial.size();
227 simulatedParticlesInitial.push_back(inputParticle);
228 for (; iinitial < simulatedParticlesInitial.size(); ++iinitial) {
229 const auto &initialParticle = simulatedParticlesInitial[iinitial];
230
231 // only simulatable particles are pushed to the container and here we
232 // only need to switch between charged/neutral.
233 SingleParticleSimulationResult result =
234 SingleParticleSimulationResult::success({});
235 if (initialParticle.charge() != 0.) {
236 result = charged.simulate(geoCtx, magCtx, generator, initialParticle);
237 } else {
238 result = neutral.simulate(geoCtx, magCtx, generator, initialParticle);
239 }
240
241 if (!result.ok()) {
242 // remove particle from output container since it was not simulated.
243 simulatedParticlesInitial.erase(
244 std::next(simulatedParticlesInitial.begin(), iinitial));
245 // record the particle as failed
246 failedParticles.push_back({initialParticle, result.error()});
247 continue;
248 }
249
250 assert(result->particle.particleId() == initialParticle.particleId() &&
251 "Particle id must not change during simulation");
252
253 copyOutputs(result.value(), simulatedParticlesInitial,
254 simulatedParticlesFinal, hits);
255 // since physics processes are independent, there can be particle id
256 // collisions within the generated secondaries. they can be resolved by
257 // renumbering within each sub-particle generation. this must happen
258 // before the particle is simulated since the particle id is used to
259 // associate generated hits back to the particle.
260 renumberTailParticleIds(simulatedParticlesInitial, iinitial);
261 }
262 }
263
264 assert(
265 (simulatedParticlesInitial.size() == simulatedParticlesFinal.size()) &&
266 "Inconsistent final sizes of the simulated particle containers");
267
268 // the overall function call succeeded, i.e. no fatal errors occurred.
269 // yet, there might have been some particle for which the propagation
270 // failed. thus, the successful result contains a list of failed particles.
271 // sounds a bit weird, but that is the way it is.
272 return failedParticles;
273 }
274
275 private:
277 bool selectParticle(const Particle &particle) const {
278 if (particle.charge() != 0.) {
279 return selectCharged(particle);
280 } else {
281 return selectNeutral(particle);
282 }
283 }

289 template <typename particles_t, typename hits_t>
290 void copyOutputs(const SimulationResult &result,
291 particles_t &particlesInitial, particles_t &particlesFinal,
292 hits_t &hits) const {
293 // initial particle state was already pushed to the container before
294 // store final particle state at the end of the simulation
295 particlesFinal.push_back(result.particle);
296 std::copy(result.hits.begin(), result.hits.end(), std::back_inserter(hits));
297
298 // move generated secondaries that should be simulated to the output
299 std::copy_if(
300 result.generatedParticles.begin(), result.generatedParticles.end(),
301 std::back_inserter(particlesInitial),
302 [this](const Particle &particle) { return selectParticle(particle); });
303 }

319 template <typename particles_t>
320 static void renumberTailParticleIds(particles_t &particles,
321 std::size_t lastValid) {
322 // iterate over adjacent pairs; potentially modify the second element.
323 // assume e.g. a primary particle 2 with generation=subparticle=0 that
324 // generates two secondaries during simulation. we have the following
325 // ids (decoded as vertex|particle|generation|subparticle)
326 //
327 // v|2|0|0, v|2|1|0, v|2|1|0
328 //
329 // where v represents the vertex numbers. this will be renumbered to
330 //
331 // v|2|0|0, v|2|1|0, v|2|1|1
332 //
333 // if each secondary then generates two tertiaries we could have e.g.
334 //
335 // v|2|0|0, v|2|1|0, v|2|1|1, v|2|2|0, v|2|2|1, v|2|2|0, v|2|2|1
336 //
337 // which would then be renumbered to
338 //
339 // v|2|0|0, v|2|1|0, v|2|1|1, v|2|2|0, v|2|2|1, v|2|2|2, v|2|2|3
340 //
341 for (auto j = lastValid; (j + 1u) < particles.size(); ++j) {
342 const auto prevId = particles[j].particleId();
343 auto currId = particles[j + 1u].particleId();
344 // NOTE primary/secondary vertex and particle are assumed to be equal
345 // only renumber within one generation
346 if (prevId.generation() != currId.generation()) {
347 continue;
348 }
349 // ensure the sub-particle is strictly monotonic within a generation
350 if (prevId.subParticle() < currId.subParticle()) {
351 continue;
352 }
353 // sub-particle numbering must be non-zero
354 currId = currId.withSubParticle(prevId.subParticle() + 1u);
355 particles[j + 1u] = particles[j + 1u].withParticleId(currId);
356 }
357 }
