20// This object can be iterated to produce up to two sequences of integer
21// indices, corresponding to the half-open integer ranges [begin1, end1[ and
22// [begin2, end2[.
23//
24// The goal is to emulate the effect of enumerating a range of neighbor
25// indices on an axis (which may go out of bounds and wrap around since we
26// have AxisBoundaryType::Closed), inserting them into an std::vector, and
27// discarding duplicates, without paying the price of duplicate removal
28// and dynamic memory allocation in hot magnetic field interpolation code.
29//
30class NeighborHoodIndices {
31 public:
32 NeighborHoodIndices() = default;
33
34 NeighborHoodIndices(std::size_t begin, std::size_t end)
35 : m_begin1(begin), m_end1(end), m_begin2(end), m_end2(end) {}
36
37 NeighborHoodIndices(std::size_t begin1, std::size_t end1, std::size_t begin2,
38 std::size_t end2)
39 : m_begin1(begin1), m_end1(end1), m_begin2(begin2), m_end2(end2) {}
40
41 class iterator {
42 public:
43 iterator() = default;
44
45 // Specialized constructor for end() iterator
46 explicit iterator(std::size_t current)
47 : m_current(current), m_wrapped(true) {}
48
49 iterator(std::size_t begin1, std::size_t end1, std::size_t begin2)
50 : m_current(begin1),
51 m_end1(end1),
52 m_begin2(begin2),
53 m_wrapped(begin1 == begin2) {}
54
55 std::size_t operator*() const { return m_current; }
56
57 iterator& operator++() {
58 ++m_current;
59 if (m_current == m_end1) {
60 m_current = m_begin2;
61 m_wrapped = true;
62 }
63 return *this;
64 }
65
66 bool operator==(const iterator& it) const {
67 return (m_current == it.m_current) && (m_wrapped == it.m_wrapped);
68 }
69
70 private:
71 std::size_t m_current = 0, m_end1 = 0, m_begin2 = 0;
72 bool m_wrapped = false;
73 };
74
75 iterator begin() const { return iterator(m_begin1, m_end1, m_begin2); }
76
77 iterator end() const { return iterator(m_end2); }
78
79 // Number of indices that will be produced if this sequence is iterated
80 std::size_t size() const { return (m_end1 - m_begin1) + (m_end2 - m_begin2); }
81
82 // Collect the sequence of indices into an std::vector
83 std::vector<std::size_t> collect() const {
84 std::vector<std::size_t> result;
85 result.reserve(this->size());
86 for (std::size_t idx : *this) {
87 result.push_back(idx);
88 }
89 return result;
90 }
91
92 private:
93 std::size_t m_begin1 = 0, m_end1 = 0, m_begin2 = 0, m_end2 = 0;
94};

203 {
204 if (idx <= 0 || idx >= (getNBins() + 1)) {
205 return NeighborHoodIndices();
206 }
207 constexpr int min = 1;

209 const int itmin = std::clamp(static_cast<int>(idx) + sizes.first, min, max);

224 NeighborHoodIndices neighborHoodIndices(std::size_t idx,
225 std::pair<int, int> sizes = {-1,
226 1}) const
227 requires(bdt == AxisBoundaryType::Closed)
228 {
229 // Handle invalid indices
230 if (idx <= 0 || idx >= (getNBins() + 1)) {
231 return NeighborHoodIndices();
232 }
233
234 // Handle corner case where user requests more neighbours than the number
235 // of bins on the axis. All bins are returned in this case.
236
237 const int max = getNBins();
238 sizes.first = std::clamp(sizes.first, -max, max);
239 sizes.second = std::clamp(sizes.second, -max, max);
240 if (std::abs(sizes.first - sizes.second) >= max) {
241 sizes.first = 1 - idx;
242 sizes.second = max - idx;
243 }
244
245 // If the entire index range is not covered, we must wrap the range of
246 // targeted neighbor indices into the range of valid bin indices. This may
247 // split the range of neighbor indices in two parts:
248 //
249 // Before wraparound - [ XXXXX]XXX
250 // After wraparound - [ XXXX XXXX ]
251 //
252 const int itmin = idx + sizes.first;
253 const int itmax = idx + sizes.second;
254 const std::size_t itfirst = wrapBin(itmin);
255 const std::size_t itlast = wrapBin(itmax);
256 if (itfirst <= itlast) {
257 return NeighborHoodIndices(itfirst, itlast + 1);
258 } else {
259 return NeighborHoodIndices(itfirst, max + 1, 1, itlast + 1);
260 }
261 }

368 double getMin() const override { return m_min; }
369

374

384
387 std::vector<double> getBinEdges() const override {
388 std::vector<double> binEdges;
389 for (std::size_t i = 1; i <= m_bins; i++) {
390 binEdges.push_back(getBinLowerBound(i));
391 }
392 binEdges.push_back(getBinUpperBound(m_bins));
393 return binEdges;
394 }

404 protected:

407 private:

413 double m_width{};
