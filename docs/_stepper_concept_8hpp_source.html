25template <typename Stepper, typename State = typename Stepper::State>
26concept CommonStepper = requires {
27 typename Stepper::State;
28 typename Stepper::Jacobian;
29 typename Stepper::Covariance;
30 typename Stepper::BoundState;
31
32 requires requires(const Stepper& s, State& t) {
33 { s.transportCovarianceToCurvilinear(t) } -> std::same_as<void>;
34
35 requires requires(
36 const std::tuple_element_t<0, typename Stepper::BoundState>& par) {
37 { s.initialize(t, par) } -> std::same_as<void>;
38 };
39
40 requires requires(const Surface& sf, bool b,
41 const FreeToBoundCorrection& corr) {
42 {
43 s.boundState(t, sf, b, corr)
44 } -> std::same_as<Result<typename Stepper::BoundState>>;
45 { s.transportCovarianceToBound(t, sf, corr) } -> std::same_as<void>;
46 };
47
48 requires requires(bool b) {
49 {
50 s.curvilinearState(t, b)
51 } -> std::same_as<typename Stepper::BoundState>;
52 };
53
54 requires requires(const Surface& sf, std::uint8_t ui, Direction d,
55 const BoundaryTolerance& bt, double sc,
56 ConstrainedStep::Type st, const Logger& l) {
57 { s.updateSurfaceStatus(t, sf, ui, d, bt, sc, st, l) };
58 };
59
60 requires requires(const ConstrainedStep::Type st) {
61 { s.releaseStepSize(t, st) } -> std::same_as<void>;
62
63 requires requires(double d) {
64 { s.updateStepSize(t, d, st) } -> std::same_as<void>;
65 };
66 };
67 };
68
69 requires requires(const Stepper& s, const State& t) {
70 { s.position(t) } -> std::same_as<Vector3>;
71 { s.direction(t) } -> std::same_as<Vector3>;
72 { s.qOverP(t) } -> std::same_as<double>;
73 { s.absoluteMomentum(t) } -> std::same_as<double>;
74 { s.momentum(t) } -> std::same_as<Vector3>;
75 { s.charge(t) } -> std::same_as<double>;
76 { s.time(t) } -> std::same_as<double>;
77 { s.outputStepSize(t) } -> std::same_as<std::string>;
78
79 requires requires(const ConstrainedStep::Type st) {
80 { s.getStepSize(t, st) } -> std::same_as<double>;
81 };
82 };
83};

86template <typename Stepper, typename State = typename Stepper::State>
87concept SingleStepper =
88 CommonStepper<Stepper, State> && requires(const Stepper& s, State& t) {
89 requires requires(const FreeVector& fv, const BoundVector& bv,
90 const BoundSquareMatrix& bm, const Surface& sf) {
91 { s.update(t, fv, bv, bm, sf) } -> std::same_as<void>;
92 };
93
94 requires requires(const Vector3& v1, const Vector3& v2, double d1,
95 double d2) {
96 { s.update(t, v1, v2, d1, d2) } -> std::same_as<void>;
97 { s.getField(t, v1) } -> std::same_as<Result<Vector3>>;
98 };
99 };

102template <typename Stepper, typename State = typename Stepper::State>
103concept MultiStepper = CommonStepper<Stepper, State> && requires {
104 // TODO for now we do not check if the ComponentProxy does fulfill a concept
105 typename Stepper::ComponentProxy;
106
107 // TODO for now we do not check if the ConstComponentProxy does fulfill a
108 // concept
109 typename Stepper::ConstComponentProxy;
110
111 requires requires(const Stepper& s, State& t) {
112 { s.numberComponents(t) } -> std::same_as<std::size_t>;
113 { s.clearComponents(t) } -> std::same_as<void>;
114 { s.removeMissedComponents(t) } -> std::same_as<void>;
115 };
116};
