117 explicit PropagatorOptions(const PropagatorPlainOptions& pOptions)

126 explicit operator PropagatorPlainOptions() const {

129 static_cast<const PurePropagatorPlainOptions&>(*this);
130 pOptions.stepping = static_cast<const StepperPlainOptions&>(stepping);
131 pOptions.navigation = static_cast<const NavigatorPlainOptions&>(navigation);

141 template <typename extended_actor_list_t>
142 PropagatorOptions<stepper_options_t, navigator_options_t,

145 PropagatorOptions<stepper_options_t, navigator_options_t,
146 extended_actor_list_t>
147 eoptions(geoContext, magFieldContext);
148
149 // Copy the base options
150 static_cast<PurePropagatorPlainOptions&>(eoptions) =
151 static_cast<const PurePropagatorPlainOptions&>(*this);
152
153 // Stepper / Navigator options
154 eoptions.stepping = stepping;
155 eoptions.navigation = navigation;
156
157 // Action / Abort list
158 eoptions.actorList = extendedActorList;
159
160 // And return the options
161 return eoptions;
162 }
