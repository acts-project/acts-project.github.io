118 constexpr static bool pathLengthOrder(

131 constexpr static bool closestOrder(
132 const Intersection& aIntersection,
133 const Intersection& bIntersection) noexcept {
134 using enum IntersectionStatus;
135
136 if ((aIntersection.status() == unreachable) &&
137 (bIntersection.status() != unreachable)) {
138 return false;
139 }
140 if ((aIntersection.status() != unreachable) &&
141 (bIntersection.status() == unreachable)) {
142 return true;
143 }
144 // both are reachable or onSurface now
145 auto a = aIntersection.pathLength();
146 auto b = bIntersection.pathLength();
147 return std::abs(a) < std::abs(b);
148 }
149
155 constexpr static bool closestForwardOrder(
156 const Intersection& aIntersection,
157 const Intersection& bIntersection) noexcept {
158 auto a = aIntersection.pathLength();
159 auto b = bIntersection.pathLength();
160 return std::signbit(a) == std::signbit(b) ? std::abs(a) < std::abs(b)
161 : a > b;
162 }
163
164 private:
166 std::array<double, DIM> m_position{};
168 double m_pathLength = std::numeric_limits<double>::infinity();
170 IntersectionStatus m_status = IntersectionStatus::unreachable;
171
172 constexpr Intersection() noexcept = default;
173};
