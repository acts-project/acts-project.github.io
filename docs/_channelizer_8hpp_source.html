26 public:
37 Acts::Result<std::vector<Segmentizer::ChannelSegment>> channelize(
38 const Hit& hit, const Acts::Surface& surface,
39 const Acts::GeometryContext& gctx, const Acts::Vector3& driftDir,
40 const Acts::BinUtility& segmentation, double thickness) const {
41 auto driftedSegment = m_surfaceDrift.toReadout(
42 gctx, surface, thickness, hit.position(), hit.direction(), driftDir);
43
44 auto maskedSegmentRes = m_surfaceMask.apply(surface, driftedSegment);
45
46 if (!maskedSegmentRes.ok()) {
47 return maskedSegmentRes.error();
48 }
49
50 // Now Channelize
51 auto segments =
52 m_segmentizer.segments(gctx, surface, segmentation, *maskedSegmentRes);
53
54 // Go from 2D-path to 3D-path by applying thickness
55 const auto path2D = std::accumulate(
56 segments.begin(), segments.end(), 0.0,
57 [](double sum, const auto& seg) { return sum + seg.activation; });
58
59 for (auto& seg : segments) {
60 auto r = path2D != 0.0 ? (seg.activation / path2D) : 1.0;
61 auto segThickness = r * thickness;
62
63 seg.activation = std::hypot(segThickness, seg.activation);
64 }
65
66 return segments;
67 }
68};
