60template <TrackProxyConcept track_proxy_t>
61Result<typename track_proxy_t::ConstTrackStateProxy> findFirstMeasurementState(
62 const track_proxy_t &track) {
63 using TrackStateProxy = typename track_proxy_t::ConstTrackStateProxy;
64
65 // TODO specialize if track is forward linked
66
67 std::optional<TrackStateProxy> firstMeasurementOpt;
68
69 for (const auto &trackState : track.trackStatesReversed()) {
70 bool isMeasurement =
71 trackState.typeFlags().test(TrackStateFlag::MeasurementFlag);
72 bool isOutlier = trackState.typeFlags().test(TrackStateFlag::OutlierFlag);
73
74 if (isMeasurement && !isOutlier) {
75 firstMeasurementOpt = trackState;
76 }
77 }
78
79 if (firstMeasurementOpt) {
80 return Result<TrackStateProxy>::success(*firstMeasurementOpt);
81 }
82
83 return Result<TrackStateProxy>::failure(
84 TrackExtrapolationError::CompatibleTrackStateNotFound);
85}

90template <TrackProxyConcept track_proxy_t>
91Result<typename track_proxy_t::ConstTrackStateProxy> findLastMeasurementState(
92 const track_proxy_t &track) {
93 using TrackStateProxy = typename track_proxy_t::ConstTrackStateProxy;
94
95 for (const auto &trackState : track.trackStatesReversed()) {
96 bool isMeasurement =
97 trackState.typeFlags().test(TrackStateFlag::MeasurementFlag);
98 bool isOutlier = trackState.typeFlags().test(TrackStateFlag::OutlierFlag);
99
100 if (isMeasurement && !isOutlier) {
101 return TrackStateProxy{trackState};
102 }
103 }
104
105 return Result<TrackStateProxy>::failure(
106 TrackExtrapolationError::CompatibleTrackStateNotFound);

120template <TrackProxyConcept track_proxy_t,
121 typename smoother_t = GainMatrixSmoother>
122Result<void> smoothTrack(
123 const GeometryContext &geoContext, track_proxy_t &track,
124 const Logger &logger = *getDefaultLogger("TrackSmoother", Logging::INFO),
125 smoother_t smoother = GainMatrixSmoother()) {
126 auto &trackContainer = track.container();
127 auto &trackStateContainer = trackContainer.trackStateContainer();
128
129 auto last = findLastMeasurementState(track);
130 if (!last.ok()) {
131 ACTS_DEBUG("no last track state found");
132 return last.error();
133 }
134
135 auto smoothingResult =
136 smoother(geoContext, trackStateContainer, last->index(), logger);
137
138 if (!smoothingResult.ok()) {
139 ACTS_DEBUG("Smoothing track " << track.index() << " failed with error "
140 << smoothingResult.error());
141 return smoothingResult.error();
142 }
143
144 return Result<void>::success();
145}

156template <TrackContainerFrontend track_container_t>
157Result<void> smoothTracks(
158 const GeometryContext &geoContext, const track_container_t &trackContainer,
159 const Logger &logger = *getDefaultLogger("TrackSmoother", Logging::INFO)) {
160 Result<void> result = Result<void>::success();
161
162 for (const auto &track : trackContainer) {
163 auto smoothingResult = smoothTrack(geoContext, track, logger);
164
165 // Only keep the first error
166 if (!smoothingResult.ok() && result.ok()) {
167 result = smoothingResult.error();
168 }
169 }
170
171 return result;

186template <TrackProxyConcept track_proxy_t>
187Result<std::pair<typename track_proxy_t::ConstTrackStateProxy, double>>
188findTrackStateForExtrapolation(
189 const GeometryContext &geoContext, const track_proxy_t &track,
190 const Surface &referenceSurface, TrackExtrapolationStrategy strategy,
191 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
192 Logging::INFO)) {
193 using TrackStateProxy = typename track_proxy_t::ConstTrackStateProxy;
194
195 auto intersect = [&](const TrackStateProxy &state) -> Intersection3D {
196 assert(state.hasSmoothed() || state.hasFiltered());
197
198 FreeVector freeVector;
199 if (state.hasSmoothed()) {
200 freeVector = MultiTrajectoryHelpers::freeSmoothed(geoContext, state);
201 } else {
202 freeVector = MultiTrajectoryHelpers::freeFiltered(geoContext, state);
203 }
204
205 return referenceSurface
206 .intersect(geoContext, freeVector.template segment<3>(eFreePos0),
207 freeVector.template segment<3>(eFreeDir0),
208 BoundaryTolerance::None(), s_onSurfaceTolerance)
209 .closest();
210 };
211
212 switch (strategy) {
213 case TrackExtrapolationStrategy::first: {
214 ACTS_VERBOSE("looking for first track state");
215
216 auto first = findFirstMeasurementState(track);
217 if (!first.ok()) {
218 ACTS_DEBUG("no first track state found");
219 return first.error();
220 }
221
222 Intersection3D intersection = intersect(*first);
223 if (!intersection.isValid()) {
224 ACTS_DEBUG("no intersection found");
225 return Result<std::pair<TrackStateProxy, double>>::failure(
226 TrackExtrapolationError::ReferenceSurfaceUnreachable);
227 }
228
229 ACTS_VERBOSE("found intersection at " << intersection.pathLength());
230 return std::pair(*first, intersection.pathLength());
231 }
232
233 case TrackExtrapolationStrategy::last: {
234 ACTS_VERBOSE("looking for last track state");
235
236 auto last = findLastMeasurementState(track);
237 if (!last.ok()) {
238 ACTS_DEBUG("no last track state found");
239 return last.error();
240 }
241
242 Intersection3D intersection = intersect(*last);
243 if (!intersection.isValid()) {
244 ACTS_DEBUG("no intersection found");
245 return Result<std::pair<TrackStateProxy, double>>::failure(
246 TrackExtrapolationError::ReferenceSurfaceUnreachable);
247 }
248
249 ACTS_VERBOSE("found intersection at " << intersection.pathLength());
250 return std::pair(*last, intersection.pathLength());
251 }
252
253 case TrackExtrapolationStrategy::firstOrLast: {
254 ACTS_VERBOSE("looking for first or last track state");
255
256 auto first = findFirstMeasurementState(track);
257 if (!first.ok()) {
258 ACTS_DEBUG("no first track state found");
259 return first.error();
260 }
261
262 auto last = findLastMeasurementState(track);
263 if (!last.ok()) {
264 ACTS_DEBUG("no last track state found");
265 return last.error();
266 }
267
268 Intersection3D intersectionFirst = intersect(*first);
269 Intersection3D intersectionLast = intersect(*last);
270
271 double absDistanceFirst = std::abs(intersectionFirst.pathLength());
272 double absDistanceLast = std::abs(intersectionLast.pathLength());
273
274 if (intersectionFirst.isValid() && absDistanceFirst <= absDistanceLast) {
275 ACTS_VERBOSE("using first track state with intersection at "
276 << intersectionFirst.pathLength());
277 return std::pair(*first, intersectionFirst.pathLength());
278 }
279
280 if (intersectionLast.isValid() && absDistanceLast <= absDistanceFirst) {
281 ACTS_VERBOSE("using last track state with intersection at "
282 << intersectionLast.pathLength());
283 return std::pair(*last, intersectionLast.pathLength());
284 }
285
286 ACTS_DEBUG("no intersection found");
287 return Result<std::pair<TrackStateProxy, double>>::failure(
288 TrackExtrapolationError::ReferenceSurfaceUnreachable);
289 }
290 }
291
292 // unreachable
293 return Result<std::pair<TrackStateProxy, double>>::failure(
294 TrackExtrapolationError::CompatibleTrackStateNotFound);
295}

311template <TrackProxyConcept track_proxy_t, typename propagator_t,
312 typename propagator_options_t>
313Result<void> extrapolateTrackToReferenceSurface(
314 track_proxy_t &track, const Surface &referenceSurface,
315 const propagator_t &propagator, propagator_options_t options,
316 TrackExtrapolationStrategy strategy,
317 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
318 Logging::INFO)) {
319 auto findResult = findTrackStateForExtrapolation(
320 options.geoContext, track, referenceSurface, strategy, logger);
321
322 if (!findResult.ok()) {
323 ACTS_DEBUG("failed to find track state for extrapolation");
324 return findResult.error();
325 }
326
327 auto &[trackState, distance] = *findResult;
328
329 options.direction = Direction::fromScalarZeroAsPositive(distance);
330
331 BoundTrackParameters parameters = track.createParametersFromState(trackState);
332 ACTS_VERBOSE("extrapolating track to reference surface at distance "
333 << distance << " with direction " << options.direction
334 << " with starting parameters " << parameters);
335
336 auto propagateResult =
337 propagator.template propagate<BoundTrackParameters, propagator_options_t,
338 ForcedSurfaceReached>(
339 parameters, referenceSurface, options);
340
341 if (!propagateResult.ok()) {
342 ACTS_DEBUG("failed to extrapolate track: " << propagateResult.error());
343 return propagateResult.error();
344 }
345
346 track.setReferenceSurface(referenceSurface.getSharedPtr());
347 track.parameters() = propagateResult->endParameters.value().parameters();
348 track.covariance() =
349 propagateResult->endParameters.value().covariance().value();
350
351 return Result<void>::success();
352}

368template <TrackContainerFrontend track_container_t, typename propagator_t,
369 typename propagator_options_t>
370Result<void> extrapolateTracksToReferenceSurface(
371 const track_container_t &trackContainer, const Surface &referenceSurface,
372 const propagator_t &propagator, propagator_options_t options,
373 TrackExtrapolationStrategy strategy,
374 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
375 Logging::INFO)) {
376 Result<void> result = Result<void>::success();
377
378 for (const auto &track : trackContainer) {
379 auto extrapolateResult = extrapolateTrackToReferenceSurface(
380 track, referenceSurface, propagator, options, strategy, logger);
381
382 // Only keep the first error
383 if (!extrapolateResult.ok() && result.ok()) {
384 result = extrapolateResult.error();
385 }
386 }
387
388 return result;
389}

395template <TrackProxyConcept track_proxy_t>
396void calculateTrackQuantities(track_proxy_t track)
397 requires(!track_proxy_t::ReadOnly)
398{
399 track.chi2() = 0;
400 track.nDoF() = 0;
401
402 track.nHoles() = 0;
403 track.nMeasurements() = 0;
404 track.nSharedHits() = 0;
405 track.nOutliers() = 0;
406
407 for (const auto &trackState : track.trackStatesReversed()) {
408 ConstTrackStateType typeFlags = trackState.typeFlags();
409
410 if (typeFlags.test(Acts::TrackStateFlag::HoleFlag)) {
411 track.nHoles()++;
412 } else if (typeFlags.test(Acts::TrackStateFlag::OutlierFlag)) {
413 track.nOutliers()++;
414 } else if (typeFlags.test(Acts::TrackStateFlag::MeasurementFlag)) {
415 if (typeFlags.test(Acts::TrackStateFlag::SharedHitFlag)) {
416 track.nSharedHits()++;
417 }
418 track.nMeasurements()++;
419 track.chi2() += trackState.chi2();
420 track.nDoF() += trackState.calibratedSize();
421 }
422 }
423}

432template <TrackProxyConcept track_proxy_t>
433void trimTrackFront(track_proxy_t track, bool trimHoles, bool trimOutliers,
434 bool trimMaterial, bool trimOtherNoneMeasurement)
435 requires(!track_proxy_t::ReadOnly)
436{
437 using TrackStateProxy = typename track_proxy_t::TrackStateProxy;
438
439 // TODO specialize if track is forward linked
440
441 std::optional<TrackStateProxy> front;
442
443 for (TrackStateProxy trackState : track.trackStatesReversed()) {
444 TrackStateType typeFlags = trackState.typeFlags();
445 bool isHole = typeFlags.test(TrackStateFlag::HoleFlag);
446 bool isOutlier = typeFlags.test(TrackStateFlag::OutlierFlag);
447 bool isMaterial = typeFlags.test(TrackStateFlag::MaterialFlag) &&
448 !typeFlags.test(TrackStateFlag::MeasurementFlag);
449 bool isOtherNoneMeasurement =
450 !typeFlags.test(TrackStateFlag::MeasurementFlag) && !isHole &&
451 !isOutlier && !isMaterial;
452 if (trimHoles && isHole) {
453 continue;
454 }
455 if (trimOutliers && isOutlier) {
456 continue;
457 }
458 if (trimMaterial && isMaterial) {
459 continue;
460 }
461 if (trimOtherNoneMeasurement && isOtherNoneMeasurement) {
462 continue;
463 }
464
465 front = trackState;
466 }
467
468 if (front.has_value()) {
469 front.value().previous() = TrackStateProxy::kInvalid;
470 }
471}

544template <std::size_t nMeasurementDim,
545 TrackStateProxyConcept track_state_proxy_t>
546std::pair<ActsVector<nMeasurementDim>, ActsSquareMatrix<nMeasurementDim>>
547calculatePredictedResidual(track_state_proxy_t trackState) {
548 using MeasurementVector = ActsVector<nMeasurementDim>;
549 using MeasurementMatrix = ActsSquareMatrix<nMeasurementDim>;
550
551 if (!trackState.hasPredicted()) {
552 throw std::invalid_argument("track state has no predicted parameters");
553 }
554 if (!trackState.hasCalibrated()) {
555 throw std::invalid_argument("track state has no calibrated parameters");
556 }
557
558 auto subspaceHelper =
559 trackState.template projectorSubspaceHelper<nMeasurementDim>();
560
561 auto measurement = trackState.template calibrated<nMeasurementDim>();
562 auto measurementCovariance =
563 trackState.template calibratedCovariance<nMeasurementDim>();
564 MeasurementVector predicted =
565 subspaceHelper.projectVector(trackState.predicted());
566 MeasurementMatrix predictedCovariance =
567 subspaceHelper.projectMatrix(trackState.predictedCovariance());
568
569 MeasurementVector residual = measurement - predicted;
570 MeasurementMatrix residualCovariance =
571 measurementCovariance + predictedCovariance;
572
573 return {residual, residualCovariance};
574}

654template <TrackStateProxyConcept track_state_proxy_t>
655double calculatePredictedChi2(track_state_proxy_t trackState) {
656 if (!trackState.hasPredicted()) {
657 throw std::invalid_argument("track state has no predicted parameters");
658 }
659 if (!trackState.hasCalibrated()) {
660 throw std::invalid_argument("track state has no calibrated parameters");
661 }
662
663 return visit_measurement(
664 trackState.calibratedSize(),
665 [&]<std::size_t measdim>(
666 std::integral_constant<std::size_t, measdim>) -> double {
667 auto [residual, residualCovariance] =
668 calculatePredictedResidual<measdim>(trackState);
669
670 return (residual.transpose() * residualCovariance.inverse() * residual)
671 .eval()(0, 0);
672 });
673}

731template <TrackStateProxyConcept track_state_proxy_t>
732std::pair<BoundVector, BoundMatrix> calculateUnbiasedParametersCovariance(
733 track_state_proxy_t trackState) {
734 if (!trackState.hasSmoothed()) {
735 throw std::invalid_argument("track state has no smoothed parameters");
736 }
737 if (!trackState.hasCalibrated()) {
738 throw std::invalid_argument("track state has no calibrated parameters");
739 }
740
741 return visit_measurement(
742 trackState.calibratedSize(),
743 [&]<std::size_t measdim>(std::integral_constant<std::size_t, measdim>) {
744 FixedBoundSubspaceHelper<measdim> subspaceHelper =
745 trackState.template projectorSubspaceHelper<measdim>();
746
747 // TODO use subspace helper for projection instead
748 auto H = subspaceHelper.projector();
749 auto s = trackState.smoothed();
750 auto C = trackState.smoothedCovariance();
751 auto m = trackState.template calibrated<measdim>();
752 auto V = trackState.template calibratedCovariance<measdim>();
753 auto K =
754 (C * H.transpose() * (H * C * H.transpose() - V).inverse()).eval();
755 BoundVector unbiasedParamsVec = s + K * (m - H * s);
756 BoundMatrix unbiasedParamsCov = C - K * H * C;
757 return std::make_pair(unbiasedParamsVec, unbiasedParamsCov);
758 });
759}
