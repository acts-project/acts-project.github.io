61template <TrackProxyConcept track_proxy_t>
62Result<typename track_proxy_t::ConstTrackStateProxy> findFirstMeasurementState(
63 const track_proxy_t &track) {
64 using TrackStateProxy = typename track_proxy_t::ConstTrackStateProxy;
65
66 // TODO specialize if track is forward linked
67
68 std::optional<TrackStateProxy> firstMeasurementOpt;
69
70 for (const auto &trackState : track.trackStatesReversed()) {
71 if (trackState.typeFlags().isMeasurement()) {
72 firstMeasurementOpt = trackState;
73 }
74 }
75
76 if (firstMeasurementOpt) {
77 return Result<TrackStateProxy>::success(*firstMeasurementOpt);
78 }
79
80 return Result<TrackStateProxy>::failure(
81 TrackExtrapolationError::CompatibleTrackStateNotFound);
82}

88Result<typename track_proxy_t::ConstTrackStateProxy> findLastMeasurementState(

98 return Result<TrackStateProxy>::failure(

113template <TrackProxyConcept track_proxy_t,
114 typename smoother_t = GainMatrixSmoother>
115Result<void> smoothTrack(
116 const GeometryContext &geoContext, track_proxy_t &track,
117 const Logger &logger = *getDefaultLogger("TrackSmoother", Logging::INFO),
118 smoother_t smoother = GainMatrixSmoother()) {
119 auto &trackContainer = track.container();
120 auto &trackStateContainer = trackContainer.trackStateContainer();
121
122 auto last = findLastMeasurementState(track);
123 if (!last.ok()) {
124 ACTS_DEBUG("no last track state found");
125 return last.error();
126 }
127
128 auto smoothingResult =
129 smoother(geoContext, trackStateContainer, last->index(), logger);
130
131 if (!smoothingResult.ok()) {
132 ACTS_DEBUG("Smoothing track " << track.index() << " failed with error "
133 << smoothingResult.error());
134 return smoothingResult.error();
135 }
136
137 return Result<void>::success();
138}

149template <TrackContainerFrontend track_container_t>
150Result<void> smoothTracks(
151 const GeometryContext &geoContext, const track_container_t &trackContainer,
152 const Logger &logger = *getDefaultLogger("TrackSmoother", Logging::INFO)) {
153 Result<void> result = Result<void>::success();
154
155 for (const auto &track : trackContainer) {
156 auto smoothingResult = smoothTrack(geoContext, track, logger);
157
158 // Only keep the first error
159 if (!smoothingResult.ok() && result.ok()) {
160 result = smoothingResult.error();
161 }
162 }
163
164 return result;

179template <TrackProxyConcept track_proxy_t>
180Result<std::pair<typename track_proxy_t::ConstTrackStateProxy, double>>
181findTrackStateForExtrapolation(
182 const GeometryContext &geoContext, const track_proxy_t &track,
183 const Surface &referenceSurface, TrackExtrapolationStrategy strategy,
184 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
185 Logging::INFO)) {
186 using TrackStateProxy = typename track_proxy_t::ConstTrackStateProxy;
187
188 auto intersect = [&](const TrackStateProxy &state) -> Intersection3D {
189 assert(state.hasSmoothed() || state.hasFiltered());
190
191 FreeVector freeVector;
192 if (state.hasSmoothed()) {
193 freeVector = MultiTrajectoryHelpers::freeSmoothed(geoContext, state);
194 } else {
195 freeVector = MultiTrajectoryHelpers::freeFiltered(geoContext, state);
196 }
197
198 return referenceSurface
199 .intersect(geoContext, freeVector.template segment<3>(eFreePos0),
200 freeVector.template segment<3>(eFreeDir0),
201 BoundaryTolerance::None(), s_onSurfaceTolerance)
202 .closest();
203 };
204
205 switch (strategy) {
206 case TrackExtrapolationStrategy::first: {
207 ACTS_VERBOSE("looking for first track state");
208
209 auto first = findFirstMeasurementState(track);
210 if (!first.ok()) {
211 ACTS_DEBUG("no first track state found");
212 return first.error();
213 }
214
215 Intersection3D intersection = intersect(*first);
216 if (!intersection.isValid()) {
217 ACTS_DEBUG("no intersection found");
218 return Result<std::pair<TrackStateProxy, double>>::failure(
219 TrackExtrapolationError::ReferenceSurfaceUnreachable);
220 }
221
222 ACTS_VERBOSE("found intersection at " << intersection.pathLength());
223 return std::pair(*first, intersection.pathLength());
224 }
225
226 case TrackExtrapolationStrategy::last: {
227 ACTS_VERBOSE("looking for last track state");
228
229 auto last = findLastMeasurementState(track);
230 if (!last.ok()) {
231 ACTS_DEBUG("no last track state found");
232 return last.error();
233 }
234
235 Intersection3D intersection = intersect(*last);
236 if (!intersection.isValid()) {
237 ACTS_DEBUG("no intersection found");
238 return Result<std::pair<TrackStateProxy, double>>::failure(
239 TrackExtrapolationError::ReferenceSurfaceUnreachable);
240 }
241
242 ACTS_VERBOSE("found intersection at " << intersection.pathLength());
243 return std::pair(*last, intersection.pathLength());
244 }
245
246 case TrackExtrapolationStrategy::firstOrLast: {
247 ACTS_VERBOSE("looking for first or last track state");
248
249 auto first = findFirstMeasurementState(track);
250 if (!first.ok()) {
251 ACTS_DEBUG("no first track state found");
252 return first.error();
253 }
254
255 auto last = findLastMeasurementState(track);
256 if (!last.ok()) {
257 ACTS_DEBUG("no last track state found");
258 return last.error();
259 }
260
261 Intersection3D intersectionFirst = intersect(*first);
262 Intersection3D intersectionLast = intersect(*last);
263
264 double absDistanceFirst = std::abs(intersectionFirst.pathLength());
265 double absDistanceLast = std::abs(intersectionLast.pathLength());
266
267 if (intersectionFirst.isValid() && absDistanceFirst <= absDistanceLast) {
268 ACTS_VERBOSE("using first track state with intersection at "
269 << intersectionFirst.pathLength());
270 return std::pair(*first, intersectionFirst.pathLength());
271 }
272
273 if (intersectionLast.isValid() && absDistanceLast <= absDistanceFirst) {
274 ACTS_VERBOSE("using last track state with intersection at "
275 << intersectionLast.pathLength());
276 return std::pair(*last, intersectionLast.pathLength());
277 }
278
279 ACTS_DEBUG("no intersection found");
280 return Result<std::pair<TrackStateProxy, double>>::failure(
281 TrackExtrapolationError::ReferenceSurfaceUnreachable);
282 }
283 }
284
285 // unreachable
286 return Result<std::pair<TrackStateProxy, double>>::failure(
287 TrackExtrapolationError::CompatibleTrackStateNotFound);
288}

304template <TrackProxyConcept track_proxy_t, typename propagator_t,
305 typename propagator_options_t>
306Result<void> extrapolateTrackToReferenceSurface(
307 track_proxy_t &track, const Surface &referenceSurface,
308 const propagator_t &propagator, propagator_options_t options,
309 TrackExtrapolationStrategy strategy,
310 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
311 Logging::INFO)) {
312 auto findResult = findTrackStateForExtrapolation(
313 options.geoContext, track, referenceSurface, strategy, logger);
314
315 if (!findResult.ok()) {
316 ACTS_DEBUG("failed to find track state for extrapolation");
317 return findResult.error();
318 }
319
320 auto &[trackState, distance] = *findResult;
321
322 options.direction = Direction::fromScalarZeroAsPositive(distance);
323
324 BoundTrackParameters parameters = track.createParametersFromState(trackState);
325 ACTS_VERBOSE("extrapolating track to reference surface at distance "
326 << distance << " with direction " << options.direction
327 << " with starting parameters " << parameters);
328
329 auto propagateResult =
330 propagator.template propagate<BoundTrackParameters, propagator_options_t,
331 ForcedSurfaceReached>(
332 parameters, referenceSurface, options);
333
334 if (!propagateResult.ok()) {
335 ACTS_DEBUG("failed to extrapolate track: " << propagateResult.error());
336 return propagateResult.error();
337 }
338
339 track.setReferenceSurface(referenceSurface.getSharedPtr());
340 track.parameters() = propagateResult->endParameters.value().parameters();
341 track.covariance() =
342 propagateResult->endParameters.value().covariance().value();
343
344 return Result<void>::success();
345}

361template <TrackContainerFrontend track_container_t, typename propagator_t,
362 typename propagator_options_t>
363Result<void> extrapolateTracksToReferenceSurface(
364 const track_container_t &trackContainer, const Surface &referenceSurface,
365 const propagator_t &propagator, propagator_options_t options,
366 TrackExtrapolationStrategy strategy,
367 const Logger &logger = *getDefaultLogger("TrackExtrapolation",
368 Logging::INFO)) {
369 Result<void> result = Result<void>::success();
370
371 for (const auto &track : trackContainer) {
372 auto extrapolateResult = extrapolateTrackToReferenceSurface(
373 track, referenceSurface, propagator, options, strategy, logger);
374
375 // Only keep the first error
376 if (!extrapolateResult.ok() && result.ok()) {
377 result = extrapolateResult.error();
378 }
379 }
380
381 return result;

388template <TrackProxyConcept track_proxy_t>
389void calculateTrackQuantities(track_proxy_t track)
390 requires(!track_proxy_t::ReadOnly)
391{
392 track.chi2() = 0;
393 track.nDoF() = 0;
394
395 track.nHoles() = 0;
396 track.nMeasurements() = 0;
397 track.nSharedHits() = 0;
398 track.nOutliers() = 0;
399
400 for (const auto &trackState : track.trackStatesReversed()) {
401 ConstTrackStateTypeMap typeFlags = trackState.typeFlags();
402
403 if (typeFlags.isHole()) {
404 track.nHoles()++;
405 } else if (typeFlags.isOutlier()) {
406 track.nOutliers()++;
407 } else if (typeFlags.isMeasurement()) {
408 if (typeFlags.isSharedHit()) {
409 track.nSharedHits()++;
410 }
411 track.nMeasurements()++;
412 track.chi2() += trackState.chi2();
413 track.nDoF() += trackState.calibratedSize();
414 }
415 }
416}

425template <TrackProxyConcept track_proxy_t>
426void trimTrackFront(track_proxy_t track, bool trimHoles, bool trimOutliers,
427 bool trimMaterial, bool trimOtherNoneMeasurement)
428 requires(!track_proxy_t::ReadOnly)
429{
430 using TrackStateProxy = typename track_proxy_t::TrackStateProxy;
431
432 // TODO specialize if track is forward linked
433
434 std::optional<TrackStateProxy> front;
435
436 for (TrackStateProxy trackState : track.trackStatesReversed()) {
437 TrackStateTypeMap typeFlags = trackState.typeFlags();
438 bool isHole = typeFlags.isHole();
439 bool isOutlier = typeFlags.isOutlier();
440 bool isMaterial = typeFlags.isMaterial();
441 bool isOtherNoneMeasurement =
442 !typeFlags.hasMeasurement() && !isHole && !isOutlier && !isMaterial;
443 if (trimHoles && isHole) {
444 continue;
445 }
446 if (trimOutliers && isOutlier) {
447 continue;
448 }
449 if (trimMaterial && isMaterial) {
450 continue;
451 }
452 if (trimOtherNoneMeasurement && isOtherNoneMeasurement) {
453 continue;
454 }
455
456 front = trackState;
457 }
458
459 if (front.has_value()) {
460 front.value().previous() = TrackStateProxy::kInvalid;
461 }
462}

491 }
492 if (trimOutliers && isOutlier) {

533template <std::size_t nMeasurementDim,
534 TrackStateProxyConcept track_state_proxy_t>
535std::pair<ActsVector<nMeasurementDim>, ActsSquareMatrix<nMeasurementDim>>
536calculatePredictedResidual(track_state_proxy_t trackState) {
537 using MeasurementVector = ActsVector<nMeasurementDim>;
538 using MeasurementMatrix = ActsSquareMatrix<nMeasurementDim>;
539
540 if (!trackState.hasPredicted()) {
541 throw std::invalid_argument("track state has no predicted parameters");
542 }
543 if (!trackState.hasCalibrated()) {
544 throw std::invalid_argument("track state has no calibrated parameters");
545 }
546
547 auto subspaceHelper =
548 trackState.template projectorSubspaceHelper<nMeasurementDim>();
549
550 auto measurement = trackState.template calibrated<nMeasurementDim>();
551 auto measurementCovariance =
552 trackState.template calibratedCovariance<nMeasurementDim>();
553 MeasurementVector predicted =
554 subspaceHelper.projectVector(trackState.predicted());
555 MeasurementMatrix predictedCovariance =
556 subspaceHelper.projectMatrix(trackState.predictedCovariance());
557
558 MeasurementVector residual = measurement - predicted;
559 MeasurementMatrix residualCovariance =
560 measurementCovariance + predictedCovariance;
561
562 return {residual, residualCovariance};
563}

643template <TrackStateProxyConcept track_state_proxy_t>
644double calculatePredictedChi2(track_state_proxy_t trackState) {
645 if (!trackState.hasPredicted()) {
646 throw std::invalid_argument("track state has no predicted parameters");
647 }
648 if (!trackState.hasCalibrated()) {
649 throw std::invalid_argument("track state has no calibrated parameters");
650 }
651
652 return visit_measurement(
653 trackState.calibratedSize(),
654 [&]<std::size_t measdim>(
655 std::integral_constant<std::size_t, measdim>) -> double {
656 auto [residual, residualCovariance] =
657 calculatePredictedResidual<measdim>(trackState);
658
659 return (residual.transpose() * residualCovariance.inverse() * residual)
660 .eval()(0, 0);
661 });
662}

720template <TrackStateProxyConcept track_state_proxy_t>
721std::pair<BoundVector, BoundMatrix> calculateUnbiasedParametersCovariance(
722 track_state_proxy_t trackState) {
723 if (!trackState.hasSmoothed()) {
724 throw std::invalid_argument("track state has no smoothed parameters");
725 }
726 if (!trackState.hasCalibrated()) {
727 throw std::invalid_argument("track state has no calibrated parameters");
728 }
729
730 return visit_measurement(
731 trackState.calibratedSize(),
732 [&]<std::size_t measdim>(std::integral_constant<std::size_t, measdim>) {
733 FixedBoundSubspaceHelper<measdim> subspaceHelper =
734 trackState.template projectorSubspaceHelper<measdim>();
735
736 // TODO use subspace helper for projection instead
737 auto H = subspaceHelper.projector();
738 auto s = trackState.smoothed();
739 auto C = trackState.smoothedCovariance();
740 auto m = trackState.template calibrated<measdim>();
741 auto V = trackState.template calibratedCovariance<measdim>();
742 auto K =
743 (C * H.transpose() * (H * C * H.transpose() - V).inverse()).eval();
744 BoundVector unbiasedParamsVec = s + K * (m - H * s);
745 BoundMatrix unbiasedParamsCov = C - K * H * C;
746 return std::make_pair(unbiasedParamsVec, unbiasedParamsCov);
747 });
748}
