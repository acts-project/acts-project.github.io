316 bool prepareCurvilinearState(State& state) const {
317 // test whether the accumulated path has still its initial value.
318 if (state.pathAccumulated != 0) {
319 return true;
320 }
321
322 // dr/ds :
323 state.derivative.template head<3>() = direction(state);
324 // dt / ds
325 state.derivative(eFreeTime) = fastHypot(
326 1., state.particleHypothesis.mass() / absoluteMomentum(state));
327 // d (dr/ds) / ds : == 0
328 state.derivative.template segment<3>(4) = Acts::Vector3::Zero().transpose();
329 // d qop / ds == 0
330 state.derivative(eFreeQOverP) = 0.;
331
332 return true;
333 }

398 // This is simply ignored if `nullptr`.
403 Result<double> step(State& state, Direction propDir,
404 const IVolumeMaterial* material) const {
405 static_cast<void>(material);
406
407 // use the adjusted step size
408 const auto h = state.stepSize.value() * propDir;
409 const auto m = state.particleHypothesis.mass();
410 const auto p = absoluteMomentum(state);
411 // time propagates along distance as 1/b = sqrt(1 + m/p)
412 const auto dtds = fastHypot(1., m / p);
413 // Update the track parameters according to the equations of motion
414 Vector3 dir = direction(state);
415 state.pars.template segment<3>(eFreePos0) += h * dir;
416 state.pars[eFreeTime] += h * dtds;
417
418 // Propagate the jacobian
419 if (state.covTransport) {
420 // The step transport matrix in global coordinates
421 FreeMatrix D = FreeMatrix::Identity();
422 D.block<3, 3>(0, 4) = ActsSquareMatrix<3>::Identity() * h;
423 // Extend the calculation by the time propagation
424 // Evaluate dt/dlambda
425 D(3, 7) = h * m * m * state.pars[eFreeQOverP] / dtds;
426 // Set the derivative factor the time
427 state.derivative(3) = dtds;
428 // Update jacobian and derivative
429 state.jacTransport = D * state.jacTransport;
430 state.derivative.template head<3>() = dir;
431 }
432
433 // state the path length
434 state.pathAccumulated += h;
435 ++state.nSteps;
436 ++state.nStepTrials;
437
438 ++state.statistics.nAttemptedSteps;
439 ++state.statistics.nSuccessfulSteps;
440 if (propDir != Direction::fromScalarZeroAsPositive(h)) {
441 ++state.statistics.nReverseSteps;
442 }
443 state.statistics.pathLength += h;
444 state.statistics.absolutePathLength += std::abs(h);
445
446 return h;
447 }
