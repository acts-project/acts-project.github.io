238struct SeedFinderOptions {
239 // location of beam in x,y plane.
240 // used as offset for Space Points
241 Vector2 beamPos{0 * UnitConstants::mm, 0 * UnitConstants::mm};
242 // field induction
243 float bFieldInZ = 2 * UnitConstants::T;
244
245 // derived quantities
246 float pTPerHelixRadius = std::numeric_limits<float>::quiet_NaN();
247 float minHelixDiameter2 = std::numeric_limits<float>::quiet_NaN();
248 float pT2perRadius = std::numeric_limits<float>::quiet_NaN();
249 float sigmapT2perRadius = std::numeric_limits<float>::quiet_NaN();
250 float multipleScattering2 = std::numeric_limits<float>::quiet_NaN();
251
252 bool isInInternalUnits = true;
253
254 template <typename Config>
255 SeedFinderOptions calculateDerivedQuantities(const Config& config) const {
256 using namespace UnitLiterals;
257
258 SeedFinderOptions options = *this;
259 // bFieldInZ is in (pT/radius) natively, no need for conversion
260 options.pTPerHelixRadius = options.bFieldInZ;
261 options.minHelixDiameter2 =
262 std::pow(config.minPt * 2 / options.pTPerHelixRadius, 2) *
263 config.helixCutTolerance;
264 options.pT2perRadius =
265 std::pow(config.highland / options.pTPerHelixRadius, 2);
266 options.sigmapT2perRadius =
267 options.pT2perRadius * std::pow(2 * config.sigmaScattering, 2);
268 options.multipleScattering2 =
269 config.maxScatteringAngle2 * std::pow(config.sigmaScattering, 2);
270 return options;
271 }
272};
