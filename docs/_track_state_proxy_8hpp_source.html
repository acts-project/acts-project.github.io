490 template <TrackStateProxyConcept track_state_proxy_t>
491 void copyFrom(const track_state_proxy_t& other,
492 TrackStatePropMask mask = TrackStatePropMask::All,
493 bool onlyAllocated = true)
494 requires(!ReadOnly)
495 {
496 using PM = TrackStatePropMask;
497
498 if (onlyAllocated) {
499 auto dest = getMask();
500 auto src = other.getMask() &
501 mask; // combine what we have with what we want to copy
502
503 if (ACTS_CHECK_BIT(src, PM::Calibrated)) {
504 // on-demand allocate calibrated
505 dest |= PM::Calibrated;
506 }
507
508 if ((static_cast<std::underlying_type_t<TrackStatePropMask>>(
509 (src ^ dest) & src) != 0 ||
510 dest == TrackStatePropMask::None ||
511 src == TrackStatePropMask::None) &&
512 mask != TrackStatePropMask::None) {
513 throw std::runtime_error(
514 "Attempt track state copy with incompatible allocations");
515 }
516
517 // we're sure now this has correct allocations, so just copy
518 if (ACTS_CHECK_BIT(src, PM::Predicted)) {
519 predicted() = other.predicted();
520 predictedCovariance() = other.predictedCovariance();
521 }
522
523 if (ACTS_CHECK_BIT(src, PM::Filtered)) {
524 filtered() = other.filtered();
525 filteredCovariance() = other.filteredCovariance();
526 }
527
528 if (ACTS_CHECK_BIT(src, PM::Smoothed)) {
529 smoothed() = other.smoothed();
530 smoothedCovariance() = other.smoothedCovariance();
531 }
532
533 if (other.hasUncalibratedSourceLink()) {
534 setUncalibratedSourceLink(other.getUncalibratedSourceLink());
535 }
536
537 if (ACTS_CHECK_BIT(src, PM::Jacobian)) {
538 jacobian() = other.jacobian();
539 }
540
541 if (ACTS_CHECK_BIT(src, PM::Calibrated)) {
542 visit_measurement(other.calibratedSize(), [&](auto N) {
543 constexpr int measdim = decltype(N)::value;
544 allocateCalibrated(
545 other.template calibrated<measdim>().eval(),
546 other.template calibratedCovariance<measdim>().eval());
547 });
548
549 setProjectorSubspaceIndices(other.projectorSubspaceIndices());
550 }
551 } else {
552 if (ACTS_CHECK_BIT(mask, PM::Predicted) &&
553 has<detail_tsp::kPredictedKey>() &&
554 other.template has<detail_tsp::kPredictedKey>()) {
555 predicted() = other.predicted();
556 predictedCovariance() = other.predictedCovariance();
557 }
558
559 if (ACTS_CHECK_BIT(mask, PM::Filtered) &&
560 has<detail_tsp::kFilteredKey>() &&
561 other.template has<detail_tsp::kFilteredKey>()) {
562 filtered() = other.filtered();
563 filteredCovariance() = other.filteredCovariance();
564 }
565
566 if (ACTS_CHECK_BIT(mask, PM::Smoothed) &&
567 has<detail_tsp::kSmoothedKey>() &&
568 other.template has<detail_tsp::kSmoothedKey>()) {
569 smoothed() = other.smoothed();
570 smoothedCovariance() = other.smoothedCovariance();
571 }
572
573 if (other.hasUncalibratedSourceLink()) {
574 setUncalibratedSourceLink(other.getUncalibratedSourceLink());
575 }
576
577 if (ACTS_CHECK_BIT(mask, PM::Jacobian) &&
578 has<detail_tsp::kJacobianKey>() &&
579 other.template has<detail_tsp::kJacobianKey>()) {
580 jacobian() = other.jacobian();
581 }
582
583 // NOTE: we should not check hasCalibrated on this, since it
584 // may be not yet allocated
585 if (ACTS_CHECK_BIT(mask, PM::Calibrated) &&
586 other.template has<detail_tsp::kCalibratedKey>()) {
587 visit_measurement(other.calibratedSize(), [&](auto N) {
588 constexpr int measdim = decltype(N)::value;
589 allocateCalibrated(
590 other.template calibrated<measdim>().eval(),
591 other.template calibratedCovariance<measdim>().eval());
592 });
593
594 setProjectorSubspaceIndices(other.projectorSubspaceIndices());
595 }
596 }
597
598 chi2() = other.chi2();
599 pathLength() = other.pathLength();
600 typeFlags() = other.typeFlags();
601
602 if (other.hasReferenceSurface()) {
603 setReferenceSurface(other.referenceSurface().getSharedPtr());
604 }
605
606 m_traj->copyDynamicFrom(m_istate, other.container(), other.index());
607 }
