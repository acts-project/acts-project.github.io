34template <typename trajectory_type>
35trajectory_type spline(const trajectory_type& inputsRaw, std::size_t nPoints,
36 bool keepOriginalHits = false) {
37 trajectory_type output;
38 if (inputsRaw.empty()) {
39 return output;
40 }
41
42 using InputVectorType = typename trajectory_type::value_type;
43
44 std::vector<Vector3> inputs;
45 // If input type is a vector of Vector3 we can use it directly
46 if constexpr (std::is_same_v<trajectory_type, std::vector<Vector3>>) {
47 inputs = inputsRaw;
48 } else {
49 inputs.reserve(inputsRaw.size());
50 for (const auto& input : inputsRaw) {
51 inputs.push_back(Vector3(input[0], input[1], input[2]));
52 }
53 }
54
55 // Don't do anything if we have less than 3 points or less interpolation
56 // points than input points
57 if (inputsRaw.size() < 3 || nPoints <= inputsRaw.size()) {
58 return inputsRaw;
59 } else {
60 Eigen::MatrixXd points(3, inputs.size());
61 for (std::size_t i = 0; i < inputs.size(); ++i) {
62 points.col(i) = inputs[i].transpose();
63 }
64
65 // MARK: fpeMaskBegin(FLTDIV, 1, #4024)
66 // MARK: fpeMaskBegin(FLTINV, 1, #4024)
67 Eigen::Spline<double, 3> spline3D =
68 Eigen::SplineFitting<Eigen::Spline<double, 3>>::Interpolate(points, 2);
69 // MARK: fpeMaskEnd(FLTDIV)
70 // MARK: fpeMaskEnd(FLTINV)
71
72 double step = 1. / (nPoints - 1);
73 for (std::size_t i = 0; i < nPoints; ++i) {
74 double t = i * step;
75 InputVectorType point;
76 point[0] = spline3D(t)[0];
77 point[1] = spline3D(t)[1];
78 point[2] = spline3D(t)[2];
79 output.push_back(point);
80 }
81 }
82 // If we want to keep the original hits, we add them to the output
83 // (first and last are there anyway)
84 if (keepOriginalHits) {
85 output.insert(output.begin(), inputsRaw.begin() + 1, inputsRaw.end() - 1);
86 // We need to sort the output in distance to first
87 std::ranges::sort(output, [&inputs](const auto& a, const auto& b) {
88 const auto ifront = inputs.front();
89 double da2 = (a[0] - ifront[0]) * (a[0] - ifront[0]) +
90 (a[1] - ifront[1]) * (a[1] - ifront[1]) +
91 (a[2] - ifront[2]) * (a[2] - ifront[2]);
92 double db2 = (b[0] - ifront[0]) * (b[0] - ifront[0]) +
93 (b[1] - ifront[1]) * (b[1] - ifront[1]) +
94 (b[2] - ifront[2]) * (b[2] - ifront[2]);
95 return da2 < db2;
96 });
97 }
98
99 return output;
100}
