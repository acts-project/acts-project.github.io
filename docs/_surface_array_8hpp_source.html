145 void fill(const GeometryContext& gctx,
146 const SurfaceVector& surfaces) override {
147 for (const Surface* surface : surfaces) {
148 const std::size_t globalBin = fillSurfaceToBinMapping(gctx, *surface);
149 if (globalBin == 0) {

232 bool isValidBin(std::size_t bin) const override {
233 std::array<std::size_t, 2> indices = m_grid.localBinsFromGlobalBin(bin);
234 std::array<std::size_t, 2> nBins = m_grid.numLocalBins();
235 for (std::size_t i = 0; i < indices.size(); ++i) {
236 std::size_t idx = indices.at(i);
237 if (idx <= 0 || idx >= nBins.at(i) + 1) {
238 return false;
239 }
240 }
241 return true;
242 }

244 private:
246 std::size_t fillSurfaceToBinMapping(const GeometryContext& gctx,
247 const Surface& surface) {
248 const Vector3 pos = surface.referencePosition(gctx, AxisDirection::AxisR);
249 const Vector3 normal = m_representative->normal(gctx, pos);
250 const std::size_t globalBin = findGlobalBin(pos, normal, m_tolerance);
251 if (globalBin != 0) {
252 m_grid.at(globalBin).push_back(&surface);
253 }
254 return globalBin;
255 };
256
258 void fillBinToSurfaceMapping(const GeometryContext& gctx,
259 const Surface& surface, std::size_t startBin) {
260 const std::array<std::size_t, 2> startIndices =
261 m_grid.localBinsFromGlobalBin(startBin);
262 const auto startNeighborIndices =
263 m_grid.neighborHoodIndices(startIndices, 1u);
264
265 std::set<std::size_t> visited({startBin});
266 std::vector<std::size_t> queue(startNeighborIndices.begin(),
267 startNeighborIndices.end());
268
269 while (!queue.empty()) {
270 const std::size_t current = queue.back();
271 queue.pop_back();
272 if (visited.contains(current)) {
273 continue;
274 }
275
276 const std::array<std::size_t, 2> currentIndices =
277 m_grid.localBinsFromGlobalBin(current);
278 visited.insert(current);
279
280 const std::array<double, 2> gridLocal =
281 m_grid.binCenter(currentIndices);
282 const Vector2 surfaceLocal = gridToSurfaceLocal(gridLocal);
283 const Vector3 normal = m_representative->normal(gctx, surfaceLocal);
284 const Vector3 global =
285 m_representative->localToGlobal(gctx, surfaceLocal, normal);
286
287 const Intersection3D intersection =
288 surface.intersect(gctx, global, normal, BoundaryTolerance::None())
289 .closest();
290 if (!intersection.isValid() ||
291 std::abs(intersection.pathLength()) > m_tolerance) {
292 continue;
293 }
294 m_grid.at(current).push_back(&surface);
295
296 const auto neighborIndices =
297 m_grid.neighborHoodIndices(currentIndices, 1u);
298 queue.insert(queue.end(), neighborIndices.begin(),
299 neighborIndices.end());
300 }
301 };
302
303 void populateNeighborCache() {
304 // calculate neighbors for every bin and store in map
305 for (std::size_t i = 0; i < m_grid.size(); i++) {
306 if (!isValidBin(i)) {
307 continue;
308 }
309 const std::array<std::size_t, 2> indices =
310 m_grid.localBinsFromGlobalBin(i);
311 std::vector<const Surface*>& neighbors = m_neighborMap.at(i);
312 neighbors.clear();
313
314 for (std::size_t idx : m_grid.neighborHoodIndices(indices, 1u)) {
315 const std::vector<const Surface*>& binContent = m_grid.at(idx);
316 std::copy(binContent.begin(), binContent.end(),
317 std::back_inserter(neighbors));
318 }
319
320 std::ranges::sort(neighbors);
321 auto last = std::ranges::unique(neighbors);
322 neighbors.erase(last.begin(), last.end());
323 neighbors.shrink_to_fit();
324 }
325 }

327 Vector3 getBinCenterImpl(const GeometryContext& gctx,
328 std::size_t bin) const {
329 const std::array<double, 2> gridLocal =
330 m_grid.binCenter(m_grid.localBinsFromGlobalBin(bin));
331 const Vector2 surfaceLocal = gridToSurfaceLocal(gridLocal);
332 return m_representative->localToGlobal(gctx, surfaceLocal);
333 }

335 const CylinderBounds* getCylinderBounds() const {
336 return dynamic_cast<const CylinderBounds*>(&m_representative->bounds());
337 }
338
339 Vector2 gridToSurfaceLocal(std::array<double, 2> gridLocal) const {
340 Vector2 surfaceLocal = Eigen::Map<Vector2>(gridLocal.data());
341 if (const CylinderBounds* bounds = getCylinderBounds();
342 bounds != nullptr) {
343 surfaceLocal[0] *= bounds->get(CylinderBounds::eR);
344 }
345 return surfaceLocal;
346 }
347 std::array<double, 2> surfaceToGridLocal(Vector2 local) const {
348 std::array<double, 2> gridLocal = {local[0], local[1]};
349 if (const CylinderBounds* bounds = getCylinderBounds();
350 bounds != nullptr) {
351 gridLocal[0] /= bounds->get(CylinderBounds::eR);
352 }
353 return gridLocal;
354 }

356 std::size_t findGlobalBin(const Vector3& position, const Vector3& direction,
357 double tolerance) const {
358 auto gctx = GeometryContext::dangerouslyDefaultConstruct();
359
360 const Intersection3D intersection =
361 m_representative
362 ->intersect(gctx, position, direction,
363 BoundaryTolerance::Infinite())
364 .closest();
365 if (!intersection.isValid() ||
366 std::abs(intersection.pathLength()) > tolerance) {
367 return 0; // overflow bin
368 }
369 const Vector2 surfaceLocal =
370 m_representative
371 ->globalToLocal(gctx, intersection.position(), direction)
372 .value();
373 const std::array<double, 2> gridLocal = surfaceToGridLocal(surfaceLocal);
374 return m_grid.globalBinFromPosition(gridLocal);
375 }

377 std::shared_ptr<RegularSurface> m_representative;
378 double m_tolerance{};
379 Grid_t m_grid;
380 std::vector<AxisDirection> m_binValues;
381 std::vector<SurfaceVector> m_neighborMap;
382 };
