42class FpeMonitor {
43 public:
44 struct Buffer {
45 explicit Buffer(std::size_t bufferSize)
46 : m_data{std::make_unique<std::byte[]>(bufferSize)},
47 m_size{bufferSize} {}
48
49 Buffer(const Buffer &) = delete;
50 Buffer(Buffer &&other) noexcept
51 : m_data(std::move(other.m_data)),
52 m_size(other.m_size),
53 m_offset(other.m_offset) {
54 other.m_size = 0;
55 other.m_offset = 0;
56 }
57
58 std::pair<void *, std::size_t> next() const {
59 return {m_data.get() + m_offset, m_size - m_offset};
60 }
61
62 void pushOffset(std::size_t offset) {
63 assert(m_offset + offset < m_size);
64 m_offset = offset;
65 }
66
67 void reset() { m_offset = 0; }
68
69 std::size_t size() const { return m_size; }
70 std::size_t offset() const { return m_offset; }
71
72 std::byte *data() { return m_data.get(); }
73
74 private:
75 std::unique_ptr<std::byte[]> m_data;
76 std::size_t m_size{};
77 std::size_t m_offset{};
78 };
