59 constexpr static int hexToInt(std::string_view hex) {
60 constexpr auto hexCharToInt = [](char c) {
61 if (c >= '0' && c <= '9') {
62 return c - '0';
63 } else if (c >= 'a' && c <= 'f') {
64 return c - 'a' + 10;
65 } else if (c >= 'A' && c <= 'F') {
66 return c - 'A' + 10;
67 } else {
68 throw std::invalid_argument("Invalid hex character");
69 }
70 };
71
72 int value = 0;
73 for (char c : hex) {
74 value = (value << 4) + hexCharToInt(c);
75 }
76 return value;
77 };

82 constexpr explicit Color(std::string_view hex) {
83 if (hex[0] == '#' && hex.size() == 7) {
84 rgb[0] = hexToInt(hex.substr(1, 2)); // Extract R component
85 rgb[1] = hexToInt(hex.substr(3, 2)); // Extract G component
86 rgb[2] = hexToInt(hex.substr(5, 2)); // Extract B component
87 } else {
88 throw std::invalid_argument{

90 }
