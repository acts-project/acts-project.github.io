28inline BoundVector reflectBoundParameters(const BoundVector& boundParams) {
29 BoundVector reflected = boundParams;
30 auto [phi, theta] =
31 detail::normalizePhiTheta(boundParams[eBoundPhi] - std::numbers::pi,
32 std::numbers::pi - boundParams[eBoundTheta]);
33 reflected[eBoundPhi] = phi;
34 reflected[eBoundTheta] = theta;
35 reflected[eBoundQOverP] = -boundParams[eBoundQOverP];
36 return reflected;
37}
