81 bool degenerate() const {

219 RangeXD<Dims, Type, Vector> operator&(
220 const RangeXD<Dims, Type, Vector>& o) const {
221 RangeXD<Dims, Type> res;
222
223 for (std::size_t i = 0; i < Dims; ++i) {
224 res.min(i) = std::max(min(i), o.min(i));
225 res.max(i) = std::min(max(i), o.max(i));
226 }
227
228 return res;
229 }

240 RangeXD<Dims, Type, Vector>& operator&=(

259 bool operator&&(const RangeXD<Dims, Type, Vector>& r) const {

275 std::string toString(void) const {
276 std::stringstream s;
277
278 for (std::size_t i = 0; i < Dims; ++i) {
279 s << min(i) << " <= v[" << i << "] <= " << max(i);
280 if (i != Dims - 1) {
281 s << ", ";
282 }
283 }
284
285 return s.str();
