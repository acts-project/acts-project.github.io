43 template <typename traj_t>
44 Result<void> operator()(const GeometryContext& gctx, traj_t& trajectory,
45 std::size_t entryIndex,
46 const Logger& logger = getDummyLogger()) const {
47 static_cast<void>(gctx);
48
49 ACTS_VERBOSE("Invoked GainMatrixSmoother on entry index: " << entryIndex);
50
51 // For the last state: smoothed is filtered - also: switch to next
52 ACTS_VERBOSE("Getting previous track state");
53 auto prev_ts = trajectory.getTrackState(entryIndex);
54
55 prev_ts.shareFrom(TrackStatePropMask::Filtered,
56 TrackStatePropMask::Smoothed);
57
58 // make sure there is more than one track state
59 if (!prev_ts.hasPrevious()) {
60 ACTS_VERBOSE("Only one track state given, smoothing terminates early");
61 return Result<void>::success();
62 }
63
64 ACTS_VERBOSE("Start smoothing from previous track state at index: "
65 << prev_ts.previous());
66
67 // default-constructed error represents success, i.e. an invalid error code
68 std::error_code error;
69 trajectory.applyBackwards(prev_ts.previous(), [&, this](auto ts) {
70 // should have filtered and predicted, this should also include the
71 // covariances.
72 assert(ts.hasFiltered());
73 assert(ts.hasPredicted());
74
75 // previous trackstate should have smoothed and predicted
76 assert(prev_ts.hasSmoothed());
77 assert(prev_ts.hasPredicted());
78 assert(prev_ts.hasJacobian());
79
80 ACTS_VERBOSE("Calculate smoothing matrix:");
81 ACTS_VERBOSE("Filtered covariance:\n" << ts.filteredCovariance());
82 ACTS_VERBOSE("Jacobian:\n" << prev_ts.jacobian());
83
84 // ensure the track state has a smoothed component
85 ts.addComponents(TrackStatePropMask::Smoothed);
86
87 if (auto res = calculate(AnyMutableTrackStateProxy{ts},
88 AnyConstTrackStateProxy{prev_ts}, logger);
89 !res.ok()) {
90 error = res.error();
91 return false;
92 }
93
94 prev_ts = ts;
95 return true; // continue execution
96 });
97
98 return error ? Result<void>::failure(error) : Result<void>::success();
99 }

103 Acts::Delegate<TrackStateTraits<kMeasurementSizeMax, false>::Parameters(
104 void*)>;
106 using GetCovariance =
107 Acts::Delegate<TrackStateTraits<kMeasurementSizeMax, false>::Covariance(
108 void*)>;
109
117 Result<void> calculate(AnyMutableTrackStateProxy ts,
118 AnyConstTrackStateProxy prev_ts,
119 const Logger& logger) const;
