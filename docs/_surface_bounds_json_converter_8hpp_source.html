58template <typename bounds_t>
59std::shared_ptr<const bounds_t> fromJson(const nlohmann::json& j) {
60 const std::size_t kValues = bounds_t::BoundValues::eSize;
61 std::array<double, kValues> bValues{};
62 std::vector<double> bVector = j["values"];
63 std::copy_n(bVector.begin(), kValues, bValues.begin());
64 return std::make_shared<const bounds_t>(bValues);
65}
