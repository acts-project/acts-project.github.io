100 TrackContainerFrontend track_container_t>

102 const start_parameters_t& sParameters,
103 const GsfOptions<traj_t>& options,
104 const std::vector<const Surface*>& sSequence,

107 static_assert(
108 std::is_same_v<DirectNavigator, typename propagator_t::Navigator>);

112 using Actors = ActorList<GsfActor>;

116

121 .m_cfg.bethe_heitler_approx = m_betheHeitlerApproximation.get();

124 };

153 template <typename source_link_it_t, typename start_parameters_t,
154 TrackContainerFrontend track_container_t>
155 auto fit(source_link_it_t begin, source_link_it_t end,
156 const start_parameters_t& sParameters,
157 const GsfOptions<traj_t>& options,
158 track_container_t& trackContainer) const {
159 // Check if we have the correct navigator
160 static_assert(std::is_same_v<Navigator, typename propagator_t::Navigator>);
161
162 // Initialize the forward propagation with the DirectNavigator
163 auto fwdPropInitializer = [&](const auto& opts) {
164 using Actors = ActorList<GsfActor, EndOfWorldReached>;
165 using PropagatorOptions = typename propagator_t::template Options<Actors>;
166
167 PropagatorOptions propOptions(opts.geoContext, opts.magFieldContext);
168
169 propOptions.setPlainOptions(opts.propagatorPlainOptions);
170
171 if (options.useExternalSurfaces) {
172 for (auto it = begin; it != end; ++it) {
173 propOptions.navigation.insertExternalSurface(
174 *options.extensions.surfaceAccessor(*it));
175 }
176 }
177
178 propOptions.actorList.template get<GsfActor>()
179 .m_cfg.bethe_heitler_approx = m_betheHeitlerApproximation.get();
180
181 return propOptions;
182 };
183
184 // Initialize the backward propagation with the DirectNavigator
185 auto bwdPropInitializer = [&](const auto& opts) {
186 using Actors = ActorList<GsfActor, EndOfWorldReached>;
187 using PropagatorOptions = typename propagator_t::template Options<Actors>;
188
189 PropagatorOptions propOptions(opts.geoContext, opts.magFieldContext);
190
191 propOptions.setPlainOptions(opts.propagatorPlainOptions);
192
193 if (options.useExternalSurfaces) {
194 for (auto it = begin; it != end; ++it) {
195 propOptions.navigation.insertExternalSurface(
196 *options.extensions.surfaceAccessor(*it));
197 }
198 }
199
200 propOptions.actorList.template get<GsfActor>()
201 .m_cfg.bethe_heitler_approx = m_betheHeitlerApproximation.get();
202
203 return propOptions;
204 };
205
206 return fit_impl(begin, end, sParameters, options, fwdPropInitializer,
207 bwdPropInitializer, trackContainer);
208 }

221 template <typename source_link_it_t, typename start_parameters_t,
222 typename fwd_prop_initializer_t, typename bwd_prop_initializer_t,
223 TrackContainerFrontend track_container_t>
224 Acts::Result<typename track_container_t::TrackProxy> fit_impl(
225 source_link_it_t begin, source_link_it_t end,
226 const start_parameters_t& sParameters, const GsfOptions<traj_t>& options,
227 const fwd_prop_initializer_t& fwdPropInitializer,
228 const bwd_prop_initializer_t& bwdPropInitializer,
229 track_container_t& trackContainer) const {
230 // return or abort utility
231 auto return_error_or_abort = [&](auto error) {
232 if (options.abortOnError) {
233 std::abort();
234 }
235 return error;
236 };
237
238 // Define directions based on input propagation direction. This way we can
239 // refer to 'forward' and 'backward' regardless of the actual direction.
240 const auto gsfForward = options.propagatorPlainOptions.direction;
241 const auto gsfBackward = gsfForward.invert();
242
243 // Check if the start parameters are on the start surface
244 IntersectionStatus intersectionStatusStartSurface =
245 sParameters.referenceSurface()
246 .intersect(GeometryContext::dangerouslyDefaultConstruct(),
247 sParameters.position(
248 GeometryContext::dangerouslyDefaultConstruct()),
249 sParameters.direction(), BoundaryTolerance::None())
250 .closest()
251 .status();
252
253 if (intersectionStatusStartSurface != IntersectionStatus::onSurface) {
254 ACTS_DEBUG(
255 "Surface intersection of start parameters WITH bound-check failed");
256 }
257
258 // To be able to find measurements later, we put them into a map
259 // We need to copy input SourceLinks anyway, so the map can own them.
260 ACTS_VERBOSE("Preparing " << std::distance(begin, end)
261 << " input measurements");
262 std::map<GeometryIdentifier, SourceLink> inputMeasurements;
263 for (auto it = begin; it != end; ++it) {
264 SourceLink sl = *it;
265 inputMeasurements.emplace(
266 options.extensions.surfaceAccessor(sl)->geometryId(), std::move(sl));
267 }
268
269 ACTS_VERBOSE(
270 "Gsf: Final measurement map size: " << inputMeasurements.size());
271
272 if (sParameters.covariance() == std::nullopt) {
273 return GsfError::StartParametersHaveNoCovariance;
274 }
275
277 // Forward pass
279 ACTS_VERBOSE("+-----------------------------+");
280 ACTS_VERBOSE("| Gsf: Do forward propagation |");
281 ACTS_VERBOSE("+-----------------------------+");
282
283 auto fwdResult = [&]() {
284 auto fwdPropOptions = fwdPropInitializer(options);
285
286 // Catch the actor and set the measurements
287 auto& actor = fwdPropOptions.actorList.template get<GsfActor>();
288 actor.setOptions(options);
289 actor.m_cfg.inputMeasurements = &inputMeasurements;
290 actor.m_cfg.numberMeasurements = inputMeasurements.size();
291 actor.m_cfg.inReversePass = false;
292 actor.m_cfg.logger = m_actorLogger.get();
293
294 fwdPropOptions.direction = gsfForward;
295
296 // If necessary convert to MultiComponentBoundTrackParameters
297 using IsMultiParameters =
298 detail::IsMultiComponentBoundParameters<start_parameters_t>;
299
300 // dirty optional because parameters are not default constructible
301 std::optional<MultiComponentBoundTrackParameters> params;
302
303 // This allows the initialization with single- and multicomponent start
304 // parameters
305 if constexpr (!IsMultiParameters::value) {
306 params = MultiComponentBoundTrackParameters(
307 sParameters.referenceSurface().getSharedPtr(),
308 sParameters.parameters(), *sParameters.covariance(),
309 sParameters.particleHypothesis());
310 } else {
311 params = sParameters;
312 }
313
314 auto state = m_propagator.makeState(fwdPropOptions);
315
316 // Type deduction for propagation result to pass on errors
317 using OptionsType = decltype(fwdPropOptions);
318 using StateType = decltype(state);
319 using PropagationResultType =
320 decltype(m_propagator.propagate(std::declval<StateType&>()));
321 using ResultType = decltype(m_propagator.makeResult(
322 std::declval<StateType&&>(), std::declval<PropagationResultType>(),
323 std::declval<const OptionsType&>(), false));
324
325 auto initRes = m_propagator.initialize(state, *params);
326 if (!initRes.ok()) {
327 return ResultType::failure(initRes.error());
328 }
329
330 auto& r = state.template get<typename GsfActor::result_type>();
331 r.fittedStates = &trackContainer.trackStateContainer();
332
333 auto propagationResult = m_propagator.propagate(state);
334
335 return m_propagator.makeResult(std::move(state), propagationResult,
336 fwdPropOptions, false);
337 }();
338
339 if (!fwdResult.ok()) {
340 return return_error_or_abort(fwdResult.error());
341 }
342
343 const auto& fwdGsfResult =
344 fwdResult->template get<typename GsfActor::result_type>();
345
346 if (fwdGsfResult.measurementStates == 0) {
347 return return_error_or_abort(GsfError::NoMeasurementStatesCreatedForward);
348 }
349
350 ACTS_VERBOSE("Finished forward propagation");
351 ACTS_VERBOSE("- visited surfaces: " << fwdGsfResult.visitedSurfaces.size());
352 ACTS_VERBOSE("- processed states: " << fwdGsfResult.processedStates);
353 ACTS_VERBOSE("- measurement states: " << fwdGsfResult.measurementStates);
354
355 std::size_t nInvalidBetheHeitler = fwdGsfResult.nInvalidBetheHeitler.val();
356 double maxPathXOverX0 = fwdGsfResult.maxPathXOverX0.val();
357
359 // Backward pass
361 ACTS_VERBOSE("+------------------------------+");
362 ACTS_VERBOSE("| Gsf: Do backward propagation |");
363 ACTS_VERBOSE("+------------------------------+");
364
365 auto bwdResult = [&]() {
366 auto bwdPropOptions = bwdPropInitializer(options);
367
368 auto& actor = bwdPropOptions.actorList.template get<GsfActor>();
369 actor.setOptions(options);
370 actor.m_cfg.inputMeasurements = &inputMeasurements;
371 actor.m_cfg.inReversePass = true;
372 actor.m_cfg.logger = m_actorLogger.get();
373
374 bwdPropOptions.direction = gsfBackward;
375
376 const Surface& target = options.referenceSurface
377 ? *options.referenceSurface
378 : sParameters.referenceSurface();
379
380 std::vector<
381 std::tuple<double, BoundVector, std::optional<BoundSquareMatrix>>>
382 inflatedParamVector;
383 assert(!fwdGsfResult.lastMeasurementComponents.empty());
384 assert(fwdGsfResult.lastMeasurementSurface != nullptr);
385 for (auto& [w, p, cov] : fwdGsfResult.lastMeasurementComponents) {
386 inflatedParamVector.emplace_back(
387 w, p, cov * options.reverseFilteringCovarianceScaling);
388 }
389
390 MultiComponentBoundTrackParameters inflatedParams(
391 fwdGsfResult.lastMeasurementSurface->getSharedPtr(),
392 std::move(inflatedParamVector), sParameters.particleHypothesis());
393
394 auto state = m_propagator.template makeState<decltype(bwdPropOptions),
395 MultiStepperSurfaceReached>(
396 target, bwdPropOptions);
397
398 // Type deduction for propagation result to pass on errors
399 using OptionsType = decltype(bwdPropOptions);
400 using StateType = decltype(state);
401 using PropagationResultType =
402 decltype(m_propagator.propagate(std::declval<StateType&>()));
403 using ResultType = decltype(m_propagator.makeResult(
404 std::declval<StateType&&>(), std::declval<PropagationResultType>(),
405 target, std::declval<const OptionsType&>()));
406
407 auto initRes = m_propagator.initialize(state, inflatedParams);
408 if (!initRes.ok()) {
409 return ResultType::failure(initRes.error());
410 }
411
412 assert((fwdGsfResult.lastMeasurementTip != kTrackIndexInvalid &&
413 "tip is invalid"));
414
415 auto proxy = trackContainer.trackStateContainer().getTrackState(
416 fwdGsfResult.lastMeasurementTip);
417 proxy.shareFrom(TrackStatePropMask::Filtered,
418 TrackStatePropMask::Smoothed);
419
420 auto& r = state.template get<typename GsfActor::result_type>();
421 r.fittedStates = &trackContainer.trackStateContainer();
422 r.currentTip = fwdGsfResult.lastMeasurementTip;
423 r.visitedSurfaces.push_back(&proxy.referenceSurface());
424 r.surfacesVisitedBwdAgain.push_back(&proxy.referenceSurface());
425 r.measurementStates++;
426 r.processedStates++;
427
428 auto propagationResult = m_propagator.propagate(state);
429
430 return m_propagator.makeResult(std::move(state), propagationResult,
431 target, bwdPropOptions);
432 }();
433
434 if (!bwdResult.ok()) {
435 return return_error_or_abort(bwdResult.error());
436 }
437
438 auto& bwdGsfResult =
439 bwdResult->template get<typename GsfActor::result_type>();
440
441 if (bwdGsfResult.measurementStates == 0) {
442 return return_error_or_abort(
443 GsfError::NoMeasurementStatesCreatedBackward);
444 }
445
446 // For the backward pass we want the counters at in end (= at the
447 // interaction point) and not at the last measurement surface
448 bwdGsfResult.nInvalidBetheHeitler.update();
449 bwdGsfResult.maxPathXOverX0.update();
450 bwdGsfResult.sumPathXOverX0.update();
451 nInvalidBetheHeitler += bwdGsfResult.nInvalidBetheHeitler.val();
452 maxPathXOverX0 =
453 std::max(maxPathXOverX0, bwdGsfResult.maxPathXOverX0.val());
454
455 if (nInvalidBetheHeitler > 0) {
456 ACTS_WARNING("Encountered " << nInvalidBetheHeitler
457 << " cases where x/X0 exceeds the range "
458 "of the Bethe-Heitler-Approximation. The "
459 "maximum x/X0 encountered was "
460 << maxPathXOverX0
461 << ". Enable DEBUG output "
462 "for more information.");
463 }
464
466 // Create Kalman Result
468 ACTS_VERBOSE("Gsf - States summary:");
469 ACTS_VERBOSE("- Fwd measurement states: " << fwdGsfResult.measurementStates
470 << ", holes: "
471 << fwdGsfResult.measurementHoles);
472 ACTS_VERBOSE("- Bwd measurement states: " << bwdGsfResult.measurementStates
473 << ", holes: "
474 << bwdGsfResult.measurementHoles);
475
476 // TODO should this be warning level? it happens quite often... Investigate!
477 if (bwdGsfResult.measurementStates != fwdGsfResult.measurementStates) {
478 ACTS_DEBUG("Fwd and bwd measurement states do not match");
479 }
480
481 // Go through the states and assign outliers / unset smoothed if surface not
482 // passed in backward pass
483 const auto& foundBwd = bwdGsfResult.surfacesVisitedBwdAgain;
484 std::size_t measurementStatesFinal = 0;
485
486 for (auto state : fwdGsfResult.fittedStates->reverseTrackStateRange(
487 fwdGsfResult.currentTip)) {
488 const bool found =
489 rangeContainsValue(foundBwd, &state.referenceSurface());
490 if (!found && state.typeFlags().test(MeasurementFlag)) {
491 state.typeFlags().set(OutlierFlag);
492 state.typeFlags().reset(MeasurementFlag);
493 state.unset(TrackStatePropMask::Smoothed);
494 }
495
496 measurementStatesFinal +=
497 static_cast<std::size_t>(state.typeFlags().test(MeasurementFlag));
498 }
499
500 if (measurementStatesFinal == 0) {
501 return return_error_or_abort(GsfError::NoMeasurementStatesCreatedFinal);
502 }
503
504 auto track = trackContainer.makeTrack();
505 track.tipIndex() = fwdGsfResult.lastMeasurementTip;
506
507 if (options.referenceSurface) {
508 const auto& params = *bwdResult->endParameters;
509
510 const auto [finalPars, finalCov] = detail::mergeGaussianMixture(
511 params.components(), params.referenceSurface(),
512 options.componentMergeMethod, [](auto& t) {
513 return std::tie(std::get<0>(t), std::get<1>(t), *std::get<2>(t));
514 });
515
516 track.parameters() = finalPars;
517 track.covariance() = finalCov;
518
519 track.setReferenceSurface(params.referenceSurface().getSharedPtr());
520
521 if (trackContainer.hasColumn(
522 hashString(GsfConstants::kFinalMultiComponentStateColumn))) {
523 ACTS_DEBUG("Add final multi-component state to track");
524 track.template component<GsfConstants::FinalMultiComponentState>(
525 GsfConstants::kFinalMultiComponentStateColumn) = std::move(params);
526 }
527 }
528
529 if (trackContainer.hasColumn(
530 hashString(GsfConstants::kFwdMaxMaterialXOverX0))) {
531 track.template component<double>(GsfConstants::kFwdMaxMaterialXOverX0) =
532 fwdGsfResult.maxPathXOverX0.val();
533 }
534 if (trackContainer.hasColumn(
535 hashString(GsfConstants::kFwdSumMaterialXOverX0))) {
536 track.template component<double>(GsfConstants::kFwdSumMaterialXOverX0) =
537 fwdGsfResult.sumPathXOverX0.val();
538 }
539
540 calculateTrackQuantities(track);
541
542 return track;
543 }
544};
