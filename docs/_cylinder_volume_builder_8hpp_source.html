46struct VolumeConfig {
47 bool present{false};
48 bool wrapping{false};
49 double rMin;
50 double rMax;
51 double zMin;
52 double zMax;
53 LayerVector layers;
54 MutableTrackingVolumeVector volumes;
55
57 VolumeConfig()
58 : rMin(std::numeric_limits<double>::max()),
59 rMax(std::numeric_limits<double>::lowest()),
60 zMin(std::numeric_limits<double>::max()),
61 zMax(std::numeric_limits<double>::lowest()),
62 layers() {}
63
68 void adaptZ(const VolumeConfig& lConfig) {
69 if (lConfig.present) {
70 zMin = std::min(zMin, lConfig.zMin);
71 zMax = std::max(zMax, lConfig.zMax);
72 }
73 }
74
79 void adaptR(const VolumeConfig& lConfig) {
80 if (lConfig.present) {
81 rMin = std::min(rMin, lConfig.rMin);
82 rMax = std::max(rMax, lConfig.rMax);
83 }
84 }
85
90 void adapt(const VolumeConfig& lConfig) {
91 adaptZ(lConfig);
92 adaptR(lConfig);
93 }
94
101 void midPointAttachZ(VolumeConfig& lConfig) {
102 if (lConfig.zMin >= zMax) {
103 double zMid = 0.5 * (lConfig.zMin + zMax);
104 lConfig.zMin = zMid;
105 zMax = zMid;
106 } else {
107 double zMid = 0.5 * (zMin + lConfig.zMax);
108 lConfig.zMax = zMid;
109 zMin = zMid;
110 }
111 }
112
117 void attachZ(const VolumeConfig& lConfig) {
118 if (lConfig.zMin >= zMax) {
119 zMax = lConfig.zMin;
120 } else {
121 zMin = lConfig.zMax;
122 }
123 }
124
129 bool overlapsInR(const VolumeConfig& vConfig) const {
130 if (!present) {
131 return false;
132 }
133 return std::max(rMin, vConfig.rMin) <= std::min(rMax, vConfig.rMax);
134 }
135
140 bool overlapsInZ(const VolumeConfig& vConfig) const {
141 if (!present) {
142 return false;
143 }
144 return std::max(zMin, vConfig.zMin) <= std::min(zMax, vConfig.zMax);
145 }
146
151 bool wraps(const VolumeConfig& vConfig) const {
152 if ((zMax <= vConfig.zMin) || (zMin >= vConfig.zMax)) {
153 return true;
154 }
155 return containsInR(vConfig);
156 }
157
162 bool contains(const VolumeConfig& vConfig) const {
163 return (containsInR(vConfig) && containsInZ(vConfig));
164 }
165
170 bool containsInR(const VolumeConfig& vConfig) const {
171 return (rMin >= vConfig.rMax);
172 }
173
178 bool containsInZ(const VolumeConfig& vConfig) const {
179 return (vConfig.zMin > zMin && vConfig.zMax < zMax);
180 }
181
184 std::string toString() const {
186 std::stringstream sl;
187 sl << rMin << ", " << rMax << " / " << zMin << ", " << zMax;
188 return sl.str();
189 }
190};

193struct WrappingConfig {
194 public:
196 VolumeConfig nVolumeConfig;
198 VolumeConfig cVolumeConfig;
200 VolumeConfig pVolumeConfig;
201
203 VolumeConfig containerVolumeConfig;
204
206 VolumeConfig existingVolumeConfig;
208 VolumeConfig fGapVolumeConfig;
210 VolumeConfig sGapVolumeConfig;
211
214 VolumeConfig externalVolumeConfig;
215
217 WrappingCondition wCondition = Undefined;
219 std::string wConditionScreen = "[left untouched]";
220
222 WrappingConfig() = default;
223
225 void configureContainerVolume() {
226 // set the container to be present
227 containerVolumeConfig.present = true;
228 std::string wConditionAddon = "";
229 // if we have more than one config present
230 if ((nVolumeConfig.present && cVolumeConfig.present) ||
231 (cVolumeConfig.present && pVolumeConfig.present) ||
232 (nVolumeConfig.present && pVolumeConfig.present)) {
233 wCondition = Wrapping;
234 wConditionScreen = "grouped to ";
235 }
236 // adapt the new volume config to the existing configs
237 if (nVolumeConfig.present) {
238 containerVolumeConfig.adapt(nVolumeConfig);
239 wConditionScreen += "[n]";
240 }
241 if (cVolumeConfig.present) {
242 containerVolumeConfig.adapt(cVolumeConfig);
243 wConditionScreen += "[c]";
244 }
245 if (pVolumeConfig.present) {
246 containerVolumeConfig.adapt(pVolumeConfig);
247 wConditionScreen += "[p]";
248 }
249 // adapt the external one
250 if (externalVolumeConfig.present) {
251 containerVolumeConfig.adapt(externalVolumeConfig);
252 }
253 // attach the volume configs
254 if (nVolumeConfig.present && cVolumeConfig.present) {
255 nVolumeConfig.midPointAttachZ(cVolumeConfig);
256 }
257 if (cVolumeConfig.present && pVolumeConfig.present) {
258 cVolumeConfig.midPointAttachZ(pVolumeConfig);
259 }
260 // adapt r afterwards
261 // - easy if no existing volume
262 // - possible if no central volume
263 if (!existingVolumeConfig.present || !cVolumeConfig.present) {
264 nVolumeConfig.adaptR(containerVolumeConfig);
265 cVolumeConfig.adaptR(containerVolumeConfig);
266 pVolumeConfig.adaptR(containerVolumeConfig);
267 }
268 }
269
271 void wrapInsertAttach() {
272 // action is only needed if an existing volume
273 // is present
274 if (existingVolumeConfig.present) {
275 // 0 - simple attachment case
276 if (!cVolumeConfig.present) {
277 // check if it can be easily attached
278 if (nVolumeConfig.present &&
279 nVolumeConfig.zMax < existingVolumeConfig.zMin) {
280 nVolumeConfig.attachZ(existingVolumeConfig);
281 // will attach the new volume(s)
282 wCondition = Attaching;
283 wConditionScreen = "[n attached]";
284 }
285 if (pVolumeConfig.present &&
286 pVolumeConfig.zMin > existingVolumeConfig.zMax) {
287 pVolumeConfig.attachZ(existingVolumeConfig);
288 // will attach the new volume(s)
289 wCondition = Attaching;
290 wConditionScreen = "[p attached]";
291 }
292 // see if inner glue volumes are needed
293 if (containerVolumeConfig.rMin > existingVolumeConfig.rMin) {
294 nVolumeConfig.rMin = existingVolumeConfig.rMin;
295 pVolumeConfig.rMin = existingVolumeConfig.rMin;
296 } else {
297 fGapVolumeConfig.present = true;
298 // get the zMin/zMax boundaries
299 fGapVolumeConfig.adaptZ(existingVolumeConfig);
300 fGapVolumeConfig.rMin = containerVolumeConfig.rMin;
301 fGapVolumeConfig.rMax = existingVolumeConfig.rMin;
302 }
303 // see if outer glue volumes are needed
304 if (containerVolumeConfig.rMax < existingVolumeConfig.rMax) {
305 nVolumeConfig.rMax = existingVolumeConfig.rMax;
306 pVolumeConfig.rMax = existingVolumeConfig.rMax;
307 } else {
308 sGapVolumeConfig.present = true;
309 // get the zMin/zMax boundaries
310 sGapVolumeConfig.adaptZ(existingVolumeConfig);
311 sGapVolumeConfig.rMin = existingVolumeConfig.rMax;
312 sGapVolumeConfig.rMax = containerVolumeConfig.rMax;
313 }
314 } else {
315 // full wrapping or full insertion case
316 if (existingVolumeConfig.rMax < containerVolumeConfig.rMin) {
317 // Full wrapping case
318 // - set the rMin
319 nVolumeConfig.rMin = existingVolumeConfig.rMax;
320 cVolumeConfig.rMin = existingVolumeConfig.rMax;
321 pVolumeConfig.rMin = existingVolumeConfig.rMax;
322 // - set the rMax
323 nVolumeConfig.rMax = containerVolumeConfig.rMax;
324 cVolumeConfig.rMax = containerVolumeConfig.rMax;
325 pVolumeConfig.rMax = containerVolumeConfig.rMax;
326 // will wrap the new volume(s) around existing
327 wCondition = Wrapping;
328 wConditionScreen = "[fully wrapped]";
329 } else if (existingVolumeConfig.rMin > containerVolumeConfig.rMax) {
330 // full insertion case
331 // set the rMax
332 nVolumeConfig.rMax = existingVolumeConfig.rMin;
333 cVolumeConfig.rMax = existingVolumeConfig.rMin;
334 pVolumeConfig.rMax = existingVolumeConfig.rMin;
335 // set the rMin
336 nVolumeConfig.rMin = containerVolumeConfig.rMin;
337 cVolumeConfig.rMin = containerVolumeConfig.rMin;
338 pVolumeConfig.rMin = containerVolumeConfig.rMin;
339 // will insert the new volume(s) into existing
340 wCondition = Inserting;
341 wConditionScreen = "[fully inserted]";
342 } else if (cVolumeConfig.wraps(existingVolumeConfig)) {
343 // central wrapping case
344 // set the rMax
345 nVolumeConfig.rMax = containerVolumeConfig.rMax;
346 cVolumeConfig.rMax = containerVolumeConfig.rMax;
347 pVolumeConfig.rMax = containerVolumeConfig.rMax;
348 // set the rMin
349 nVolumeConfig.rMin = existingVolumeConfig.rMin;
350 cVolumeConfig.rMin = existingVolumeConfig.rMax;
351 pVolumeConfig.rMin = existingVolumeConfig.rMin;
352 // set the Central Wrapping
353 wCondition = CentralWrapping;
354 wConditionScreen = "[centrally wrapped]";
355 } else if (existingVolumeConfig.wraps(cVolumeConfig)) {
356 // central insertion case
357 // set the rMax
358 nVolumeConfig.rMax = containerVolumeConfig.rMax;
359 cVolumeConfig.rMax = existingVolumeConfig.rMin;
360 pVolumeConfig.rMax = containerVolumeConfig.rMax;
361 // set the rMin
362 nVolumeConfig.rMin = containerVolumeConfig.rMin;
363 cVolumeConfig.rMin = containerVolumeConfig.rMin;
364 pVolumeConfig.rMin = containerVolumeConfig.rMin;
365 // set the Central Wrapping
366 wCondition = CentralWrapping;
367 wConditionScreen = "[centrally inserted]";
368 } else if ((existingVolumeConfig.rMax > containerVolumeConfig.rMin &&
369 existingVolumeConfig.rMin < containerVolumeConfig.rMin) ||
370 (existingVolumeConfig.rMax > containerVolumeConfig.rMax &&
371 existingVolumeConfig.rMin < containerVolumeConfig.rMax)) {
372 // The volumes are overlapping this shouldn't be happening return an
373 // error
374 throw std::invalid_argument(
375 "Volumes are overlapping, this shouldn't be happening. Please "
376 "check your geometry building.");
377 }
378
379 // check if gaps are needed
380 //
381 // the gap reference is either the container for FULL wrapping,
382 // insertion
383 // or it is the centralVolume for central wrapping, insertion
384 VolumeConfig referenceVolume =
385 (wCondition == Wrapping || wCondition == Inserting)
386 ? containerVolumeConfig
387 : cVolumeConfig;
388 // - at the negative sector
389 if (existingVolumeConfig.zMin > referenceVolume.zMin) {
390 fGapVolumeConfig.present = true;
391 fGapVolumeConfig.adaptR(existingVolumeConfig);
392 fGapVolumeConfig.zMin = referenceVolume.zMin;
393 fGapVolumeConfig.zMax = existingVolumeConfig.zMin;
394 } else {
395 // adapt lower z boundary
396 if (nVolumeConfig.present) {
397 nVolumeConfig.zMin = existingVolumeConfig.zMin;
398 } else if (cVolumeConfig.present) {
399 cVolumeConfig.zMin = existingVolumeConfig.zMin;
400 }
401 }
402 // - at the positive sector
403 if (existingVolumeConfig.zMax < referenceVolume.zMax) {
404 sGapVolumeConfig.present = true;
405 sGapVolumeConfig.adaptR(existingVolumeConfig);
406 sGapVolumeConfig.zMin = existingVolumeConfig.zMax;
407 sGapVolumeConfig.zMax = referenceVolume.zMax;
408 } else {
409 // adapt higher z boundary
410 if (pVolumeConfig.present) {
411 pVolumeConfig.zMax = existingVolumeConfig.zMax;
412 } else if (cVolumeConfig.present) {
413 cVolumeConfig.zMax = existingVolumeConfig.zMax;
414 }
415 }
416 }
417 }
418 return;
419 }
420
423 std::string toString() const {
424 // for screen output
425 std::stringstream sl;
426 if (containerVolumeConfig.present) {
427 sl << "New container built with configuration: "
428 << containerVolumeConfig.toString() << '\n';
429 }
430 // go through the new ones first
431 if (nVolumeConfig.present) {
432 sl << " - n: Negative Endcap, current configuration: "
433 << nVolumeConfig.toString() << '\n';
434 }
435 if (cVolumeConfig.present) {
436 sl << " - c: Barrel, current configuration: "
437 << cVolumeConfig.toString() << '\n';
438 }
439 if (pVolumeConfig.present) {
440 sl << " - p: Negative Endcap, current configuration: "
441 << pVolumeConfig.toString() << '\n';
442 }
443 if (existingVolumeConfig.present) {
444 sl << "Existing volume with configuration: "
445 << existingVolumeConfig.toString() << '\n';
446 if (fGapVolumeConfig.present) {
447 sl << " - g1: First gap volume, configuration : "
448 << fGapVolumeConfig.toString() << '\n';
449 }
450 if (sGapVolumeConfig.present) {
451 sl << " - g2: Second gap volume, configuration : "
452 << sGapVolumeConfig.toString() << '\n';
453 }
454 if (wCondition != Undefined) {
455 sl << "WrappingCondition = " << wCondition << '\n';
456 }
457 }
458 return sl.str();
459 }
