179 Particle &correctEnergy(double delta) {
180 const auto newEnergy = std::hypot(m_mass, m_absMomentum) + delta;
181 if (newEnergy <= m_mass) {
182 m_absMomentum = 0.;
183 } else {
184 m_absMomentum = std::sqrt(newEnergy * newEnergy - m_mass * m_mass);
185 }
186 return *this;
187 }

332 Acts::Result<Acts::BoundTrackParameters> boundParameters(
333 const Acts::GeometryContext &gctx) const {
334 if (!hasReferenceSurface()) {
335 return Acts::Result<Acts::BoundTrackParameters>::failure(
336 std::error_code());
337 }
338 Acts::Result<Acts::Vector2> localResult =
339 m_referenceSurface->globalToLocal(gctx, position(), direction());
340 if (!localResult.ok()) {
341 return localResult.error();
342 }
343 Acts::BoundVector params;
344 params << localResult.value(), phi(), theta(), qOverP(), time();
345 return Acts::BoundTrackParameters(referenceSurface()->getSharedPtr(),
346 params, std::nullopt, hypothesis());
347 }
