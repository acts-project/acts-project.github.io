31BinUtility adjustBinUtility(const BinUtility& bu,
32 const CylinderVolumeBounds& cBounds,
33 const Transform3& transform) {
34 // Default constructor
35 BinUtility uBinUtil(transform);
36 // The parameters from the cylinder bounds
37 double minR = cBounds.get(CylinderVolumeBounds::eMinR);
38 double maxR = cBounds.get(CylinderVolumeBounds::eMaxR);
39 double minPhi = -cBounds.get(CylinderVolumeBounds::eHalfPhiSector);
40 double maxPhi = cBounds.get(CylinderVolumeBounds::eHalfPhiSector);
41 double minZ = -cBounds.get(CylinderVolumeBounds::eHalfLengthZ);
42 double maxZ = cBounds.get(CylinderVolumeBounds::eHalfLengthZ);
43 // Retrieve the binning data
44 const std::vector<BinningData>& bData = bu.binningData();
45 // Loop over the binning data and adjust the dimensions
46 for (auto& bd : bData) {
47 // The binning value
48 AxisDirection bval = bd.binvalue;
49 // Throw exceptions is stuff doesn't make sense:
50 // - not the right binning value
51 // - not equidistant
52 if (bd.type == arbitrary) {
53 throw std::invalid_argument("Arbitrary binning can not be adjusted.");
54 } else if (bval != AxisDirection::AxisR && bval != AxisDirection::AxisPhi &&
55 bval != AxisDirection::AxisZ) {
56 throw std::invalid_argument("Cylinder volume binning must be: phi, r, z");
57 }
58 float min = 0;
59 float max = 0;
60 // Perform the value adjustment
61 if (bval == AxisDirection::AxisPhi) {
62 min = minPhi;
63 max = maxPhi;
64 } else if (bval == AxisDirection::AxisR) {
65 min = minR;
66 max = maxR;
67 } else if (bval == AxisDirection::AxisZ) {
68 min = minZ;
69 max = maxZ;
70 }
71 // Create the updated BinningData
72 BinningData uBinData(bd.option, bval, bd.bins(), min, max);
73 uBinUtil += BinUtility(uBinData);
74 }
75
76 return uBinUtil;
77}

196BinUtility adjustBinUtility(const BinUtility& bu, const Volume& volume) {
197 auto cyBounds =
198 dynamic_cast<const CylinderVolumeBounds*>(&(volume.volumeBounds()));
199 auto cutcylBounds =
200 dynamic_cast<const CutoutCylinderVolumeBounds*>(&(volume.volumeBounds()));
201 auto cuBounds =
202 dynamic_cast<const CuboidVolumeBounds*>(&(volume.volumeBounds()));
203
204 if (cyBounds != nullptr) {
205 // Cylinder bounds
206 return adjustBinUtility(bu, *cyBounds, volume.transform());
207
208 } else if (cutcylBounds != nullptr) {
209 // Cutout Cylinder bounds
210 return adjustBinUtility(bu, *cutcylBounds, volume.transform());
211
212 } else if (cuBounds != nullptr) {
213 // Cuboid bounds
214 return adjustBinUtility(bu, *cuBounds, volume.transform());
215 }
216
217 throw std::invalid_argument(
218 "Bin adjustment not implemented for this volume yet!");
219}
