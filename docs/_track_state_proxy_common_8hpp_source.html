321 BoundSubspaceIndices boundSubspace = projectorSubspaceIndices();

349 template <std::ranges::sized_range index_range_t>
350 void setProjectorSubspaceIndices(const index_range_t& subspaceIndices)
351 requires(!read_only &&
352 std::convertible_to<std::ranges::range_value_t<index_range_t>,
353 std::uint8_t>)
354 {
355 assert(derived().template has<detail_tsp::kProjectorKey>());
356 assert(subspaceIndices.size() <= eBoundSize);
357 BoundSubspaceIndices boundSubspace{};
358 std::transform(subspaceIndices.begin(), subspaceIndices.end(),
359 boundSubspace.begin(),
360 [](auto i) { return static_cast<std::uint8_t>(i); });
361 derived()
362 .template component<SerializedSubspaceIndices,
363 detail_tsp::kProjectorKey>() =
364 serializeSubspaceIndices(boundSubspace);
365 }

500 template <typename val_t, typename cov_t>
501 void allocateCalibrated(const Eigen::DenseBase<val_t>& val,
502 const Eigen::DenseBase<cov_t>& cov)
503 requires(!read_only && Concepts::eigen_base_is_fixed_size<val_t> &&
504 Concepts::eigen_bases_have_same_num_rows<val_t, cov_t> &&
505 Concepts::eigen_base_is_square<cov_t> &&
506 Eigen::PlainObjectBase<val_t>::RowsAtCompileTime <=
507 static_cast<std::underlying_type_t<BoundIndices>>(eBoundSize))
508 {
509 constexpr std::size_t measdim =
510 static_cast<std::size_t>(val_t::RowsAtCompileTime);
511 derived().allocateCalibrated(measdim);
512 calibrated<measdim>() = val;
513 calibratedCovariance<measdim>() = cov;
514 }
