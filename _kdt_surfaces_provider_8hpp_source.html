31template <std::size_t kDIM = 2u, std::size_t bSize = 100u>
32class KdtSurfaces {
33 public:
35 using KDTS =
36 KDTree<kDIM, std::shared_ptr<Surface>, double, std::array, bSize>;
37
39 using Query = std::array<double, kDIM>;
40
42 using Entry = std::pair<Query, std::shared_ptr<Surface>>;
43
50 KdtSurfaces(const GeometryContext& gctx,
51 const std::vector<std::shared_ptr<Surface>>& surfaces,
52 const std::array<AxisDirection, kDIM>& casts,
53 const std::shared_ptr<IReferenceGenerator>& rgen =
54 std::make_shared<PolyhedronReferenceGenerator>())
55 : m_kdt(nullptr), m_casts(casts), m_rGenerator(rgen) {
56 // Simple check if the dimension is correct
57 if (kDIM == 0u) {
58 throw std::invalid_argument(
59 "KdtSurfaces: dimension and/or cast rules are incorrect.");
60 }
61 // Fill the tree from surfaces
62 std::vector<Entry> kdtEntries;
63 kdtEntries.reserve(surfaces.size());
64 for (auto& s : surfaces) {
65 // Generate the references and the center of gravity from it
66 const auto references = m_rGenerator->references(gctx, *s);
67 std::vector<Query> castedReferences;
68 castedReferences.reserve(references.size());
69 for (const auto& r : references) {
70 // Now cast into the correct fill position
71 Query rc = {};
72 fillCasts(r, rc, std::make_integer_sequence<std::size_t, kDIM>{});
73 castedReferences.push_back(rc);
74 }
75 // Calculate the center of gravity in casted frame
76 kdtEntries.push_back({cog(castedReferences), s});
77 }
78 // Create the KDTree
79 m_kdt = std::make_unique<KDTS>(std::move(kdtEntries));
80 }
81
87 std::vector<std::shared_ptr<Surface>> surfaces(
88 const RangeXD<kDIM, double>& range) const {
89 // Strip the surfaces
90 std::vector<std::shared_ptr<Surface>> surfacePtrs;
91 auto surfaceQuery = m_kdt->rangeSearchWithKey(range);
92 std::ranges::for_each(
93 surfaceQuery, [&](auto& surf) { surfacePtrs.push_back(surf.second); });
94 return surfacePtrs;
95 }
96
102 std::vector<std::shared_ptr<Surface>> surfaces(const Extent& extent) const {
103 RangeXD<kDIM, double> qRange;
104 for (auto [ibv, v] : enumerate(m_casts)) {
105 qRange[ibv] = extent.range(v);
106 }
107 return surfaces(qRange);
108 }
109
110 private:
112 std::unique_ptr<KDTS> m_kdt = nullptr;
113
115 std::array<AxisDirection, kDIM> m_casts = {};
116
118 std::shared_ptr<IReferenceGenerator> m_rGenerator{nullptr};
119
123 template <typename Array, std::size_t... idx>
124 void fillCasts(const Vector3& position, Array& a,
125 std::index_sequence<idx...> /*indices*/) const {
126 ((a[idx] = VectorHelpers::cast(position, m_casts[idx])), ...);
127 }
128
140 Query cog(const std::vector<Query>& cQueries) const {
141 // If there is only one position, return it
142 if (cQueries.size() == 1) {
143 return cQueries.front();
144 }
145 // Build the center of gravity of the n positions
146 Query c{};
147 float weight = 1. / cQueries.size();
148 for (auto& q : cQueries) {
149 std::transform(c.begin(), c.end(), q.begin(), c.begin(),
150 std::plus<double>());
151 }
152 std::ranges::for_each(c, [&](auto& v) { v *= weight; });
153 return c;
154 }
155};

161template <std::size_t kDIM = 2u, std::size_t bSize = 100u>
162class KdtSurfacesProvider : public ISurfacesProvider {
163 public:
169 KdtSurfacesProvider(std::shared_ptr<KdtSurfaces<kDIM, bSize>> kdts,
170 const Extent& kregion)
171 : m_kdt(std::move(kdts)), m_region(kregion) {
173 if (m_kdt == nullptr) {
174 throw std::invalid_argument(
175 "KdtSurfacesProvider: no KDTree structure provided.");
176 }
177 }
178
182 std::vector<std::shared_ptr<Surface>> surfaces(
183 [[maybe_unused]] const GeometryContext& gctx) const final {
184 return m_kdt->surfaces(m_region);
185 }
186
187 private:
188 std::shared_ptr<KdtSurfaces<kDIM, bSize, reference_generator>> m_kdt =
189 nullptr;
191 Extent m_region;
192};
