58template <typename T>
59class TransitiveConstPointer {
60 public:
61 using element_type = T;
62 TransitiveConstPointer() = default;
63 explicit TransitiveConstPointer(T* ptr) : m_ptr{ptr} {}
64
65 template <typename U>
66 explicit TransitiveConstPointer(const TransitiveConstPointer<U>& other)
67 : m_ptr{other.ptr()} {}
68
69 template <typename U>
70 TransitiveConstPointer& operator=(const TransitiveConstPointer<U>& other) {
71 m_ptr = other.m_ptr;
72 return *this;
73 }
74
75 template <typename U>
76 bool operator==(const TransitiveConstPointer<U>& other) const {
77 return m_ptr == other.m_ptr;
78 }
79
80 const T* operator->() const { return m_ptr; }
81
82 T* operator->() { return m_ptr; }
83
84 template <typename U>
85 friend class TransitiveConstPointer;
86
87 const T& operator*() const { return *m_ptr; }
88
89 T& operator*() { return *m_ptr; }
90
91 explicit operator bool() const { return m_ptr != nullptr; }
92
93 private:
94 T* ptr() const { return m_ptr; }
95
96 T* m_ptr{nullptr};
97};

101template <std::size_t Size, bool ReadOnlyMaps = true>
102struct FixedSizeTypes {
103 constexpr static auto Flags = Eigen::ColMajor | Eigen::AutoAlign;
104
105 // single items
106 using Coefficients = Eigen::Matrix<double, Size, 1, Flags>;
107 using Covariance = Eigen::Matrix<double, Size, Size, Flags>;
108 using CoefficientsMap = Eigen::Map<ConstIf<Coefficients, ReadOnlyMaps>>;
109 using CovarianceMap = Eigen::Map<ConstIf<Covariance, ReadOnlyMaps>>;
110
111 using DynamicCoefficients = Eigen::Matrix<double, Eigen::Dynamic, 1, Flags>;
112 using DynamicCovariance =
113 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, Flags>;
114 using DynamicCoefficientsMap =
115 Eigen::Map<ConstIf<DynamicCoefficients, ReadOnlyMaps>>;
116 using DynamicCovarianceMap =
117 Eigen::Map<ConstIf<DynamicCovariance, ReadOnlyMaps>>;
118};

642 requires(!ReadOnly &&
643 std::convertible_to<std::ranges::range_value_t<index_range_t>,

907 template <TrackStateProxyConcept track_state_proxy_t>
908 void copyFrom(const track_state_proxy_t& other,
909 TrackStatePropMask mask = TrackStatePropMask::All,
910 bool onlyAllocated = true)
911 requires(!ReadOnly)
912 {
913 using PM = TrackStatePropMask;
914
915 if (onlyAllocated) {
916 auto dest = getMask();
917 auto src = other.getMask() &
918 mask; // combine what we have with what we want to copy
919
920 if (ACTS_CHECK_BIT(src, PM::Calibrated)) {
921 // on-demand allocate calibrated
922 dest |= PM::Calibrated;
923 }
924
925 if ((static_cast<std::underlying_type_t<TrackStatePropMask>>(
926 (src ^ dest) & src) != 0 ||
927 dest == TrackStatePropMask::None ||
928 src == TrackStatePropMask::None) &&
929 mask != TrackStatePropMask::None) {
930 throw std::runtime_error(
931 "Attempt track state copy with incompatible allocations");
932 }
933
934 // we're sure now this has correct allocations, so just copy
935 if (ACTS_CHECK_BIT(src, PM::Predicted)) {
936 predicted() = other.predicted();
937 predictedCovariance() = other.predictedCovariance();
938 }
939
940 if (ACTS_CHECK_BIT(src, PM::Filtered)) {
941 filtered() = other.filtered();
942 filteredCovariance() = other.filteredCovariance();
943 }
944
945 if (ACTS_CHECK_BIT(src, PM::Smoothed)) {
946 smoothed() = other.smoothed();
947 smoothedCovariance() = other.smoothedCovariance();
948 }
949
950 if (other.hasUncalibratedSourceLink()) {
951 setUncalibratedSourceLink(other.getUncalibratedSourceLink());
952 }
953
954 if (ACTS_CHECK_BIT(src, PM::Jacobian)) {
955 jacobian() = other.jacobian();
956 }
957
958 if (ACTS_CHECK_BIT(src, PM::Calibrated)) {
959 visit_measurement(other.calibratedSize(), [&](auto N) {
960 constexpr int measdim = decltype(N)::value;
961 allocateCalibrated(
962 other.template calibrated<measdim>().eval(),
963 other.template calibratedCovariance<measdim>().eval());
964 });
965
966 setProjectorSubspaceIndices(other.projectorSubspaceIndices());
967 }
968 } else {
969 if (ACTS_CHECK_BIT(mask, PM::Predicted) &&
970 has<detail_tsp::kPredictedKey>() &&
971 other.template has<detail_tsp::kPredictedKey>()) {
972 predicted() = other.predicted();
973 predictedCovariance() = other.predictedCovariance();
974 }
975
976 if (ACTS_CHECK_BIT(mask, PM::Filtered) &&
977 has<detail_tsp::kFilteredKey>() &&
978 other.template has<detail_tsp::kFilteredKey>()) {
979 filtered() = other.filtered();
980 filteredCovariance() = other.filteredCovariance();
981 }
982
983 if (ACTS_CHECK_BIT(mask, PM::Smoothed) &&
984 has<detail_tsp::kSmoothedKey>() &&
985 other.template has<detail_tsp::kSmoothedKey>()) {
986 smoothed() = other.smoothed();
987 smoothedCovariance() = other.smoothedCovariance();
988 }
989
990 if (other.hasUncalibratedSourceLink()) {
991 setUncalibratedSourceLink(other.getUncalibratedSourceLink());
992 }
993
994 if (ACTS_CHECK_BIT(mask, PM::Jacobian) &&
995 has<detail_tsp::kJacobianKey>() &&
996 other.template has<detail_tsp::kJacobianKey>()) {
997 jacobian() = other.jacobian();
998 }
999
1000 // NOTE: we should not check hasCalibrated on this, since it
1001 // may be not yet allocated
1002 if (ACTS_CHECK_BIT(mask, PM::Calibrated) &&
1003 other.template has<detail_tsp::kCalibratedKey>()) {
1004 visit_measurement(other.calibratedSize(), [&](auto N) {
1005 constexpr int measdim = decltype(N)::value;
1006 allocateCalibrated(
1007 other.template calibrated<measdim>().eval(),
1008 other.template calibratedCovariance<measdim>().eval());
1009 });
1010
1011 setProjectorSubspaceIndices(other.projectorSubspaceIndices());
1012 }
1013 }
1014
1015 chi2() = other.chi2();
1016 pathLength() = other.pathLength();
1017 typeFlags() = other.typeFlags();
1018
1019 if (other.hasReferenceSurface()) {
1020 setReferenceSurface(other.referenceSurface().getSharedPtr());
1021 }
1022
1023 m_traj->copyDynamicFrom(m_istate, other.container(), other.index());
1024 }
