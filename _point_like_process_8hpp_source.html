54template <detail::PointLikeProcessConcept physics_t,
55 typename input_particle_selector_t,
56 typename output_particle_selector_t,
57 typename child_particle_selector_t = output_particle_selector_t>
58struct PointLikeProcess {
60 physics_t physics;
62 input_particle_selector_t selectInputParticle;
64 output_particle_selector_t selectOutputParticle;
66 child_particle_selector_t selectChildParticle;
67
68 template <class generator_t>
69 std::pair<double, double> generatePathLimits(generator_t& generator,
70 const Particle& particle) const {
71 return physics.generatePathLimits(generator, particle);
72 }
73
74 template <class generator_t>
75 bool run(generator_t& rng, Particle& particle,
76 std::vector<Particle>& generatedParticles) const {
77 if (!selectInputParticle(particle)) {
78 return false;
79 }
80
81 std::vector<Particle> children;
82 physics.run(rng, particle, children);
83
84 std::copy_if(std::begin(children), std::end(children),
85 std::back_inserter(generatedParticles), selectChildParticle);
86
87 return !selectOutputParticle(particle);
88 }
89};
