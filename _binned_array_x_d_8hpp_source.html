35template <class T>
36class BinnedArrayXD : public BinnedArray<T> {
38 using TAP = std::pair<T, Vector3>;
39
40 public:
43 explicit BinnedArrayXD(T object)
44 : BinnedArray<T>(),
45 m_objectGrid(
46 1, std::vector<std::vector<T>>(1, std::vector<T>(1, nullptr))),
47 m_arrayObjects({object}),
48 m_binUtility(nullptr) {
50 m_objectGrid[0][0][0] = object;

52
58 BinnedArrayXD(const std::vector<TAP>& tapvector,
59 std::unique_ptr<const BinUtility> bu)
60 : BinnedArray<T>(),
61 m_objectGrid(bu->bins(2),
62 std::vector<std::vector<T>>(
63 bu->bins(1), std::vector<T>(bu->bins(0), nullptr))),
64 m_arrayObjects(),
65 m_binUtility(std::move(bu)) {
67 m_arrayObjects.reserve(tapvector.size());
69 for (auto& tap : tapvector) {
71 if (m_binUtility->inside(tap.second)) {
72 // butil to the array store - if the bingen
73 // dimension is smaller 1,2 it will provide 0
74 auto bins = m_binUtility->binTriple(tap.second);
76 m_objectGrid[bins[2]][bins[1]][bins[0]] = tap.first;
78 if (!rangeContainsValue(m_arrayObjects, tap.first)) {
79 m_arrayObjects.push_back(tap.first);
80 }
81 }
82 }
83 }
84
89 BinnedArrayXD(const std::vector<std::vector<std::vector<T>>>& grid,
90 std::unique_ptr<const BinUtility> bu)
91 : BinnedArray<T>(),
92 m_objectGrid(grid),
93 m_arrayObjects(),
94 m_binUtility(std::move(bu)) {
95 // get the total dimension
96 std::size_t objects =
97 m_binUtility->bins(0) * m_binUtility->bins(1) * m_binUtility->bins(2);
99 m_arrayObjects.reserve(objects);
101 for (auto& o2 : m_objectGrid) {
102 for (auto& o1 : o2) {
103 for (auto& o0 : o1) {
105 if (o0 && !rangeContainsValue(m_arrayObjects, o0)) {
106 m_arrayObjects.push_back(o0);
107 }
108 }
109 }
110 }
111 }

131 T object(const Vector2& lposition,
132 std::array<std::size_t, 3>& bins) const final {
133 if (m_binUtility) {
134 std::size_t bdim = m_binUtility->dimensions();
135 bins[2] = bdim > 2 ? m_binUtility->bin(lposition, 2) : 0;
136 bins[1] = bdim > 1 ? m_binUtility->bin(lposition, 1) : 0;
137 bins[0] = m_binUtility->bin(lposition, 0);
138 return m_objectGrid[bins[2]][bins[1]][bins[0]];
139 }
140 return m_objectGrid[0][0][0];
141 }
