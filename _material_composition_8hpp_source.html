37class ElementFraction {
38 public:
43 constexpr ElementFraction(unsigned int e, float f)
44 : m_element(static_cast<std::uint8_t>(e)),
45 m_fraction(static_cast<std::uint8_t>(
46 f * std::numeric_limits<std::uint8_t>::max())) {
47 assert((0u < e) && ("The atomic number must be positive"));
48 assert((0.0f <= f) && (f <= 1.0f) && "Relative fraction must be in [0,1]");
49 }
50
54 constexpr explicit ElementFraction(unsigned int e, unsigned int w)
55 : m_element(static_cast<std::uint8_t>(e)),
56 m_fraction(static_cast<std::uint8_t>(w)) {
57 assert((0u < e) && ("The atomic number must be positive"));
58 assert((w < 256u) && "Integer weight must be in [0,256)");
59 }
60
62 ElementFraction() = delete;
64 ElementFraction(ElementFraction&&) = default;
66 ElementFraction(const ElementFraction&) = default;
67 ~ElementFraction() = default;
70 ElementFraction& operator=(ElementFraction&&) = default;
73 ElementFraction& operator=(const ElementFraction&) = default;
74
77 constexpr std::uint8_t element() const { return m_element; }
80 constexpr float fraction() const {
81 return static_cast<float>(m_fraction) /
82 std::numeric_limits<std::uint8_t>::max();
83 }
84
85 private:
86 // element atomic number
87 std::uint8_t m_element;
88 // element fraction in the compound scaled to the [0,256) range.
89 std::uint8_t m_fraction;

91 friend constexpr bool operator==(ElementFraction lhs, ElementFraction rhs) {

93 (lhs.m_element == rhs.m_element);

106 }

112class MaterialComposition {
113 public:
115 MaterialComposition() = default;
120 explicit MaterialComposition(std::vector<ElementFraction> elements)
121 : m_elements(std::move(elements)) {
122 std::ranges::sort(m_elements, std::less<ElementFraction>{});
123 // compute the total weight first
124 unsigned total = 0u;
125 for (const auto& element : m_elements) {
126 total += element.m_fraction;
127 }
128 // compute scale factor into the [0, 256) range
129 float scale = float{std::numeric_limits<std::uint8_t>::max()} / total;
130 for (auto& element : m_elements) {
131 element.m_fraction =
132 static_cast<std::uint8_t>(element.m_fraction * scale);
133 }
134 }
135
137 MaterialComposition(MaterialComposition&&) = default;
139 MaterialComposition(const MaterialComposition&) = default;
140 ~MaterialComposition() = default;
143 MaterialComposition& operator=(MaterialComposition&&) = default;
146 MaterialComposition& operator=(const MaterialComposition&) = default;
147
150 auto begin() const { return m_elements.begin(); }
153 auto end() const { return m_elements.end(); }
154
156 explicit operator bool() const { return !m_elements.empty(); }
159 std::size_t size() const { return m_elements.size(); }
160
161 private:
162 std::vector<ElementFraction> m_elements;
163
164 friend inline bool operator==(const MaterialComposition& lhs,
165 const MaterialComposition& rhs) {
166 return lhs.m_elements == rhs.m_elements;
167 }
168
170 friend std::ostream& operator<<(std::ostream& os,
171 const MaterialComposition& mc) {
172 os << "MaterialComposition(elements=[";
173 for (std::size_t i = 0; i < mc.m_elements.size(); ++i) {
174 if (i > 0) {
175 os << ", ";
176 }
177 os << mc.m_elements[i];
178 }
179 os << "])";
180 return os;
181 }
182};
