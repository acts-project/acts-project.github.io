30inline Eigen::Matrix<T, 3, 1> makeDirectionFromPhiEta(T phi, T eta) {
31 const auto coshEtaInv = 1 / std::cosh(eta);
32 return {
33 std::cos(phi) * coshEtaInv,
34 std::sin(phi) * coshEtaInv,

93template <typename InputVector>
94inline auto createCurvilinearUnitU(
95 const Eigen::MatrixBase<InputVector>& direction) {
96 EIGEN_STATIC_ASSERT_FIXED_SIZE(InputVector);
97 EIGEN_STATIC_ASSERT_VECTOR_ONLY(InputVector);
98 static_assert(3 <= InputVector::RowsAtCompileTime,
99 "Direction vector must be at least three-dimensional.");
100
101 using OutputVector = typename InputVector::PlainObject;
102 using OutputScalar = typename InputVector::Scalar;
103
104 OutputVector unitU = OutputVector::Zero();
105 // explicit version of U = Z x T
106 unitU[0] = -direction[1];
107 unitU[1] = direction[0];
108 const auto scale = unitU.template head<2>().norm();
109 // if the absolute scale is tiny, the initial direction vector is aligned with
110 // the z-axis. the ZxT product is ill-defined since any vector in the x-y
111 // plane would be orthogonal to the direction. fix the U unit vector along the
112 // x-axis to avoid this numerical instability.
113 if (scale < (16 * std::numeric_limits<OutputScalar>::epsilon())) {
114 unitU[0] = 1;
115 unitU[1] = 0;
116 } else {
117 unitU.template head<2>() /= scale;
118 }
119 return unitU;
120}

135template <typename InputVector>
136inline auto createCurvilinearUnitVectors(
137 const Eigen::MatrixBase<InputVector>& direction) {
138 EIGEN_STATIC_ASSERT_FIXED_SIZE(InputVector);
139 EIGEN_STATIC_ASSERT_VECTOR_ONLY(InputVector);
140 static_assert(3 <= InputVector::RowsAtCompileTime,
141 "Direction vector must be at least three-dimensional.");
142
143 using OutputVector = typename InputVector::PlainObject;
144
145 std::pair<OutputVector, OutputVector> unitVectors;
146 unitVectors.first = createCurvilinearUnitU(direction);
147 unitVectors.second = direction.cross(unitVectors.first);
148 unitVectors.second.normalize();
149 return unitVectors;
150}
