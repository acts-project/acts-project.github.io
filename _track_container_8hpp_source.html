37template <TrackContainerBackend track_container_t,

39 template <typename> class holder_t = detail::RefHolder>
40class TrackContainer {
41 public:
43 static constexpr bool ReadOnly =
44 IsReadOnlyTrackContainer<track_container_t>::value;
45
48 static constexpr bool TrackStateReadOnly =
49 IsReadOnlyMultiTrajectory<traj_t>::value;
50
51 static_assert(ReadOnly == TrackStateReadOnly,
52 "Either both track container and track state container need to "
53 "be readonly or both have to be readwrite");
54
56 using IndexType = TrackIndexType;
57
59 static constexpr IndexType kInvalid = kTrackIndexInvalid;
60
63 using TrackProxy =
64 Acts::TrackProxy<track_container_t, traj_t, holder_t, false>;

68 using ConstTrackProxy =
69 Acts::TrackProxy<track_container_t, traj_t, holder_t, true>;
70
71 static_assert(ConstTrackProxyConcept<ConstTrackProxy>,
72 "ConstTrackProxy must fulfill the TrackProxyConcept");
73 static_assert(ReadOnly || MutableTrackProxyConcept<TrackProxy>,
74 "TrackProxy must fulfill the TrackProxyConcept");
75
77 using TrackContainerBackend = track_container_t;
79 using TrackStateContainerBackend = traj_t;
80
82 using TrackStateProxy = typename MultiTrajectory<traj_t>::TrackStateProxy;
84 using ConstTrackStateProxy =
85 typename MultiTrajectory<traj_t>::ConstTrackStateProxy;

88 using size_type = IndexType;
90 using iterator =
91 detail::ContainerIterator<TrackContainer, TrackProxy, IndexType, false>;

94 detail::ContainerIterator<TrackContainer, ConstTrackProxy, IndexType,

432 auto reverseTrackStateRange(IndexType itrack)

451 auto forwardTrackStateRange(IndexType itrack)
452 requires(!ReadOnly)
453 {
454 auto stem = component<IndexType, hashString("stemIndex")>(itrack);
455 if (stem == kInvalid) {
456 throw std::invalid_argument{"Track has no stem index"};
457 }
458 return m_traj->forwardTrackStateRange(stem);
459 }
460
465 auto forwardTrackStateRange(IndexType itrack) const {
466 auto stem = component<IndexType, hashString("stemIndex")>(itrack);
467 if (stem == kInvalid) {
468 throw std::invalid_argument{"Track has no stem index"};
469 }
470 return m_traj->forwardTrackStateRange(stem);
471 }

473 private:
474 template <typename T>
475 void copyDynamicFrom(IndexType dstIdx, const T& src, IndexType srcIdx)
476 requires(!ReadOnly)
477 {
478 const auto& dynamicKeys = src.dynamicKeys_impl();
479 for (const auto key : dynamicKeys) {
480 std::any srcPtr = src.component_impl(key, srcIdx);
481 container().copyDynamicFrom_impl(dstIdx, key, srcPtr);
482 }
483 }

485 const_if_t<ReadOnly, holder_t<track_container_t>> m_container;
486 const_if_t<ReadOnly, holder_t<traj_t>> m_traj;
487};
